47    private static final Predicate<Integer> TRUE = Predicates.alwaysTrue();
48    private static final Predicate<Integer> FALSE = Predicates.alwaysFalse();
49    private static final Predicate<Integer> NEVER_REACHED =
50        new Predicate<Integer>() {
51          @Override
52          public boolean apply(Integer i) {
53            throw new AssertionFailedError("This predicate should never have been evaluated");
54          }
55        };

57    /** Instantiable predicate with reasonable hashCode() and equals() methods. */
58    static class IsOdd implements Predicate<Integer>, Serializable {
59      private static final long serialVersionUID = 0x150ddL;
60  
61      @Override
62      public boolean apply(Integer i) {
63        return (i.intValue() & 1) == 1;
64      }
65  
66      @Override
67      public int hashCode() {
68        return 0x150dd;
69      }
70  
71      @Override
72      public boolean equals(Object obj) {
73        return obj instanceof IsOdd;
74      }
75  
76      @Override
77      public String toString() {
78        return "IsOdd";
79      }
80    }
81  
82    /**
83     * Generates a new Predicate per call.
84     *
85     * <p>Creating a new Predicate each time helps catch cases where code is using {@code x == y}
86     * instead of {@code x.equals(y)}.
87     */
88    private static IsOdd isOdd() {
89      return new IsOdd();
90    }

928       Predicate<? super Integer> expected, Predicate<? super Integer> actual) {

934   private static <T> void assertEvalsLike(
935       Predicate<? super T> expected, Predicate<? super T> actual, T input) {
936     Boolean expectedResult = null;
937     RuntimeException expectedRuntimeException = null;
938     try {
939       expectedResult = expected.apply(input);
940     } catch (RuntimeException e) {
941       expectedRuntimeException = e;
942     }
943 
944     Boolean actualResult = null;
945     RuntimeException actualRuntimeException = null;
946     try {
947       actualResult = actual.apply(input);
948     } catch (RuntimeException e) {
949       actualRuntimeException = e;
950     }
951 
952     assertEquals(expectedResult, actualResult);
953     if (expectedRuntimeException != null) {
954       assertNotNull(actualRuntimeException);
955       assertEquals(expectedRuntimeException.getClass(), actualRuntimeException.getClass());
956     }
957   }
