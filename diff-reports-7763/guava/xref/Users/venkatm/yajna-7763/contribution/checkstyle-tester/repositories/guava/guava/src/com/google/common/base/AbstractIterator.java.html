15  package com.google.common.base;
16  
17  import static com.google.common.base.Preconditions.checkState;
18  
19  import com.google.common.annotations.GwtCompatible;
20  import com.google.errorprone.annotations.CanIgnoreReturnValue;
21  import java.util.Iterator;
22  import java.util.NoSuchElementException;
23  import org.checkerframework.checker.nullness.qual.Nullable;
24  
25  /**
26   * Note this class is a copy of {@link com.google.common.collect.AbstractIterator} (for dependency
27   * reasons).
28   */
29  @GwtCompatible
30  abstract class AbstractIterator<T> implements Iterator<T> {
31    private State state = State.NOT_READY;
32  
33    protected AbstractIterator() {}
34  
35    private enum State {
36      READY,
37      NOT_READY,
38      DONE,
39      FAILED,
40    }
41  
42    private @Nullable T next;
43  
44    protected abstract T computeNext();
45  
46    @CanIgnoreReturnValue
47    protected final @Nullable T endOfData() {
48      state = State.DONE;
49      return null;
50    }
51  
52    @Override
53    public final boolean hasNext() {
54      checkState(state != State.FAILED);
55      switch (state) {
56        case DONE:
57          return false;
58        case READY:
59          return true;
60        default:
61      }
62      return tryToComputeNext();
63    }
64  
65    private boolean tryToComputeNext() {
66      state = State.FAILED; // temporary pessimism
67      next = computeNext();
68      if (state != State.DONE) {
69        state = State.READY;
70        return true;
71      }
72      return false;
73    }
74  
75    @Override
76    public final T next() {
77      if (!hasNext()) {
78        throw new NoSuchElementException();
79      }
80      state = State.NOT_READY;
81      T result = next;
82      next = null;
83      return result;
84    }
85  
86    @Override
87    public final void remove() {
88      throw new UnsupportedOperationException();
89    }
90  }
