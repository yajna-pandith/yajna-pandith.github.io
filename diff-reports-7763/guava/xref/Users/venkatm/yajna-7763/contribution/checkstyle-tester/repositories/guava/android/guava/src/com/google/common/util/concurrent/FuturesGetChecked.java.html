40  /** Static methods used to implement {@link Futures#getChecked(Future, Class)}. */
41  @GwtIncompatible
42  final class FuturesGetChecked {
43    @CanIgnoreReturnValue
44    static <V, X extends Exception> V getChecked(Future<V> future, Class<X> exceptionClass) throws X {
45      return getChecked(bestGetCheckedTypeValidator(), future, exceptionClass);
46    }
47  
48    /** Implementation of {@link Futures#getChecked(Future, Class)}. */
49    @CanIgnoreReturnValue
50    @VisibleForTesting
51    static <V, X extends Exception> V getChecked(
52        GetCheckedTypeValidator validator, Future<V> future, Class<X> exceptionClass) throws X {
53      validator.validateClass(exceptionClass);
54      try {
55        return future.get();
56      } catch (InterruptedException e) {
57        currentThread().interrupt();
58        throw newWithCause(exceptionClass, e);
59      } catch (ExecutionException e) {
60        wrapAndThrowExceptionOrError(e.getCause(), exceptionClass);
61        throw new AssertionError();
62      }
63    }
64  
65    /** Implementation of {@link Futures#getChecked(Future, Class, long, TimeUnit)}. */
66    @CanIgnoreReturnValue
67    static <V, X extends Exception> V getChecked(
68        Future<V> future, Class<X> exceptionClass, long timeout, TimeUnit unit) throws X {
69      // TODO(cpovirk): benchmark a version of this method that accepts a GetCheckedTypeValidator
70      bestGetCheckedTypeValidator().validateClass(exceptionClass);
71      try {
72        return future.get(timeout, unit);
73      } catch (InterruptedException e) {
74        currentThread().interrupt();
75        throw newWithCause(exceptionClass, e);
76      } catch (TimeoutException e) {
77        throw newWithCause(exceptionClass, e);
78      } catch (ExecutionException e) {
79        wrapAndThrowExceptionOrError(e.getCause(), exceptionClass);
80        throw new AssertionError();
81      }
82    }

84    @VisibleForTesting
85    interface GetCheckedTypeValidator {
86      void validateClass(Class<? extends Exception> exceptionClass);
87    }

89    private static GetCheckedTypeValidator bestGetCheckedTypeValidator() {
90      return GetCheckedTypeValidatorHolder.BEST_VALIDATOR;
91    }

95      return GetCheckedTypeValidatorHolder.WeakSetValidator.INSTANCE;
96    }

104   /**
105    * Provides a check of whether an exception type is valid for use with {@link
106    * FuturesGetChecked#getChecked(Future, Class)}, possibly using caching.
107    *
108    * <p>Uses reflection to gracefully fall back to when certain implementations aren't available.
109    */
110   @VisibleForTesting
111   static class GetCheckedTypeValidatorHolder {
112     static final String CLASS_VALUE_VALIDATOR_NAME =
113         GetCheckedTypeValidatorHolder.class.getName() + "$ClassValueValidator";
114 
115     static final GetCheckedTypeValidator BEST_VALIDATOR = getBestValidator();
116 
117     @IgnoreJRERequirement // getChecked falls back to another implementation if necessary
118     @J2ObjCIncompatible // ClassValue
119     enum ClassValueValidator implements GetCheckedTypeValidator {
120       INSTANCE;
121 
122       /*
123        * Static final fields are presumed to be fastest, based on our experience with
124        * UnsignedBytesBenchmark. TODO(cpovirk): benchmark this
125        */
126       private static final ClassValue<Boolean> isValidClass =
127           new ClassValue<Boolean>() {
128             @Override
129             protected Boolean computeValue(Class<?> type) {
130               checkExceptionClassValidity(type.asSubclass(Exception.class));
131               return true;
132             }
133           };
134 
135       @Override
136       public void validateClass(Class<? extends Exception> exceptionClass) {
137         isValidClass.get(exceptionClass); // throws if invalid; returns safely (and caches) if valid
138       }
139     }
140 
141     enum WeakSetValidator implements GetCheckedTypeValidator {
142       INSTANCE;
143 
144       /*
145        * Static final fields are presumed to be fastest, based on our experience with
146        * UnsignedBytesBenchmark. TODO(cpovirk): benchmark this
147        */
148       /*
149        * A CopyOnWriteArraySet<WeakReference> is faster than a newSetFromMap of a MapMaker map with
150        * weakKeys() and concurrencyLevel(1), even up to at least 12 cached exception types.
151        */
152       private static final Set<WeakReference<Class<? extends Exception>>> validClasses =
153           new CopyOnWriteArraySet<>();
154 
155       @Override
156       public void validateClass(Class<? extends Exception> exceptionClass) {
157         for (WeakReference<Class<? extends Exception>> knownGood : validClasses) {
158           if (exceptionClass.equals(knownGood.get())) {
159             return;
160           }
161           // TODO(cpovirk): if reference has been cleared, remove it?
162         }
163         checkExceptionClassValidity(exceptionClass);
164 
165         /*
166          * It's very unlikely that any loaded Futures class will see getChecked called with more
167          * than a handful of exceptions. But it seems prudent to set a cap on how many we'll cache.
168          * This avoids out-of-control memory consumption, and it keeps the cache from growing so
169          * large that doing the lookup is noticeably slower than redoing the work would be.
170          *
171          * Ideally we'd have a real eviction policy, but until we see a problem in practice, I hope
172          * that this will suffice. I have not even benchmarked with different size limits.
173          */
174         if (validClasses.size() > 1000) {
175           validClasses.clear();
176         }
177 
178         validClasses.add(new WeakReference<Class<? extends Exception>>(exceptionClass));
179       }
180     }
181 
182     /**
183      * Returns the ClassValue-using validator, or falls back to the "weak Set" implementation if
184      * unable to do so.
185      */
186     static GetCheckedTypeValidator getBestValidator() {
187       try {
188         Class<?> theClass = Class.forName(CLASS_VALUE_VALIDATOR_NAME);
189         return (GetCheckedTypeValidator) theClass.getEnumConstants()[0];
190       } catch (Throwable t) { // ensure we really catch *everything*
191         return weakSetValidator();
192       }
193     }
194   }
195 
196   // TODO(cpovirk): change parameter order to match other helper methods (Class, Throwable)?
197   private static <X extends Exception> void wrapAndThrowExceptionOrError(
198       Throwable cause, Class<X> exceptionClass) throws X {
199     if (cause instanceof Error) {
200       throw new ExecutionError((Error) cause);
201     }
202     if (cause instanceof RuntimeException) {
203       throw new UncheckedExecutionException(cause);
204     }
205     throw newWithCause(exceptionClass, cause);
206   }

213   private static boolean hasConstructorUsableByGetChecked(
214       Class<? extends Exception> exceptionClass) {
215     try {
216       Exception unused = newWithCause(exceptionClass, new Exception());
217       return true;
218     } catch (Exception e) {
219       return false;
220     }
221   }
222 
223   private static <X extends Exception> X newWithCause(Class<X> exceptionClass, Throwable cause) {
224     // getConstructors() guarantees this as long as we don't modify the array.
225     @SuppressWarnings({"unchecked", "rawtypes"})
226     List<Constructor<X>> constructors = (List) Arrays.asList(exceptionClass.getConstructors());
227     for (Constructor<X> constructor : preferringStrings(constructors)) {
228       @NullableDecl X instance = newFromConstructor(constructor, cause);
229       if (instance != null) {
230         if (instance.getCause() == null) {
231           instance.initCause(cause);
232         }
233         return instance;
234       }
235     }
236     throw new IllegalArgumentException(
237         "No appropriate constructor for exception of type "
238             + exceptionClass
239             + " in response to chained exception",
240         cause);
241   }
242 
243   private static <X extends Exception> List<Constructor<X>> preferringStrings(
244       List<Constructor<X>> constructors) {
245     return WITH_STRING_PARAM_FIRST.sortedCopy(constructors);
246   }
247 
248   private static final Ordering<Constructor<?>> WITH_STRING_PARAM_FIRST =
249       Ordering.natural()
250           .onResultOf(
251               new Function<Constructor<?>, Boolean>() {
252                 @Override
253                 public Boolean apply(Constructor<?> input) {
254                   return asList(input.getParameterTypes()).contains(String.class);
255                 }
256               })
257           .reverse();
258 
259   @NullableDecl
260   private static <X> X newFromConstructor(Constructor<X> constructor, Throwable cause) {
261     Class<?>[] paramTypes = constructor.getParameterTypes();
262     Object[] params = new Object[paramTypes.length];
263     for (int i = 0; i < paramTypes.length; i++) {
264       Class<?> paramType = paramTypes[i];
265       if (paramType.equals(String.class)) {
266         params[i] = cause.toString();
267       } else if (paramType.equals(Throwable.class)) {
268         params[i] = cause;
269       } else {
270         return null;
271       }
272     }
273     try {
274       return constructor.newInstance(params);
275     } catch (IllegalArgumentException
276         | InstantiationException
277         | IllegalAccessException
278         | InvocationTargetException e) {
279       return null;
280     }
281   }
282 
283   @VisibleForTesting
284   static boolean isCheckedException(Class<? extends Exception> type) {
285     return !RuntimeException.class.isAssignableFrom(type);
286   }
287 
288   @VisibleForTesting
289   static void checkExceptionClassValidity(Class<? extends Exception> exceptionClass) {
290     checkArgument(
291         isCheckedException(exceptionClass),
292         "Futures.getChecked exception type (%s) must not be a RuntimeException",
293         exceptionClass);
294     checkArgument(
295         hasConstructorUsableByGetChecked(exceptionClass),
296         "Futures.getChecked exception type (%s) must be an accessible class with an accessible "
297             + "constructor whose parameters (if any) must be of type String and/or Throwable",
298         exceptionClass);
299   }
