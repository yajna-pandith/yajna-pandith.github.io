73    /**
74     * Asserts that {@link BloomFilter#approximateElementCount} is within 1 percent of the expected
75     * value.
76     */
77    private static void assertApproximateElementCountGuess(BloomFilter<?> bf, int sizeGuess) {
78      assertThat(bf.approximateElementCount()).isAtLeast((long) (sizeGuess * 0.99));
79      assertThat(bf.approximateElementCount()).isAtMost((long) (sizeGuess * 1.01));
80    }

82    public void testCreateAndCheckMitz32BloomFilterWithKnownFalsePositives() {
83      int numInsertions = 1000000;
84      BloomFilter<String> bf =
85          BloomFilter.create(
86              Funnels.unencodedCharsFunnel(),
87              numInsertions,
88              0.03,
89              BloomFilterStrategies.MURMUR128_MITZ_32);
90  
91      // Insert "numInsertions" even numbers into the BF.
92      for (int i = 0; i < numInsertions * 2; i += 2) {
93        bf.put(Integer.toString(i));
94      }
95      assertApproximateElementCountGuess(bf, numInsertions);
96  
97      // Assert that the BF "might" have all of the even numbers.
98      for (int i = 0; i < numInsertions * 2; i += 2) {
99        assertTrue(bf.mightContain(Integer.toString(i)));
100     }
101 
102     // Now we check for known false positives using a set of known false positives.
103     // (These are all of the false positives under 900.)
104     ImmutableSet<Integer> falsePositives =
105         ImmutableSet.of(
106             49, 51, 59, 163, 199, 321, 325, 363, 367, 469, 545, 561, 727, 769, 773, 781);
107     for (int i = 1; i < 900; i += 2) {
108       if (!falsePositives.contains(i)) {
109         assertFalse("BF should not contain " + i, bf.mightContain(Integer.toString(i)));
110       }
111     }
112 
113     // Check that there are exactly 29824 false positives for this BF.
114     int knownNumberOfFalsePositives = 29824;
115     int numFpp = 0;
116     for (int i = 1; i < numInsertions * 2; i += 2) {
117       if (bf.mightContain(Integer.toString(i))) {
118         numFpp++;
119       }
120     }
121     assertEquals(knownNumberOfFalsePositives, numFpp);
122     double expectedReportedFpp = (double) knownNumberOfFalsePositives / numInsertions;
123     double actualReportedFpp = bf.expectedFpp();
124     assertEquals(expectedReportedFpp, actualReportedFpp, 0.00015);
125   }

277   public void testOptimalSize() {
278     for (int n = 1; n < 1000; n++) {
279       for (double fpp = Double.MIN_VALUE; fpp < 1.0; fpp += 0.001) {
280         assertTrue(BloomFilter.optimalNumOfBits(n, fpp) >= 0);

329   public void testExpectedFpp() {
330     BloomFilter<Object> bf = BloomFilter.create(HashTestUtils.BAD_FUNNEL, 10, 0.03);
331     double fpp = bf.expectedFpp();
332     assertEquals(0.0, fpp);
333     // usually completed in less than 200 iterations
334     while (fpp != 1.0) {
335       boolean changed = bf.put(new Object());
336       double newFpp = bf.expectedFpp();
337       // if changed, the new fpp is strictly higher, otherwise it is the same
338       assertTrue(changed ? newFpp > fpp : newFpp == fpp);
339       fpp = newFpp;
340     }
341   }

522   public void testNoRaceConditions() throws Exception {
523     final BloomFilter<Integer> bloomFilter =
524         BloomFilter.create(Funnels.integerFunnel(), 15_000_000, 0.01);
525 
526     // This check has to be BEFORE the loop because the random insertions can
527     // flip GOLDEN_PRESENT_KEY to true even if it wasn't explicitly inserted
528     // (false positive).
529     assertThat(bloomFilter.mightContain(GOLDEN_PRESENT_KEY)).isFalse();
530     for (int i = 0; i < NUM_PUTS; i++) {
531       bloomFilter.put(getNonGoldenRandomKey());
532     }
533     bloomFilter.put(GOLDEN_PRESENT_KEY);
534 
535     int numThreads = 12;
536     final double safetyFalsePositiveRate = 0.1;
537     final Stopwatch stopwatch = Stopwatch.createStarted();
538 
539     Runnable task =
540         new Runnable() {
541           @Override
542           public void run() {
543             do {
544               // We can't have a GOLDEN_NOT_PRESENT_KEY because false positives are
545               // possible! It's false negatives that can't happen.
546               assertThat(bloomFilter.mightContain(GOLDEN_PRESENT_KEY)).isTrue();
547 
548               int key = getNonGoldenRandomKey();
549               // We can't check that the key is mightContain() == false before the
550               // put() because the key could have already been generated *or* the
551               // bloom filter might say true even when it's not there (false
552               // positive).
553               bloomFilter.put(key);
554               // False negative should *never* happen.
555               assertThat(bloomFilter.mightContain(key)).isTrue();
556 
557               // If this check ever fails, that means we need to either bump the
558               // number of expected insertions or don't run the test for so long.
559               // Don't forget, the bloom filter slowly saturates over time and the
560               // expected false positive probability goes up!
561               assertThat(bloomFilter.expectedFpp()).isLessThan(safetyFalsePositiveRate);
562             } while (stopwatch.elapsed(TimeUnit.SECONDS) < 1);
563           }
564         };
565 
566     List<Throwable> exceptions = runThreadsAndReturnExceptions(numThreads, task);
567 
568     assertThat(exceptions).isEmpty();
569   }
570 
571   private static List<Throwable> runThreadsAndReturnExceptions(int numThreads, Runnable task) {
572     List<Thread> threads = new ArrayList<>(numThreads);
573     final List<Throwable> exceptions = new ArrayList<>(numThreads);
574     for (int i = 0; i < numThreads; i++) {
575       Thread thread = new Thread(task);
576       thread.setUncaughtExceptionHandler(
577           new UncaughtExceptionHandler() {
578             @Override
579             public void uncaughtException(Thread t, Throwable e) {
580               exceptions.add(e);
581             }
582           });
583       threads.add(thread);
584     }
585     for (Thread t : threads) {
586       t.start();
587     }
588     for (Thread t : threads) {
589       Uninterruptibles.joinUninterruptibly(t);
590     }
591     return exceptions;
592   }

594   private static int getNonGoldenRandomKey() {
595     int key;
596     do {
597       key = random.get().nextInt();
598     } while (key == GOLDEN_PRESENT_KEY);
599     return key;
600   }
