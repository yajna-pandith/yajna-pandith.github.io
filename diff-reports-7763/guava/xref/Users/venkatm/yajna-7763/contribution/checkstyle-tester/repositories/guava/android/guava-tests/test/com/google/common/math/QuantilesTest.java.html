54    /*
55     * Since Quantiles provides a fluent-style API, each test covers a chain of methods resulting in
56     * the computation of one or more quantiles (or in an error) rather than individual methods. The
57     * tests are divided into three sections:
58     * 1. Tests on a hardcoded dataset for chains starting with median(), quartiles(), and scale(10);
59     * 2. Tests on hardcoded datasets include non-finite values for chains starting with scale(10);
60     * 3. Tests on a mechanically generated dataset for chains starting with percentiles();
61     * 4. Tests of illegal usages of the API.
62     */
63  
64    /*
65     * Covering every combination would lead to an explosion in the number of tests. So we cover only:
66     * - median with compute taking a double-collection and with computeInPlace;
67     * - quartiles with index and with indexes taking int-varargs, and with compute taking a
68     *   double-collection and with computeInPlace;
69     * - scale with index and with indexes taking int-varargs, and with all overloads of compute
70     *   taking a double-collection and with computeInPlace;
71     * - scale with indexes taking integer-collection with compute taking a double-collection and with
72     *   computeInPlace;
73     * - (except that, for non-finite values, we don't do all combinations exhaustively);
74     * - percentiles with index and with indexes taking int-varargs, and with compute taking a
75     *   double-collection and with computeInPlace.
76     */
77  
78    private static final double ALLOWED_ERROR = 1.0e-10;
79  
80    /**
81     * A {@link Correspondence} which accepts finite values within {@link #ALLOWED_ERROR} of each
82     * other.
83     */
84    private static final Correspondence<Number, Number> FINITE_QUANTILE_CORRESPONDENCE =
85        Correspondence.tolerance(ALLOWED_ERROR);
86  
87    /**
88     * A {@link Correspondence} which accepts either finite values within {@link #ALLOWED_ERROR} of
89     * each other or identical non-finite values.
90     */
91    private static final Correspondence<Double, Double> QUANTILE_CORRESPONDENCE =
92        Correspondence.from(
93            new BinaryPredicate<Double, Double>() {
94              @Override
95              public boolean apply(@NullableDecl Double actual, @NullableDecl Double expected) {
96                // Test for equality to allow non-finite values to match; otherwise, use the finite
97                // test.
98                return actual.equals(expected)
99                    || FINITE_QUANTILE_CORRESPONDENCE.compare(actual, expected);
100             }
101           },
102           "is identical to or " + FINITE_QUANTILE_CORRESPONDENCE);

255   public void testScale_indexes_largeVarargs_compute_doubleCollection() {
256     int scale = Integer.MAX_VALUE;
257     int otherIndex = (Integer.MAX_VALUE - 1) / 3; // this divides exactly
258     // For the otherIndex calculation, we have q=Integer.MAX_VALUE, k=(Integer.MAX_VALUE-1)/3, and
259     // N=16. Therefore k*(N-1)/q = 5-5/Integer.MAX_VALUE, which has floor 4 and fractional part
260     // (1-5/Integer.MAX_VALUE).
261     double otherValue = 16.0 * 5.0 / Integer.MAX_VALUE + 25.0 * (1.0 - 5.0 / Integer.MAX_VALUE);
262     assertThat(
263             Quantiles.scale(scale).indexes(0, scale, otherIndex).compute(SIXTEEN_SQUARES_DOUBLES))
264         .comparingValuesUsing(QUANTILE_CORRESPONDENCE)
265         .containsExactly(
266             0, SIXTEEN_SQUARES_MIN, scale, SIXTEEN_SQUARES_MAX, otherIndex, otherValue);
267   }

489   // 3. Tests on a mechanically generated dataset for chains starting with percentiles():
490 
491   private static final int PSEUDORANDOM_DATASET_SIZE = 9951;
492   private static final ImmutableList<Double> PSEUDORANDOM_DATASET = generatePseudorandomDataset();
493   private static final ImmutableList<Double> PSEUDORANDOM_DATASET_SORTED =
494       Ordering.natural().immutableSortedCopy(PSEUDORANDOM_DATASET);
495 
496   private static ImmutableList<Double> generatePseudorandomDataset() {
497     Random random = new Random(2211275185798966364L);
498     ImmutableList.Builder<Double> largeDatasetBuilder = ImmutableList.builder();
499     for (int i = 0; i < PSEUDORANDOM_DATASET_SIZE; i++) {
500       largeDatasetBuilder.add(random.nextGaussian());
501     }
502     return largeDatasetBuilder.build();
503   }
504 
505   private static double expectedLargeDatasetPercentile(int index) {
506     // We have q=100, k=index, and N=9951. Therefore k*(N-1)/q is 99.5*index. If index is even, that
507     // is an integer 199*index/2. If index is odd, that is halfway between floor(199*index/2) and
508     // ceil(199*index/2).
509     if (index % 2 == 0) {
510       int position = IntMath.divide(199 * index, 2, UNNECESSARY);
511       return PSEUDORANDOM_DATASET_SORTED.get(position);
512     } else {
513       int positionFloor = IntMath.divide(199 * index, 2, FLOOR);
514       int positionCeil = IntMath.divide(199 * index, 2, CEILING);
515       double lowerValue = PSEUDORANDOM_DATASET_SORTED.get(positionFloor);
516       double upperValue = PSEUDORANDOM_DATASET_SORTED.get(positionCeil);
517       return (lowerValue + upperValue) / 2.0;
518     }
519   }
