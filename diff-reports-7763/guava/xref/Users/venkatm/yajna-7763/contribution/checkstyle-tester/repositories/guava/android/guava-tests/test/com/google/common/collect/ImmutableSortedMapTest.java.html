121   public abstract static class AbstractMapTests<K, V> extends SortedMapInterfaceTest<K, V> {
122     public AbstractMapTests() {
123       super(false, false, false, false, false);
124     }
125 
126     @Override
127     protected SortedMap<K, V> makeEmptyMap() {
128       throw new UnsupportedOperationException();
129     }
130 
131     private static final Joiner joiner = Joiner.on(", ");
132 
133     @Override
134     protected void assertMoreInvariants(Map<K, V> map) {
135       // TODO: can these be moved to MapInterfaceTest?
136       for (Entry<K, V> entry : map.entrySet()) {
137         assertEquals(entry.getKey() + "=" + entry.getValue(), entry.toString());
138       }
139 
140       assertEquals("{" + joiner.join(map.entrySet()) + "}", map.toString());
141       assertEquals("[" + joiner.join(map.entrySet()) + "]", map.entrySet().toString());
142       assertEquals("[" + joiner.join(map.keySet()) + "]", map.keySet().toString());
143       assertEquals("[" + joiner.join(map.values()) + "]", map.values().toString());
144 
145       assertEquals(Sets.newHashSet(map.entrySet()), map.entrySet());
146       assertEquals(Sets.newHashSet(map.keySet()), map.keySet());
147     }
148   }
