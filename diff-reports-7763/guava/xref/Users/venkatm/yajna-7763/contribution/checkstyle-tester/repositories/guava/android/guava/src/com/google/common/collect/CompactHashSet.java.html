43  /**
44   * CompactHashSet is an implementation of a Set. All optional operations (adding and removing) are
45   * supported. The elements can be any objects.
46   *
47   * <p>{@code contains(x)}, {@code add(x)} and {@code remove(x)}, are all (expected and amortized)
48   * constant time operations. Expected in the hashtable sense (depends on the hash function doing a
49   * good job of distributing the elements to the buckets to a distribution not far from uniform), and
50   * amortized since some operations can trigger a hash table resize.
51   *
52   * <p>Unlike {@code java.util.HashSet}, iteration is only proportional to the actual {@code size()},
53   * which is optimal, and <i>not</i> the size of the internal hashtable, which could be much larger
54   * than {@code size()}. Furthermore, this structure only depends on a fixed number of arrays; {@code
55   * add(x)} operations <i>do not</i> create objects for the garbage collector to deal with, and for
56   * every element added, the garbage collector will have to traverse {@code 1.5} references on
57   * average, in the marking phase, not {@code 5.0} as in {@code java.util.HashSet}.
58   *
59   * <p>If there are no removals, then {@link #iterator iteration} order is the same as insertion
60   * order. Any removal invalidates any ordering guarantees.
61   *
62   * <p>This class should not be assumed to be universally superior to {@code java.util.HashSet}.
63   * Generally speaking, this class reduces object allocation and memory consumption at the price of
64   * moderately increased constant factors of CPU. Only use this class when there is a specific reason
65   * to prioritize memory over CPU.
66   *
67   * @author Dimitris Andreou
68   * @author Jon Noack
69   */
70  @GwtIncompatible // not worth using in GWT for now
71  class CompactHashSet<E> extends AbstractSet<E> implements Serializable {
72    // TODO(user): cache all field accesses in local vars

76      return new CompactHashSet<>();

79    /**
80     * Creates a <i>mutable</i> {@code CompactHashSet} instance containing the elements of the given
81     * collection in unspecified order.
82     *
83     * @param collection the elements that the set should contain
84     * @return a new {@code CompactHashSet} containing those elements (minus duplicates)
85     */
86    public static <E> CompactHashSet<E> create(Collection<? extends E> collection) {
87      CompactHashSet<E> set = createWithExpectedSize(collection.size());
88      set.addAll(collection);
89      return set;
90    }

92    /**
93     * Creates a <i>mutable</i> {@code CompactHashSet} instance containing the given elements in
94     * unspecified order.
95     *
96     * @param elements the elements that the set should contain
97     * @return a new {@code CompactHashSet} containing those elements (minus duplicates)
98     */
99    @SafeVarargs
100   public static <E> CompactHashSet<E> create(E... elements) {
101     CompactHashSet<E> set = createWithExpectedSize(elements.length);
102     Collections.addAll(set, elements);
103     return set;
104   }

106   /**
107    * Creates a {@code CompactHashSet} instance, with a high enough "initial capacity" that it
108    * <i>should</i> hold {@code expectedSize} elements without growth.
109    *
110    * @param expectedSize the number of elements you expect to add to the returned set
111    * @return a new, empty {@code CompactHashSet} with enough capacity to hold {@code expectedSize}
112    *     elements without resizing
113    * @throws IllegalArgumentException if {@code expectedSize} is negative

116     return new CompactHashSet<>(expectedSize);

119   /**
120    * The hashtable. Its values are indexes to the elements and entries arrays.
121    *
122    * <p>Currently, the UNSET value means "null pointer", and any positive value x is the actual
123    * index + 1.
124    *
125    * <p>Its size must be a power of two.
126    */
127   @MonotonicNonNullDecl private transient Object table;
128 
129   /**
130    * Contains the logical entries, in the range of [0, size()). The high bits of each int are the
131    * part of the smeared hash of the element not covered by the hashtable mask, whereas the low bits
132    * are the "next" pointer (pointing to the next entry in the bucket chain), which will always be
133    * less than or equal to the hashtable mask.
134    *
135    * <pre>
136    * hash  = aaaaaaaa
137    * mask  = 0000ffff
138    * next  = 0000bbbb
139    * entry = aaaabbbb
140    * </pre>
141    *
142    * <p>The pointers in [size(), entries.length) are all "null" (UNSET).
143    */
144   @MonotonicNonNullDecl private transient int[] entries;
145 
146   /**
147    * The elements contained in the set, in the range of [0, size()). The elements in [size(),
148    * elements.length) are all {@code null}.
149    */
150   @VisibleForTesting @MonotonicNonNullDecl transient Object[] elements;
151 
152   /**
153    * Keeps track of metadata like the number of hash table bits and modifications of this data
154    * structure (to make it possible to throw ConcurrentModificationException in the iterator). Note
155    * that we choose not to make this volatile, so we do less of a "best effort" to track such
156    * errors, for better performance.
157    */
158   private transient int metadata;
159 
160   /** The number of elements contained in the set. */
161   private transient int size;

177   /** Pseudoconstructor for serialization support. */
178   void init(int expectedSize) {
179     Preconditions.checkArgument(expectedSize >= 0, "Expected size must be >= 0");
180 
181     // Save expectedSize for use in allocArrays()
182     this.metadata = Math.max(1, Math.min(CompactHashing.MAX_SIZE, expectedSize));
183   }

185   /** Returns whether arrays need to be allocated. */
186   @VisibleForTesting
187   boolean needsAllocArrays() {
188     return table == null;
189   }
190 
191   /** Handle lazy allocation of arrays. */
192   @CanIgnoreReturnValue
193   int allocArrays() {
194     Preconditions.checkState(needsAllocArrays(), "Arrays already allocated");
195 
196     int expectedSize = metadata;
197     int buckets = CompactHashing.tableSize(expectedSize);
198     this.table = CompactHashing.createTable(buckets);
199     setHashTableMask(buckets - 1);
200 
201     this.entries = new int[expectedSize];
202     this.elements = new Object[expectedSize];
203 
204     return expectedSize;
205   }
206 
207   /** Stores the hash table mask as the number of bits needed to represent an index. */
208   private void setHashTableMask(int mask) {
209     int hashTableBits = Integer.SIZE - Integer.numberOfLeadingZeros(mask);
210     metadata =
211         CompactHashing.maskCombine(metadata, hashTableBits, CompactHashing.HASH_TABLE_BITS_MASK);
212   }
213 
214   /** Gets the hash table mask using the stored number of hash table bits. */
215   private int hashTableMask() {
216     return (1 << (metadata & CompactHashing.HASH_TABLE_BITS_MASK)) - 1;
217   }
218 
219   void incrementModCount() {
220     metadata += CompactHashing.MODIFICATION_COUNT_INCREMENT;
221   }

223   @CanIgnoreReturnValue
224   @Override
225   public boolean add(@NullableDecl E object) {
226     if (needsAllocArrays()) {
227       allocArrays();
228     }
229     int[] entries = this.entries;
230     Object[] elements = this.elements;
231 
232     int newEntryIndex = this.size; // current size, and pointer to the entry to be appended
233     int newSize = newEntryIndex + 1;
234     int hash = smearedHash(object);
235     int mask = hashTableMask();
236     int tableIndex = hash & mask;
237     int next = CompactHashing.tableGet(table, tableIndex);
238     if (next == UNSET) { // uninitialized bucket
239       if (newSize > mask) {
240         // Resize and add new entry
241         mask = resizeTable(mask, CompactHashing.newCapacity(mask), hash, newEntryIndex);
242       } else {
243         CompactHashing.tableSet(table, tableIndex, newEntryIndex + 1);
244       }
245     } else {
246       int entryIndex;
247       int entry;
248       int hashPrefix = CompactHashing.getHashPrefix(hash, mask);
249       do {
250         entryIndex = next - 1;
251         entry = entries[entryIndex];
252         if (CompactHashing.getHashPrefix(entry, mask) == hashPrefix
253             && Objects.equal(object, elements[entryIndex])) {
254           return false;
255         }
256         next = CompactHashing.getNext(entry, mask);
257       } while (next != UNSET);
258       if (newSize > mask) {
259         // Resize and add new entry
260         mask = resizeTable(mask, CompactHashing.newCapacity(mask), hash, newEntryIndex);
261       } else {
262         entries[entryIndex] = CompactHashing.maskCombine(entry, newEntryIndex + 1, mask);
263       }
264     }
265     resizeMeMaybe(newSize);
266     insertEntry(newEntryIndex, object, hash, mask);
267     this.size = newSize;
268     incrementModCount();
269     return true;
270   }
271 
272   /**
273    * Creates a fresh entry with the specified object at the specified position in the entry arrays.
274    */
275   void insertEntry(int entryIndex, @NullableDecl E object, int hash, int mask) {
276     this.entries[entryIndex] = CompactHashing.maskCombine(hash, UNSET, mask);
277     this.elements[entryIndex] = object;
278   }
279 
280   /** Resizes the entries storage if necessary. */
281   private void resizeMeMaybe(int newSize) {
282     int entriesSize = entries.length;
283     if (newSize > entriesSize) {
284       // 1.5x but round up to nearest odd (this is optimal for memory consumption on Android)
285       int newCapacity =
286           Math.min(CompactHashing.MAX_SIZE, (entriesSize + Math.max(1, entriesSize >>> 1)) | 1);
287       if (newCapacity != entriesSize) {
288         resizeEntries(newCapacity);
289       }
290     }
291   }
292 
293   /**
294    * Resizes the internal entries array to the specified capacity, which may be greater or less than
295    * the current capacity.
296    */
297   void resizeEntries(int newCapacity) {
298     this.entries = Arrays.copyOf(entries, newCapacity);
299     this.elements = Arrays.copyOf(elements, newCapacity);
300   }
301 
302   @CanIgnoreReturnValue
303   private int resizeTable(int mask, int newCapacity, int targetHash, int targetEntryIndex) {
304     Object newTable = CompactHashing.createTable(newCapacity);
305     int newMask = newCapacity - 1;
306 
307     if (targetEntryIndex != UNSET) {
308       // Add target first; it must be last in the chain because its entry hasn't yet been created
309       CompactHashing.tableSet(newTable, targetHash & newMask, targetEntryIndex + 1);
310     }
311 
312     Object table = this.table;
313     int[] entries = this.entries;
314 
315     // Loop over current hashtable
316     for (int tableIndex = 0; tableIndex <= mask; tableIndex++) {
317       int next = CompactHashing.tableGet(table, tableIndex);
318       while (next != UNSET) {
319         int entryIndex = next - 1;
320         int entry = entries[entryIndex];
321 
322         // Rebuild hash using entry hashPrefix and tableIndex ("hashSuffix")
323         int hash = CompactHashing.getHashPrefix(entry, mask) | tableIndex;
324 
325         int newTableIndex = hash & newMask;
326         int newNext = CompactHashing.tableGet(newTable, newTableIndex);
327         CompactHashing.tableSet(newTable, newTableIndex, next);
328         entries[entryIndex] = CompactHashing.maskCombine(hash, newNext, newMask);
329 
330         next = CompactHashing.getNext(entry, mask);
331       }
332     }
333 
334     this.table = newTable;
335     setHashTableMask(newMask);
336     return newMask;
337   }

339   @Override
340   public boolean contains(@NullableDecl Object object) {
341     if (needsAllocArrays()) {
342       return false;
343     }
344     int hash = smearedHash(object);
345     int mask = hashTableMask();
346     int next = CompactHashing.tableGet(table, hash & mask);
347     if (next == UNSET) {
348       return false;
349     }
350     int hashPrefix = CompactHashing.getHashPrefix(hash, mask);
351     do {
352       int entryIndex = next - 1;
353       int entry = entries[entryIndex];
354       if (CompactHashing.getHashPrefix(entry, mask) == hashPrefix
355           && Objects.equal(object, elements[entryIndex])) {
356         return true;
357       }
358       next = CompactHashing.getNext(entry, mask);
359     } while (next != UNSET);
360     return false;
361   }

363   @CanIgnoreReturnValue
364   @Override
365   public boolean remove(@NullableDecl Object object) {
366     if (needsAllocArrays()) {
367       return false;
368     }
369     int mask = hashTableMask();
370     int index =
371         CompactHashing.remove(
372             object, /* value= */ null, mask, table, entries, elements, /* values= */ null);
373     if (index == -1) {
374       return false;
375     }
376 
377     moveLastEntry(index, mask);
378     size--;
379     incrementModCount();
380 
381     return true;
382   }
383 
384   /**
385    * Moves the last entry in the entry array into {@code dstIndex}, and nulls out its old position.
386    */
387   void moveLastEntry(int dstIndex, int mask) {
388     int srcIndex = size() - 1;
389     if (dstIndex < srcIndex) {
390       // move last entry to deleted spot
391       @NullableDecl Object object = elements[srcIndex];
392       elements[dstIndex] = object;
393       elements[srcIndex] = null;
394 
395       // move the last entry to the removed spot, just like we moved the element
396       entries[dstIndex] = entries[srcIndex];
397       entries[srcIndex] = 0;
398 
399       // also need to update whoever's "next" pointer was pointing to the last entry place
400       int tableIndex = smearedHash(object) & mask;
401       int next = CompactHashing.tableGet(table, tableIndex);
402       int srcNext = srcIndex + 1;
403       if (next == srcNext) {
404         // we need to update the root pointer
405         CompactHashing.tableSet(table, tableIndex, dstIndex + 1);
406       } else {
407         // we need to update a pointer in an entry
408         int entryIndex;
409         int entry;
410         do {
411           entryIndex = next - 1;
412           entry = entries[entryIndex];
413           next = CompactHashing.getNext(entry, mask);
414         } while (next != srcNext);
415         // here, entries[entryIndex] points to the old entry location; update it
416         entries[entryIndex] = CompactHashing.maskCombine(entry, dstIndex + 1, mask);
417       }
418     } else {
419       elements[dstIndex] = null;
420       entries[dstIndex] = 0;
421     }
422   }

424   int firstEntryIndex() {
425     return isEmpty() ? -1 : 0;
426   }
427 
428   int getSuccessor(int entryIndex) {
429     return (entryIndex + 1 < size) ? entryIndex + 1 : -1;
430   }

432   /**
433    * Updates the index an iterator is pointing to after a call to remove: returns the index of the
434    * entry that should be looked at after a removal on indexRemoved, with indexBeforeRemove as the
435    * index that *was* the next entry that would be looked at.
436    */
437   int adjustAfterRemove(int indexBeforeRemove, @SuppressWarnings("unused") int indexRemoved) {
438     return indexBeforeRemove - 1;
439   }

441   @Override
442   public Iterator<E> iterator() {
443     return new Iterator<E>() {
444       int expectedMetadata = metadata;
445       int currentIndex = firstEntryIndex();
446       int indexToRemove = -1;
447 
448       @Override
449       public boolean hasNext() {
450         return currentIndex >= 0;
451       }
452 
453       @SuppressWarnings("unchecked") // known to be Es
454       @Override
455       public E next() {
456         checkForConcurrentModification();
457         if (!hasNext()) {
458           throw new NoSuchElementException();
459         }
460         indexToRemove = currentIndex;
461         E result = (E) elements[currentIndex];
462         currentIndex = getSuccessor(currentIndex);
463         return result;
464       }
465 
466       @Override
467       public void remove() {
468         checkForConcurrentModification();
469         checkRemove(indexToRemove >= 0);
470         incrementExpectedModCount();
471         CompactHashSet.this.remove(elements[indexToRemove]);
472         currentIndex = adjustAfterRemove(currentIndex, indexToRemove);
473         indexToRemove = -1;
474       }
475 
476       void incrementExpectedModCount() {
477         expectedMetadata += CompactHashing.MODIFICATION_COUNT_INCREMENT;
478       }
479 
480       private void checkForConcurrentModification() {
481         if (metadata != expectedMetadata) {
482           throw new ConcurrentModificationException();
483         }
484       }
485     };
486   }

498   @Override

500     if (needsAllocArrays()) {

506   @CanIgnoreReturnValue
507   @Override
508   public <T> T[] toArray(T[] a) {
509     if (needsAllocArrays()) {
510       if (a.length > 0) {
511         a[0] = null;
512       }
513       return a;
514     }
515     return ObjectArrays.toArrayImpl(elements, 0, size, a);
516   }

518   /**
519    * Ensures that this {@code CompactHashSet} has the smallest representation in memory, given its
520    * current size.
521    */
522   public void trimToSize() {
523     if (needsAllocArrays()) {
524       return;
525     }
526     int size = this.size;
527     if (size < entries.length) {
528       resizeEntries(size);
529     }
530     int minimumTableSize = CompactHashing.tableSize(size);
531     int mask = hashTableMask();
532     if (minimumTableSize < mask) { // smaller table size will always be less than current mask
533       resizeTable(mask, minimumTableSize, UNSET, UNSET);
534     }
535   }
536 
537   @Override
538   public void clear() {
539     if (needsAllocArrays()) {
540       return;
541     }
542     incrementModCount();
543     Arrays.fill(elements, 0, size, null);
544     CompactHashing.tableClear(table);
545     Arrays.fill(entries, 0, size, 0);
546     this.size = 0;
547   }
548 
549   private void writeObject(ObjectOutputStream stream) throws IOException {
550     stream.defaultWriteObject();
551     stream.writeInt(size);
552     for (int i = firstEntryIndex(); i >= 0; i = getSuccessor(i)) {
553       stream.writeObject(elements[i]);
554     }
555   }
556 
557   @SuppressWarnings("unchecked")
558   private void readObject(ObjectInputStream stream) throws IOException, ClassNotFoundException {
559     stream.defaultReadObject();
560     int elementCount = stream.readInt();
561     if (elementCount < 0) {
562       throw new InvalidObjectException("Invalid size: " + elementCount);
563     }
564     init(elementCount);
565     for (int i = 0; i < elementCount; i++) {
566       E element = (E) stream.readObject();
567       add(element);
568     }
569   }
