15  package com.google.common.util.concurrent;

17  import static com.google.common.base.Preconditions.checkNotNull;
18  import static com.google.common.util.concurrent.Futures.getDone;
19  import static com.google.common.util.concurrent.MoreExecutors.rejectionPropagatingExecutor;
20  
21  import com.google.common.annotations.GwtCompatible;
22  import com.google.common.base.Function;
23  import com.google.errorprone.annotations.ForOverride;
24  import java.util.concurrent.CancellationException;
25  import java.util.concurrent.ExecutionException;
26  import java.util.concurrent.Executor;
27  import org.checkerframework.checker.nullness.compatqual.NullableDecl;
28  
29  /** Implementations of {@code Futures.transform*}. */
30  @GwtCompatible
31  abstract class AbstractTransformFuture<I, O, F, T> extends FluentFuture.TrustedFuture<O>
32      implements Runnable {
33    static <I, O> ListenableFuture<O> create(
34        ListenableFuture<I> input,
35        AsyncFunction<? super I, ? extends O> function,
36        Executor executor) {
37      checkNotNull(executor);
38      AsyncTransformFuture<I, O> output = new AsyncTransformFuture<>(input, function);
39      input.addListener(output, rejectionPropagatingExecutor(executor, output));
40      return output;
41    }
42  
43    static <I, O> ListenableFuture<O> create(
44        ListenableFuture<I> input, Function<? super I, ? extends O> function, Executor executor) {
45      checkNotNull(function);
46      TransformFuture<I, O> output = new TransformFuture<>(input, function);
47      input.addListener(output, rejectionPropagatingExecutor(executor, output));
48      return output;
49    }
50  
51    /*
52     * In certain circumstances, this field might theoretically not be visible to an afterDone() call
53     * triggered by cancel(). For details, see the comments on the fields of TimeoutFuture.
54     */
55    @NullableDecl ListenableFuture<? extends I> inputFuture;
56    @NullableDecl F function;
57  
58    AbstractTransformFuture(ListenableFuture<? extends I> inputFuture, F function) {
59      this.inputFuture = checkNotNull(inputFuture);
60      this.function = checkNotNull(function);
61    }
62  
63    @Override
64    public final void run() {
65      ListenableFuture<? extends I> localInputFuture = inputFuture;
66      F localFunction = function;
67      if (isCancelled() | localInputFuture == null | localFunction == null) {
68        return;
69      }
70      inputFuture = null;
71  
72      if (localInputFuture.isCancelled()) {
73        @SuppressWarnings("unchecked")
74        boolean unused =
75            setFuture((ListenableFuture<O>) localInputFuture); // Respects cancellation cause setting
76        return;
77      }
78  
79      /*
80       * Any of the setException() calls below can fail if the output Future is cancelled between now
81       * and then. This means that we're silently swallowing an exception -- maybe even an Error. But
82       * this is no worse than what FutureTask does in that situation. Additionally, because the
83       * Future was cancelled, its listeners have been run, so its consumers will not hang.
84       *
85       * Contrast this to the situation we have if setResult() throws, a situation described below.
86       */
87      I sourceResult;
88      try {
89        sourceResult = getDone(localInputFuture);
90      } catch (CancellationException e) {
91        // TODO(user): verify future behavior - unify logic with getFutureValue in AbstractFuture. This
92        // code should be unreachable with correctly implemented Futures.
93        // Cancel this future and return.
94        // At this point, inputFuture is cancelled and outputFuture doesn't exist, so the value of
95        // mayInterruptIfRunning is irrelevant.
96        cancel(false);
97        return;
98      } catch (ExecutionException e) {
99        // Set the cause of the exception as this future's exception.
100       setException(e.getCause());
101       return;
102     } catch (RuntimeException e) {
103       // Bug in inputFuture.get(). Propagate to the output Future so that its consumers don't hang.
104       setException(e);
105       return;
106     } catch (Error e) {
107       /*
108        * StackOverflowError, OutOfMemoryError (e.g., from allocating ExecutionException), or
109        * something. Try to treat it like a RuntimeException. If we overflow the stack again, the
110        * resulting Error will propagate upward up to the root call to set().
111        */
112       setException(e);
113       return;
114     }
115 
116     T transformResult;
117     try {
118       transformResult = doTransform(localFunction, sourceResult);
119     } catch (Throwable t) {
120       // This exception is irrelevant in this thread, but useful for the client.
121       setException(t);
122       return;
123     } finally {
124       function = null;
125     }
126 
127     /*
128      * If set()/setValue() throws an Error, we let it propagate. Why? The most likely Error is a
129      * StackOverflowError (from deep transform(..., directExecutor()) nesting), and calling
130      * setException(stackOverflowError) would fail:
131      *
132      * - If the stack overflowed before set()/setValue() could even store the result in the output
133      * Future, then a call setException() would likely also overflow.
134      *
135      * - If the stack overflowed after set()/setValue() stored its result, then a call to
136      * setException() will be a no-op because the Future is already done.
137      *
138      * Both scenarios are bad: The output Future might never complete, or, if it does complete, it
139      * might not run some of its listeners. The likely result is that the app will hang. (And of
140      * course stack overflows are bad news in general. For example, we may have overflowed in the
141      * middle of defining a class. If so, that class will never be loadable in this process.) The
142      * best we can do (since logging may overflow the stack) is to let the error propagate. Because
143      * it is an Error, it won't be caught and logged by AbstractFuture.executeListener. Instead, it
144      * can propagate through many layers of AbstractTransformFuture up to the root call to set().
145      *
146      * https://github.com/google/guava/issues/2254
147      *
148      * Other kinds of Errors are possible:
149      *
150      * - OutOfMemoryError from allocations in setFuture(): The calculus here is similar to
151      * StackOverflowError: We can't reliably call setException(error).
152      *
153      * - Any kind of Error from a listener. Even if we could distinguish that case (by exposing some
154      * extra state from AbstractFuture), our options are limited: A call to setException() would be
155      * a no-op. We could log, but if that's what we really want, we should modify
156      * AbstractFuture.executeListener to do so, since that method would have the ability to continue
157      * to execute other listeners.
158      *
159      * What about RuntimeException? If there is a bug in set()/setValue() that produces one, it will
160      * propagate, too, but only as far as AbstractFuture.executeListener, which will catch and log
161      * it.
162      */
163     setResult(transformResult);
164   }
165 
166   /** Template method for subtypes to actually run the transform. */
167   @ForOverride
168   @NullableDecl
169   abstract T doTransform(F function, @NullableDecl I result) throws Exception;
170 
171   /** Template method for subtypes to actually set the result. */
172   @ForOverride
173   abstract void setResult(@NullableDecl T result);
174 
175   @Override
176   protected final void afterDone() {
177     maybePropagateCancellationTo(inputFuture);
178     this.inputFuture = null;
179     this.function = null;
180   }
181 
182   @Override
183   protected String pendingToString() {
184     ListenableFuture<? extends I> localInputFuture = inputFuture;
185     F localFunction = function;
186     String superString = super.pendingToString();
187     String resultString = "";
188     if (localInputFuture != null) {
189       resultString = "inputFuture=[" + localInputFuture + "], ";
190     }
191     if (localFunction != null) {
192       return resultString + "function=[" + localFunction + "]";
193     } else if (superString != null) {
194       return resultString + superString;
195     }
196     return null;
197   }
198 
199   /**
200    * An {@link AbstractTransformFuture} that delegates to an {@link AsyncFunction} and {@link
201    * #setFuture(ListenableFuture)}.
202    */
203   private static final class AsyncTransformFuture<I, O>
204       extends AbstractTransformFuture<
205           I, O, AsyncFunction<? super I, ? extends O>, ListenableFuture<? extends O>> {
206     AsyncTransformFuture(
207         ListenableFuture<? extends I> inputFuture, AsyncFunction<? super I, ? extends O> function) {
208       super(inputFuture, function);
209     }
210 
211     @Override
212     ListenableFuture<? extends O> doTransform(
213         AsyncFunction<? super I, ? extends O> function, @NullableDecl I input) throws Exception {
214       ListenableFuture<? extends O> outputFuture = function.apply(input);
215       checkNotNull(
216           outputFuture,
217           "AsyncFunction.apply returned null instead of a Future. "
218               + "Did you mean to return immediateFuture(null)? %s",
219           function);
220       return outputFuture;
221     }
222 
223     @Override
224     void setResult(ListenableFuture<? extends O> result) {
225       setFuture(result);
226     }
227   }
228 
229   /**
230    * An {@link AbstractTransformFuture} that delegates to a {@link Function} and {@link
231    * #set(Object)}.
232    */
233   private static final class TransformFuture<I, O>
234       extends AbstractTransformFuture<I, O, Function<? super I, ? extends O>, O> {
235     TransformFuture(
236         ListenableFuture<? extends I> inputFuture, Function<? super I, ? extends O> function) {
237       super(inputFuture, function);
238     }
239 
240     @Override
241     @NullableDecl
242     O doTransform(Function<? super I, ? extends O> function, @NullableDecl I input) {
243       return function.apply(input);
244     }
245 
246     @Override
247     void setResult(@NullableDecl O result) {
248       set(result);
249     }
250   }
