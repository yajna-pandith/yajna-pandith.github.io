35  /**
36   * A future whose value is derived from a collection of input futures.
37   *
38   * @param <InputT> the type of the individual inputs
39   * @param <OutputT> the type of the output (i.e. this) future
40   */
41  @GwtCompatible
42  abstract class AggregateFuture<InputT, OutputT> extends AggregateFutureState<OutputT> {
43    private static final Logger logger = Logger.getLogger(AggregateFuture.class.getName());
44  
45    /**
46     * The input futures. After {@link #init}, this field is read only by {@link #afterDone()} (to
47     * propagate cancellation) and {@link #toString()}. To access the futures' <i>values</i>, {@code
48     * AggregateFuture} attaches listeners that hold references to one or more inputs. And in the case
49     * of {@link CombinedFuture}, the user-supplied callback usually has its own references to inputs.
50     */
51    /*
52     * In certain circumstances, this field might theoretically not be visible to an afterDone() call
53     * triggered by cancel(). For details, see the comments on the fields of TimeoutFuture.
54     */
55    private @Nullable ImmutableCollection<? extends ListenableFuture<? extends InputT>> futures;
56  
57    private final boolean allMustSucceed;
58    private final boolean collectsValues;
59  
60    AggregateFuture(
61        ImmutableCollection<? extends ListenableFuture<? extends InputT>> futures,
62        boolean allMustSucceed,
63        boolean collectsValues) {
64      super(futures.size());
65      this.futures = checkNotNull(futures);
66      this.allMustSucceed = allMustSucceed;
67      this.collectsValues = collectsValues;
68    }
69  
70    @Override
71    protected final void afterDone() {
72      super.afterDone();
73  
74      ImmutableCollection<? extends Future<?>> localFutures = futures;
75      releaseResources(OUTPUT_FUTURE_DONE); // nulls out `futures`
76  
77      if (isCancelled() & localFutures != null) {
78        boolean wasInterrupted = wasInterrupted();
79        for (Future<?> future : localFutures) {
80          future.cancel(wasInterrupted);
81        }
82      }
83      /*
84       * We don't call clearSeenExceptions() until processCompleted(). Prior to that, it may be needed
85       * again if some outstanding input fails.
86       */
87    }
88  
89    @Override
90    protected final String pendingToString() {
91      ImmutableCollection<? extends Future<?>> localFutures = futures;
92      if (localFutures != null) {
93        return "futures=" + localFutures;
94      }
95      return super.pendingToString();
96    }
97  
98    /**
99     * Must be called at the end of each subclass's constructor. This method performs the "real"
100    * initialization; we can't put this in the constructor because, in the case where futures are
101    * already complete, we would not initialize the subclass before calling {@link
102    * #collectValueFromNonCancelledFuture}. As this is called after the subclass is constructed,
103    * we're guaranteed to have properly initialized the subclass.
104    */
105   final void init() {
106     // Corner case: List is empty.
107     if (futures.isEmpty()) {
108       handleAllCompleted();
109       return;
110     }
111 
112     // NOTE: If we ever want to use a custom executor here, have a look at CombinedFuture as we'll
113     // need to handle RejectedExecutionException
114 
115     if (allMustSucceed) {
116       // We need fail fast, so we have to keep track of which future failed so we can propagate
117       // the exception immediately
118 
119       // Register a listener on each Future in the list to update the state of this future.
120       // Note that if all the futures on the list are done prior to completing this loop, the last
121       // call to addListener() will callback to setOneValue(), transitively call our cleanup
122       // listener, and set this.futures to null.
123       // This is not actually a problem, since the foreach only needs this.futures to be non-null
124       // at the beginning of the loop.
125       int i = 0;
126       for (final ListenableFuture<? extends InputT> future : futures) {
127         final int index = i++;
128         future.addListener(
129             new Runnable() {
130               @Override
131               public void run() {
132                 try {
133                   if (future.isCancelled()) {
134                     // Clear futures prior to cancelling children. This sets our own state but lets
135                     // the input futures keep running, as some of them may be used elsewhere.
136                     futures = null;
137                     cancel(false);
138                   } else {
139                     collectValueFromNonCancelledFuture(index, future);
140                   }
141                 } finally {
142                   /*
143                    * "null" means: There is no need to access `futures` again during
144                    * `processCompleted` because we're reading each value during a call to
145                    * handleOneInputDone.
146                    */
147                   decrementCountAndMaybeComplete(null);
148                 }
149               }
150             },
151             directExecutor());
152       }
153     } else {
154       /*
155        * We'll call the user callback or collect the values only when all inputs complete,
156        * regardless of whether some failed. This lets us avoid calling expensive methods like
157        * Future.get() when we don't need to (specifically, for whenAllComplete().call*()), and it
158        * lets all futures share the same listener.
159        *
160        * We store `localFutures` inside the listener because `this.futures` might be nulled out by
161        * the time the listener runs for the final future -- at which point we need to check all
162        * inputs for exceptions *if* we're collecting values. If we're not, then the listener doesn't
163        * need access to the futures again, so we can just pass `null`.
164        *
165        * TODO(b/112550045): Allocating a single, cheaper listener is (I think) only an optimization.
166        * If we make some other optimizations, this one will no longer be necessary. The optimization
167        * could actually hurt in some cases, as it forces us to keep all inputs in memory until the
168        * final input completes.
169        */
170       final ImmutableCollection<? extends Future<? extends InputT>> localFutures =
171           collectsValues ? futures : null;
172       Runnable listener =
173           new Runnable() {
174             @Override
175             public void run() {
176               decrementCountAndMaybeComplete(localFutures);
177             }
178           };
179       for (ListenableFuture<? extends InputT> future : futures) {
180         future.addListener(listener, directExecutor());
181       }
182     }
183   }
184 
185   /**
186    * Fails this future with the given Throwable if {@link #allMustSucceed} is true. Also, logs the
187    * throwable if it is an {@link Error} or if {@link #allMustSucceed} is {@code true}, the
188    * throwable did not cause this future to fail, and it is the first time we've seen that
189    * particular Throwable.
190    */
191   private void handleException(Throwable throwable) {
192     checkNotNull(throwable);
193 
194     if (allMustSucceed) {
195       // As soon as the first one fails, make that failure the result of the output future.
196       // The results of all other inputs are then ignored (except for logging any failures).
197       boolean completedWithFailure = setException(throwable);
198       if (!completedWithFailure) {
199         // Go up the causal chain to see if we've already seen this cause; if we have, even if
200         // it's wrapped by a different exception, don't log it.
201         boolean firstTimeSeeingThisException = addCausalChain(getOrInitSeenExceptions(), throwable);
202         if (firstTimeSeeingThisException) {
203           log(throwable);
204           return;
205         }
206       }
207     }
208 
209     /*
210      * TODO(cpovirk): Should whenAllComplete().call*() log errors, too? Currently, it doesn't call
211      * handleException() at all.
212      */
213     if (throwable instanceof Error) {
214       /*
215        * TODO(cpovirk): Do we really want to log this if we called setException(throwable) and it
216        * returned true? This was intentional (CL 46470009), but it seems odd compared to how we
217        * normally handle Error.
218        *
219        * Similarly, do we really want to log the same Error more than once?
220        */
221       log(throwable);
222     }
223   }
224 
225   private static void log(Throwable throwable) {
226     String message =
227         (throwable instanceof Error)
228             ? "Input Future failed with Error"
229             : "Got more than one input Future failure. Logging failures after the first";
230     logger.log(SEVERE, message, throwable);
231   }
232 
233   @Override
234   final void addInitialException(Set<Throwable> seen) {
235     checkNotNull(seen);
236     if (!isCancelled()) {
237       // TODO(cpovirk): Think about whether we could/should use Verify to check this.
238       boolean unused = addCausalChain(seen, tryInternalFastPathGetFailure());
239     }
240   }
241 
242   /**
243    * Collects the result (success or failure) of one input future. The input must not have been
244    * cancelled. For details on when this is called, see {@link #collectOneValue}.
245    */
246   private void collectValueFromNonCancelledFuture(int index, Future<? extends InputT> future) {
247     try {
248       // We get the result, even if collectOneValue is a no-op, so that we can fail fast.
249       collectOneValue(index, getDone(future));
250     } catch (ExecutionException e) {
251       handleException(e.getCause());
252     } catch (Throwable t) {
253       handleException(t);
254     }
255   }
256 
257   private void decrementCountAndMaybeComplete(
258       @Nullable
259           ImmutableCollection<? extends Future<? extends InputT>>
260               futuresIfNeedToCollectAtCompletion) {
261     int newRemaining = decrementRemainingAndGet();
262     checkState(newRemaining >= 0, "Less than 0 remaining futures");
263     if (newRemaining == 0) {
264       processCompleted(futuresIfNeedToCollectAtCompletion);
265     }
266   }
267 
268   private void processCompleted(
269       @Nullable
270           ImmutableCollection<? extends Future<? extends InputT>>
271               futuresIfNeedToCollectAtCompletion) {
272     if (futuresIfNeedToCollectAtCompletion != null) {
273       int i = 0;
274       for (Future<? extends InputT> future : futuresIfNeedToCollectAtCompletion) {
275         if (!future.isCancelled()) {
276           collectValueFromNonCancelledFuture(i, future);
277         }
278         i++;
279       }
280     }
281     clearSeenExceptions();
282     handleAllCompleted();
283     /*
284      * Null out fields, including some used in handleAllCompleted() above (like
285      * `CollectionFuture.values`). This might be a no-op: If this future completed during
286      * handleAllCompleted(), they will already have been nulled out. But in the case of
287      * whenAll*().call*(), this future may be pending until the callback runs -- or even longer in
288      * the case of callAsync(), which waits for the callback's returned future to complete.
289      */
290     releaseResources(ALL_INPUT_FUTURES_PROCESSED);
291   }
292 
293   /**
294    * Clears fields that are no longer needed after this future has completed -- or at least all its
295    * inputs have completed (more precisely, after {@link #handleAllCompleted()} has been called).
296    * Often called multiple times (that is, both when the inputs complete and when the output
297    * completes).
298    *
299    * <p>This is similar to our proposed {@code afterCommit} method but not quite the same. See the
300    * description of CL 265462958.
301    */
302   // TODO(user): Write more tests for memory retention.
303   @ForOverride
304   @OverridingMethodsMustInvokeSuper
305   void releaseResources(ReleaseResourcesReason reason) {
306     checkNotNull(reason);
307     /*
308      * All elements of `futures` are completed, or this future has already completed and read
309      * `futures` into a local variable (in preparation for propagating cancellation to them). In
310      * either case, no one needs to read `futures` for cancellation purposes later. (And
311      * cancellation purposes are the main reason to access `futures`, as discussed in its docs.)
312      */
313     this.futures = null;
314   }
315 
316   enum ReleaseResourcesReason {
317     OUTPUT_FUTURE_DONE,
318     ALL_INPUT_FUTURES_PROCESSED,
319   }
320 
321   /**
322    * If {@code allMustSucceed} is true, called as each future completes; otherwise, if {@code
323    * collectsValues} is true, called for each future when all futures complete.
324    */
325   abstract void collectOneValue(int index, @Nullable InputT returnValue);
326 
327   abstract void handleAllCompleted();
328 
329   /** Adds the chain to the seen set, and returns whether all the chain was new to us. */
330   private static boolean addCausalChain(Set<Throwable> seen, Throwable t) {
331     for (; t != null; t = t.getCause()) {
332       boolean firstTimeSeen = seen.add(t);
333       if (!firstTimeSeen) {
334         /*
335          * We've seen this, so we've seen its causes, too. No need to re-add them. (There's one case
336          * where this isn't true, but we ignore it: If we record an exception, then someone calls
337          * initCause() on it, and then we examine it again, we'll conclude that we've seen the whole
338          * chain before when it fact we haven't. But this should be rare.)
339          */
340         return false;
341       }
342     }
343     return true;
344   }
