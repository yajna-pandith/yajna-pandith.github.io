249   /**
250    * This test attempts to cause a future to wait for longer than it was requested to from a timed
251    * get() call. As measurements of time are prone to flakiness, it tries to assert based on ranges
252    * derived from observing how much time actually passed for various operations.
253    */
254   @SuppressWarnings({"DeprecatedThreadMethods", "ThreadPriorityCheck"})
255   public void testToString_delayedTimeout() throws Exception {
256     TimedWaiterThread thread =
257         new TimedWaiterThread(new AbstractFuture<Object>() {}, 2, TimeUnit.SECONDS);
258     thread.start();
259     thread.awaitWaiting();
260     thread.suspend();
261     // Sleep for enough time to add 1500 milliseconds of overwait to the get() call.
262     long toWaitMillis = 3500 - TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - thread.startTime);
263     Thread.sleep(toWaitMillis);
264     thread.setPriority(Thread.MAX_PRIORITY);
265     thread.resume();
266     thread.join();
267     // It's possible to race and suspend the thread just before the park call actually takes effect,
268     // causing the thread to be suspended for 3.5 seconds, and then park itself for 2 seconds after
269     // being resumed. To avoid a flake in this scenario, calculate how long that thread actually
270     // waited and assert based on that time. Empirically, the race where the thread ends up waiting
271     // for 5.5 seconds happens about 2% of the time.
272     boolean longWait = TimeUnit.NANOSECONDS.toSeconds(thread.timeSpentBlocked) >= 5;
273     // Count how long it actually took to return; we'll accept any number between the expected delay
274     // and the approximate actual delay, to be robust to variance in thread scheduling.
275     char overWaitNanosFirstDigit =
276         Long.toString(
277                 thread.timeSpentBlocked - TimeUnit.MILLISECONDS.toNanos(longWait ? 5000 : 3000))
278             .charAt(0);
279     if (overWaitNanosFirstDigit < '4') {
280       overWaitNanosFirstDigit = '9';
281     }
282     String nanosRegex = "[4-" + overWaitNanosFirstDigit + "][0-9]+";
283     assertWithMessage(
284             "Spent " + thread.timeSpentBlocked + " ns blocked; slept for " + toWaitMillis + " ms")
285         .that(thread.exception)
286         .hasMessageThat()
287         .matches(
288             "Waited 2 seconds \\(plus "
289                 + (longWait ? "3" : "1")
290                 + " seconds, "
291                 + nanosRegex
292                 + " nanoseconds delay\\).*");
293   }

389     final CyclicBarrier barrier =
390         new CyclicBarrier(
391             6 // for the setter threads
392                 + 50 // for the listeners
393                 + 50 // for the blocking get threads,
394                 + 1); // for the main thread
395     final ExecutorService executor = Executors.newFixedThreadPool(barrier.getParties());
396     final AtomicReference<AbstractFuture<String>> currentFuture = Atomics.newReference();
397     final AtomicInteger numSuccessfulSetCalls = new AtomicInteger();
398     Callable<Void> completeSucessFullyRunnable =
399         new Callable<Void>() {
400           @Override
401           public Void call() {
402             if (currentFuture.get().set("set")) {
403               numSuccessfulSetCalls.incrementAndGet();
404             }
405             awaitUnchecked(barrier);
406             return null;
407           }
408         };
409     Callable<Void> completeExceptionallyRunnable =
410         new Callable<Void>() {
411           Exception failureCause = new Exception("setException");
412 
413           @Override
414           public Void call() {
415             if (currentFuture.get().setException(failureCause)) {
416               numSuccessfulSetCalls.incrementAndGet();
417             }
418             awaitUnchecked(barrier);
419             return null;
420           }
421         };
422     Callable<Void> cancelRunnable =
423         new Callable<Void>() {
424           @Override
425           public Void call() {
426             if (currentFuture.get().cancel(true)) {
427               numSuccessfulSetCalls.incrementAndGet();
428             }
429             awaitUnchecked(barrier);
430             return null;
431           }
432         };
433     Callable<Void> setFutureCompleteSucessFullyRunnable =
434         new Callable<Void>() {
435           ListenableFuture<String> future = Futures.immediateFuture("setFuture");
436 
437           @Override
438           public Void call() {
439             if (currentFuture.get().setFuture(future)) {
440               numSuccessfulSetCalls.incrementAndGet();
441             }
442             awaitUnchecked(barrier);
443             return null;
444           }
445         };
446     Callable<Void> setFutureCompleteExceptionallyRunnable =
447         new Callable<Void>() {
448           ListenableFuture<String> future =
449               Futures.immediateFailedFuture(new Exception("setFuture"));
450 
451           @Override
452           public Void call() {
453             if (currentFuture.get().setFuture(future)) {
454               numSuccessfulSetCalls.incrementAndGet();
455             }
456             awaitUnchecked(barrier);
457             return null;
458           }
459         };
460     Callable<Void> setFutureCancelRunnable =
461         new Callable<Void>() {
462           ListenableFuture<String> future = Futures.immediateCancelledFuture();
463 
464           @Override
465           public Void call() {
466             if (currentFuture.get().setFuture(future)) {
467               numSuccessfulSetCalls.incrementAndGet();
468             }
469             awaitUnchecked(barrier);
470             return null;
471           }
472         };
473     final Set<Object> finalResults = Collections.synchronizedSet(Sets.newIdentityHashSet());
474     Runnable collectResultsRunnable =
475         new Runnable() {
476           @Override
477           public void run() {
478             try {
479               String result = Uninterruptibles.getUninterruptibly(currentFuture.get());
480               finalResults.add(result);
481             } catch (ExecutionException e) {
482               finalResults.add(e.getCause());
483             } catch (CancellationException e) {
484               finalResults.add(CancellationException.class);
485             } finally {
486               awaitUnchecked(barrier);
487             }
488           }
489         };
490     Runnable collectResultsTimedGetRunnable =
491         new Runnable() {
492           @Override
493           public void run() {
494             Future<String> future = currentFuture.get();
495             while (true) {
496               try {
497                 String result = Uninterruptibles.getUninterruptibly(future, 0, TimeUnit.SECONDS);
498                 finalResults.add(result);
499                 break;
500               } catch (ExecutionException e) {
501                 finalResults.add(e.getCause());
502                 break;
503               } catch (CancellationException e) {
504                 finalResults.add(CancellationException.class);
505                 break;
506               } catch (TimeoutException e) {
507                 // loop
508               }
509             }
510             awaitUnchecked(barrier);
511           }
512         };
513     List<Callable<?>> allTasks = new ArrayList<>();
514     allTasks.add(completeSucessFullyRunnable);
515     allTasks.add(completeExceptionallyRunnable);
516     allTasks.add(cancelRunnable);
517     allTasks.add(setFutureCompleteSucessFullyRunnable);
518     allTasks.add(setFutureCompleteExceptionallyRunnable);
519     allTasks.add(setFutureCancelRunnable);
520     for (int k = 0; k < 50; k++) {
521       // For each listener we add a task that submits it to the executor directly for the blocking
522       // get usecase and another task that adds it as a listener to the future to exercise both
523       // racing addListener calls and addListener calls completing after the future completes.
524       final Runnable listener =
525           k % 2 == 0 ? collectResultsRunnable : collectResultsTimedGetRunnable;
526       allTasks.add(Executors.callable(listener));
527       allTasks.add(
528           new Callable<Void>() {
529             @Override
530             public Void call() throws Exception {
531               currentFuture.get().addListener(listener, executor);
532               return null;
533             }
534           });
535     }
536     assertEquals(allTasks.size() + 1, barrier.getParties());
537     for (int i = 0; i < 1000; i++) {
538       Collections.shuffle(allTasks);
539       final AbstractFuture<String> future = new AbstractFuture<String>() {};
540       currentFuture.set(future);
541       for (Callable<?> task : allTasks) {
542         @SuppressWarnings("unused") // go/futurereturn-lsc
543         Future<?> possiblyIgnoredError = executor.submit(task);
544       }
545       awaitUnchecked(barrier);
546       assertThat(future.isDone()).isTrue();
547       // inspect state and ensure it is correct!
548       // asserts that all get calling threads received the same value
549       Object result = Iterables.getOnlyElement(finalResults);
550       if (result == CancellationException.class) {
551         assertTrue(future.isCancelled());
552         if (future.wasInterrupted()) {
553           // We were cancelled, it is possible that setFuture could have succeeded too.
554           assertThat(numSuccessfulSetCalls.get()).isIn(Range.closed(1, 2));

556           assertThat(numSuccessfulSetCalls.get()).isEqualTo(1);
557         }
558       } else {
559         assertThat(numSuccessfulSetCalls.get()).isEqualTo(1);
560       }
561       // reset for next iteration
562       numSuccessfulSetCalls.set(0);
563       finalResults.clear();
564     }
565     executor.shutdown();
566   }

570     final int size = 50;
571     final CyclicBarrier barrier =
572         new CyclicBarrier(
573             2 // for the setter threads
574                 + size // for the listeners
575                 + size // for the get threads,
576                 + 1); // for the main thread

578     final AtomicReference<AbstractFuture<String>> currentFuture = Atomics.newReference();
579     final AtomicReference<AbstractFuture<String>> setFutureFuture = Atomics.newReference();
580     final AtomicBoolean setFutureSetSucess = new AtomicBoolean();
581     final AtomicBoolean setFutureCompletionSucess = new AtomicBoolean();
582     final AtomicBoolean cancellationSucess = new AtomicBoolean();
583     Runnable cancelRunnable =
584         new Runnable() {
585           @Override
586           public void run() {
587             cancellationSucess.set(currentFuture.get().cancel(true));
588             awaitUnchecked(barrier);
589           }
590         };

592         new Runnable() {
593           @Override
594           public void run() {
595             AbstractFuture<String> future = setFutureFuture.get();

597             setFutureCompletionSucess.set(future.set("hello-async-world"));
598             awaitUnchecked(barrier);
599           }
600         };
601     final Set<Object> finalResults = Collections.synchronizedSet(Sets.newIdentityHashSet());
602     Runnable collectResultsRunnable =
603         new Runnable() {
604           @Override
605           public void run() {
606             try {
607               String result = Uninterruptibles.getUninterruptibly(currentFuture.get());
608               finalResults.add(result);
609             } catch (ExecutionException e) {
610               finalResults.add(e.getCause());
611             } catch (CancellationException e) {
612               finalResults.add(CancellationException.class);
613             } finally {
614               awaitUnchecked(barrier);
615             }
616           }
617         };
618     Runnable collectResultsTimedGetRunnable =
619         new Runnable() {
620           @Override
621           public void run() {
622             Future<String> future = currentFuture.get();
623             while (true) {
624               try {
625                 String result = Uninterruptibles.getUninterruptibly(future, 0, TimeUnit.SECONDS);
626                 finalResults.add(result);
627                 break;
628               } catch (ExecutionException e) {
629                 finalResults.add(e.getCause());
630                 break;
631               } catch (CancellationException e) {
632                 finalResults.add(CancellationException.class);
633                 break;
634               } catch (TimeoutException e) {
635                 // loop
636               }
637             }
638             awaitUnchecked(barrier);
639           }
640         };
641     List<Runnable> allTasks = new ArrayList<>();
642     allTasks.add(cancelRunnable);
643     allTasks.add(setFutureCompleteSucessFullyRunnable);
644     for (int k = 0; k < size; k++) {
645       // For each listener we add a task that submits it to the executor directly for the blocking
646       // get usecase and another task that adds it as a listener to the future to exercise both
647       // racing addListener calls and addListener calls completing after the future completes.
648       final Runnable listener =
649           k % 2 == 0 ? collectResultsRunnable : collectResultsTimedGetRunnable;
650       allTasks.add(listener);
651       allTasks.add(
652           new Runnable() {
653             @Override
654             public void run() {
655               currentFuture.get().addListener(listener, executor);
656             }
657           });
658     }
659     assertEquals(allTasks.size() + 1, barrier.getParties()); // sanity check
660     for (int i = 0; i < 1000; i++) {
661       Collections.shuffle(allTasks);
662       final AbstractFuture<String> future = new AbstractFuture<String>() {};
663       final AbstractFuture<String> setFuture = new AbstractFuture<String>() {};
664       currentFuture.set(future);
665       setFutureFuture.set(setFuture);
666       for (Runnable task : allTasks) {
667         executor.execute(task);
668       }
669       awaitUnchecked(barrier);
670       assertThat(future.isDone()).isTrue();
671       // inspect state and ensure it is correct!
672       // asserts that all get calling threads received the same value
673       Object result = Iterables.getOnlyElement(finalResults);
674       if (result == CancellationException.class) {
675         assertTrue(future.isCancelled());
676         assertTrue(cancellationSucess.get());
677         // cancellation can interleave in 3 ways
678         // 1. prior to setFuture
679         // 2. after setFuture before set() on the future assigned
680         // 3. after setFuture and set() are called but before the listener completes.
681         if (!setFutureSetSucess.get() || !setFutureCompletionSucess.get()) {
682           // If setFuture fails or set on the future fails then it must be because that future was
683           // cancelled
684           assertTrue(setFuture.isCancelled());

1144   private static int findStackFrame(ExecutionException e, String clazz, String method) {
1145     StackTraceElement[] elements = e.getStackTrace();
1146     for (int i = 0; i < elements.length; i++) {
1147       StackTraceElement element = elements[i];
1148       if (element.getClassName().equals(clazz) && element.getMethodName().equals(method)) {
1149         return i;
1150       }
1151     }
1152     AssertionFailedError failure =
1153         new AssertionFailedError(
1154             "Expected element " + clazz + "." + method + " not found in stack trace");
1155     failure.initCause(e);
1156     throw failure;
1157   }

1245     private final CountDownLatch completedIteration = new CountDownLatch(10);
