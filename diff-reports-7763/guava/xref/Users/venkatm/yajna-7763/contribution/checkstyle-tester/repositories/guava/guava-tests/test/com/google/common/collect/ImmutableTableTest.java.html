47    public void testToImmutableTable() {
48      Collector<Cell<String, String, Integer>, ?, ImmutableTable<String, String, Integer>> collector =
49          ImmutableTable.toImmutableTable(Cell::getRowKey, Cell::getColumnKey, Cell::getValue);
50      Equivalence<ImmutableTable<String, String, Integer>> equivalence =
51          Equivalence.equals()
52              .<Cell<String, String, Integer>>pairwise()
53              .onResultOf(ImmutableTable::cellSet);
54      CollectorTester.of(collector, equivalence)
55          .expectCollects(
56              new ImmutableTable.Builder<String, String, Integer>()
57                  .put("one", "uno", 1)
58                  .put("two", "dos", 2)
59                  .put("three", "tres", 3)
60                  .build(),
61              Tables.immutableCell("one", "uno", 1),
62              Tables.immutableCell("two", "dos", 2),
63              Tables.immutableCell("three", "tres", 3));
64    }

247     Table.Cell<Character, Integer, String> mutableCell =

258

310   private static <R, C, V> void validateTableCopies(Table<R, C, V> original) {
311     Table<R, C, V> copy = ImmutableTable.copyOf(original);
312     assertEquals(original, copy);
313     validateViewOrdering(original, copy);
314 
315     Table<R, C, V> built = ImmutableTable.<R, C, V>builder().putAll(original).build();
316     assertEquals(original, built);
317     validateViewOrdering(original, built);
318   }
319 
320   private static <R, C, V> void validateViewOrdering(Table<R, C, V> original, Table<R, C, V> copy) {
321     assertThat(copy.cellSet()).containsExactlyElementsIn(original.cellSet()).inOrder();
322     assertThat(copy.rowKeySet()).containsExactlyElementsIn(original.rowKeySet()).inOrder();
323     assertThat(copy.values()).containsExactlyElementsIn(original.values()).inOrder();
324   }

326   public void testCopyOf() {
327     Table<Character, Integer, String> table = TreeBasedTable.create();
328     validateTableCopies(table);
329     table.put('b', 2, "foo");
330     validateTableCopies(table);
331     table.put('b', 1, "bar");
332     table.put('a', 2, "baz");
333     validateTableCopies(table);
334     // Even though rowKeySet, columnKeySet, and cellSet have the same
335     // iteration ordering, row has an inconsistent ordering.
336     assertThat(table.row('b').keySet()).containsExactly(1, 2).inOrder();
337     assertThat(ImmutableTable.copyOf(table).row('b').keySet()).containsExactly(2, 1).inOrder();
338   }

367   public void testBuilder_orderRowsAndColumnsBy_putAll() {
368     Table<Character, Integer, String> table = HashBasedTable.create();
369     table.put('b', 2, "foo");
370     table.put('b', 1, "bar");
371     table.put('a', 2, "baz");
372     ImmutableTable.Builder<Character, Integer, String> builder = ImmutableTable.builder();
373     Table<Character, Integer, String> copy =
374         builder
375             .orderRowsBy(Ordering.natural())
376             .orderColumnsBy(Ordering.natural())
377             .putAll(table)
378             .build();
379     assertThat(copy.rowKeySet()).containsExactly('a', 'b').inOrder();
380     assertThat(copy.columnKeySet()).containsExactly(1, 2).inOrder();
381     assertThat(copy.values()).containsExactly("baz", "bar", "foo").inOrder();
382     assertThat(copy.row('b').keySet()).containsExactly(1, 2).inOrder();
383   }
