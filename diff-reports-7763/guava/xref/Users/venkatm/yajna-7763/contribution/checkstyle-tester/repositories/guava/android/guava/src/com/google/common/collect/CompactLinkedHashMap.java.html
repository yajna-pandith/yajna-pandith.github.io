17  package com.google.common.collect;
18  
19  import com.google.common.annotations.GwtIncompatible;
20  import com.google.common.annotations.VisibleForTesting;
21  import com.google.errorprone.annotations.CanIgnoreReturnValue;
22  import java.util.Arrays;
23  import java.util.LinkedHashMap;
24  import java.util.Map;
25  import org.checkerframework.checker.nullness.compatqual.NullableDecl;
26  
27  /**
28   * CompactLinkedHashMap is an implementation of a Map with insertion or LRU iteration order,
29   * maintained with a doubly linked list through the entries. All optional operations (put and
30   * remove) are supported. Null keys and values are supported.
31   *
32   * <p>{@code containsKey(k)}, {@code put(k, v)} and {@code remove(k)} are all (expected and
33   * amortized) constant time operations. Expected in the hashtable sense (depends on the hash
34   * function doing a good job of distributing the elements to the buckets to a distribution not far
35   * from uniform), and amortized since some operations can trigger a hash table resize.
36   *
37   * <p>As compared with {@link java.util.LinkedHashMap}, this structure places significantly reduced
38   * load on the garbage collector by only using a constant number of internal objects.
39   *
40   * <p>This class should not be assumed to be universally superior to {@code
41   * java.util.LinkedHashMap}. Generally speaking, this class reduces object allocation and memory
42   * consumption at the price of moderately increased constant factors of CPU. Only use this class
43   * when there is a specific reason to prioritize memory over CPU.
44   *
45   * @author Louis Wasserman
46   */
47  @GwtIncompatible // not worth using in GWT for now
48  class CompactLinkedHashMap<K, V> extends CompactHashMap<K, V> {
49    // TODO(lowasser): implement removeEldestEntry so this can be used as a drop-in replacement
50  
51    /** Creates an empty {@code CompactLinkedHashMap} instance. */
52    public static <K, V> CompactLinkedHashMap<K, V> create() {
53      return new CompactLinkedHashMap<>();
54    }
55  
56    /**
57     * Creates a {@code CompactLinkedHashMap} instance, with a high enough "initial capacity" that it
58     * <i>should</i> hold {@code expectedSize} elements without rebuilding internal data structures.
59     *
60     * @param expectedSize the number of elements you expect to add to the returned set
61     * @return a new, empty {@code CompactLinkedHashMap} with enough capacity to hold {@code
62     *     expectedSize} elements without resizing
63     * @throws IllegalArgumentException if {@code expectedSize} is negative
64     */
65    public static <K, V> CompactLinkedHashMap<K, V> createWithExpectedSize(int expectedSize) {
66      return new CompactLinkedHashMap<>(expectedSize);
67    }
68  
69    private static final int ENDPOINT = -2;
70  
71    /**
72     * Contains the link pointers corresponding with the entries, in the range of [0, size()). The
73     * high 32 bits of each long is the "prev" pointer, whereas the low 32 bits is the "succ" pointer
74     * (pointing to the next entry in the linked list). The pointers in [size(), entries.length) are
75     * all "null" (UNSET).
76     *
77     * <p>A node with "prev" pointer equal to {@code ENDPOINT} is the first node in the linked list,
78     * and a node with "next" pointer equal to {@code ENDPOINT} is the last node.
79     */
80    @VisibleForTesting @NullableDecl transient long[] links;
81  
82    /** Pointer to the first node in the linked list, or {@code ENDPOINT} if there are no entries. */
83    private transient int firstEntry;
84  
85    /** Pointer to the last node in the linked list, or {@code ENDPOINT} if there are no entries. */
86    private transient int lastEntry;
87  
88    private final boolean accessOrder;
89  
90    CompactLinkedHashMap() {
91      this(CompactHashing.DEFAULT_SIZE);
92    }
93  
94    CompactLinkedHashMap(int expectedSize) {
95      this(expectedSize, false);
96    }
97  
98    CompactLinkedHashMap(int expectedSize, boolean accessOrder) {
99      super(expectedSize);
100     this.accessOrder = accessOrder;
101   }
102 
103   @Override
104   void init(int expectedSize) {
105     super.init(expectedSize);
106     this.firstEntry = ENDPOINT;
107     this.lastEntry = ENDPOINT;
108   }
109 
110   @Override
111   int allocArrays() {
112     int expectedSize = super.allocArrays();
113     this.links = new long[expectedSize];
114     return expectedSize;
115   }
116 
117   @Override
118   Map<K, V> createHashFloodingResistantDelegate(int tableSize) {
119     return new LinkedHashMap<K, V>(tableSize, 1.0f, accessOrder);
120   }
121 
122   @Override
123   @CanIgnoreReturnValue
124   Map<K, V> convertToHashFloodingResistantImplementation() {
125     Map<K, V> result = super.convertToHashFloodingResistantImplementation();
126     links = null;
127     return result;
128   }
129 
130   private int getPredecessor(int entry) {
131     return ((int) (links[entry] >>> 32)) - 1;
132   }
133 
134   @Override
135   int getSuccessor(int entry) {
136     return ((int) links[entry]) - 1;
137   }
138 
139   private void setSuccessor(int entry, int succ) {
140     long succMask = (~0L) >>> 32;
141     links[entry] = (links[entry] & ~succMask) | ((succ + 1) & succMask);
142   }
143 
144   private void setPredecessor(int entry, int pred) {
145     long predMask = ~0L << 32;
146     links[entry] = (links[entry] & ~predMask) | ((long) (pred + 1) << 32);
147   }
148 
149   private void setSucceeds(int pred, int succ) {
150     if (pred == ENDPOINT) {
151       firstEntry = succ;
152     } else {
153       setSuccessor(pred, succ);
154     }
155 
156     if (succ == ENDPOINT) {
157       lastEntry = pred;
158     } else {
159       setPredecessor(succ, pred);
160     }
161   }
162 
163   @Override
164   void insertEntry(int entryIndex, @NullableDecl K key, @NullableDecl V value, int hash, int mask) {
165     super.insertEntry(entryIndex, key, value, hash, mask);
166     setSucceeds(lastEntry, entryIndex);
167     setSucceeds(entryIndex, ENDPOINT);
168   }
169 
170   @Override
171   void accessEntry(int index) {
172     if (accessOrder) {
173       // delete from previous position...
174       setSucceeds(getPredecessor(index), getSuccessor(index));
175       // ...and insert at the end.
176       setSucceeds(lastEntry, index);
177       setSucceeds(index, ENDPOINT);
178       incrementModCount();
179     }
180   }
181 
182   @Override
183   void moveLastEntry(int dstIndex, int mask) {
184     int srcIndex = size() - 1;
185     super.moveLastEntry(dstIndex, mask);
186 
187     setSucceeds(getPredecessor(dstIndex), getSuccessor(dstIndex));
188     if (dstIndex < srcIndex) {
189       setSucceeds(getPredecessor(srcIndex), dstIndex);
190       setSucceeds(dstIndex, getSuccessor(srcIndex));
191     }
192     links[srcIndex] = 0;
193   }
194 
195   @Override
196   void resizeEntries(int newCapacity) {
197     super.resizeEntries(newCapacity);
198     links = Arrays.copyOf(links, newCapacity);
199   }
200 
201   @Override
202   int firstEntryIndex() {
203     return firstEntry;
204   }
205 
206   @Override
207   int adjustAfterRemove(int indexBeforeRemove, int indexRemoved) {
208     return (indexBeforeRemove >= size()) ? indexRemoved : indexBeforeRemove;
209   }
210 
211   @Override
212   public void clear() {
213     if (needsAllocArrays()) {
214       return;
215     }
216     this.firstEntry = ENDPOINT;
217     this.lastEntry = ENDPOINT;
218     if (links != null) {
219       Arrays.fill(links, 0, size(), 0);
220     }
221     super.clear();
222   }
223 }
