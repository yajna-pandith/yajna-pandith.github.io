248   public void testPublicSuffix() {
249     for (String name : PS) {
250       final InternetDomainName domain = InternetDomainName.from(name);
251       assertTrue(name, domain.isPublicSuffix());
252       assertTrue(name, domain.hasPublicSuffix());
253       assertFalse(name, domain.isUnderPublicSuffix());
254       assertFalse(name, domain.isTopPrivateDomain());
255       assertEquals(domain, domain.publicSuffix());
256     }
257 
258     for (String name : NO_PS) {
259       final InternetDomainName domain = InternetDomainName.from(name);
260       assertFalse(name, domain.isPublicSuffix());
261       assertFalse(name, domain.hasPublicSuffix());
262       assertFalse(name, domain.isUnderPublicSuffix());
263       assertFalse(name, domain.isTopPrivateDomain());
264       assertNull(domain.publicSuffix());
265     }
266 
267     for (String name : NON_PS) {
268       final InternetDomainName domain = InternetDomainName.from(name);
269       assertFalse(name, domain.isPublicSuffix());
270       assertTrue(name, domain.hasPublicSuffix());
271       assertTrue(name, domain.isUnderPublicSuffix());
272     }
273   }

418     final Iterable<String> validCases = Iterables.concat(VALID_NAME, PS, NO_PS, NON_PS);
419     final Iterable<String> invalidCases =

431   public void testToString() {
432     for (String inputName : SOMEWHERE_UNDER_PS) {
433       InternetDomainName domain = InternetDomainName.from(inputName);
434 
435       /*
436        * We would ordinarily use constants for the expected results, but
437        * doing it by derivation allows us to reuse the test case definitions
438        * used in other tests.
439        */
440 
441       String expectedName = Ascii.toLowerCase(inputName);
442       expectedName = expectedName.replaceAll("[\u3002\uFF0E\uFF61]", ".");
443 
444       if (expectedName.endsWith(".")) {
445         expectedName = expectedName.substring(0, expectedName.length() - 1);
446       }
447 
448       assertEquals(expectedName, domain.toString());
449     }
450   }
