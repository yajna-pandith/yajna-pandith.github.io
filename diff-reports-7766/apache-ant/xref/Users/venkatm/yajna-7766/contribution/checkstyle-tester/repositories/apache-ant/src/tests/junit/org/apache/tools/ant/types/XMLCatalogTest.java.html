80      @Test
81      public void testEmptyCatalog() {
82          try {
83              InputSource result = catalog.resolveEntity("PUBLIC ID ONE",
84                                                         "i/dont/exist.dtd");
85              assertNull("Empty catalog should return null", result);
86          } catch (Exception e) {
87              fail("resolveEntity() failed!" + e.toString());
88          }
89  
90          try {
91              Source result = catalog.resolve("i/dont/exist.dtd", null);
92              String expected = toURLString(new File(project.getBaseDir() +
93                                                     "/i/dont/exist.dtd"));
94              String resultStr =
95                  fileURLPartWithoutLeadingSlashes((SAXSource)result);
96              assertTrue("Empty catalog should return input with a system ID like "
97                         + expected + " but was " + resultStr,
98                         expected.endsWith(resultStr));
99          } catch (Exception e) {
100             fail("resolve() failed!" + e.toString());
101         }
102     }
103 
104     private static String fileURLPartWithoutLeadingSlashes(SAXSource result)
105         throws MalformedURLException {
106         //
107         // These shenanigans are necessary b/c Norm Walsh's resolver
108         // has a different idea of how file URLs are created on windoze
109         // ie file://c:/foo instead of file:///c:/foo
110         //
111         String resultStr =
112             new URL(result.getInputSource().getSystemId()).getFile();
113         // on Sun's Java6 this returns an unexpected number of four
114         // leading slashes, at least on Linux - strip all of them
115         while (resultStr.startsWith("/")) {
116             resultStr = resultStr.substring(1);
117         }
118         return resultStr;
119     }

126         dtd.setLocation("i/dont/exist.dtd");

137         assertTrue("Nonexistent Catalog entry return input with a system ID like "
138                    + expected + " but was " + resultStr,
139                    expected.endsWith(resultStr));
140     }

170     @Test
171     public void testCircularReferenceCheck() throws IOException, SAXException {
172 
173         // catalog <--> catalog
174         project.addReference("catalog", catalog);
175         catalog.setRefid(new Reference(project, "catalog"));
176 
177         try {
178             InputSource result = catalog.resolveEntity("PUBLIC ID ONE",
179                                                        "i/dont/exist.dtd");
180             fail("Can make XMLCatalog a Reference to itself.");
181         } catch (BuildException be) {
182             assertEquals("This data type contains a circular reference.",
183                          be.getMessage());
184         } catch (Exception e) {
185             fail("resolveEntity() failed!" + e.toString());
186         }
187 
188         // catalog1 --> catalog2 --> catalog3 --> catalog1
189         XMLCatalog catalog1 = newCatalog();
190         project.addReference("catalog1", catalog1);
191         XMLCatalog catalog2 = newCatalog();
192         project.addReference("catalog2", catalog2);
193         XMLCatalog catalog3 = newCatalog();
194         project.addReference("catalog3", catalog3);
195 
196         catalog3.setRefid(new Reference(project, "catalog1"));
197         catalog2.setRefid(new Reference(project, "catalog3"));
198         catalog1.setRefid(new Reference(project, "catalog2"));
199 
200         try {
201             catalog1.resolveEntity("PUBLIC ID ONE",
202                                                         "i/dont/exist.dtd");
203             fail("Can make circular reference");
204         } catch (BuildException be) {
205             assertEquals("This data type contains a circular reference.",
206                     be.getMessage());
207         }
208     }

352     @Test
353     public void testClasspath() throws IOException, TransformerException, SAXException {
354 
355 
356         String publicId = "-//stevo//DTD doc 1.0//EN";
357         String dtdLoc = "testcases/taskdefs/optional/xml/doc.dtd";
358         String path1 = project.getBaseDir().toString() + "/src/etc";
359 
360         ResourceLocation dtd = new ResourceLocation();
361         dtd.setPublicId(publicId);
362         dtd.setLocation(dtdLoc);
363         catalog.addDTD(dtd);
364         File dtdFile = project.resolveFile("src/etc/" + dtdLoc);
365 
366         String uri = "http://foo.com/bar/blah.xml";
367         String uriLoc = "etc/testcases/taskdefs/optional/xml/about.xml";
368         String path2 = project.getBaseDir().toString() + "/src";
369 
370         ResourceLocation entity = new ResourceLocation();
371         entity.setPublicId(uri);
372         entity.setLocation(uriLoc);
373         catalog.addEntity(entity);
374         File xmlFile = project.resolveFile("src/" + uriLoc);
375 
376         Path aPath = new Path(project, path1);
377         aPath.append(new Path(project, path2));
378         catalog.setClasspath(aPath);
379 
380         InputSource isResult = catalog.resolveEntity(publicId,
381                                                    "nap:chemical+brothers");
382         assertNotNull(isResult);
383         String resultStr1 = new URL(isResult.getSystemId()).getFile();
384         assertTrue(toURLString(dtdFile).endsWith(resultStr1));
385 
386         Source result = catalog.resolve(uri, null);
387         assertNotNull(result);
388         String resultStr = new URL(result.getSystemId()).getFile();
389         assertTrue(toURLString(xmlFile).endsWith(resultStr));
390 
391     }
