435     /**
436      * Iterate through the generated command line comparing it to reference one.
437      * @param sTestCmdLine          The reference command line;
438      * @param sGeneratedCmdLine     The generated command line;
439      */
440     private void checkCommandLines(String[] sTestCmdLine, String[] sGeneratedCmdLine) {
441         int testLength = sTestCmdLine.length;
442         int genLength = sGeneratedCmdLine.length;
443 
444         int genIndex = 0;
445         int testIndex = 0;
446 
447         while (testIndex < testLength) {
448             try {
449                 if (sGeneratedCmdLine[genIndex].equals("")) {
450                     genIndex++;
451                     continue;
452                 }
453                 assertEquals("arg # " + testIndex,
454                         sTestCmdLine[testIndex],
455                         sGeneratedCmdLine[genIndex]);
456                 testIndex++;
457                 genIndex++;
458             } catch (ArrayIndexOutOfBoundsException aioob) {
459                 fail("missing arg " + sTestCmdLine[testIndex]);
460             }
461         }
462 
463         // Count the number of empty strings
464         int cnt = 0;
465         for (int i = 0; i < genLength; i++) {
466             if (sGeneratedCmdLine[i].equals("")) {
467                 cnt++;
468             }
469         }
470         if (genLength - cnt > sTestCmdLine.length) {
471             // We have extra elements
472             fail("extra args");
473         }
474     }
