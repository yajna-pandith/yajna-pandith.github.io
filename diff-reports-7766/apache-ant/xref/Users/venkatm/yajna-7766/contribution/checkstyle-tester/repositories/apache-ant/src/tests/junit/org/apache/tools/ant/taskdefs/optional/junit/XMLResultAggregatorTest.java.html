19  package org.apache.tools.ant.taskdefs.optional.junit;
20  
21  import static org.junit.Assert.assertTrue;
22  
23  import java.io.File;
24  import java.io.FileOutputStream;
25  import java.io.PrintWriter;
26  import java.security.Permission;
27  
28  import org.apache.tools.ant.DefaultLogger;
29  import org.apache.tools.ant.Project;
30  import org.apache.tools.ant.taskdefs.Delete;
31  import org.apache.tools.ant.types.FileSet;
32  import org.junit.Assume;
33  import org.junit.Test;
34  
35  public class XMLResultAggregatorTest {
36  
37  	@Test
38      public void testFrames() throws Exception {
39          // For now, skip this test on JDK 6 (and below); see below for why:
40          try {
41              Class.forName("java.nio.file.Files");
42          } catch (ClassNotFoundException x) {
43              Assume.assumeNoException("Skip test on JDK 6 and below", x);
44          }
45          final File d = new File(System.getProperty("java.io.tmpdir"), "XMLResultAggregatorTest");
46          if (d.exists()) {
47              new Delete() {{removeDir(d);}}; // is there no utility method for this?
48          }
49          assertTrue(d.getAbsolutePath(), d.mkdir());
50          File xml = new File(d, "x.xml");
51          PrintWriter pw = new PrintWriter(new FileOutputStream(xml));
52          try {
53              pw.println("<testsuite errors='0' failures='0' name='my.UnitTest' tests='1'>");
54              pw.println(" <testcase classname='my.UnitTest' name='testSomething'/>");
55              pw.println("</testsuite>");
56              pw.flush();
57          } finally {
58              pw.close();
59          }
60          XMLResultAggregator task = new XMLResultAggregator();
61          task.setTodir(d);
62          AggregateTransformer report = task.createReport();
63          report.setTodir(d);
64          FileSet fs = new FileSet();
65          fs.setFile(xml);
66          task.addFileSet(fs);
67          Project project = new Project();
68          DefaultLogger logger = new DefaultLogger();
69          logger.setOutputPrintStream(System.out);
70          logger.setErrorPrintStream(System.err);
71          logger.setMessageOutputLevel(Project.MSG_INFO);
72          project.addBuildListener(logger);
73          project.init();
74          task.setProject(project);
75          /* getResourceAsStream override unnecessary on JDK 7. Ought to work around JAXP #6723276 in JDK 6, but causes a TypeCheckError in FunctionCall for reasons TBD:
76          Thread.currentThread().setContextClassLoader(new ClassLoader(ClassLoader.getSystemClassLoader().getParent()) {
77              public InputStream getResourceAsStream(String name) {
78                  if (name.startsWith("META-INF/services/")) {
79                      return new ByteArrayInputStream(new byte[0]);
80                  }
81                  return super.getResourceAsStream(name);
82              }
83          });
84          */
85          // Use the JRE's Xerces, not lib/optional/xerces.jar:
86          Thread.currentThread().setContextClassLoader(ClassLoader.getSystemClassLoader().getParent());
87          // Tickle #51668:
88          System.setSecurityManager(new SecurityManager() {public void checkPermission(Permission perm) {}});
89          task.execute();
90          assertTrue(new File(d, "index.html").isFile());
91      }
