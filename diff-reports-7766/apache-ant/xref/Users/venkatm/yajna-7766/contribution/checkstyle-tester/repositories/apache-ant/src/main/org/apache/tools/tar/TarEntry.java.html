34  /**
35   * This class represents an entry in a Tar archive. It consists
36   * of the entry's header, as well as the entry's File. Entries
37   * can be instantiated in one of three ways, depending on how
38   * they are to be used.
39   * <p>
40   * TarEntries that are created from the header bytes read from
41   * an archive are instantiated with the TarEntry( byte[] )
42   * constructor. These entries will be used when extracting from
43   * or listing the contents of an archive. These entries have their
44   * header filled in using the header bytes. They also set the File
45   * to null, since they reference an archive entry not a file.
46   * <p>
47   * TarEntries that are created from Files that are to be written
48   * into an archive are instantiated with the TarEntry( File )
49   * constructor. These entries have their header filled in using
50   * the File's information. They also keep a reference to the File
51   * for convenience when writing entries.
52   * <p>
53   * Finally, TarEntries can be constructed from nothing but a name.
54   * This allows the programmer to construct the entry by hand, for
55   * instance when only an InputStream is available for writing to
56   * the archive, and the header information is constructed from
57   * other information. In this case the header fields are set to
58   * defaults and the File is set to null.
59   *
60   * <p>
61   * The C structure for a Tar Entry's header is:
62   * <pre>
63   * struct header {
64   * char name[NAMSIZ];
65   * char mode[8];
66   * char uid[8];
67   * char gid[8];
68   * char size[12];
69   * char mtime[12];
70   * char chksum[8];
71   * char linkflag;
72   * char linkname[NAMSIZ];
73   * char magic[8];
74   * char uname[TUNMLEN];
75   * char gname[TGNMLEN];
76   * char devmajor[8];
77   * char devminor[8];
78   * } header;
79   * All unused bytes are set to null.
80   * New-style GNU tar files are slightly different from the above.
81   * For values of size larger than 077777777777L (11 7s)
82   * or uid and gid larger than 07777777L (7 7s)
83   * the sign bit of the first byte is set, and the rest of the
84   * field is the binary representation of the number.
85   * See TarUtils.parseOctalOrBinary.
86   * </pre>
87   * 
88   * <p>
89   * The C structure for a old GNU Tar Entry's header is:
90   * <pre>
91   * struct oldgnu_header {
92   * char unused_pad1[345]; // TarConstants.PAD1LEN_GNU       - offset 0
93   * char atime[12];        // TarConstants.ATIMELEN_GNU      - offset 345
94   * char ctime[12];        // TarConstants.CTIMELEN_GNU      - offset 357
95   * char offset[12];       // TarConstants.OFFSETLEN_GNU     - offset 369
96   * char longnames[4];     // TarConstants.LONGNAMESLEN_GNU  - offset 381
97   * char unused_pad2;      // TarConstants.PAD2LEN_GNU       - offset 385
98   * struct sparse sp[4];   // TarConstants.SPARSELEN_GNU     - offset 386
99   * char isextended;       // TarConstants.ISEXTENDEDLEN_GNU - offset 482
100  * char realsize[12];     // TarConstants.REALSIZELEN_GNU   - offset 483
101  * char unused_pad[17];   // TarConstants.PAD3LEN_GNU       - offset 495
102  * };
103  * </pre>
104  * Whereas, "struct sparse" is:
105  * <pre>
106  * struct sparse {
107  * char offset[12];   // offset 0
108  * char numbytes[12]; // offset 12
109  * };
110  * </pre>
111  *
112  */
113

177     /**
178      * Construct an empty entry and prepares the header values.
179      */
180     private TarEntry() {
181         this.magic = MAGIC_POSIX;
182         this.version = VERSION_POSIX;
183         this.name = "";
184         this.linkName = "";
185 
186         String user = System.getProperty("user.name", "");
187 
188         if (user.length() > MAX_NAMELEN) {
189             user = user.substring(0, MAX_NAMELEN);
190         }
191 
192         this.userId = 0;
193         this.groupId = 0;
194         this.userName = user;

196         this.file = null;

218         this();

221         boolean isDir = name.endsWith("/");
222 
223         this.devMajor = 0;
224         this.devMinor = 0;
225         this.name = name;
226         this.mode = isDir ? DEFAULT_DIR_MODE : DEFAULT_FILE_MODE;
227         this.linkFlag = isDir ? LF_DIR : LF_NORMAL;
228         this.userId = 0;
229         this.groupId = 0;
230         this.size = 0;
231         this.modTime = (new Date()).getTime() / MILLIS_PER_SECOND;
232         this.linkName = "";
233         this.userName = "";
234         this.groupName = "";
235     }

263     /**
264      * Construct an entry for a file. File is set to file, and the
265      * header is constructed from information from the file.
266      *
267      * @param file The file that the entry represents.
268      * @param fileName the name to be used for the entry.
269      */
270     public TarEntry(File file, String fileName) {
271         this();
272 
273         this.file = file;
274 
275         this.linkName = "";
276 
277         if (file.isDirectory()) {
278             this.mode = DEFAULT_DIR_MODE;
279             this.linkFlag = LF_DIR;
280 
281             int nameLength = fileName.length();
282             if (nameLength == 0 || fileName.charAt(nameLength - 1) != '/') {
283                 this.name = fileName + "/";
284             } else {
285                 this.name = fileName;
286             }
287             this.size = 0;
288         } else {
289             this.mode = DEFAULT_FILE_MODE;
290             this.linkFlag = LF_NORMAL;
291             this.size = file.length();
292             this.name = fileName;
293         }
294 
295         this.modTime = file.lastModified() / MILLIS_PER_SECOND;
296         this.devMajor = 0;
297         this.devMinor = 0;
298     }

763     /**
764      * If this entry represents a file, and the file is a directory, return
765      * an array of TarEntries for this entry's children.
766      *
767      * @return An array of TarEntry's for this entry's children.
768      */
769     public TarEntry[] getDirectoryEntries() {
770         if (file == null || !file.isDirectory()) {
771             return new TarEntry[0];
772         }
773 
774         String[]   list = file.list();
775         TarEntry[] result = new TarEntry[list.length];
776 
777         for (int i = 0; i < list.length; ++i) {
778             result[i] = new TarEntry(new File(file, list[i]));
779         }
780 
781         return result;
782     }

784     /**
785      * Write an entry's header information to a header buffer.
786      *
787      * <p>This method does not use the star/GNU tar/BSD tar extensions.</p>
788      *
789      * @param outbuf The tar entry header buffer to fill in.
790      */
791     public void writeEntryHeader(byte[] outbuf) {
792         try {
793             writeEntryHeader(outbuf, TarUtils.DEFAULT_ENCODING, false);
794         } catch (IOException ex) {
795             try {
796                 writeEntryHeader(outbuf, TarUtils.FALLBACK_ENCODING, false);
797             } catch (IOException ex2) {
798                 // impossible
799                 throw new RuntimeException(ex2);
800             }
801         }
802     }

804     /**
805      * Write an entry's header information to a header buffer.
806      *
807      * @param outbuf The tar entry header buffer to fill in.
808      * @param encoding encoding to use when writing the file name.
809      * @param starMode whether to use the star/GNU tar/BSD tar
810      * extension for numeric fields if their value doesn't fit in the
811      * maximum size of standard tar archives
812      */
813     public void writeEntryHeader(byte[] outbuf, ZipEncoding encoding,
814                                  boolean starMode) throws IOException {
815         int offset = 0;
816 
817         offset = TarUtils.formatNameBytes(name, outbuf, offset, NAMELEN,
818                                           encoding);
819         offset = writeEntryHeaderField(mode, outbuf, offset, MODELEN, starMode);
820         offset = writeEntryHeaderField(userId, outbuf, offset, UIDLEN,
821                                        starMode);
822         offset = writeEntryHeaderField(groupId, outbuf, offset, GIDLEN,
823                                        starMode);
824         offset = writeEntryHeaderField(size, outbuf, offset, SIZELEN, starMode);
825         offset = writeEntryHeaderField(modTime, outbuf, offset, MODTIMELEN,
826                                        starMode);
827 
828         int csOffset = offset;
829 
830         for (int c = 0; c < CHKSUMLEN; ++c) {
831             outbuf[offset++] = (byte) ' ';
832         }
833 
834         outbuf[offset++] = linkFlag;
835         offset = TarUtils.formatNameBytes(linkName, outbuf, offset, NAMELEN,
836                                           encoding);
837         offset = TarUtils.formatNameBytes(magic, outbuf, offset, PURE_MAGICLEN);
838         offset = TarUtils.formatNameBytes(version, outbuf, offset, VERSIONLEN);
839         offset = TarUtils.formatNameBytes(userName, outbuf, offset, UNAMELEN,
840                                           encoding);
841         offset = TarUtils.formatNameBytes(groupName, outbuf, offset, GNAMELEN,
842                                           encoding);
843         offset = writeEntryHeaderField(devMajor, outbuf, offset, DEVLEN,
844                                        starMode);
845         offset = writeEntryHeaderField(devMinor, outbuf, offset, DEVLEN,
846                                        starMode);
847 
848         while (offset < outbuf.length) {
849             outbuf[offset++] = 0;
850         }
851 
852         long chk = TarUtils.computeCheckSum(outbuf);
853 
854         TarUtils.formatCheckSumOctalBytes(chk, outbuf, csOffset, CHKSUMLEN);
855     }
856 
857     private int writeEntryHeaderField(long value, byte[] outbuf, int offset,
858                                       int length, boolean starMode) {
859         if (!starMode && (value < 0
860                           || value >= (1l << (3 * (length - 1))))) {
861             // value doesn't fit into field when written as octal
862             // number, will be written to PAX header or causes an
863             // error
864             return TarUtils.formatLongOctalBytes(0, outbuf, offset, length);
865         }
866         return TarUtils.formatLongOctalOrBinaryBytes(value, outbuf, offset,
867                                                      length);
868     }

902     private void parseTarHeader(byte[] header, ZipEncoding encoding,
903                                 final boolean oldStyle)
904         throws IOException {
905         int offset = 0;
906 
907         name = oldStyle ? TarUtils.parseName(header, offset, NAMELEN)
908             : TarUtils.parseName(header, offset, NAMELEN, encoding);
909         offset += NAMELEN;
910         mode = (int) TarUtils.parseOctalOrBinary(header, offset, MODELEN);
911         offset += MODELEN;
912         userId = (int) TarUtils.parseOctalOrBinary(header, offset, UIDLEN);
913         offset += UIDLEN;
914         groupId = (int) TarUtils.parseOctalOrBinary(header, offset, GIDLEN);
915         offset += GIDLEN;
916         size = TarUtils.parseOctalOrBinary(header, offset, SIZELEN);
917         offset += SIZELEN;
918         modTime = TarUtils.parseOctalOrBinary(header, offset, MODTIMELEN);
919         offset += MODTIMELEN;
920         offset += CHKSUMLEN;
921         linkFlag = header[offset++];
922         linkName = oldStyle ? TarUtils.parseName(header, offset, NAMELEN)
923             : TarUtils.parseName(header, offset, NAMELEN, encoding);
924         offset += NAMELEN;
925         magic = TarUtils.parseName(header, offset, PURE_MAGICLEN);
926         offset += PURE_MAGICLEN;
927         version = TarUtils.parseName(header, offset, VERSIONLEN);
928         offset += VERSIONLEN;
929         userName = oldStyle ? TarUtils.parseName(header, offset, UNAMELEN)
930             : TarUtils.parseName(header, offset, UNAMELEN, encoding);
931         offset += UNAMELEN;
932         groupName = oldStyle ? TarUtils.parseName(header, offset, GNAMELEN)
933             : TarUtils.parseName(header, offset, GNAMELEN, encoding);
934         offset += GNAMELEN;
935         devMajor = (int) TarUtils.parseOctalOrBinary(header, offset, DEVLEN);
936         offset += DEVLEN;
937         devMinor = (int) TarUtils.parseOctalOrBinary(header, offset, DEVLEN);
938         offset += DEVLEN;
939 
940         int type = evaluateType(header);
941         switch (type) {
942         case FORMAT_OLDGNU: {
943             offset += ATIMELEN_GNU;
944             offset += CTIMELEN_GNU;
945             offset += OFFSETLEN_GNU;
946             offset += LONGNAMESLEN_GNU;
947             offset += PAD2LEN_GNU;
948             offset += SPARSELEN_GNU;
949             isExtended = TarUtils.parseBoolean(header, offset);
950             offset += ISEXTENDEDLEN_GNU;
951             realSize = TarUtils.parseOctal(header, offset, REALSIZELEN_GNU);
952             offset += REALSIZELEN_GNU;
953             break;
954         }
955         case FORMAT_POSIX:
956         default: {
957             String prefix = oldStyle
958                 ? TarUtils.parseName(header, offset, PREFIXLEN)
959                 : TarUtils.parseName(header, offset, PREFIXLEN, encoding);
960             // SunOS tar -E does not add / to directory names, so fix
961             // up to be consistent
962             if (isDirectory() && !name.endsWith("/")){
963                 name = name + "/";
964             }
965             if (prefix.length() > 0){
966                 name = prefix + "/" + name;
967             }
968         }
969         }
970     }

972     /**
973      * Strips Windows' drive letter as well as any leading slashes,
974      * turns path separators into forward slahes.
975      */
976     private static String normalizeFileName(String fileName,
977                                             boolean preserveLeadingSlashes) {
978         String osname = System.getProperty("os.name").toLowerCase(Locale.ENGLISH);
979 
980         if (osname != null) {
981 
982             // Strip off drive letters!
983             // REVIEW Would a better check be "(File.separator == '\')"?
984 
985             if (osname.startsWith("windows")) {
986                 if (fileName.length() > 2) {
987                     char ch1 = fileName.charAt(0);
988                     char ch2 = fileName.charAt(1);
989 
990                     if (ch2 == ':'
991                         && ((ch1 >= 'a' && ch1 <= 'z')
992                             || (ch1 >= 'A' && ch1 <= 'Z'))) {
993                         fileName = fileName.substring(2);
994                     }
995                 }
996             } else if (osname.indexOf("netware") > -1) {
997                 int colon = fileName.indexOf(':');
998                 if (colon != -1) {
999                     fileName = fileName.substring(colon + 1);
1000                 }
1001             }
1002         }
1003 
1004         fileName = fileName.replace(File.separatorChar, '/');
1005 
1006         // No absolute pathnames
1007         // Windows (and Posix?) paths can start with "\\NetworkDrive\",
1008         // so we loop on starting /'s.
1009         while (!preserveLeadingSlashes && fileName.startsWith("/")) {
1010             fileName = fileName.substring(1);
1011         }
1012         return fileName;
1013     }

1015     /**
1016      * Evaluate an entry's header format from a header buffer.
1017      *
1018      * @param header The tar entry header buffer to evaluate the format for.
1019      * @return format type
1020      */
1021     private int evaluateType(byte[] header) {
1022         if (matchAsciiBuffer(GNU_TMAGIC, header, MAGIC_OFFSET, PURE_MAGICLEN)) {
1023             return FORMAT_OLDGNU;
1024         }
1025         if (matchAsciiBuffer(MAGIC_POSIX, header, MAGIC_OFFSET, PURE_MAGICLEN)) {
1026             return FORMAT_POSIX;
1027         }
1028         return 0;
1029     }

1031     /**
1032      * Check if buffer contents matches Ascii String.
1033      * 
1034      * @param expected
1035      * @param buffer
1036      * @param offset
1037      * @param length
1038      * @return {@code true} if buffer is the same as the expected string
1039      */
1040     private static boolean matchAsciiBuffer(String expected, byte[] buffer,
1041                                             int offset, int length){
1042         byte[] buffer1;
1043         try {
1044             buffer1 = expected.getBytes("ASCII");
1045         } catch (UnsupportedEncodingException e) {
1046             throw new RuntimeException(e); // Should not happen
1047         }
1048         return isEqual(buffer1, 0, buffer1.length, buffer, offset, length,
1049                        false);
1050     }

1052     /**
1053      * Compare byte buffers, optionally ignoring trailing nulls
1054      * 
1055      * @param buffer1
1056      * @param offset1
1057      * @param length1
1058      * @param buffer2
1059      * @param offset2
1060      * @param length2
1061      * @param ignoreTrailingNulls
1062      * @return {@code true} if buffer1 and buffer2 have same contents, having regard to trailing nulls
1063      */
1064     private static boolean isEqual(
1065             final byte[] buffer1, final int offset1, final int length1,
1066             final byte[] buffer2, final int offset2, final int length2,
1067             boolean ignoreTrailingNulls){
1068         int minLen=length1 < length2 ? length1 : length2;
1069         for (int i=0; i < minLen; i++){
1070             if (buffer1[offset1+i] != buffer2[offset2+i]){
1071                 return false;
1072             }
1073         }
1074         if (length1 == length2){
1075             return true;
1076         }
1077         if (ignoreTrailingNulls){
1078             if (length1 > length2){
1079                 for(int i = length2; i < length1; i++){
1080                     if (buffer1[offset1+i] != 0){
1081                         return false;
1082                     }
1083                 }
1084             } else {
1085                 for(int i = length1; i < length2; i++){
1086                     if (buffer2[offset2+i] != 0){
1087                         return false;
1088                     }
1089                 }
1090             }
1091             return true;
1092         }
1093         return false;
1094     }
