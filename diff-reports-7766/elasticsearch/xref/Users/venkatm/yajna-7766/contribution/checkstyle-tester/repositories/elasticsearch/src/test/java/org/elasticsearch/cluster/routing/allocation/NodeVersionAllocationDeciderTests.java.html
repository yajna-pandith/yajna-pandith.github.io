155     @Test
156     public void testRandom() {
157         AllocationService service = createAllocationService(settingsBuilder()
158                 .put("cluster.routing.allocation.concurrent_recoveries", 10)
159                 .put(ClusterRebalanceAllocationDecider.CLUSTER_ROUTING_ALLOCATION_ALLOW_REBALANCE, "always")
160                 .put("cluster.routing.allocation.cluster_concurrent_rebalance", -1)
161                 .build());
162 
163         logger.info("Building initial routing table");
164         MetaData.Builder builder = MetaData.builder();
165         RoutingTable.Builder rtBuilder = RoutingTable.builder();
166         int numIndices = between(1, 20);
167         for (int i = 0; i < numIndices; i++) {
168             builder.put(IndexMetaData.builder("test_" + i).numberOfShards(between(1, 5)).numberOfReplicas(between(0, 2)));
169         }
170         MetaData metaData = builder.build();
171 
172         for (int i = 0; i < numIndices; i++) {
173             rtBuilder.addAsNew(metaData.index("test_" + i));
174         }
175         RoutingTable routingTable = rtBuilder.build();
176 
177         ClusterState clusterState = ClusterState.builder(org.elasticsearch.cluster.ClusterName.DEFAULT).metaData(metaData).routingTable(routingTable).build();
178         assertThat(routingTable.shardsWithState(UNASSIGNED).size(), equalTo(routingTable.allShards().size()));
179         List<DiscoveryNode> nodes = new ArrayList<>();
180         int nodeIdx = 0;
181         int iters = scaledRandomIntBetween(10, 100);
182         for (int i = 0; i < iters; i++) {
183             DiscoveryNodes.Builder nodesBuilder = DiscoveryNodes.builder();
184             int numNodes = between(1, 20);
185             if (nodes.size() > numNodes) {
186                 Collections.shuffle(nodes, getRandom());
187                 nodes = nodes.subList(0, numNodes);
188             } else {
189                 for (int j = nodes.size(); j < numNodes; j++) {
190                     if (frequently()) {
191                         nodes.add(newNode("node" + (nodeIdx++), randomBoolean() ? getPreviousVersion() : Version.CURRENT));
192                     } else {
193                         nodes.add(newNode("node" + (nodeIdx++), randomVersion()));
194                     }
195                 }
196             }
197             for (DiscoveryNode node : nodes) {
198                nodesBuilder.put(node);
199             }
200             clusterState = ClusterState.builder(clusterState).nodes(nodesBuilder).build();
201             clusterState = stabilize(clusterState, service);
202         }
203     }

273     private ClusterState stabilize(ClusterState clusterState, AllocationService service) {
274         logger.trace("RoutingNodes: {}", clusterState.routingNodes().prettyPrint());
275 
276         RoutingTable routingTable = service.reroute(clusterState).routingTable();
277         clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
278         RoutingNodes routingNodes = clusterState.routingNodes();
279         assertRecoveryNodeVersions(routingNodes);
280 
281         logger.info("complete rebalancing");
282         RoutingTable prev = routingTable;
283         boolean stable = false;
284         for (int i = 0; i < 1000; i++) {   // at most 200 iters - this should be enough for all tests
285             logger.trace("RoutingNodes: {}", clusterState.getRoutingNodes().prettyPrint());
286             routingTable = service.applyStartedShards(clusterState, routingNodes.shardsWithState(INITIALIZING)).routingTable();
287             clusterState = ClusterState.builder(clusterState).routingTable(routingTable).build();
288             routingNodes = clusterState.routingNodes();
289             if (stable = (routingTable == prev)) {
290                 break;
291             }
292             assertRecoveryNodeVersions(routingNodes);
293             prev = routingTable;
294         }
295         logger.info("stabilized success [{}]", stable);
296         assertThat(stable, is(true));
297         return clusterState;
298     }

300     private final void assertRecoveryNodeVersions(RoutingNodes routingNodes) {
301         logger.trace("RoutingNodes: {}", routingNodes.prettyPrint());
302 
303         List<MutableShardRouting> mutableShardRoutings = routingNodes.shardsWithState(ShardRoutingState.RELOCATING);
304         for (MutableShardRouting r : mutableShardRoutings) {
305             String toId = r.relocatingNodeId();
306             String fromId = r.currentNodeId();
307             assertThat(fromId, notNullValue());
308             assertThat(toId, notNullValue());
309             logger.trace("From: " + fromId + " with Version: " + routingNodes.node(fromId).node().version() + " to: " + toId + " with Version: " + routingNodes.node(toId).node().version());
310             assertTrue(routingNodes.node(toId).node().version().onOrAfter(routingNodes.node(fromId).node().version()));
311         }
312 
313         mutableShardRoutings = routingNodes.shardsWithState(ShardRoutingState.INITIALIZING);
314         for (MutableShardRouting r : mutableShardRoutings) {
315             if (r.initializing() && r.relocatingNodeId() == null && !r.primary()) {
316                 MutableShardRouting primary = routingNodes.activePrimary(r);
317                 assertThat(primary, notNullValue());
318                 String fromId = primary.currentNodeId();
319                 String toId = r.currentNodeId();
320                 logger.trace("From: " + fromId + " with Version: " + routingNodes.node(fromId).node().version() + " to: " + toId + " with Version: " + routingNodes.node(toId).node().version());
321                 assertTrue(routingNodes.node(toId).node().version().onOrAfter(routingNodes.node(fromId).node().version()));
322             }
323         }
324     }

326     public void testFailRecoverFromPre132WithCompression() {
327         final boolean compress = randomBoolean();
328         AllocationService service = createAllocationService(settingsBuilder()
329                 .put("cluster.routing.allocation.concurrent_recoveries", 10)
330                 .put(ClusterRebalanceAllocationDecider.CLUSTER_ROUTING_ALLOCATION_ALLOW_REBALANCE, "INDICES_ALL_ACTIVE")
331                 .put("cluster.routing.allocation.cluster_concurrent_rebalance", -1)
332                 .put(RecoverySettings.INDICES_RECOVERY_COMPRESS, compress)
333                 .build());
334 
335         logger.info("Building initial routing table");
336 
337         MetaData metaData = MetaData.builder()
338                 .put(IndexMetaData.builder("test").numberOfShards(1).numberOfReplicas(1))
339                 .build();
340 
341         RoutingTable routingTable = RoutingTable.builder()
342                 .addAsNew(metaData.index("test"))
343                 .build();
344 
345         ClusterState clusterState = ClusterState.builder(org.elasticsearch.cluster.ClusterName.DEFAULT).metaData(metaData).routingTable(routingTable).build();
346 
347         assertThat(routingTable.index("test").shards().size(), equalTo(1));
348         for (int i = 0; i < routingTable.index("test").shards().size(); i++) {
349             assertThat(routingTable.index("test").shard(i).shards().size(), equalTo(2));
350             for (ShardRouting shard : routingTable.index("test").shard(i).shards()) {
351                 assertEquals(shard.state(), UNASSIGNED);
352                 assertNull(shard.currentNodeId());
353             }
354         }
355         Version version = randomVersion();
356         clusterState = ClusterState.builder(clusterState).nodes(DiscoveryNodes.builder()
357                 .put(newNode("old0", version))).build();
358         clusterState = stabilize(clusterState, service);
359         routingTable = clusterState.routingTable();
360         for (int i = 0; i < routingTable.index("test").shards().size(); i++) {
361             assertEquals(routingTable.index("test").shard(i).shards().size(), 2);
362             for (ShardRouting shard : routingTable.index("test").shard(i).shards()) {
363                 if (shard.primary()) {
364                     assertEquals(shard.state(), STARTED);
365                     assertEquals(shard.currentNodeId(), "old0");
366                 } else {
367                     assertEquals(shard.state(), UNASSIGNED);
368                     assertNull(shard.currentNodeId());
369                 }
370             }
371         }
372 
373         clusterState = ClusterState.builder(clusterState).nodes(DiscoveryNodes.builder()
374                 .put(newNode("old0",  version))
375                 .put(newNode("new0"))).build();
376 
377         clusterState = stabilize(clusterState, service);
378         routingTable = clusterState.routingTable();
379         for (int i = 0; i < routingTable.index("test").shards().size(); i++) {
380             assertEquals(routingTable.index("test").shard(i).shards().size(), 2);
381             for (ShardRouting shard : routingTable.index("test").shard(i).shards()) {
382                 if (shard.primary()) {
383                     assertEquals(shard.state(), STARTED);
384                     assertEquals(shard.currentNodeId(), "old0");
385                 } else {
386                     if (version.before(Version.V_1_3_2) && compress) { // can't recover from pre 1.3.2 with compression enabled
387                         assertEquals(shard.state(), UNASSIGNED);
388                         assertNull(shard.currentNodeId());
389                     } else {
390                         assertEquals(shard.state(), STARTED);
391                         assertEquals(shard.currentNodeId(), "new0");
392                     }
393                 }
394             }
395 
396 
397         }
