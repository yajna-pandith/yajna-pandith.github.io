73      private Set<String> prepareData(int numShards) throws Exception {
74          Set<String> fullExpectedIds = Sets.newTreeSet();
75  
76          ImmutableSettings.Builder settingsBuilder = settingsBuilder()
77                  .put(indexSettings())
78                  .put("routing.hash.type", "simple");
79  
80          if (numShards > 0) {
81              settingsBuilder.put(SETTING_NUMBER_OF_SHARDS, numShards);
82          }
83  
84          client().admin().indices().create(createIndexRequest("test")
85                  .settings(settingsBuilder))
86                  .actionGet();
87  
88          ensureGreen();
89          for (int i = 0; i < 100; i++) {
90              index(Integer.toString(i), "test", i);
91              fullExpectedIds.add(Integer.toString(i));
92          }
93          refresh();
94          return fullExpectedIds;
95      }
96  
97      private void index(String id, String nameValue, int age) throws IOException {
98          client().index(Requests.indexRequest("test").type("type1").id(id).source(source(id, nameValue, age))).actionGet();
99      }
100 
101     private XContentBuilder source(String id, String nameValue, int age) throws IOException {
102         StringBuilder multi = new StringBuilder().append(nameValue);
103         for (int i = 0; i < age; i++) {
104             multi.append(" ").append(nameValue);
105         }
106         return jsonBuilder().startObject()
107                 .field("id", id)
108                 .field("nid", Integer.parseInt(id))
109                 .field("name", nameValue + id)
110                 .field("age", age)
111                 .field("multi", multi.toString())
112                 .field("_boost", age * 10)
113                 .endObject();
114     }

283             assertThat("make sure we don't have duplicates", expectedIds.remove(hit.id()), notNullValue());

286         searchResponse = client().searchScroll(searchScrollRequest(searchResponse.getScrollId())).actionGet();

344                 .query(termQuery("multi", "test"))
345                 .from(0).size(20).explain(true)

354         assertThat(searchResponse.getFacets().facet(QueryFacet.class, "all").getCount(), equalTo(100l));
355     }

358     public void testFailedSearchWithWrongQuery() throws Exception {
359         prepareData();
360 
361         NumShards test = getNumShards("test");
362 
363         logger.info("Start Testing failed search with wrong query");
364         try {
365             SearchResponse searchResponse = client().search(searchRequest("test").source("{ xxx }".getBytes(Charsets.UTF_8))).actionGet();
366             assertThat(searchResponse.getTotalShards(), equalTo(test.numPrimaries));
367             assertThat(searchResponse.getSuccessfulShards(), equalTo(0));
368             assertThat(searchResponse.getFailedShards(), equalTo(test.numPrimaries));
369             fail("search should fail");
370         } catch (ElasticsearchException e) {
371             assertThat(e.unwrapCause(), instanceOf(SearchPhaseExecutionException.class));
372             // all is well
373         }
374         logger.info("Done Testing failed search");
375     }
