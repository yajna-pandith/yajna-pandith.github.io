63      @Test
64      public void testIndexWithFewDocuments() throws Exception {
65          Settings nodeSettings = ImmutableSettings.builder()
66                  .put("node.add_id_to_custom_path", false)
67                  .put("node.enable_custom_paths", true)
68                  .put("gateway.type", "local") // don't delete things!
69                  .build();
70  
71          internalCluster().startNodesAsync(3, nodeSettings).get();
72          final String IDX = "test";
73          final Path dataPath = newTempDir().toPath();
74  
75          Settings idxSettings = ImmutableSettings.builder()
76                  .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
77                  .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 2)
78                  .put(IndexMetaData.SETTING_DATA_PATH, dataPath.toAbsolutePath().toString())
79                  .put(IndexMetaData.SETTING_SHADOW_REPLICAS, true)
80                  .put(IndexMetaData.SETTING_SHARED_FILESYSTEM, true)
81                  .build();
82  
83          prepareCreate(IDX).setSettings(idxSettings).addMapping("doc", "foo", "type=string").get();
84          ensureGreen(IDX);
85  
86          // So basically, the primary should fail and the replica will need to
87          // replay the translog, this is what this tests
88          client().prepareIndex(IDX, "doc", "1").setSource("foo", "bar").get();
89          client().prepareIndex(IDX, "doc", "2").setSource("foo", "bar").get();
90  
91          // Check that we can get doc 1 and 2, because we are doing realtime
92          // gets and getting from the primary
93          GetResponse gResp1 = client().prepareGet(IDX, "doc", "1").setRealtime(true).setFields("foo").get();
94          GetResponse gResp2 = client().prepareGet(IDX, "doc", "2").setRealtime(true).setFields("foo").get();
95          assertThat(gResp1.getField("foo").getValue().toString(), equalTo("bar"));
96          assertThat(gResp2.getField("foo").getValue().toString(), equalTo("bar"));
97  
98          flushAndRefresh(IDX);
99          client().prepareIndex(IDX, "doc", "3").setSource("foo", "bar").get();
100         client().prepareIndex(IDX, "doc", "4").setSource("foo", "bar").get();
101         refresh();
102 
103         // Check that we can get doc 1 and 2 without realtime
104         gResp1 = client().prepareGet(IDX, "doc", "1").setRealtime(false).setFields("foo").get();
105         gResp2 = client().prepareGet(IDX, "doc", "2").setRealtime(false).setFields("foo").get();
106         assertThat(gResp1.getField("foo").getValue().toString(), equalTo("bar"));
107         assertThat(gResp2.getField("foo").getValue().toString(), equalTo("bar"));
108 
109         logger.info("--> restarting all nodes");
110         if (randomBoolean()) {
111             logger.info("--> rolling restart");
112             internalCluster().rollingRestart();
113         } else {
114             logger.info("--> full restart");
115             internalCluster().fullRestart();
116         }
117 
118         client().admin().cluster().prepareHealth().setWaitForNodes("3").get();
119         ensureGreen(IDX);
120         flushAndRefresh(IDX);
121 
122         logger.info("--> performing query");
123         SearchResponse resp = client().prepareSearch(IDX).setQuery(matchAllQuery()).get();
124         assertHitCount(resp, 4);
125 
126         logger.info("--> deleting index");
127         assertAcked(client().admin().indices().prepareDelete(IDX));
128     }

259     public void testPrimaryRelocationWithConcurrentIndexing() throws Exception {
260         Settings nodeSettings = ImmutableSettings.builder()
261                 .put("node.add_id_to_custom_path", false)
262                 .put("node.enable_custom_paths", true)
263                 .build();
264 
265         String node1 = internalCluster().startNode(nodeSettings);
266         Path dataPath = newTempDir().toPath();
267         final String IDX = "test";
268 
269         Settings idxSettings = ImmutableSettings.builder()
270                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
271                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1)
272                 .put(IndexMetaData.SETTING_DATA_PATH, dataPath.toAbsolutePath().toString())
273                 .put(IndexMetaData.SETTING_SHADOW_REPLICAS, true)
274                 .put(IndexMetaData.SETTING_SHARED_FILESYSTEM, true)
275                 .build();
276 
277         prepareCreate(IDX).setSettings(idxSettings).addMapping("doc", "foo", "type=string").get();
278         ensureYellow(IDX);
279         // Node1 has the primary, now node2 has the replica
280         String node2 = internalCluster().startNode(nodeSettings);
281         ensureGreen(IDX);
282         flushAndRefresh(IDX);
283         String node3 = internalCluster().startNode(nodeSettings);
284         final AtomicInteger counter = new AtomicInteger(0);
285         final CountDownLatch started = new CountDownLatch(1);
286 
287         final int numPhase1Docs = scaledRandomIntBetween(25, 200);
288         final int numPhase2Docs = scaledRandomIntBetween(25, 200);
289         final CountDownLatch phase1finished = new CountDownLatch(1);
290         final CountDownLatch phase2finished = new CountDownLatch(1);
291 
292         Thread thread = new Thread() {
293             @Override
294             public void run() {
295                 started.countDown();
296                 while (counter.get() < (numPhase1Docs + numPhase2Docs)) {
297                     final IndexResponse indexResponse = client().prepareIndex(IDX, "doc",
298                             Integer.toString(counter.incrementAndGet())).setSource("foo", "bar").get();
299                     assertTrue(indexResponse.isCreated());
300                     final int docCount = counter.get();
301                     if (docCount == numPhase1Docs) {
302                         phase1finished.countDown();
303                     }
304                 }
305                 logger.info("--> stopping indexing thread");
306                 phase2finished.countDown();
307             }
308         };
309         thread.start();
310         started.await();
311         phase1finished.await(); // wait for a certain number of documents to be indexed
312         logger.info("--> excluding {} from allocation", node1);
313         // now prevent primary from being allocated on node 1 move to node_3
314         Settings build = ImmutableSettings.builder().put("index.routing.allocation.exclude._name", node1).build();
315         client().admin().indices().prepareUpdateSettings(IDX).setSettings(build).execute().actionGet();
316         // wait for more documents to be indexed post-recovery, also waits for
317         // indexing thread to stop
318         phase2finished.await();
319         ensureGreen(IDX);
320         thread.join();
321         logger.info("--> performing query");
322         flushAndRefresh();
323 
324         SearchResponse resp = client().prepareSearch(IDX).setQuery(matchAllQuery()).get();
325         assertHitCount(resp, counter.get());
326         assertHitCount(resp, numPhase1Docs + numPhase2Docs);
327     }

329     @Test
330     public void testPrimaryRelocationWhereRecoveryFails() throws Exception {
331         Settings nodeSettings = ImmutableSettings.builder()
332                 .put("node.add_id_to_custom_path", false)
333                 .put("node.enable_custom_paths", true)
334                 .put(TransportModule.TRANSPORT_SERVICE_TYPE_KEY, MockTransportService.class.getName())
335                 .build();
336 
337         String node1 = internalCluster().startNode(nodeSettings);
338         Path dataPath = newTempDir().toPath();
339         final String IDX = "test";
340 
341         Settings idxSettings = ImmutableSettings.builder()
342                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
343                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 1)
344                 .put(IndexMetaData.SETTING_DATA_PATH, dataPath.toAbsolutePath().toString())
345                 .put(IndexMetaData.SETTING_SHADOW_REPLICAS, true)
346                 .put(IndexMetaData.SETTING_SHARED_FILESYSTEM, true)
347                 .build();
348 
349         prepareCreate(IDX).setSettings(idxSettings).addMapping("doc", "foo", "type=string").get();
350         ensureYellow(IDX);
351         // Node1 has the primary, now node2 has the replica
352         String node2 = internalCluster().startNode(nodeSettings);
353         ensureGreen(IDX);
354         flushAndRefresh(IDX);
355         String node3 = internalCluster().startNode(nodeSettings);
356         final AtomicInteger counter = new AtomicInteger(0);
357         final CountDownLatch started = new CountDownLatch(1);
358 
359         final int numPhase1Docs = scaledRandomIntBetween(25, 200);
360         final int numPhase2Docs = scaledRandomIntBetween(25, 200);
361         final int numPhase3Docs = scaledRandomIntBetween(25, 200);
362         final CountDownLatch phase1finished = new CountDownLatch(1);
363         final CountDownLatch phase2finished = new CountDownLatch(1);
364         final CountDownLatch phase3finished = new CountDownLatch(1);
365 
366         final AtomicBoolean keepFailing = new AtomicBoolean(true);
367 
368         MockTransportService mockTransportService = ((MockTransportService) internalCluster().getInstance(TransportService.class, node1));
369         mockTransportService.addDelegate(internalCluster().getInstance(Discovery.class, node3).localNode(),
370                 new MockTransportService.DelegateTransport(mockTransportService.original()) {
371 
372                     @Override
373                     public void sendRequest(DiscoveryNode node, long requestId, String action,
374                                             TransportRequest request, TransportRequestOptions options)
375                             throws IOException, TransportException {
376                         if (keepFailing.get() && action.equals(RecoveryTarget.Actions.TRANSLOG_OPS)) {
377                             logger.info("--> failing translog ops");
378                             throw new ElasticsearchException("failing on purpose");
379                         }
380                         super.sendRequest(node, requestId, action, request, options);
381                     }
382                 });
383 
384         Thread thread = new Thread() {
385             @Override
386             public void run() {
387                 started.countDown();
388                 while (counter.get() < (numPhase1Docs + numPhase2Docs + numPhase3Docs)) {
389                     final IndexResponse indexResponse = client().prepareIndex(IDX, "doc",
390                             Integer.toString(counter.incrementAndGet())).setSource("foo", "bar").get();
391                     assertTrue(indexResponse.isCreated());
392                     final int docCount = counter.get();
393                     if (docCount == numPhase1Docs) {
394                         phase1finished.countDown();
395                     } else if (docCount == (numPhase1Docs + numPhase2Docs)) {
396                         phase2finished.countDown();
397                     }
398                 }
399                 logger.info("--> stopping indexing thread");
400                 phase3finished.countDown();
401             }
402         };
403         thread.start();
404         started.await();
405         phase1finished.await(); // wait for a certain number of documents to be indexed
406         logger.info("--> excluding {} from allocation", node1);
407         // now prevent primary from being allocated on node 1 move to node_3
408         Settings build = ImmutableSettings.builder().put("index.routing.allocation.exclude._name", node1).build();
409         client().admin().indices().prepareUpdateSettings(IDX).setSettings(build).execute().actionGet();
410         // wait for more documents to be indexed post-recovery, also waits for
411         // indexing thread to stop
412         phase2finished.await();
413         // stop failing
414         keepFailing.set(false);
415         // wait for more docs to be indexed
416         phase3finished.await();
417         ensureGreen(IDX);
418         thread.join();
419         logger.info("--> performing query");
420         flushAndRefresh();
421 
422         SearchResponse resp = client().prepareSearch(IDX).setQuery(matchAllQuery()).get();
423         assertHitCount(resp, counter.get());
424     }
