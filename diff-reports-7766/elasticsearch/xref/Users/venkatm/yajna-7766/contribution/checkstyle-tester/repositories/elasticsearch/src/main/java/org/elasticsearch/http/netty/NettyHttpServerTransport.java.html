138     @Inject
139     public NettyHttpServerTransport(Settings settings, NetworkService networkService, BigArrays bigArrays) {
140         super(settings);
141         this.networkService = networkService;
142         this.bigArrays = bigArrays;
143 
144         if (settings.getAsBoolean("netty.epollBugWorkaround", false)) {
145             System.setProperty("org.jboss.netty.epollBugWorkaround", "true");
146         }
147 
148         ByteSizeValue maxContentLength = componentSettings.getAsBytesSize("max_content_length", settings.getAsBytesSize("http.max_content_length", new ByteSizeValue(100, ByteSizeUnit.MB)));
149         this.maxChunkSize = componentSettings.getAsBytesSize("max_chunk_size", settings.getAsBytesSize("http.max_chunk_size", new ByteSizeValue(8, ByteSizeUnit.KB)));
150         this.maxHeaderSize = componentSettings.getAsBytesSize("max_header_size", settings.getAsBytesSize("http.max_header_size", new ByteSizeValue(8, ByteSizeUnit.KB)));
151         this.maxInitialLineLength = componentSettings.getAsBytesSize("max_initial_line_length", settings.getAsBytesSize("http.max_initial_line_length", new ByteSizeValue(4, ByteSizeUnit.KB)));
152         // don't reset cookies by default, since I don't think we really need to
153         // note, parsing cookies was fixed in netty 3.5.1 regarding stack allocation, but still, currently, we don't need cookies
154         this.resetCookies = componentSettings.getAsBoolean("reset_cookies", settings.getAsBoolean("http.reset_cookies", false));
155         this.maxCumulationBufferCapacity = componentSettings.getAsBytesSize("max_cumulation_buffer_capacity", null);
156         this.maxCompositeBufferComponents = componentSettings.getAsInt("max_composite_buffer_components", -1);
157         this.workerCount = componentSettings.getAsInt("worker_count", EsExecutors.boundedNumberOfProcessors(settings) * 2);
158         this.blockingServer = settings.getAsBoolean("http.blocking_server", settings.getAsBoolean(TCP_BLOCKING_SERVER, settings.getAsBoolean(TCP_BLOCKING, false)));
159         this.port = componentSettings.get("port", settings.get("http.port", "9200-9300"));
160         this.bindHost = componentSettings.get("bind_host", settings.get("http.bind_host", settings.get("http.host")));
161         this.publishHost = componentSettings.get("publish_host", settings.get("http.publish_host", settings.get("http.host")));
162         this.publishPort = componentSettings.getAsInt("publish_port", settings.getAsInt("http.publish_port", 0));
163         this.tcpNoDelay = componentSettings.get("tcp_no_delay", settings.get(TCP_NO_DELAY, "true"));
164         this.tcpKeepAlive = componentSettings.get("tcp_keep_alive", settings.get(TCP_KEEP_ALIVE, "true"));
165         this.reuseAddress = componentSettings.getAsBoolean("reuse_address", settings.getAsBoolean(TCP_REUSE_ADDRESS, NetworkUtils.defaultReuseAddress()));
166         this.tcpSendBufferSize = componentSettings.getAsBytesSize("tcp_send_buffer_size", settings.getAsBytesSize(TCP_SEND_BUFFER_SIZE, TCP_DEFAULT_SEND_BUFFER_SIZE));
167         this.tcpReceiveBufferSize = componentSettings.getAsBytesSize("tcp_receive_buffer_size", settings.getAsBytesSize(TCP_RECEIVE_BUFFER_SIZE, TCP_DEFAULT_RECEIVE_BUFFER_SIZE));
168         this.detailedErrorsEnabled = settings.getAsBoolean(SETTING_HTTP_DETAILED_ERRORS_ENABLED, true);
169 
170         long defaultReceiverPredictor = 512 * 1024;
171         if (JvmInfo.jvmInfo().mem().directMemoryMax().bytes() > 0) {
172             // we can guess a better default...
173             long l = (long) ((0.3 * JvmInfo.jvmInfo().mem().directMemoryMax().bytes()) / workerCount);
174             defaultReceiverPredictor = Math.min(defaultReceiverPredictor, Math.max(l, 64 * 1024));
175         }
176 
177         // See AdaptiveReceiveBufferSizePredictor#DEFAULT_XXX for default values in netty..., we can use higher ones for us, even fixed one
178         ByteSizeValue receivePredictorMin = componentSettings.getAsBytesSize("receive_predictor_min", componentSettings.getAsBytesSize("receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
179         ByteSizeValue receivePredictorMax = componentSettings.getAsBytesSize("receive_predictor_max", componentSettings.getAsBytesSize("receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
180         if (receivePredictorMax.bytes() == receivePredictorMin.bytes()) {
181             receiveBufferSizePredictorFactory = new FixedReceiveBufferSizePredictorFactory((int) receivePredictorMax.bytes());
182         } else {
183             receiveBufferSizePredictorFactory = new AdaptiveReceiveBufferSizePredictorFactory((int) receivePredictorMin.bytes(), (int) receivePredictorMin.bytes(), (int) receivePredictorMax.bytes());
184         }
185 
186         this.compression = settings.getAsBoolean(SETTING_HTTP_COMPRESSION, false);
187         this.compressionLevel = settings.getAsInt(SETTING_HTTP_COMPRESSION_LEVEL, 6);
188         this.pipelining = settings.getAsBoolean(SETTING_PIPELINING, DEFAULT_SETTING_PIPELINING);
189         this.pipeliningMaxEvents = settings.getAsInt(SETTING_PIPELINING_MAX_EVENTS, DEFAULT_SETTING_PIPELINING_MAX_EVENTS);
190 
191         // validate max content length
192         if (maxContentLength.bytes() > Integer.MAX_VALUE) {
193             logger.warn("maxContentLength[" + maxContentLength + "] set to high value, resetting it to [100mb]");
194             maxContentLength = new ByteSizeValue(100, ByteSizeUnit.MB);
195         }
196         this.maxContentLength = maxContentLength;
197 
198         logger.debug("using max_chunk_size[{}], max_header_size[{}], max_initial_line_length[{}], max_content_length[{}], receive_predictor[{}->{}], pipelining[{}], pipelining_max_events[{}]",
199                 maxChunkSize, maxHeaderSize, maxInitialLineLength, this.maxContentLength, receivePredictorMin, receivePredictorMax, pipelining, pipeliningMaxEvents);
200     }

210     @Override
211     protected void doStart() throws ElasticsearchException {
212         this.serverOpenChannels = new OpenChannelsHandler(logger);
213 
214         if (blockingServer) {
215             serverBootstrap = new ServerBootstrap(new OioServerSocketChannelFactory(
216                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_boss")),
217                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_worker"))
218             ));
219         } else {
220             serverBootstrap = new ServerBootstrap(new NioServerSocketChannelFactory(
221                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_boss")),
222                     Executors.newCachedThreadPool(daemonThreadFactory(settings, "http_server_worker")),
223                     workerCount));
224         }
225 
226         serverBootstrap.setPipelineFactory(configureServerChannelPipelineFactory());
227 
228         if (!"default".equals(tcpNoDelay)) {
229             serverBootstrap.setOption("child.tcpNoDelay", Booleans.parseBoolean(tcpNoDelay, null));
230         }
231         if (!"default".equals(tcpKeepAlive)) {
232             serverBootstrap.setOption("child.keepAlive", Booleans.parseBoolean(tcpKeepAlive, null));
233         }
234         if (tcpSendBufferSize != null && tcpSendBufferSize.bytes() > 0) {
235             serverBootstrap.setOption("child.sendBufferSize", tcpSendBufferSize.bytes());
236         }
237         if (tcpReceiveBufferSize != null && tcpReceiveBufferSize.bytes() > 0) {
238             serverBootstrap.setOption("child.receiveBufferSize", tcpReceiveBufferSize.bytes());
239         }
240         serverBootstrap.setOption("receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
241         serverBootstrap.setOption("child.receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
242         if (reuseAddress != null) {
243             serverBootstrap.setOption("reuseAddress", reuseAddress);
244             serverBootstrap.setOption("child.reuseAddress", reuseAddress);
245         }
246 
247         // Bind and start to accept incoming connections.
248         InetAddress hostAddressX;
249         try {
250             hostAddressX = networkService.resolveBindHostAddress(bindHost);
251         } catch (IOException e) {
252             throw new BindHttpException("Failed to resolve host [" + bindHost + "]", e);
253         }
254         final InetAddress hostAddress = hostAddressX;
255 
256         PortsRange portsRange = new PortsRange(port);
257         final AtomicReference<Exception> lastException = new AtomicReference<>();
258         boolean success = portsRange.iterate(new PortsRange.PortCallback() {
259             @Override
260             public boolean onPortNumber(int portNumber) {
261                 try {
262                     serverChannel = serverBootstrap.bind(new InetSocketAddress(hostAddress, portNumber));
263                 } catch (Exception e) {
264                     lastException.set(e);
265                     return false;
266                 }
267                 return true;
268             }
269         });
270         if (!success) {
271             throw new BindHttpException("Failed to bind to [" + port + "]", lastException.get());
272         }
273 
274         InetSocketAddress boundAddress = (InetSocketAddress) serverChannel.getLocalAddress();
275         InetSocketAddress publishAddress;
276         if (0 == publishPort) {
277             publishPort = boundAddress.getPort();
278         }
279         try {
280             publishAddress = new InetSocketAddress(networkService.resolvePublishHostAddress(publishHost), publishPort);
281         } catch (Exception e) {
282             throw new BindTransportException("Failed to resolve publish address", e);
283         }
284         this.boundAddress = new BoundTransportAddress(new InetSocketTransportAddress(boundAddress), new InetSocketTransportAddress(publishAddress));
285     }

287     @Override
288     protected void doStop() throws ElasticsearchException {
289         if (serverChannel != null) {
290             serverChannel.close().awaitUninterruptibly();
291             serverChannel = null;
292         }
293 
294         if (serverOpenChannels != null) {
295             serverOpenChannels.close();
296             serverOpenChannels = null;
297         }
298 
299         if (serverBootstrap != null) {
300             serverBootstrap.releaseExternalResources();
301             serverBootstrap = null;
302         }
303     }

332     protected void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
333         if (e.getCause() instanceof ReadTimeoutException) {
334             if (logger.isTraceEnabled()) {
335                 logger.trace("Connection timeout [{}]", ctx.getChannel().getRemoteAddress());
336             }
337             ctx.getChannel().close();
338         } else {
339             if (!lifecycle.started()) {
340                 // ignore
341                 return;
342             }
343             if (!NetworkExceptionHelper.isCloseConnectionException(e.getCause())) {
344                 logger.warn("Caught exception while handling client http traffic, closing connection {}", e.getCause(), ctx.getChannel());
345                 ctx.getChannel().close();
346             } else {
347                 logger.debug("Caught exception while handling client http traffic, closing connection {}", e.getCause(), ctx.getChannel());
348                 ctx.getChannel().close();
349             }
350         }
351     }

357     protected static class HttpChannelPipelineFactory implements ChannelPipelineFactory {
358 
359         protected final NettyHttpServerTransport transport;
360         protected final HttpRequestHandler requestHandler;
361 
362         public HttpChannelPipelineFactory(NettyHttpServerTransport transport, boolean detailedErrorsEnabled) {
363             this.transport = transport;
364             this.requestHandler = new HttpRequestHandler(transport, detailedErrorsEnabled);
365         }
366 
367         @Override
368         public ChannelPipeline getPipeline() throws Exception {
369             ChannelPipeline pipeline = Channels.pipeline();
370             pipeline.addLast("openChannels", transport.serverOpenChannels);
371             HttpRequestDecoder requestDecoder = new HttpRequestDecoder(
372                     (int) transport.maxInitialLineLength.bytes(),
373                     (int) transport.maxHeaderSize.bytes(),
374                     (int) transport.maxChunkSize.bytes()
375             );
376             if (transport.maxCumulationBufferCapacity != null) {
377                 if (transport.maxCumulationBufferCapacity.bytes() > Integer.MAX_VALUE) {
378                     requestDecoder.setMaxCumulationBufferCapacity(Integer.MAX_VALUE);
379                 } else {
380                     requestDecoder.setMaxCumulationBufferCapacity((int) transport.maxCumulationBufferCapacity.bytes());
381                 }
382             }
383             if (transport.maxCompositeBufferComponents != -1) {
384                 requestDecoder.setMaxCumulationBufferComponents(transport.maxCompositeBufferComponents);
385             }
386             pipeline.addLast("decoder", requestDecoder);
387             pipeline.addLast("decoder_compress", new ESHttpContentDecompressor(transport.compression));
388             HttpChunkAggregator httpChunkAggregator = new HttpChunkAggregator((int) transport.maxContentLength.bytes());
389             if (transport.maxCompositeBufferComponents != -1) {
390                 httpChunkAggregator.setMaxCumulationBufferComponents(transport.maxCompositeBufferComponents);
391             }
392             pipeline.addLast("aggregator", httpChunkAggregator);
393             pipeline.addLast("encoder", new ESHttpResponseEncoder());
394             if (transport.compression) {
395                 pipeline.addLast("encoder_compress", new HttpContentCompressor(transport.compressionLevel));
396             }
397             if (transport.pipelining) {
398                 pipeline.addLast("pipelining", new HttpPipeliningHandler(transport.pipeliningMaxEvents));
399             }
400             pipeline.addLast("handler", requestHandler);
401             return pipeline;
402         }
