757     private PackageConfig makePackageConfig(String name, String namespace, PackageConfig parent,
758             String defaultResultType, ResultTypeConfig[] results, List<InterceptorConfig> interceptors,
759             List<InterceptorStackConfig> interceptorStacks, Set<String> globalAllowedMethods, boolean strictMethodInvocation) {
760         PackageConfig.Builder builder = new PackageConfig.Builder(name);
761         if (namespace != null) {
762             builder.namespace(namespace);
763         }
764         if (parent != null) {
765             builder.addParent(parent);
766         }
767         if (defaultResultType != null) {
768             builder.defaultResultType(defaultResultType);
769         }
770         if (results != null) {
771             for (ResultTypeConfig result : results) {
772                 builder.addResultTypeConfig(result);
773             }
774         }
775         if (interceptors != null) {
776             for (InterceptorConfig ref : interceptors) {
777                 builder.addInterceptorConfig(ref);
778             }
779         }
780         if (interceptorStacks != null) {
781             for (InterceptorStackConfig ref : interceptorStacks) {
782                 builder.addInterceptorStackConfig(ref);
783             }
784         }
785 
786         if (globalAllowedMethods != null) {
787             builder.addGlobalAllowedMethods(globalAllowedMethods);
788         }
789 
790         if (!strictMethodInvocation) {
791             builder.strictMethodInvocation(strictMethodInvocation);
792         }
793 
794         return new MyPackageConfig(builder.build());
795     }
