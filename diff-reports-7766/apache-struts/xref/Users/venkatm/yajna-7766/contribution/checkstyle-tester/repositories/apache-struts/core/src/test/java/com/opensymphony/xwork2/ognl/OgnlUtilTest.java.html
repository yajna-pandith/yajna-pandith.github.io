137     public void testClearExpressionCache() throws OgnlException {
138         ognlUtil.setEnableExpressionCache("true");
139         // Test that the expression cache is functioning as expected.
140         Object expr0 = ognlUtil.compile("test");
141         Object expr1 = ognlUtil.compile("test");
142         Object expr2 = ognlUtil.compile("test");
143         // Cache in effect, so expr0, expr1, expr2 should be the same.
144         assertSame(expr0, expr1);
145         assertSame(expr0, expr2);
146         assertTrue("Expression cache empty before clear ?", ognlUtil.expressionCacheSize() > 0);
147         // Clear the Epxression cache and confirm subsequent requests are new.
148         ognlUtil.clearExpressionCache();
149         assertEquals("Expression cache not empty after clear ?", 0, ognlUtil.expressionCacheSize());
150         Object expr3 = ognlUtil.compile("test");
151         Object expr4 = ognlUtil.compile("test");
152         Object expr5 = ognlUtil.compile("test");
153         // Cache cleared, expr3 should be a new instance.
154         assertNotSame(expr0, expr3);
155         // Cache still in effect, so expr3, expr4, expr5 should be the same.
156         assertSame(expr3, expr4);
157         assertSame(expr3, expr5);
158         assertTrue("Expression cache empty after usage ?", ognlUtil.expressionCacheSize() > 0);
159     }
