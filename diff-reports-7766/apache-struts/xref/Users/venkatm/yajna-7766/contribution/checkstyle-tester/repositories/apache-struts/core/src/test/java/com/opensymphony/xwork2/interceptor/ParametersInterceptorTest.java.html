51      public void testParameterNameAware() {
52          ParametersInterceptor pi = createParametersInterceptor();
53          final Map<String, Object> actual = injectValueStackFactory(pi);
54          ValueStack stack = createStubValueStack(actual);
55          final Map<String, Object> expected = new HashMap<String, Object>() {
56              {
57                  put("fooKey", "fooValue");
58                  put("barKey", "barValue");
59              }
60          };
61          Object a = new ParameterNameAware() {
62              public boolean acceptableParameterName(String parameterName) {
63                  return expected.containsKey(parameterName);
64              }
65          };
66          final Map<String, Object> parameters = new HashMap<String, Object>() {
67              {
68                  put("fooKey", "fooValue");
69                  put("barKey", "barValue");
70                  put("error-key", "error");
71                  put("error key", "error");
72                  put("error:key", "error");
73                  put("error+key", "error");
74                  put("test%test", "test%test");
75              }
76          };
77          pi.setParameters(a, stack, HttpParameters.create(parameters).build());
78          assertEquals(expected, actual);
79      }
