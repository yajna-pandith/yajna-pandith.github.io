33      public void testNulls() throws IntrospectionException, InvocationTargetException, NoSuchMethodException,
34              JSONException, InstantiationException, IllegalAccessException {
35          JSONPopulator populator = new JSONPopulator();
36          OtherBean bean = new OtherBean();
37          Map jsonMap = new HashMap();
38  
39          jsonMap.put("intField", null);
40          jsonMap.put("booleanField", null);
41          jsonMap.put("charField", null);
42          jsonMap.put("longField", null);
43          jsonMap.put("floatField", null);
44          jsonMap.put("doubleField", null);
45          jsonMap.put("byteField", null);
46  
47          populator.populateObject(bean, jsonMap);
48          assertNull(bean.getIntField());
49          assertNull(bean.isBooleanField());
50          assertNull(bean.getCharField());
51          assertNull(bean.getLongField());
52          assertNull(bean.getDoubleField());
53          assertNull(bean.getByteField());
54      }

56      public void testPrimitiveBean() throws Exception {
57          StringReader stringReader = new StringReader(TestUtils.readContent(JSONInterceptorTest.class
58                  .getResource("json-7.txt")));
59          Object json = JSONUtil.deserialize(stringReader);
60          assertNotNull(json);
61          assertTrue(json instanceof Map);
62          Map jsonMap = (Map) json;
63          JSONPopulator populator = new JSONPopulator();
64          Bean bean = new Bean();
65          populator.populateObject(bean, jsonMap);
66          assertTrue(bean.isBooleanField());
67          assertEquals("test\u000E\u000f", bean.getStringField());
68          assertEquals(10, bean.getIntField());
69          assertEquals('s', bean.getCharField());
70          assertEquals(10.1d, bean.getDoubleField(), 0d);
71          assertEquals(3, bean.getByteField());
72          assertEquals(new BigDecimal(111111.5d), bean.getBigDecimal());
73          assertEquals(new BigInteger("111111"), bean.getBigInteger());
74      }

76      public void testObjectBean() throws Exception {
77          String text = TestUtils.readContent(JSONInterceptorTest.class.getResource("json-7.txt"));
78          Object json = JSONUtil.deserialize(text);
79          assertNotNull(json);
80          assertTrue(json instanceof Map);
81          Map jsonMap = (Map) json;
82          JSONPopulator populator = new JSONPopulator();
83          WrapperClassBean bean = new WrapperClassBean();
84          populator.populateObject(bean, jsonMap);
85          assertEquals(Boolean.TRUE, bean.getBooleanField());
86          assertEquals(true, bean.isPrimitiveBooleanField1());
87          assertEquals(false, bean.isPrimitiveBooleanField2());
88          assertEquals(false, bean.isPrimitiveBooleanField3());
89          assertEquals("test\u000E\u000f", bean.getStringField());
90          assertEquals(new Integer(10), bean.getIntField());
91          assertEquals(0, bean.getNullIntField());
92          assertEquals(new Character('s'), bean.getCharField());
93          assertEquals(10.1d, bean.getDoubleField());
94          assertEquals(new Byte((byte) 3), bean.getByteField());
95  
96          assertEquals(2, bean.getListField().size());
97          assertEquals("1", bean.getListField().get(0).getValue());
98          assertEquals("2", bean.getListField().get(1).getValue());
99  
100         assertEquals(1, bean.getListMapField().size());
101         assertEquals(2, bean.getListMapField().get(0).size());
102         assertEquals(new Long(2073501), bean.getListMapField().get(0).get("id1"));
103         assertEquals(new Long(3), bean.getListMapField().get(0).get("id2"));
104 
105         assertEquals(2, bean.getMapListField().size());
106         assertEquals(3, bean.getMapListField().get("id1").size());
107         assertEquals(new Long(2), bean.getMapListField().get("id1").get(1));
108         assertEquals(4, bean.getMapListField().get("id2").size());
109         assertEquals(new Long(3), bean.getMapListField().get("id2").get(1));
110 
111         assertEquals(1, bean.getArrayMapField().length);
112         assertEquals(2, bean.getArrayMapField()[0].size());
113         assertEquals(new Long(2073501), bean.getArrayMapField()[0].get("id1"));
114         assertEquals(new Long(3), bean.getArrayMapField()[0].get("id2"));
115 
116         assertEquals(3, bean.getSetField().size());
117         assertEquals(true, bean.getSetField().contains("A"));
118         assertEquals(true, bean.getSetField().contains("B"));
119         assertEquals(true, bean.getSetField().contains("C"));
120 
121         assertEquals(3, bean.getSortedSetField().size());
122         assertEquals("A", bean.getSortedSetField().first());
123         assertEquals(true, bean.getSortedSetField().contains("B"));
124         assertEquals("C", bean.getSortedSetField().last());
125 
126         assertEquals(3, bean.getNavigableSetField().size());
127         assertEquals("A", bean.getNavigableSetField().first());
128         assertEquals(true, bean.getNavigableSetField().contains("B"));
129         assertEquals("C", bean.getNavigableSetField().last());
130 
131         assertEquals(3, bean.getQueueField().size());
132         assertEquals("A", bean.getQueueField().poll());
133         assertEquals("B", bean.getQueueField().poll());
134         assertEquals("C", bean.getQueueField().poll());
135 
136         assertEquals(3, bean.getDequeField().size());
137         assertEquals("A", bean.getDequeField().pollFirst());
138         assertEquals("B", bean.getDequeField().pollFirst());
139         assertEquals("C", bean.getDequeField().pollFirst());
140     }
