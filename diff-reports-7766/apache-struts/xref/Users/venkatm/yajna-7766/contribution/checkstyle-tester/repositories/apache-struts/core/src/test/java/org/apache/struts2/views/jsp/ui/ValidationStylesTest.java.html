19  package org.apache.struts2.views.jsp.ui;
20  
21  import org.apache.struts2.TestAction;
22  import org.apache.struts2.views.jsp.AbstractUITagTest;
23  
24  public class ValidationStylesTest extends AbstractUITagTest {
25      private TextFieldTag tag;
26  
27      public void testNormalStyle() throws Exception {
28          tag.setCssStyle("style");
29          tag.doStartTag();
30          tag.doEndTag();
31  
32          verify(ValidationStylesTest.class.getResource("validationstyles-1.txt"));
33      }
34  
35      public void testErrorStyle() throws Exception {
36          tag.setCssErrorStyle("errstyle");
37          tag.doStartTag();
38          tag.doEndTag();
39  
40          verify(ValidationStylesTest.class.getResource("validationstyles-2.txt"));
41      }
42  
43      public void testErrorClass() throws Exception {
44          tag.setCssErrorClass("errclass");
45          tag.doStartTag();
46          tag.doEndTag();
47  
48          verify(ValidationStylesTest.class.getResource("validationstyles-3.txt"));
49      }
50  
51      public void testStyleAndErrorStyle() throws Exception {
52          tag.setCssStyle("style");
53          tag.setCssErrorStyle("errstyle");
54          tag.doStartTag();
55          tag.doEndTag();
56  
57          verify(ValidationStylesTest.class.getResource("validationstyles-2.txt"));
58      }
59  
60       public void testStyleAndErrorClass() throws Exception {
61          tag.setCssStyle("style");
62          tag.setCssErrorClass("errclass");
63          tag.doStartTag();
64          tag.doEndTag();
65  
66          verify(ValidationStylesTest.class.getResource("validationstyles-3.txt"));
67      }
68  
69      @Override
70      protected void setUp() throws Exception {
71          super.setUp();
72          TestAction testAction = (TestAction) action;
73          tag = new TextFieldTag();
74          tag.setPageContext(pageContext);
75          tag.setId("myId");
76          tag.setLabel("mylabel");
77          tag.setName("foo");
78          tag.setValue("bar");
79          tag.setTitle("mytitle");
80  
81          testAction.addFieldError("foo", "bar error message");
82      }
83  }
