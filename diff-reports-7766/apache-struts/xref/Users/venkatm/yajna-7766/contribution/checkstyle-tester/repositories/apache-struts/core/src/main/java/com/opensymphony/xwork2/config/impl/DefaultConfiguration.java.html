117     public void addPackageConfig(String name, PackageConfig packageContext) {
118         PackageConfig check = packageContexts.get(name);
119         if (check != null) {
120             if (check.getLocation() != null && packageContext.getLocation() != null
121                     && check.getLocation().equals(packageContext.getLocation())) {
122                 LOG.debug("The package name '{}' is already been loaded by the same location and could be removed: {}",
123                         name, packageContext.getLocation());
124             } else {
125                 throw new ConfigurationException("The package name '" + name
126                         + "' at location "+packageContext.getLocation()
127                         + " is already been used by another package at location " + check.getLocation(),
128                         packageContext);
129             }
130         }
131         packageContexts.put(name, packageContext);
132     }

150     /**
151      * Calls the ConfigurationProviderFactory.getConfig() to tell it to reload the configuration and then calls
152      * buildRuntimeConfiguration().
153      *
154      * @return list of package providers
155      *
156      * @throws ConfigurationException in case of any configuration errors
157      */
158     public synchronized List<PackageProvider> reloadContainer(List<ContainerProvider> providers) throws ConfigurationException {
159         packageContexts.clear();
160         loadedFileNames.clear();
161         List<PackageProvider> packageProviders = new ArrayList<>();
162 
163         ContainerProperties props = new ContainerProperties();
164         ContainerBuilder builder = new ContainerBuilder();
165         Container bootstrap = createBootstrapContainer(providers);
166         for (final ContainerProvider containerProvider : providers)
167         {
168             bootstrap.inject(containerProvider);
169             containerProvider.init(this);
170             containerProvider.register(builder, props);
171         }
172         props.setConstants(builder);
173 
174         builder.factory(Configuration.class, new Factory<Configuration>() {
175             public Configuration create(Context context) throws Exception {
176                 return DefaultConfiguration.this;
177             }
178 
179             @Override
180             public Class<? extends Configuration> type() {
181                 return DefaultConfiguration.this.getClass();
182             }
183         });
184 
185         ActionContext oldContext = ActionContext.getContext();
186         try {
187             // Set the bootstrap container for the purposes of factory creation
188 
189             setContext(bootstrap);
190             container = builder.create(false);
191             setContext(container);
192             objectFactory = container.getInstance(ObjectFactory.class);
193 
194             // Process the configuration providers first
195             for (final ContainerProvider containerProvider : providers)
196             {
197                 if (containerProvider instanceof PackageProvider) {
198                     container.inject(containerProvider);
199                     ((PackageProvider)containerProvider).loadPackages();
200                     packageProviders.add((PackageProvider)containerProvider);
201                 }
202             }
203 
204             // Then process any package providers from the plugins
205             Set<String> packageProviderNames = container.getInstanceNames(PackageProvider.class);
206             for (String name : packageProviderNames) {
207                 PackageProvider provider = container.getInstance(PackageProvider.class, name);
208                 provider.init(this);
209                 provider.loadPackages();
210                 packageProviders.add(provider);
211             }
212 
213             rebuildRuntimeConfiguration();
214         } finally {
215             if (oldContext == null) {
216                 ActionContext.setContext(null);
217             }
218         }
219         return packageProviders;
220     }

222     protected ActionContext setContext(Container cont) {
223         ActionContext context = ActionContext.getContext();
224         if (context == null) {
225             ValueStack vs = cont.getInstance(ValueStackFactory.class).createValueStack();
226             context = new ActionContext(vs.getContext());
227             ActionContext.setContext(context);
228         }
229         return context;
230     }

297     /**
298      * <p>
299      * This builds the internal runtime configuration used by Xwork for finding and configuring Actions from the
300      * programmatic configuration data structures. All of the old runtime configuration will be discarded and rebuilt.
301      * </p>
302      *
303      * <p>
304      * It basically flattens the data structures to make the information easier to access.  It will take
305      * an {@link ActionConfig} and combine its data with all inherited dast.  For example, if the {@link ActionConfig}
306      * is in a package that contains a global result and it also contains a result, the resulting {@link ActionConfig}
307      * will have two results.
308      * </p>
309      *
310      * @return runtime configuration
311      * @throws ConfigurationException in case of any configuration errors
312      */
313     protected synchronized RuntimeConfiguration buildRuntimeConfiguration() throws ConfigurationException {
314         Map<String, Map<String, ActionConfig>> namespaceActionConfigs = new LinkedHashMap<>();
315         Map<String, String> namespaceConfigs = new LinkedHashMap<>();
316 
317         for (PackageConfig packageConfig : packageContexts.values()) {
318 
319             if (!packageConfig.isAbstract()) {
320                 String namespace = packageConfig.getNamespace();
321                 Map<String, ActionConfig> configs = namespaceActionConfigs.get(namespace);
322 
323                 if (configs == null) {
324                     configs = new LinkedHashMap<>();
325                 }
326 
327                 Map<String, ActionConfig> actionConfigs = packageConfig.getAllActionConfigs();
328 
329                 for (Object o : actionConfigs.keySet()) {
330                     String actionName = (String) o;
331                     ActionConfig baseConfig = actionConfigs.get(actionName);
332                     configs.put(actionName, buildFullActionConfig(packageConfig, baseConfig));
333                 }
334 
335                 namespaceActionConfigs.put(namespace, configs);
336                 if (packageConfig.getFullDefaultActionRef() != null) {
337                     namespaceConfigs.put(namespace, packageConfig.getFullDefaultActionRef());
338                 }
339             }
340         }
341 
342         PatternMatcher<int[]> matcher = container.getInstance(PatternMatcher.class);
343         return new RuntimeConfigurationImpl(Collections.unmodifiableMap(namespaceActionConfigs),
344                 Collections.unmodifiableMap(namespaceConfigs), matcher);
345     }

347     private void setDefaultResults(Map<String, ResultConfig> results, PackageConfig packageContext) {
348         String defaultResult = packageContext.getFullDefaultResultType();
349 
350         for (Map.Entry<String, ResultConfig> entry : results.entrySet()) {
351 
352             if (entry.getValue() == null) {
353                 ResultTypeConfig resultTypeConfig = packageContext.getAllResultTypeConfigs().get(defaultResult);
354                 entry.setValue(new ResultConfig.Builder(null, resultTypeConfig.getClassName()).build());
355             }
356         }
357     }

359     /**
360      * Builds the full runtime actionconfig with all of the defaults and inheritance
361      *
362      * @param packageContext the PackageConfig which holds the base config we're building from
363      * @param baseConfig     the ActionConfig which holds only the configuration specific to itself, without the defaults
364      *                       and inheritance
365      * @return a full ActionConfig for runtime configuration with all of the inherited and default params
366      * @throws com.opensymphony.xwork2.config.ConfigurationException
367      *
368      */
369     private ActionConfig buildFullActionConfig(PackageConfig packageContext, ActionConfig baseConfig) throws ConfigurationException {
370         Map<String, String> params = new TreeMap<>(baseConfig.getParams());
371         Map<String, ResultConfig> results = new TreeMap<>();
372 
373         if (!baseConfig.getPackageName().equals(packageContext.getName()) && packageContexts.containsKey(baseConfig.getPackageName())) {
374             results.putAll(packageContexts.get(baseConfig.getPackageName()).getAllGlobalResults());
375         } else {
376             results.putAll(packageContext.getAllGlobalResults());
377         }
378 
379        	results.putAll(baseConfig.getResults());
380 
381         setDefaultResults(results, packageContext);
382 
383         List<InterceptorMapping> interceptors = new ArrayList<>(baseConfig.getInterceptors());
384 
385         if (interceptors.size() <= 0) {
386             String defaultInterceptorRefName = packageContext.getFullDefaultInterceptorRef();
387 
388             if (defaultInterceptorRefName != null) {
389                 interceptors.addAll(InterceptorBuilder.constructInterceptorReference(new PackageConfig.Builder(packageContext), defaultInterceptorRefName,
390                         new LinkedHashMap<String, String>(), packageContext.getLocation(), objectFactory));
391             }
392         }
393 
394         String methodRegex = container.getInstance(String.class, StrutsConstants.STRUTS_SMI_METHOD_REGEX);
395         if (methodRegex == null) {
396             methodRegex = ActionConfig.DEFAULT_METHOD_REGEX;
397         }
398 
399         LOG.debug("Using pattern [{}] to match allowed methods when SMI is disabled!", methodRegex);
400 
401         return new ActionConfig.Builder(baseConfig)
402             .addParams(params)
403             .addResultConfigs(results)
404             .defaultClassName(packageContext.getDefaultClassRef())  // fill in default if non class has been provided
405             .interceptors(interceptors)
406             .setStrictMethodInvocation(packageContext.isStrictMethodInvocation())
407             .setDefaultMethodRegex(methodRegex)
408             .addExceptionMappings(packageContext.getAllExceptionMappingConfigs())
409             .build();
410     }

413     private static class RuntimeConfigurationImpl implements RuntimeConfiguration {
414 
415         private Map<String, Map<String, ActionConfig>> namespaceActionConfigs;
416         private Map<String, ActionConfigMatcher> namespaceActionConfigMatchers;
417         private NamespaceMatcher namespaceMatcher;
418         private Map<String, String> namespaceConfigs;
419 
420         public RuntimeConfigurationImpl(Map<String, Map<String, ActionConfig>> namespaceActionConfigs,
421                                         Map<String, String> namespaceConfigs,
422                                         PatternMatcher<int[]> matcher) {
423             this.namespaceActionConfigs = namespaceActionConfigs;
424             this.namespaceConfigs = namespaceConfigs;
425 
426             this.namespaceActionConfigMatchers = new LinkedHashMap<>();
427             this.namespaceMatcher = new NamespaceMatcher(matcher, namespaceActionConfigs.keySet());
428 
429             for (Map.Entry<String, Map<String, ActionConfig>> entry : namespaceActionConfigs.entrySet()) {
430                 namespaceActionConfigMatchers.put(entry.getKey(), new ActionConfigMatcher(matcher, entry.getValue(), true));
431             }
432         }
433 
434 
435         /**
436          * Gets the configuration information for an action name, or returns null if the
437          * name is not recognized.
438          *
439          * @param name      the name of the action
440          * @param namespace the namespace for the action or null for the empty namespace, ""
441          * @return the configuration information for action requested
442          */
443         public ActionConfig getActionConfig(String namespace, String name) {
444             ActionConfig config = findActionConfigInNamespace(namespace, name);
445 
446             // try wildcarded namespaces
447             if (config == null) {
448                 NamespaceMatch match = namespaceMatcher.match(namespace);
449                 if (match != null) {
450                     config = findActionConfigInNamespace(match.getPattern(), name);
451 
452                     // If config found, place all the matches found in the namespace processing in the action's parameters
453                     if (config != null) {
454                         config = new ActionConfig.Builder(config)
455                                 .addParams(match.getVariables())
456                                 .build();
457                     }
458                 }
459             }
460 
461             // fail over to empty namespace
462             if (config == null && StringUtils.isNotBlank(namespace)) {
463                 config = findActionConfigInNamespace("", name);
464             }
465 
466 
467             return config;
468         }
469 
470         private ActionConfig findActionConfigInNamespace(String namespace, String name) {
471             ActionConfig config = null;
472             if (namespace == null) {
473                 namespace = "";
474             }
475             Map<String, ActionConfig> actions = namespaceActionConfigs.get(namespace);
476             if (actions != null) {
477                 config = actions.get(name);
478                 // Check wildcards
479                 if (config == null) {
480                     config = namespaceActionConfigMatchers.get(namespace).match(name);
481                     // fail over to default action
482                     if (config == null) {
483                         String defaultActionRef = namespaceConfigs.get(namespace);
484                         if (defaultActionRef != null) {
485                             config = actions.get(defaultActionRef);
486                         }
487                     }
488                 }
489             }
490             return config;
491         }
492 
493         /**
494          * Gets the configuration settings for every action.
495          *
496          * @return a Map of namespace - > Map of ActionConfig objects, with the key being the action name
497          */
498         public Map<String, Map<String, ActionConfig>>  getActionConfigs() {
499             return namespaceActionConfigs;
500         }
501 
502         @Override
503         public String toString() {
504             StringBuilder buff = new StringBuilder("RuntimeConfiguration - actions are\n");
505 
506             for (Map.Entry<String, Map<String, ActionConfig>> entry : namespaceActionConfigs.entrySet()) {
507                 Map<String, ActionConfig> actionConfigs = entry.getValue();
508 
509                 for (String s : actionConfigs.keySet()) {
510                     buff.append(entry.getKey()).append("/").append(s).append("\n");
511                 }
512             }
513 
514             return buff.toString();
515         }
516     }

518     class ContainerProperties extends LocatableProperties {
519         private static final long serialVersionUID = -7320625750836896089L;
520 
521         @Override
522         public Object setProperty(String key, String value) {
523             String oldValue = getProperty(key);
524             if (LOG.isInfoEnabled() && oldValue != null && !oldValue.equals(value) && !defaultFrameworkBeanName.equals(oldValue)) {
525                 LOG.info("Overriding property {} - old value: {} new value: {}", key, oldValue, value);
526             }
527             return super.setProperty(key, value);
528         }

530         public void setConstants(ContainerBuilder builder) {
531             for (Object keyobj : keySet()) {
532                 String key = (String)keyobj;
533                 builder.factory(String.class, key, new LocatableConstantFactory<>(getProperty(key), getPropertyLocation(key)));
534             }
535         }
