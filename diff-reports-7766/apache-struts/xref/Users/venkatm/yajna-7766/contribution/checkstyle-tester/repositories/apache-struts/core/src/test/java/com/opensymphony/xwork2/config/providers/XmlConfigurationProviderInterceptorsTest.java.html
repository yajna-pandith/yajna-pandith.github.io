57      public void testBasicInterceptors() throws ConfigurationException {
58          final String filename = "com/opensymphony/xwork2/config/providers/xwork-test-interceptors-basic.xml";
59          ConfigurationProvider provider = buildConfigurationProvider(filename);
60  
61          // setup expectations
62          // the test interceptor with a parameter
63          Map<String, String> params = new HashMap<>();
64          params.put("foo", "expectedFoo");
65  
66          InterceptorConfig paramsInterceptor = new InterceptorConfig.Builder("test", MockInterceptor.class.getName())
67              .addParams(params).build();
68  
69          // the default interceptor stack
70          InterceptorStackConfig defaultStack = new InterceptorStackConfig.Builder("defaultStack")
71                  .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
72                  .addInterceptor(new InterceptorMapping("test", objectFactory.buildInterceptor(mockInterceptor, params)))
73                  .build();
74  
75          // the derivative interceptor stack
76          InterceptorStackConfig derivativeStack = new InterceptorStackConfig.Builder("derivativeStack")
77                  .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
78                  .addInterceptor(new InterceptorMapping("test", objectFactory.buildInterceptor(mockInterceptor, params)))
79                  .addInterceptor(new InterceptorMapping("logging", objectFactory.buildInterceptor(loggingInterceptor, new HashMap<String, String>())))
80                  .build();
81  
82          // execute the configuration
83          provider.init(configuration);
84          provider.loadPackages();
85  
86          PackageConfig pkg = configuration.getPackageConfig("default");
87          Map interceptorConfigs = pkg.getInterceptorConfigs();
88  
89          // assertions for size
90          assertEquals(5, interceptorConfigs.size());
91  
92          // assertions for interceptors
93          assertEquals(noopInterceptor, interceptorConfigs.get("noop"));
94          assertEquals(loggingInterceptor, interceptorConfigs.get("logging"));
95          assertEquals(paramsInterceptor, interceptorConfigs.get("test"));
96  
97          // assertions for interceptor stacks
98          assertEquals(defaultStack, interceptorConfigs.get("defaultStack"));
99          assertEquals(derivativeStack, interceptorConfigs.get("derivativeStack"));
100     }

102     public void testInterceptorDefaultRefs() throws ConfigurationException {
103         XmlConfigurationProvider provider = new XmlConfigurationProvider("com/opensymphony/xwork2/config/providers/xwork-test-interceptor-defaultref.xml");
104         container.inject(provider);
105         loadConfigurationProviders(provider);
106 
107         // expectations - the inherited interceptor stack
108         // default package
109         ArrayList<InterceptorMapping> interceptors = new ArrayList<>();
110         interceptors.add(new InterceptorMapping("logging", objectFactory.buildInterceptor(loggingInterceptor, new HashMap<String, String>())));
111 
112         ActionConfig actionWithOwnRef = new ActionConfig.Builder("", "ActionWithOwnRef", SimpleAction.class.getName())
113             .addInterceptors(interceptors)
114             .build();
115 
116         ActionConfig actionWithDefaultRef = new ActionConfig.Builder("", "ActionWithDefaultRef", SimpleAction.class.getName())
117             .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
118             .build();
119 
120         // sub package
121         // this should inherit
122         ActionConfig actionWithNoRef = new ActionConfig.Builder("", "ActionWithNoRef", SimpleAction.class.getName())
123             .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
124             .build();
125 
126         interceptors = new ArrayList<>();
127         interceptors.add(new InterceptorMapping("logging", objectFactory.buildInterceptor(loggingInterceptor, new HashMap<String, String>())));
128 
129         ActionConfig anotherActionWithOwnRef = new ActionConfig.Builder("", "AnotherActionWithOwnRef", SimpleAction.class.getName())
130             .addInterceptor(new InterceptorMapping("logging", objectFactory.buildInterceptor(loggingInterceptor, new HashMap<String, String>())))
131             .build();
132 
133         RuntimeConfiguration runtimeConfig = configurationManager.getConfiguration().getRuntimeConfiguration();
134 
135         // assertions
136         assertEquals(actionWithOwnRef, runtimeConfig.getActionConfig("", "ActionWithOwnRef"));
137         assertEquals(actionWithDefaultRef, runtimeConfig.getActionConfig("", "ActionWithDefaultRef"));
138 
139         assertEquals(actionWithNoRef, runtimeConfig.getActionConfig("", "ActionWithNoRef"));
140         assertEquals(anotherActionWithOwnRef, runtimeConfig.getActionConfig("", "AnotherActionWithOwnRef"));
141     }

143     public void testInterceptorInheritance() throws ConfigurationException {
144         
145         // expectations - the inherited interceptor stack
146         InterceptorStackConfig inheritedStack = new InterceptorStackConfig.Builder("subDefaultStack")
147             .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
148             .build();
149 
150         ConfigurationProvider provider = buildConfigurationProvider("com/opensymphony/xwork2/config/providers/xwork-test-interceptor-inheritance.xml");
151 
152         // assertions
153         PackageConfig defaultPkg = configuration.getPackageConfig("default");
154         assertEquals(2, defaultPkg.getInterceptorConfigs().size());
155 
156         PackageConfig subPkg = configuration.getPackageConfig("subPackage");
157         assertEquals(1, subPkg.getInterceptorConfigs().size());
158         assertEquals(3, subPkg.getAllInterceptorConfigs().size());
159         assertEquals(inheritedStack, subPkg.getInterceptorConfigs().get("subDefaultStack"));
160 
161         // expectations - the inherited interceptor stack
162         inheritedStack = new InterceptorStackConfig.Builder("subSubDefaultStack")
163                 .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
164                 .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
165                 .build();
166 
167         PackageConfig subSubPkg = configuration.getPackageConfig("subSubPackage");
168         assertEquals(1, subSubPkg.getInterceptorConfigs().size());
169         assertEquals(4, subSubPkg.getAllInterceptorConfigs().size());
170         assertEquals(inheritedStack, subSubPkg.getInterceptorConfigs().get("subSubDefaultStack"));
171     }

174     public void testInterceptorParamOverriding() throws Exception {
175 
176         Map<String, String> params = new HashMap<>();
177         params.put("foo", "expectedFoo");
178         params.put("expectedFoo", "expectedFooValue");
179 
180         InterceptorStackConfig defaultStack = new InterceptorStackConfig.Builder("defaultStack")
181             .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
182             .addInterceptor(new InterceptorMapping("test", objectFactory.buildInterceptor(mockInterceptor, params)))
183             .build();
184 
185         ArrayList<InterceptorMapping> interceptors = new ArrayList<>();
186         interceptors.addAll(defaultStack.getInterceptors());
187 
188         ActionConfig intAction = new ActionConfig.Builder("", "TestInterceptorParam", SimpleAction.class.getName())
189             .addInterceptors(interceptors)
190             .build();
191 
192         // TestInterceptorParamOverride action tests that an interceptor with a param override worked
193         HashMap<String, String> interceptorParams = new HashMap<>();
194         interceptorParams.put("expectedFoo", "expectedFooValue2");
195         interceptorParams.put("foo", "foo123");
196 
197         InterceptorStackConfig defaultStack2 = new InterceptorStackConfig.Builder("defaultStack")
198             .addInterceptor(new InterceptorMapping("noop", objectFactory.buildInterceptor(noopInterceptor, new HashMap<String, String>())))
199             .addInterceptor(new InterceptorMapping("test", objectFactory.buildInterceptor(mockInterceptor, interceptorParams)))
200             .build();
201 
202         interceptors = new ArrayList<>();
203 
204         interceptors.addAll(defaultStack2.getInterceptors());
205 
206         ActionConfig intOverAction = new ActionConfig.Builder("", "TestInterceptorParamOverride", SimpleAction.class.getName())
207             .addInterceptors(interceptors)
208             .build();
209 
210         ConfigurationProvider provider = buildConfigurationProvider("com/opensymphony/xwork2/config/providers/xwork-test-interceptor-params.xml");
211 
212 
213         PackageConfig pkg = configuration.getPackageConfig("default");
214         Map actionConfigs = pkg.getActionConfigs();
215 
216         // assertions
217         assertEquals(2, actionConfigs.size());
218         assertEquals(intAction, actionConfigs.get("TestInterceptorParam"));
219         assertEquals(intOverAction, actionConfigs.get("TestInterceptorParamOverride"));
220 
221         ActionConfig ac = (ActionConfig) actionConfigs.get("TestInterceptorParamOverride");
222         assertEquals(defaultStack.getInterceptors(), ac.getInterceptors());
223 
224         ActionConfig ac2 = (ActionConfig) actionConfigs.get("TestInterceptorParam");
225         assertEquals(defaultStack2.getInterceptors(), ac2.getInterceptors());
226 
227     }
