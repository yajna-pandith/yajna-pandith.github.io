13  import org.testng.annotations.AfterTest;
14  import org.testng.annotations.BeforeTest;
15  import org.testng.annotations.Test;
16  
17  /**
18   * CacheCreationTest.
19   * 
20   * @author Sanne Grinovero
21   * @since 4.0
22   */
23  @Test(groups = "functional", testName = "lucenedemo.CacheConfigurationTest")
24  public class CacheConfigurationTest {
25     
26     private EmbeddedCacheManager cacheManager1;
27     private EmbeddedCacheManager cacheManager2;
28     private Directory directoryNodeOne;
29     private Directory directoryNodeTwo;
30     private Cache cache1;
31     private Cache cache2;
32  
33     @BeforeTest
34     public void init() throws IOException {
35        cacheManager1 = TestCacheManagerFactory.fromXml("config-samples/lucene-demo-cache-config.xml");
36        cacheManager1.start();
37        cache1 = cacheManager1.getCache();
38        cache1.clear();
39        directoryNodeOne = DirectoryBuilder.newDirectoryInstance(cache1, cache1, cache1, "index-name").create();
40        cacheManager2 = TestCacheManagerFactory.fromXml("config-samples/lucene-demo-cache-config.xml");
41        cacheManager2.start();
42        cache2 = cacheManager2.getCache();
43        cache2.clear();
44        directoryNodeTwo = DirectoryBuilder.newDirectoryInstance(cache2, cache2, cache2, "index-name").create();
45     }
46     
47     @AfterTest
48     public void cleanup() throws IOException {
49        directoryNodeOne.close();
50        directoryNodeTwo.close();
51        cacheManager1.stop();
52        cacheManager2.stop();
53     }
54  
55     @Test
56     public void inserting() throws IOException, ParseException {
57        DemoActions node1 = new DemoActions(directoryNodeOne, cache1);
58        DemoActions node2 = new DemoActions(directoryNodeTwo, cache2);
59        node1.addNewDocument("hello?");
60        assert node1.listAllDocuments().size() == 1;
61        node1.addNewDocument("anybody there?");
62        assert node2.listAllDocuments().size() == 2;
63        Query query = node1.parseQuery("hello world");
64        List<String> valuesMatchingQuery = node2.listStoredValuesMatchingQuery(query);
65        assert valuesMatchingQuery.size() == 1;
66        assert valuesMatchingQuery.get(0).equals("hello?");
67     }
