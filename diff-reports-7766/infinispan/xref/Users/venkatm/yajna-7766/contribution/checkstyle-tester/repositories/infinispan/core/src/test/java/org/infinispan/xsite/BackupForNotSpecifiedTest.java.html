3   import org.infinispan.configuration.cache.BackupConfiguration;
4   import org.infinispan.configuration.cache.BackupFailurePolicy;
5   import org.infinispan.configuration.cache.CacheMode;
6   import org.infinispan.configuration.cache.ConfigurationBuilder;
7   import org.infinispan.configuration.global.GlobalConfigurationBuilder;
8   import org.testng.annotations.Test;
9   
10  import static org.testng.AssertJUnit.assertEquals;
11  import static org.testng.AssertJUnit.assertNull;
12  
13  /**
14   * @author Mircea Markus
15   * @since 5.2

17  @Test(groups = "xsite", testName = "xsite.BackupForNotSpecifiedTest")
18  public class BackupForNotSpecifiedTest extends AbstractXSiteTest {
19  
20     @Override
21     protected void createSites() {
22  
23        GlobalConfigurationBuilder lonGc = GlobalConfigurationBuilder.defaultClusteredBuilder();
24        lonGc.site().localSite("LON");
25        ConfigurationBuilder lonDefault = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
26        lonDefault.sites().addBackup()
27              .site("NYC")
28              .backupFailurePolicy(BackupFailurePolicy.FAIL)
29              .strategy(BackupConfiguration.BackupStrategy.SYNC)
30              .sites().addInUseBackupSite("NYC");
31        ConfigurationBuilder someCache = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
32  
33        GlobalConfigurationBuilder nycGc = GlobalConfigurationBuilder.defaultClusteredBuilder();
34        nycGc.site().localSite("NYC");
35        ConfigurationBuilder nycDefault = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
36        nycDefault.sites().addBackup()
37              .site("LON")
38              .strategy(BackupConfiguration.BackupStrategy.SYNC)
39              .sites().addInUseBackupSite("LON");
40        ConfigurationBuilder someCacheBackup = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
41        someCacheBackup.sites().backupFor().remoteCache("someCache").remoteSite("LON");
42        someCacheBackup.sites().disableBackups(true);
43  
44        createSite("LON", 2, lonGc, lonDefault);
45        createSite("NYC", 2, nycGc, nycDefault);
46  
47        startCache("LON", "backup", getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true));
48        startCache("NYC", "backup", getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, true));
49  
50        startCache("LON", "someCache", someCache);
51        startCache("NYC", "someCacheBackup", someCacheBackup);
52     }
53  
54     public void testDataGetsReplicated() {
55        cache("LON", 0).put("k_default_lon", "v_default_lon");
56        assertEquals("v_default_lon", cache("LON", 1).get("k_default_lon"));
57        assertEquals("v_default_lon", cache("NYC", 0).get("k_default_lon"));
58        assertEquals("v_default_lon", cache("NYC", 1).get("k_default_lon"));
59  
60        cache("NYC", 0).put("k_default_nyc", "v_default_nyc");
61        assertEquals("v_default_nyc", cache("NYC", 1).get("k_default_nyc"));
62        assertEquals("v_default_nyc", cache("LON", 0).get("k_default_nyc"));
63        assertEquals("v_default_nyc", cache("LON", 1).get("k_default_nyc"));
64  
65        cache("LON", "backup", 0).put("k_backup_lon", "v_backup_lon");
66        assertEquals("v_backup_lon", cache("LON", "backup", 1).get("k_backup_lon"));
67        assertEquals("v_backup_lon", cache("NYC", "backup", 0).get("k_backup_lon"));
68        assertEquals("v_backup_lon", cache("NYC", "backup", 1).get("k_backup_lon"));
69  
70        cache("NYC", "backup", 0).put("k_backup_nyc", "v_backup_nyc");
71        assertEquals("v_backup_nyc", cache("NYC", "backup", 1).get("k_backup_nyc"));
72        assertEquals("v_backup_nyc", cache("LON", "backup", 0).get("k_backup_nyc"));
73        assertEquals("v_backup_nyc", cache("LON", "backup", 1).get("k_backup_nyc"));
74  
75        cache("LON", "someCache", 0).put("k_someCache_lon", "v_someCache_lon");
76        assertEquals("v_someCache_lon", cache("LON", "someCache", 1).get("k_someCache_lon"));
77        assertEquals("v_someCache_lon", cache("NYC", "someCacheBackup", 0).get("k_someCache_lon"));
78        assertEquals("v_someCache_lon", cache("NYC", "someCacheBackup", 1).get("k_someCache_lon"));
79  
80        cache("NYC", "someCacheBackup", 0).put("k_lon_sb", "v_lon_sb");
81        assertEquals("v_lon_sb", cache("NYC", "someCacheBackup", 1).get("k_lon_sb"));
82        assertNull(cache("LON", "someCacheBackup", 0).get("k_lon_sb"));
83        assertNull(cache("LON", "someCacheBackup", 1).get("k_lon_sb"));
84        assertNull(cache("LON", "cacheBackup", 0).get("k_lon_sb"));
85        assertNull(cache("LON", "cacheBackup", 1).get("k_lon_sb"));
86     }
