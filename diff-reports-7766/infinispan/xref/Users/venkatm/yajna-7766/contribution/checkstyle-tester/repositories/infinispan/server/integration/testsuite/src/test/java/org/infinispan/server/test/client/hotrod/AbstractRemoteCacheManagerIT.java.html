215     private void doTestDefaultLoadBalanding() throws Exception {
216         InetSocketAddress hostport0 = new InetSocketAddress(getServers().get(0).getHotrodEndpoint().getInetAddress().getHostName(), getServers().get(0)
217                 .getHotrodEndpoint().getPort());
218         InetSocketAddress hostport1 = new InetSocketAddress(getServers().get(1).getHotrodEndpoint().getInetAddress().getHostName(), getServers().get(1)
219                 .getHotrodEndpoint().getPort());
220 
221         TcpTransport tt = null;
222         InetSocketAddress sock_addr = null;
223 
224         StringBuilder serverAddrSequence = new StringBuilder();
225         String hostport0String = hostport0.getAddress().getHostAddress() + ":" + hostport0.getPort();
226         String hostport1String = hostport1.getAddress().getHostAddress() + ":" + hostport1.getPort();
227         String expectedSequence1 = hostport0String + " " + hostport1String + " " + hostport0String;
228         String expectedSequence2 = hostport1String + " " + hostport0String + " " + hostport1String;
229         String expectedSequenceLocalMode = hostport0String + " " + hostport0String + " " + hostport0String;
230 
231         Configuration cfg = createRemoteCacheManagerConfigurationBuilder().build();
232 
233         RemoteCacheManager rcm = new RemoteCacheManager(cfg);
234         RemoteCache rc = rcm.getCache(TEST_CACHE_NAME);
235         RemoteCacheImpl rci = (RemoteCacheImpl) rc;
236 
237         // the factory used to create all remote operations for this class
238         OperationsFactory of = getOperationsFactoryField(rci);
239         TcpTransportFactory ttf = (TcpTransportFactory) getTransportFactoryField(of);
240 
241         // perform first simulated operation
242         tt = (TcpTransport) ttf.getTransport(null, rci.getName().getBytes());
243         sock_addr = (InetSocketAddress) tt.getServerAddress();
244         ttf.releaseTransport(tt);
245         serverAddrSequence.append(sock_addr.getAddress().getHostAddress() + ":" + sock_addr.getPort()).append(" ");
246 
247         tt = (TcpTransport) ttf.getTransport(null, rci.getName().getBytes());
248         sock_addr = (InetSocketAddress) tt.getServerAddress();
249         ttf.releaseTransport(tt);
250         serverAddrSequence.append(sock_addr.getAddress().getHostAddress() + ":" + sock_addr.getPort()).append(" ");
251 
252         tt = (TcpTransport) ttf.getTransport(null, rci.getName().getBytes());
253         sock_addr = (InetSocketAddress) tt.getServerAddress();
254         ttf.releaseTransport(tt);
255         serverAddrSequence.append(sock_addr.getAddress().getHostAddress() + ":" + sock_addr.getPort());
256 
257         if (!isLocalMode()) {
258             assertTrue(
259                     "loadbalancing server sequence expected either " + expectedSequence1 + " or " + expectedSequence2
260                             + ", actual sequence: " + serverAddrSequence.toString(),
261                     serverAddrSequence.toString().equals(expectedSequence1)
262                             || serverAddrSequence.toString().equals(expectedSequence2));
263         } else {
264             assertEquals("LOCAL mode - loadbalancing server sequence expected " + expectedSequenceLocalMode
265                     + ", actual sequence: " + serverAddrSequence.toString(),
266                     serverAddrSequence.toString(), expectedSequenceLocalMode);
267         }
268     }

290         InetSocketAddress hostport0 = new InetSocketAddress(getServers().get(0).getHotrodEndpoint().getInetAddress().getHostName(), getServers().get(0)

293                 .getHotrodEndpoint().getPort());

297 
298         // create configuration with the custom balancing strategy
299         Configuration cfg = createRemoteCacheManagerConfigurationBuilder()
300                 .balancingStrategy("org.infinispan.server.test.client.hotrod.Node0OnlyBalancingStrategy")
301                 .build();
302 
303         RemoteCacheManager rcm = new RemoteCacheManager(cfg);
304         RemoteCache rc = rcm.getCache(TEST_CACHE_NAME);
305         RemoteCacheImpl rci = (RemoteCacheImpl) rc;
306         // the factory used to create all remote operations for this class
307         OperationsFactory of = getOperationsFactoryField(rci);

331     private void assertEqualConfiguration(Configuration config, RemoteCache rc) throws Exception {
332         assertEquals(config.balancingStrategy().getName(), getRequestBalancingStrategyProperty(rc));
333 
334         // Configuration stores servers as List<ServerConfiguration>, getServerListProperty returns string "host1:port1;host2:port2..."
335         String servers = getServerListProperty(rc);
336         for (ServerConfiguration scfg : config.servers()) {
337             boolean found = false;
338             String host;
339             int port = ConfigurationProperties.DEFAULT_HOTROD_PORT;
340             Pattern patternIpv6 = Pattern.compile(IPV6_REGEX);
341             Pattern patternIpv4 = Pattern.compile(IPV4_REGEX);
342             for (String server : servers.split(";")) {
343                 Matcher matcher6 = patternIpv6.matcher(server);
344                 Matcher matcher4 = patternIpv4.matcher(server);
345                 if (matcher6.matches()) {
346                     host = matcher6.group(1);
347                     port = Integer.parseInt(matcher6.group(2));
348                 } else if (matcher4.matches()) {
349                     host = matcher4.group(1);
350                     port = Integer.parseInt(matcher4.group(2));
351                 } else {
352                     host = server;
353                 }
354                 if (scfg.host().equals(host) && scfg.port() == port)
355                     found = true;
356             }
357             if (!found)
358                 fail("The remote cache manager was configured to have server with an address " + scfg.host() + ":" + scfg.port() + ", but it doesn't.");
359         }
360 
361         assertEquals(config.forceReturnValues(), Boolean.parseBoolean(getForceReturnValueProperty(rc)));
362         assertEquals(config.tcpNoDelay(), Boolean.parseBoolean(getTcpNoDelayProperty(rc)));
363         assertEquals(config.tcpKeepAlive(), Boolean.parseBoolean(getTcpKeepAliveProperty(rc)));
364         assertEquals(config.maxRetries(), Integer.parseInt(getMaxRetries(rc)));
365 
366         // pingOnStartup and asyncExecutorFactory compared only with the configuration itself
367         assertEquals(config.pingOnStartup(), rc.getRemoteCacheManager().getConfiguration().pingOnStartup());
368         assertEquals(config.asyncExecutorFactory().factoryClass().getName(),
369                 rc.getRemoteCacheManager().getConfiguration().asyncExecutorFactory().factoryClass().getName());
370 
371         assertEquals(config.transportFactory().getName(), getTransportFactoryProperty(rc));
372         // either marshaller or marshallerClass is set
373         if (config.marshaller() != null) {
374             assertEquals(config.marshaller().getClass().getName(), getMarshallerProperty(rc));
375         } else {
376             assertEquals(config.marshallerClass().getName(), getMarshallerProperty(rc));
377         }
378 
379         // need to do some hotrod operation, only then is the hash initialized
380         rc.stats();
381         // get hash function only for distribution mode
382         if (isDistributedMode()) {
383             assertEquals(config.consistentHashImpl(3).getName(), getHashFunctionImplProperty(rc));
384         }
385 
386         assertEquals(config.keySizeEstimate(), getKeySizeEstimateProperty(rc));
387         assertEquals(config.valueSizeEstimate(), getValueSizeEstimateProperty(rc));
388     }

398     private String getServerListProperty(RemoteCache rc) throws Exception {
399 
400         RemoteCacheImpl rci = (RemoteCacheImpl) rc;
401         OperationsFactory of = getOperationsFactoryField(rci);
402         TcpTransportFactory ttf = (TcpTransportFactory) getTransportFactoryField(of);
403         Collection<SocketAddress> servers = ttf.getServers();
404         // create a list of IP address:port to return
405         StringBuffer serverList = new StringBuffer();
406         int listSize = servers.size();
407         int i = 0;
408         for (Iterator iter = servers.iterator(); iter.hasNext(); i++) {
409             InetSocketAddress addr = (InetSocketAddress) iter.next();
410             // take care to remove prepended backslash
411             String serverAddress = addr.getAddress().toString();
412 //            if (serverAddress.startsWith("/"))
413 //                serverAddress = serverAddress.substring(1);
414 //            serverList.append(serverAddress);
415             if (addr.getAddress() instanceof Inet6Address) {
416                 serverList.append('[').append(addr.getHostName()).append(']');
417             } else {
418                 serverList.append(addr.getHostName());
419             }
420             serverList.append(":");
421             serverList.append(addr.getPort());
422             if (i < listSize - 1)
423                 serverList.append(";");
424         }
425         return serverList.toString();
426     }
