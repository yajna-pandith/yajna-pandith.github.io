63  /**
64   * Factory class that builds an interceptor chain based on cache configuration.
65   *
66   * @author <a href="mailto:manik@jboss.org">Manik Surtani (manik@jboss.org)</a>
67   * @author Mircea.Markus@jboss.com
68   * @author Marko Luksa
69   * @author Pedro Ruivo
70   * @since 4.0
71   */
72  @DefaultFactoryFor(classes = InterceptorChain.class)
73  public class InterceptorChainFactory extends AbstractNamedCacheComponentFactory implements AutoInstantiableFactory {
74  
75     private static final Log log = LogFactory.getLog(InterceptorChainFactory.class);
76  
77     private CommandInterceptor createInterceptor(CommandInterceptor interceptor, Class<? extends CommandInterceptor> interceptorType) {
78        CommandInterceptor chainedInterceptor = componentRegistry.getComponent(interceptorType);
79        if (chainedInterceptor == null) {
80           chainedInterceptor = interceptor;
81           register(interceptorType, chainedInterceptor);
82        } else {
83           // wipe next/last chaining!!
84           chainedInterceptor.setNext(null);
85        }
86        return chainedInterceptor;
87     }
88  
89  
90     private void register(Class<? extends CommandInterceptor> clazz, CommandInterceptor chainedInterceptor) {
91        try {
92           componentRegistry.registerComponent(chainedInterceptor, clazz);
93        } catch (RuntimeException e) {
94           log.unableToCreateInterceptor(clazz, e);
95           throw e;
96        }
97     }

99     private boolean isUsingMarshalledValues(Configuration c) {
100       return c.storeAsBinary().enabled() && (c.storeAsBinary().storeKeysAsBinary() || c.storeAsBinary().storeValuesAsBinary());
101    }

103    public InterceptorChain buildInterceptorChain() {
104       boolean needsVersionAwareComponents = configuration.transaction().transactionMode().isTransactional() &&
105             Configurations.isVersioningEnabled(configuration);
106 
107       InterceptorChain interceptorChain = new InterceptorChain(componentRegistry.getComponentMetadataRepo());
108       // add the interceptor chain to the registry first, since some interceptors may ask for it.
109       componentRegistry.registerComponent(interceptorChain, InterceptorChain.class);
110 
111       boolean invocationBatching = configuration.invocationBatching().enabled();
112       boolean isTotalOrder = configuration.transaction().transactionProtocol().isTotalOrder();
113 
114       // load the icInterceptor first
115       if (invocationBatching) {
116          interceptorChain.appendInterceptor(createInterceptor(new BatchingInterceptor(), BatchingInterceptor.class), false);
117       }
118       interceptorChain.appendInterceptor(createInterceptor(new InvocationContextInterceptor(), InvocationContextInterceptor.class), false);
119 
120       CompatibilityModeConfiguration compatibility = configuration.compatibility();
121       if (compatibility.enabled()) {
122          Marshaller compatibilityMarshaller = compatibility.marshaller();
123          if (compatibilityMarshaller != null) {
124             componentRegistry.wireDependencies(compatibilityMarshaller);
125          }
126          interceptorChain.appendInterceptor(createInterceptor(
127                new TypeConverterInterceptor(compatibilityMarshaller), TypeConverterInterceptor.class), false);
128       }
129 
130       // add marshallable check interceptor for situations where we want to figure out before marshalling
131       // Store as binary marshalls keys/values eagerly now, so avoid extra serialization
132       if (configuration.clustering().async().asyncMarshalling()
133             || configuration.clustering().async().useReplQueue() || hasAsyncStore())
134          interceptorChain.appendInterceptor(createInterceptor(new IsMarshallableInterceptor(), IsMarshallableInterceptor.class), false);
135 
136       // load the cache management interceptor next
137       interceptorChain.appendInterceptor(createInterceptor(new CacheMgmtInterceptor(), CacheMgmtInterceptor.class), false);
138 
139       // load the state transfer lock interceptor
140       // the state transfer lock ensures that the cache member list is up-to-date
141       // so it's necessary even if state transfer is disabled
142       if (configuration.clustering().cacheMode().isDistributed() || configuration.clustering().cacheMode().isReplicated()) {
143          if (isTotalOrder) {
144             interceptorChain.appendInterceptor(createInterceptor(new TotalOrderStateTransferInterceptor(),
145                                                                  TotalOrderStateTransferInterceptor.class), false);
146          } else {
147             interceptorChain.appendInterceptor(createInterceptor(new StateTransferInterceptor(), StateTransferInterceptor.class), false);
148          }
149          if (configuration.transaction().transactionMode().isTransactional()) {
150             interceptorChain.appendInterceptor(createInterceptor(new TransactionSynchronizerInterceptor(), TransactionSynchronizerInterceptor.class), false);
151          }
152       }
153 
154       // The partition handling must run every time the state transfer interceptor retries a command (in non-transactional caches)
155       if (configuration.clustering().partitionHandling().enabled()
156             && (configuration.clustering().cacheMode().isDistributed()
157             || configuration.clustering().cacheMode().isReplicated())) {
158          interceptorChain.appendInterceptor(createInterceptor(new PartitionHandlingInterceptor(), PartitionHandlingInterceptor.class), false);
159       }
160 
161 
162       //load total order interceptor
163       if (isTotalOrder) {
164          interceptorChain.appendInterceptor(createInterceptor(new TotalOrderInterceptor(), TotalOrderInterceptor.class), false);
165       }
166 
167       // load the tx interceptor
168       if (configuration.transaction().transactionMode().isTransactional())
169          interceptorChain.appendInterceptor(createInterceptor(new TxInterceptor(), TxInterceptor.class), false);
170 
171       if (isUsingMarshalledValues(configuration)) {
172          CommandInterceptor interceptor;
173             interceptor = createInterceptor(
174                   new MarshalledValueInterceptor(), MarshalledValueInterceptor.class);
175 
176          interceptorChain.appendInterceptor(interceptor, false);
177       }
178 
179 
180       interceptorChain.appendInterceptor(createInterceptor(new NotificationInterceptor(), NotificationInterceptor.class), false);
181 
182       if (configuration.transaction().useEagerLocking()) {
183          configuration.transaction().lockingMode(LockingMode.PESSIMISTIC);
184       }
185 
186       //the total order protocol doesn't need locks
187       if (!isTotalOrder) {
188          if (configuration.transaction().transactionMode().isTransactional()) {
189             if (configuration.transaction().lockingMode() == LockingMode.PESSIMISTIC) {
190                interceptorChain.appendInterceptor(createInterceptor(new PessimisticLockingInterceptor(), PessimisticLockingInterceptor.class), false);
191             } else {
192                interceptorChain.appendInterceptor(createInterceptor(new OptimisticLockingInterceptor(), OptimisticLockingInterceptor.class), false);
193             }
194          } else {
195             interceptorChain.appendInterceptor(createInterceptor(new NonTransactionalLockingInterceptor(), NonTransactionalLockingInterceptor.class), false);
196          }
197       }
198 
199       if (configuration.sites().hasEnabledBackups() && !configuration.sites().disableBackups()) {
200          if ((configuration.transaction().transactionMode() == TransactionMode.TRANSACTIONAL)) {
201             if (configuration.transaction().lockingMode() == LockingMode.OPTIMISTIC) {
202                interceptorChain.appendInterceptor(createInterceptor(new OptimisticBackupInterceptor(), OptimisticBackupInterceptor.class), false);
203             } else {
204                interceptorChain.appendInterceptor(createInterceptor(new PessimisticBackupInterceptor(), PessimisticBackupInterceptor.class), false);
205             }
206          } else {
207             interceptorChain.appendInterceptor(createInterceptor(new NonTransactionalBackupInterceptor(), NonTransactionalBackupInterceptor.class), false);
208          }
209       }
210 
211       // This needs to be added after the locking interceptor (for tx caches) but before the wrapping interceptor.
212       if (configuration.clustering().l1().enabled()) {
213          interceptorChain.appendInterceptor(createInterceptor(new L1LastChanceInterceptor(), L1LastChanceInterceptor.class), false);
214       }
215 
216       if (configuration.clustering().hash().groups().enabled()) {
217          interceptorChain.appendInterceptor(createInterceptor(new GroupingInterceptor(), GroupingInterceptor.class), false);
218       }
219 
220       if (needsVersionAwareComponents && configuration.clustering().cacheMode().isClustered()) {
221          if (isTotalOrder) {
222             interceptorChain.appendInterceptor(createInterceptor(new TotalOrderVersionedEntryWrappingInterceptor(),
223                                                                  TotalOrderVersionedEntryWrappingInterceptor.class), false);
224          } else {
225             interceptorChain.appendInterceptor(createInterceptor(new VersionedEntryWrappingInterceptor(), VersionedEntryWrappingInterceptor.class), false);
226          }
227       } else
228          interceptorChain.appendInterceptor(createInterceptor(new EntryWrappingInterceptor(), EntryWrappingInterceptor.class), false);
229 
230       if (configuration.persistence().usingStores()) {
231          if (configuration.persistence().passivation()) {
232             if (configuration.clustering().cacheMode().isClustered())
233                interceptorChain.appendInterceptor(createInterceptor(new ClusteredActivationInterceptor(), ClusteredActivationInterceptor.class), false);
234             else
235                interceptorChain.appendInterceptor(createInterceptor(new ActivationInterceptor(), ActivationInterceptor.class), false);
236          } else {
237             if (configuration.clustering().cacheMode().isClustered())
238                interceptorChain.appendInterceptor(createInterceptor(new ClusteredCacheLoaderInterceptor(), ClusteredCacheLoaderInterceptor.class), false);
239             else
240                interceptorChain.appendInterceptor(createInterceptor(new CacheLoaderInterceptor(), CacheLoaderInterceptor.class), false);
241             switch (configuration.clustering().cacheMode()) {
242                case DIST_SYNC:
243                case DIST_ASYNC:
244                case REPL_SYNC:
245                case REPL_ASYNC:
246                   interceptorChain.appendInterceptor(createInterceptor(new DistCacheWriterInterceptor(), DistCacheWriterInterceptor.class), false);
247                   break;
248                default:
249                   interceptorChain.appendInterceptor(createInterceptor(new CacheWriterInterceptor(), CacheWriterInterceptor.class), false);
250                   break;
251             }
252          }
253       }
254 
255       if (configuration.deadlockDetection().enabled() && !isTotalOrder) {
256          interceptorChain.appendInterceptor(createInterceptor(new DeadlockDetectingInterceptor(), DeadlockDetectingInterceptor.class), false);
257       }
258 
259       if (configuration.clustering().l1().enabled()) {
260          if (configuration.transaction().transactionMode().isTransactional()) {
261             interceptorChain.appendInterceptor(createInterceptor(new L1TxInterceptor(), L1TxInterceptor.class), false);
262          }
263          else {
264             interceptorChain.appendInterceptor(createInterceptor(new L1NonTxInterceptor(), L1NonTxInterceptor.class), false);
265          }
266       }
267 
268       switch (configuration.clustering().cacheMode()) {
269          case INVALIDATION_SYNC:
270          case INVALIDATION_ASYNC:
271             interceptorChain.appendInterceptor(createInterceptor(new InvalidationInterceptor(), InvalidationInterceptor.class), false);
272             break;
273          case DIST_SYNC:
274          case REPL_SYNC:
275             if (needsVersionAwareComponents) {
276                if (isTotalOrder) {
277                   interceptorChain.appendInterceptor(createInterceptor(new TotalOrderVersionedDistributionInterceptor(),
278                                                                        TotalOrderVersionedDistributionInterceptor.class), false);
279                } else {
280                   interceptorChain.appendInterceptor(createInterceptor(new VersionedDistributionInterceptor(), VersionedDistributionInterceptor.class), false);
281                }
282                break;
283             }
284          case DIST_ASYNC:
285          case REPL_ASYNC:
286             if (configuration.transaction().transactionMode().isTransactional()) {
287                if (isTotalOrder) {
288                   interceptorChain.appendInterceptor(createInterceptor(new TotalOrderDistributionInterceptor(), TotalOrderDistributionInterceptor.class), false);
289                } else {
290                   interceptorChain.appendInterceptor(createInterceptor(new TxDistributionInterceptor(), TxDistributionInterceptor.class), false);
291                }
292             } else {
293                interceptorChain.appendInterceptor(createInterceptor(new NonTxDistributionInterceptor(), NonTxDistributionInterceptor.class), false);
294             }
295             break;
296          case LOCAL:
297             //Nothing...
298       }
299 
300       CommandInterceptor callInterceptor = createInterceptor(new CallInterceptor(), CallInterceptor.class);
301       interceptorChain.appendInterceptor(callInterceptor, false);
302       log.trace("Finished building default interceptor chain.");
303       buildCustomInterceptors(interceptorChain, configuration.customInterceptors());
304       return interceptorChain;
305    }
306 
307    private void buildCustomInterceptors(InterceptorChain interceptorChain, CustomInterceptorsConfiguration customInterceptors) {
308       for (InterceptorConfiguration config : customInterceptors.interceptors()) {
309          if (interceptorChain.containsInterceptorType(config.interceptor().getClass())) continue;
310 
311          CommandInterceptor customInterceptor = config.interceptor();
312          SecurityActions.applyProperties(customInterceptor, config.properties());
313          register(customInterceptor.getClass(), customInterceptor);
314          if (config.first())
315             interceptorChain.addInterceptor(customInterceptor, 0);
316          else if (config.last())
317             interceptorChain.addInterceptorBefore(customInterceptor, CallInterceptor.class);
318          else if (config.index() >= 0)
319             interceptorChain.addInterceptor(customInterceptor, config.index());
320          else if (config.after() != null) {
321             List<CommandInterceptor> withClassName = interceptorChain.getInterceptorsWithClass(config.after());
322             if (withClassName.isEmpty()) {
323                throw new CacheConfigurationException("Cannot add after class: " + config.after()
324                                                       + " as no such interceptor exists in the default chain");
325             }
326             interceptorChain.addInterceptorAfter(customInterceptor, withClassName.get(0).getClass());
327          } else if (config.before() != null) {
328             List<CommandInterceptor> withClassName = interceptorChain.getInterceptorsWithClass(config.before());
329             if (withClassName.isEmpty()) {
330                throw new CacheConfigurationException("Cannot add before class: " + config.after()
331                                                       + " as no such interceptor exists in the default chain");
332             }
333             interceptorChain.addInterceptorBefore(customInterceptor, withClassName.get(0).getClass());
334          } else if (config.position() == InterceptorConfiguration.Position.OTHER_THAN_FIRST_OR_LAST) {
335             interceptorChain.addInterceptor(customInterceptor, 1);
336          }
337       }
338 
339    }

341    private boolean hasAsyncStore() {
342       List<StoreConfiguration> loaderConfigs = configuration.persistence().stores();
343       for (StoreConfiguration loaderConfig : loaderConfigs) {
344          if (loaderConfig.async().enabled())
345             return true;
346       }
347       return false;
348    }
349 
350    @Override
351    public <T> T construct(Class<T> componentType) {
352       try {
353          return componentType.cast(buildInterceptorChain());
354       } catch (CacheException ce) {
355          throw ce;
356       } catch (Exception e) {
357          throw new CacheConfigurationException("Unable to build interceptor chain", e);
358       }
359    }
360 
361    public static InterceptorChainFactory getInstance(ComponentRegistry componentRegistry, Configuration configuration) {
362       InterceptorChainFactory icf = new InterceptorChainFactory();
363       icf.componentRegistry = componentRegistry;
364       icf.configuration = configuration;
365       return icf;
366    }
