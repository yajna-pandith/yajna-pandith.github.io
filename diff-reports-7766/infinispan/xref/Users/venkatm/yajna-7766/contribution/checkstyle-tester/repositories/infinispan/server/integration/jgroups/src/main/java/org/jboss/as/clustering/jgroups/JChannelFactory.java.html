85      @Override
86      public Channel createChannel(final String id) throws Exception {
87          JChannel channel = new JChannel(this);
88  
89          // We need to synchronize on shared transport,
90          // so we don't attempt to init a shared transport multiple times
91          TP transport = channel.getProtocolStack().getTransport();
92          if (transport.isSingleton()) {
93              synchronized (transport) {
94                  this.init(transport);
95              }
96          } else {
97              this.init(transport);
98          }
99  
100         // Relay protocol is added to stack programmatically, not via ProtocolStackConfigurator
101         final RelayConfiguration relayConfig = this.configuration.getRelay();
102         if (relayConfig != null) {
103             final String localSite = relayConfig.getSiteName();
104             final List<RemoteSiteConfiguration> remoteSites = this.configuration.getRelay().getRemoteSites();
105             final List<String> sites = new ArrayList<String>(remoteSites.size() + 1);
106             sites.add(localSite);
107             // Collect bridges, eliminating duplicates
108             final Map<String, RelayConfig.BridgeConfig> bridges = new HashMap<String, RelayConfig.BridgeConfig>();
109             for (final RemoteSiteConfiguration remoteSite: remoteSites) {
110                 final String siteName = remoteSite.getName();
111                 sites.add(siteName);
112                 final String cluster = remoteSite.getCluster();
113                 final String clusterName = (cluster != null) ? cluster : siteName;
114                 final RelayConfig.BridgeConfig bridge = new RelayConfig.BridgeConfig(clusterName) {
115                     @Override
116                     public JChannel createChannel() throws Exception {
117                         return (JChannel) remoteSite.getChannelFactory().createChannel(id + "/" + clusterName);
118                     }
119                 };
120                 bridges.put(clusterName, bridge);
121             }
122             // Acquire consistent order of sites
123             Collections.sort(sites);
124             final RELAY2 relay = new RELAY2().site(localSite);
125             for (short i = 0; i < sites.size(); ++i) {
126                 final String site = sites.get(i);
127                 RelayConfig.SiteConfig siteConfig = new RelayConfig.SiteConfig(site);
128                 relay.addSite(site, siteConfig);
129                 if (site.equals(localSite)) {
130                     for (RelayConfig.BridgeConfig bridge: bridges.values()) {
131                         siteConfig.addBridge(bridge);
132                     }
133                 }
134             }
135             Configurator.resolveAndAssignFields(relay, relayConfig.getProperties());
136             Configurator.resolveAndInvokePropertyMethods(relay, relayConfig.getProperties());
137             channel.getProtocolStack().addProtocol(relay);
138             relay.init();
139         }
140 
141         // Handle the Sasl protocol
142         final SaslConfiguration saslConfig = this.configuration.getSasl();
143         if (saslConfig != null) {
144            final String clusterRole = saslConfig.getClusterRole();
145            final SecurityRealm securityRealm = saslConfig.getSecurityRealm();
146            final String mech = saslConfig.getMech();
147            final SASL sasl = new SASL();
148            sasl.setMech(mech);
149            Map<String, String> props = saslConfig.getProperties();
150            if (props.containsKey("client_password")) {
151                String credential = props.get("client_password");
152                String name = props.get("client_name");
153                if (name == null) {
154                    sasl.setClientCallbackHandler(new SaslClientCallbackHandler(securityRealm.getName(), this.configuration.getEnvironment().getNodeName(), credential));
155                } else if (name.contains("@")) {
156                    sasl.setClientCallbackHandler(new SaslClientCallbackHandler(name, credential));
157                } else {
158                    sasl.setClientCallbackHandler(new SaslClientCallbackHandler(securityRealm.getName(), name, credential));
159                }
160            } else {
161                props.put("client_password", ""); // HACKY
162            }
163            Map<String, String> saslProps = props.containsKey("sasl_props") ? Util.parseCommaDelimitedProps(props.get("sasl_props")) : new HashMap<String, String>();
164            sasl.setServerCallbackHandler(new RealmAuthorizationCallbackHandler(securityRealm, mech, clusterRole !=null  ? clusterRole : id, saslProps));
165            props.put("sasl_props",  new PropertyConverters.StringProperties().toString(saslProps));
166            Configurator.resolveAndAssignFields(sasl, props);
167            Configurator.resolveAndInvokePropertyMethods(sasl, props);
168            channel.getProtocolStack().insertProtocol(sasl, ProtocolStack.BELOW, GMS.class);
169            sasl.init();
170         }
171 
172         channel.setName(this.configuration.getEnvironment().getNodeName() + "/" + id);
173 
174         TransportConfiguration.Topology topology = this.configuration.getTransport().getTopology();
175         if (topology != null) {
176             channel.setAddressGenerator(new TopologyAddressGenerator(channel, topology.getSite(), topology.getRack(), topology.getMachine()));
177         }
178 
179         MBeanServer server = this.configuration.getMBeanServer();
180         if (server != null) {
181             try {
182                 this.channels.put(channel, id);
183                 JmxConfigurator.registerChannel(channel, server, id);
184             } catch (Exception e) {
185                 ROOT_LOGGER.warn(e.getMessage(), e);
186             }
187             channel.addChannelListener(this);
188         }
189 
190         return channel;
191     }

193     private void init(TP transport) {
194         TransportConfiguration transportConfig = this.configuration.getTransport();
195         SocketBinding binding = transportConfig.getSocketBinding();
196         if (binding != null) {
197             SocketFactory factory = transport.getSocketFactory();
198             if (!(factory instanceof ManagedSocketFactory)) {
199                 transport.setSocketFactory(new ManagedSocketFactory(factory, binding.getSocketBindings()));
200             }
201         }
202         ThreadFactory threadFactory = transportConfig.getThreadFactory();
203         if (threadFactory != null) {
204             if (!(transport.getThreadFactory() instanceof ThreadFactoryAdapter)) {
205                 transport.setThreadFactory(new ThreadFactoryAdapter(threadFactory));
206             }
207         }
208         ExecutorService defaultExecutor = transportConfig.getDefaultExecutor();
209         if (defaultExecutor != null) {
210             if (!(transport.getDefaultThreadPool() instanceof ManagedExecutorService)) {
211                 transport.setDefaultThreadPool(new ManagedExecutorService(defaultExecutor));
212             }
213         }
214         ExecutorService oobExecutor = transportConfig.getOOBExecutor();
215         if (oobExecutor != null) {
216             if (!(transport.getOOBThreadPool() instanceof ManagedExecutorService)) {
217                 transport.setOOBThreadPool(new ManagedExecutorService(oobExecutor));
218             }
219         }
220         ScheduledExecutorService timerExecutor = transportConfig.getTimerExecutor();
221         if (timerExecutor != null) {
222             if (!(transport.getTimer() instanceof TimerSchedulerAdapter)) {
223                 this.setValue(transport, "timer", new TimerSchedulerAdapter(new ManagedScheduledExecutorService(timerExecutor)));
224             }
225         }
226     }

241     @Override
242     public List<org.jgroups.conf.ProtocolConfiguration> getProtocolStack() {
243         List<org.jgroups.conf.ProtocolConfiguration> configs = new ArrayList<org.jgroups.conf.ProtocolConfiguration>(this.configuration.getProtocols().size() + 1);
244         TransportConfiguration transport = this.configuration.getTransport();
245         org.jgroups.conf.ProtocolConfiguration config = this.createProtocol(transport);
246         Map<String, String> properties = config.getProperties();
247 
248         if (transport.isShared()) {
249             properties.put(Global.SINGLETON_NAME, this.configuration.getName());
250         }
251 
252         SocketBinding binding = transport.getSocketBinding();
253         if (binding != null) {
254             this.configureBindAddress(transport, config, binding);
255             this.configureServerSocket(transport, config, "bind_port", binding);
256             this.configureMulticastSocket(transport, config, "mcast_addr", "mcast_port", binding);
257         }
258 
259         SocketBinding diagnosticsSocketBinding = transport.getDiagnosticsSocketBinding();
260         boolean diagnostics = (diagnosticsSocketBinding != null);
261         properties.put("enable_diagnostics", String.valueOf(diagnostics));
262         if (diagnostics) {
263             this.configureMulticastSocket(transport, config, "diagnostics_addr", "diagnostics_port", diagnosticsSocketBinding);
264         }
265 
266         configs.add(config);
267 
268         boolean supportsMulticast = transport.hasProperty("mcast_addr");
269 
270         for (ProtocolConfiguration protocol: this.configuration.getProtocols()) {
271             config = this.createProtocol(protocol);
272             binding = protocol.getSocketBinding();
273             if (binding != null) {
274                 this.configureBindAddress(protocol, config, binding);
275                 this.configureServerSocket(protocol, config, "bind_port", binding);
276                 this.configureServerSocket(protocol, config, "start_port", binding);
277                 this.configureMulticastSocket(protocol, config, "mcast_addr", "mcast_port", binding);
278             } else if (transport.getSocketBinding() != null) {
279                 // If no socket-binding was specified, use bind address of transport
280                 this.configureBindAddress(protocol, config, transport.getSocketBinding());
281             }
282             if (!supportsMulticast) {
283                 this.setProperty(protocol, config, "use_mcast_xmit", String.valueOf(false));
284             }
285             configs.add(config);
286         }
287         return configs;
288     }

290     private void configureBindAddress(ProtocolConfiguration protocol, org.jgroups.conf.ProtocolConfiguration config, SocketBinding binding) {
291         this.setPropertyNoOverride(protocol, config, "bind_addr", binding.getSocketAddress().getAddress().getHostAddress());
292     }
293 
294     private void configureServerSocket(ProtocolConfiguration protocol, org.jgroups.conf.ProtocolConfiguration config, String property, SocketBinding binding) {
295         this.setPropertyNoOverride(protocol, config, property, String.valueOf(binding.getSocketAddress().getPort()));
296     }
297 
298     private void configureMulticastSocket(ProtocolConfiguration protocol, org.jgroups.conf.ProtocolConfiguration config, String addressProperty, String portProperty, SocketBinding binding) {
299         try {
300             InetSocketAddress mcastSocketAddress = binding.getMulticastSocketAddress();
301             this.setPropertyNoOverride(protocol, config, addressProperty, mcastSocketAddress.getAddress().getHostAddress());
302             this.setPropertyNoOverride(protocol, config, portProperty, String.valueOf(mcastSocketAddress.getPort()));
303         } catch (IllegalStateException e) {
304             ROOT_LOGGER.couldNotSetAddressAndPortNoMulticastSocket(e, config.getProtocolName(), addressProperty, config.getProtocolName(), portProperty, binding.getName());
305         }
306     }
307 
308     private void setPropertyNoOverride(ProtocolConfiguration protocol, org.jgroups.conf.ProtocolConfiguration config, String name, String value) {
309         boolean overridden = false ;
310         String propertyValue = null ;
311         // check if the property has been overridden by the user and log a message
312         try {
313             if (overridden = config.getOriginalProperties().containsKey(name))
314                propertyValue = config.getOriginalProperties().get(name);
315         }
316         catch(Exception e) {
317             ROOT_LOGGER.unableToAccessProtocolPropertyValue(e, name, protocol.getName());
318         }
319 
320         // log a warning if property tries to override
321         if (overridden) {
322             ROOT_LOGGER.unableToOverrideSocketBindingValue(name, protocol.getName(), value, propertyValue);
323         }
324         setProperty(protocol, config, name, value);
325     }
326 
327     private void setProperty(ProtocolConfiguration protocol, org.jgroups.conf.ProtocolConfiguration config, String name, String value) {
328         if (protocol.hasProperty(name)) {
329             config.getProperties().put(name, value);
330         }
331     }
332 
333     private org.jgroups.conf.ProtocolConfiguration createProtocol(final ProtocolConfiguration protocolConfig) {
334         String protocol = protocolConfig.getName();
335         final Map<String, String> properties = new HashMap<String, String>(this.configuration.getDefaults().getProperties(protocol));
336         properties.putAll(protocolConfig.getProperties());
337         return new org.jgroups.conf.ProtocolConfiguration(protocol, properties) {
338             @Override
339             public Map<String, String> getOriginalProperties() {
340                 return properties;
341             }
342         };
343     }
344 
345     private void setValue(Protocol protocol, String property, Object value) {
346         ROOT_LOGGER.setProtocolPropertyValue(protocol.getName(), property, value);
347         try {
348             protocol.setValue(property, value);
349         } catch (IllegalArgumentException e) {
350             ROOT_LOGGER.nonExistentProtocolPropertyValue(e, protocol.getName(), property, value);
351         }
352     }

364     @Override
365     public void channelClosed(Channel channel) {
366         MBeanServer server = this.configuration.getMBeanServer();
367         if (server != null) {
368             try {
369                 JmxConfigurator.unregisterChannel((JChannel) channel, server, this.channels.remove(channel));
370             } catch (Exception e) {
371                 ROOT_LOGGER.warn(e.getMessage(), e);
372             }
373         }
