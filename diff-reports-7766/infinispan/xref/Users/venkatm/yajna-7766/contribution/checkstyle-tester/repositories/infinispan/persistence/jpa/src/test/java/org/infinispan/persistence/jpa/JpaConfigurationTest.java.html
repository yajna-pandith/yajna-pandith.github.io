26  @Test(groups = "functional", testName = "persistence.JpaConfigurationTest")
27  public class JpaConfigurationTest {
28  
29     private static final String PERSISTENCE_UNIT_NAME = "org.infinispan.persistence.jpa.configurationTest";
30  
31     public void testConfigBuilder() {
32        GlobalConfiguration globalConfig = new GlobalConfigurationBuilder()
33              .globalJmxStatistics().transport().defaultTransport().build();
34  
35        Configuration cacheConfig = new ConfigurationBuilder().persistence()
36              .addStore(JpaStoreConfigurationBuilder.class)
37              .persistenceUnitName(PERSISTENCE_UNIT_NAME)
38              .entityClass(User.class).build();
39  
40  
41        StoreConfiguration storeConfiguration = cacheConfig.persistence().stores().get(0);
42        assertTrue(storeConfiguration instanceof JpaStoreConfiguration);
43        JpaStoreConfiguration jpaCacheLoaderConfig = (JpaStoreConfiguration) storeConfiguration;
44        assertEquals(PERSISTENCE_UNIT_NAME, jpaCacheLoaderConfig.persistenceUnitName());
45        assertEquals(User.class, jpaCacheLoaderConfig.entityClass());
46  
47        EmbeddedCacheManager cacheManager = new DefaultCacheManager(globalConfig);
48  
49        cacheManager.defineConfiguration("userCache", cacheConfig);
50  
51        cacheManager.start();
52        Cache<String, User> userCache = cacheManager.getCache("userCache");
53        User user = new User();
54        user.setUsername("rtsang");
55        user.setFirstName("Ray");
56        user.setLastName("Tsang");
57        userCache.put(user.getUsername(), user);
58        userCache.stop();
59        cacheManager.stop();
60     }
61  
62     protected void validateConfig(Cache<VehicleId, Vehicle> vehicleCache) {
63        StoreConfiguration config = vehicleCache.getCacheConfiguration().persistence().stores().get(0);
64  
65        assertTrue(config instanceof JpaStoreConfiguration);
66        JpaStoreConfiguration jpaConfig = (JpaStoreConfiguration) config;
67        assertEquals(1, jpaConfig.batchSize());
68        assertEquals(Vehicle.class, jpaConfig.entityClass());
69        assertEquals(PERSISTENCE_UNIT_NAME, jpaConfig.persistenceUnitName());
70     }
71  
72     public void testXmlConfig() throws IOException {
73        EmbeddedCacheManager cacheManager = new DefaultCacheManager("config/jpa-config.xml");
74  
75        Cache<VehicleId, Vehicle> vehicleCache = cacheManager.getCache("vehicleCache");
76        validateConfig(vehicleCache);
77  
78        Vehicle v = new Vehicle();
79        v.setId(new VehicleId("NC", "123456"));
80        v.setColor("BLUE");
81        vehicleCache.put(v.getId(), v);
82  
83        vehicleCache.stop();
84        cacheManager.stop();
85     }
