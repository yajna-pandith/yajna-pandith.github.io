184    public void testEventReceiveBasic() {
185       EventLogListener<Integer> eventListener = new EventLogListener<>(true);
186       Cache<Integer, String> embedded = cacheFactory.getEmbeddedCache();
187       RemoteCache<Integer, String> remote = cacheFactory.getHotRodCache();
188       remote.addClientListener(eventListener);
189       try {
190          eventListener.expectNoEvents();
191          remote.remove(1);
192          eventListener.expectNoEvents();
193          remote.put(1, "one");
194          assertEquals("one", embedded.get(1));
195          eventListener.expectOnlyCreatedEvent(1, embedded);
196          remote.put(1, "new-one");
197          assertEquals("new-one", embedded.get(1));
198          eventListener.expectOnlyModifiedEvent(1, embedded);
199          remote.remove(1);
200          eventListener.expectOnlyRemovedEvent(1, embedded);
201       } finally {
202          remote.removeClientListener(eventListener);
203       }
204    }

206    public void testEventReceiveConditional() {
207       EventLogListener<Integer> eventListener = new EventLogListener<>(true);
208       Cache<Integer, String> embedded = cacheFactory.getEmbeddedCache();
209       RemoteCache<Integer, String> remote = cacheFactory.getHotRodCache();
210       remote.addClientListener(eventListener);
211       try {
212          eventListener.expectNoEvents();
213          // Put if absent
214          remote.putIfAbsent(1, "one");
215          eventListener.expectOnlyCreatedEvent(1, embedded);
216          remote.putIfAbsent(1, "again");
217          eventListener.expectNoEvents();
218          // Replace
219          remote.replace(1, "newone");
220          eventListener.expectOnlyModifiedEvent(1, embedded);
221          // Replace with version
222          remote.replaceWithVersion(1, "one", 0);
223          eventListener.expectNoEvents();
224          VersionedValue<String> versioned = remote.getVersioned(1);
225          remote.replaceWithVersion(1, "one", versioned.getVersion());
226          eventListener.expectOnlyModifiedEvent(1, embedded);
227          // Remove with version
228          remote.removeWithVersion(1, 0);
229          eventListener.expectNoEvents();
230          versioned = remote.getVersioned(1);
231          remote.removeWithVersion(1, versioned.getVersion());
232          eventListener.expectOnlyRemovedEvent(1, embedded);
233       } finally {
234          remote.removeClientListener(eventListener);
235       }
236    }

239       EventLogWithStateListener<Integer> eventListener = new EventLogWithStateListener<>(true);
240       Cache<Integer, String> embedded = cacheFactory.getEmbeddedCache();
241       RemoteCache<Integer, String> remote = cacheFactory.getHotRodCache();

248       remote.remove(3);
249       assertNull(embedded.get(3));
250       eventListener.expectNoEvents();
251       remote.addClientListener(eventListener);
252       try {
253          eventListener.expectUnorderedEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_CREATED, 1, 2);
254          remote.remove(1);
255          eventListener.expectOnlyRemovedEvent(1, embedded);
256          remote.remove(2);
257          eventListener.expectOnlyRemovedEvent(2, embedded);
258       } finally {
259          remote.removeClientListener(eventListener);
260       }
261    }
