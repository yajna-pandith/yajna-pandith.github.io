38  /**
39   * XML writer for current Infinispan subsystem schema version.
40   * @author Paul Ferraro
41   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
42   * @author Tristan Tarrant
43   */
44  public class InfinispanSubsystemXMLWriter implements XMLElementWriter<SubsystemMarshallingContext> {
45      public static final XMLElementWriter<SubsystemMarshallingContext> INSTANCE = new InfinispanSubsystemXMLWriter();
46  
47      /**
48       * {@inheritDoc}
49       * @see org.jboss.staxmapper.XMLElementWriter#writeContent(org.jboss.staxmapper.XMLExtendedStreamWriter, java.lang.Object)
50       */
51      @Override
52      public void writeContent(XMLExtendedStreamWriter writer, SubsystemMarshallingContext context) throws XMLStreamException {
53          context.startSubsystemElement(Namespace.CURRENT.getUri(), false);
54          ModelNode model = context.getModelNode();
55          if (model.isDefined()) {
56              for (Property entry: model.get(ModelKeys.CACHE_CONTAINER).asPropertyList()) {
57  
58                  String containerName = entry.getName();
59                  ModelNode container = entry.getValue();
60  
61                  writer.writeStartElement(Element.CACHE_CONTAINER.getLocalName());
62                  writer.writeAttribute(Attribute.NAME.getLocalName(), containerName);
63                  // AS7-3488 make default-cache a non required attribute
64                  // this.writeRequired(writer, Attribute.DEFAULT_CACHE, container, ModelKeys.DEFAULT_CACHE);
65                  this.writeListAsAttribute(writer, Attribute.ALIASES, container, ModelKeys.ALIASES);
66                  this.writeOptional(writer, Attribute.DEFAULT_CACHE, container, ModelKeys.DEFAULT_CACHE);
67                  this.writeOptional(writer, Attribute.EVICTION_EXECUTOR, container, ModelKeys.EVICTION_EXECUTOR);
68                  this.writeOptional(writer, Attribute.JNDI_NAME, container, ModelKeys.JNDI_NAME);
69                  this.writeOptional(writer, Attribute.LISTENER_EXECUTOR, container, ModelKeys.LISTENER_EXECUTOR);
70                  this.writeOptional(writer, Attribute.REPLICATION_QUEUE_EXECUTOR, container, ModelKeys.REPLICATION_QUEUE_EXECUTOR);
71                  this.writeOptional(writer, Attribute.STATE_TRANSFER_EXECUTOR, container, ModelKeys.STATE_TRANSFER_EXECUTOR);
72                  this.writeOptional(writer, Attribute.START, container, ModelKeys.START);
73                  this.writeOptional(writer, Attribute.MODULE, container, ModelKeys.MODULE);
74                  this.writeOptional(writer, Attribute.STATISTICS, container, ModelKeys.STATISTICS);
75  
76                  if (container.hasDefined(ModelKeys.TRANSPORT)) {
77                      writer.writeStartElement(Element.TRANSPORT.getLocalName());
78                      ModelNode transport = container.get(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
79                      this.writeOptional(writer, Attribute.STACK, transport, ModelKeys.STACK);
80                      this.writeOptional(writer, Attribute.CLUSTER, transport, ModelKeys.CLUSTER);
81                      this.writeOptional(writer, Attribute.EXECUTOR, transport, ModelKeys.EXECUTOR);
82                      this.writeOptional(writer, Attribute.LOCK_TIMEOUT, transport, ModelKeys.LOCK_TIMEOUT);
83                      this.writeOptional(writer, Attribute.REMOTE_COMMAND_EXECUTOR, transport, ModelKeys.REMOTE_COMMAND_EXECUTOR);
84                      this.writeOptional(writer, Attribute.STRICT_PEER_TO_PEER, transport, ModelKeys.STRICT_PEER_TO_PEER);
85                      this.writeOptional(writer, Attribute.TOTAL_ORDER_EXECUTOR, transport, ModelKeys.TOTAL_ORDER_EXECUTOR);
86                      writer.writeEndElement();
87                  }
88  
89                  if (container.hasDefined(ModelKeys.SECURITY)) {
90                      writer.writeStartElement(Element.SECURITY.getLocalName());
91                      ModelNode security = container.get(ModelKeys.SECURITY, ModelKeys.SECURITY_NAME);
92                      if (security.hasDefined(ModelKeys.AUTHORIZATION)) {
93                          writer.writeStartElement(Element.AUTHORIZATION.getLocalName());
94                          ModelNode authorization = security.get(ModelKeys.AUTHORIZATION, ModelKeys.AUTHORIZATION_NAME);
95                          if (authorization.hasDefined(ModelKeys.MAPPER)) {
96                              String mapper = authorization.get(ModelKeys.MAPPER).asString();
97                              if (CommonNameRoleMapper.class.getName().equals(mapper)) {
98                                  writer.writeEmptyElement(Element.COMMON_NAME_ROLE_MAPPER.getLocalName());
99                              } else if (ClusterRoleMapper.class.getName().equals(mapper)) {
100                                 writer.writeEmptyElement(Element.CLUSTER_ROLE_MAPPER.getLocalName());
101                             } else if (IdentityRoleMapper.class.getName().equals(mapper)) {
102                                 writer.writeEmptyElement(Element.IDENTITY_ROLE_MAPPER.getLocalName());
103                             } else {
104                                 writer.writeStartElement(Element.CUSTOM_ROLE_MAPPER.getLocalName());
105                                 writer.writeAttribute(Attribute.CLASS.getLocalName(), mapper);
106                                 writer.writeEndElement();
107                             }
108                         }
109 
110                         ModelNode roles = authorization.get(ModelKeys.ROLE);
111                         for(ModelNode roleNode : roles.asList()) {
112                             ModelNode role = roleNode.get(0);
113                             writer.writeStartElement(Element.ROLE.getLocalName());
114                             AuthorizationRoleResource.NAME.marshallAsAttribute(role, writer);
115                             this.writeListAsAttribute(writer, Attribute.PERMISSIONS, role, ModelKeys.PERMISSIONS);
116                             writer.writeEndElement();
117                         }
118 
119                         writer.writeEndElement();
120                     }
121 
122                     writer.writeEndElement();
123                 }
124 
125                 // write any existent cache types
126                 if (container.get(ModelKeys.LOCAL_CACHE).isDefined()) {
127                     for (Property localCacheEntry : container.get(ModelKeys.LOCAL_CACHE).asPropertyList()) {
128                         String localCacheName = localCacheEntry.getName();
129                         ModelNode localCache = localCacheEntry.getValue();
130 
131                         writer.writeStartElement(Element.LOCAL_CACHE.getLocalName());
132                         // write identifier before other attributes
133                         writer.writeAttribute(Attribute.NAME.getLocalName(), localCacheName);
134 
135                         processCommonCacheAttributesElements(writer, localCache);
136 
137                         writer.writeEndElement();
138                     }
139                 }
140 
141                 if (container.get(ModelKeys.INVALIDATION_CACHE).isDefined()) {
142                     for (Property invalidationCacheEntry : container.get(ModelKeys.INVALIDATION_CACHE).asPropertyList()) {
143                         String invalidationCacheName = invalidationCacheEntry.getName();
144                         ModelNode invalidationCache = invalidationCacheEntry.getValue();
145 
146                         writer.writeStartElement(Element.INVALIDATION_CACHE.getLocalName());
147                         // write identifier before other attributes
148                         writer.writeAttribute(Attribute.NAME.getLocalName(), invalidationCacheName);
149 
150                         processCommonClusteredCacheAttributes(writer, invalidationCache);
151                         processCommonCacheAttributesElements(writer, invalidationCache);
152 
153                         writer.writeEndElement();
154                     }
155                 }
156 
157                 if (container.get(ModelKeys.REPLICATED_CACHE).isDefined()) {
158                     for (Property replicatedCacheEntry : container.get(ModelKeys.REPLICATED_CACHE).asPropertyList()) {
159                         String replicatedCacheName = replicatedCacheEntry.getName();
160                         ModelNode replicatedCache = replicatedCacheEntry.getValue();
161 
162                         writer.writeStartElement(Element.REPLICATED_CACHE.getLocalName());
163                         // write identifier before other attributes
164                         writer.writeAttribute(Attribute.NAME.getLocalName(), replicatedCacheName);
165 
166                         processCommonClusteredCacheAttributes(writer, replicatedCache);
167                         processCommonCacheAttributesElements(writer, replicatedCache);
168 
169                         writer.writeEndElement();
170                     }
171                 }
172 
173                 if (container.get(ModelKeys.DISTRIBUTED_CACHE).isDefined()) {
174                     for (Property distributedCacheEntry : container.get(ModelKeys.DISTRIBUTED_CACHE).asPropertyList()) {
175                         String distributedCacheName = distributedCacheEntry.getName();
176                         ModelNode distributedCache = distributedCacheEntry.getValue();
177 
178                         writer.writeStartElement(Element.DISTRIBUTED_CACHE.getLocalName());
179                         // write identifier before other attributes
180                         writer.writeAttribute(Attribute.NAME.getLocalName(), distributedCacheName);
181                         // distributed cache attributes
182                         this.writeOptional(writer, Attribute.OWNERS, distributedCache, ModelKeys.OWNERS);
183                         this.writeOptional(writer, Attribute.SEGMENTS, distributedCache, ModelKeys.SEGMENTS);
184                         this.writeOptional(writer, Attribute.CAPACITY_FACTOR, distributedCache, ModelKeys.CAPACITY_FACTOR);
185                         this.writeOptional(writer, Attribute.L1_LIFESPAN, distributedCache, ModelKeys.L1_LIFESPAN);
186 
187                         processCommonClusteredCacheAttributes(writer, distributedCache);
188                         processCommonCacheAttributesElements(writer, distributedCache);
189 
190                         writer.writeEndElement();
191                     }
192                 }
193                 writer.writeEndElement();
194             }
195         }
196         writer.writeEndElement();
197     }
198 
199     private void processCommonClusteredCacheAttributes(XMLExtendedStreamWriter writer, ModelNode cache)
200             throws XMLStreamException {
201 
202         this.writeOptional(writer, Attribute.ASYNC_MARSHALLING, cache, ModelKeys.ASYNC_MARSHALLING);
203         this.writeRequired(writer, Attribute.MODE, cache, ModelKeys.MODE);
204         this.writeOptional(writer, Attribute.QUEUE_SIZE, cache, ModelKeys.QUEUE_SIZE);
205         this.writeOptional(writer, Attribute.QUEUE_FLUSH_INTERVAL, cache, ModelKeys.QUEUE_FLUSH_INTERVAL);
206         this.writeOptional(writer, Attribute.REMOTE_TIMEOUT, cache, ModelKeys.REMOTE_TIMEOUT);
207     }
208 
209     private void processCommonCacheAttributesElements(XMLExtendedStreamWriter writer, ModelNode cache)
210             throws XMLStreamException {
211 
212         this.writeOptional(writer, Attribute.START, cache, ModelKeys.START);
213         this.writeOptional(writer, Attribute.BATCHING, cache, ModelKeys.BATCHING);
214         this.writeOptional(writer, Attribute.JNDI_NAME, cache, ModelKeys.JNDI_NAME);
215         this.writeOptional(writer, Attribute.MODULE, cache, ModelKeys.MODULE);
216         this.writeOptional(writer, Attribute.STATISTICS, cache, ModelKeys.STATISTICS);
217 
218         if (cache.get(ModelKeys.LOCKING, ModelKeys.LOCKING_NAME).isDefined()) {
219             writer.writeStartElement(Element.LOCKING.getLocalName());
220             ModelNode locking = cache.get(ModelKeys.LOCKING, ModelKeys.LOCKING_NAME);
221             this.writeOptional(writer, Attribute.ISOLATION, locking, ModelKeys.ISOLATION);
222             this.writeOptional(writer, Attribute.STRIPING, locking, ModelKeys.STRIPING);
223             this.writeOptional(writer, Attribute.ACQUIRE_TIMEOUT, locking, ModelKeys.ACQUIRE_TIMEOUT);
224             this.writeOptional(writer, Attribute.CONCURRENCY_LEVEL, locking, ModelKeys.CONCURRENCY_LEVEL);
225             writer.writeEndElement();
226         }
227 
228         if (cache.get(ModelKeys.TRANSACTION, ModelKeys.TRANSACTION_NAME).isDefined()) {
229             writer.writeStartElement(Element.TRANSACTION.getLocalName());
230             ModelNode transaction = cache.get(ModelKeys.TRANSACTION, ModelKeys.TRANSACTION_NAME);
231             this.writeOptional(writer, Attribute.STOP_TIMEOUT, transaction, ModelKeys.STOP_TIMEOUT);
232             this.writeOptional(writer, Attribute.MODE, transaction, ModelKeys.MODE);
233             this.writeOptional(writer, Attribute.LOCKING, transaction, ModelKeys.LOCKING);
234             writer.writeEndElement();
235         }
236 
237         if (cache.get(ModelKeys.EVICTION, ModelKeys.EVICTION_NAME).isDefined()) {
238             writer.writeStartElement(Element.EVICTION.getLocalName());
239             ModelNode eviction = cache.get(ModelKeys.EVICTION, ModelKeys.EVICTION_NAME);
240             this.writeOptional(writer, Attribute.STRATEGY, eviction, ModelKeys.STRATEGY);
241             this.writeOptional(writer, Attribute.MAX_ENTRIES, eviction, ModelKeys.MAX_ENTRIES);
242             writer.writeEndElement();
243         }
244 
245         if (cache.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME).isDefined()) {
246             writer.writeStartElement(Element.EXPIRATION.getLocalName());
247             ModelNode expiration = cache.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME);
248             this.writeOptional(writer, Attribute.MAX_IDLE, expiration, ModelKeys.MAX_IDLE);
249             this.writeOptional(writer, Attribute.LIFESPAN, expiration, ModelKeys.LIFESPAN);
250             this.writeOptional(writer, Attribute.INTERVAL, expiration, ModelKeys.INTERVAL);
251             writer.writeEndElement();
252         }
253 
254         if (cache.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME).isDefined()) {
255             ModelNode stateTransfer = cache.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME);
256             writer.writeStartElement(Element.STATE_TRANSFER.getLocalName());
257             this.writeOptional(writer, Attribute.AWAIT_INITIAL_TRANSFER, stateTransfer, ModelKeys.AWAIT_INITIAL_TRANSFER);
258             this.writeOptional(writer, Attribute.ENABLED, stateTransfer, ModelKeys.ENABLED);
259             this.writeOptional(writer, Attribute.TIMEOUT, stateTransfer, ModelKeys.TIMEOUT);
260             this.writeOptional(writer, Attribute.CHUNK_SIZE, stateTransfer, ModelKeys.CHUNK_SIZE);
261             writer.writeEndElement();
262         }
263 
264         if (cache.get(ModelKeys.PARTITION_HANDLING, ModelKeys.PARTITION_HANDLING_NAME).isDefined()) {
265             ModelNode partitionHandling = cache.get(ModelKeys.PARTITION_HANDLING, ModelKeys.PARTITION_HANDLING_NAME);
266             writer.writeStartElement(Element.PARTITION_HANDLING.getLocalName());
267             this.writeOptional(writer, Attribute.ENABLED, partitionHandling, ModelKeys.ENABLED);
268             writer.writeEndElement();
269         }
270 
271         if (cache.get(ModelKeys.COMPATIBILITY).isDefined()) {
272             ModelNode compatibility = cache.get(ModelKeys.COMPATIBILITY, ModelKeys.COMPATIBILITY_NAME);
273             writer.writeStartElement(Element.COMPATIBILITY.getLocalName());
274             CompatibilityResource.ENABLED.marshallAsAttribute(compatibility, writer);
275             CompatibilityResource.MARSHALLER.marshallAsAttribute(compatibility, writer);
276             writer.writeEndElement();
277         }
278 
279         if (cache.hasDefined(ModelKeys.SECURITY)) {
280             writer.writeStartElement(Element.SECURITY.getLocalName());
281             ModelNode security = cache.get(ModelKeys.SECURITY, ModelKeys.SECURITY_NAME);
282             if (security.hasDefined(ModelKeys.AUTHORIZATION)) {
283                 writer.writeStartElement(Element.AUTHORIZATION.getLocalName());
284                 ModelNode authorization = security.get(ModelKeys.AUTHORIZATION, ModelKeys.AUTHORIZATION_NAME);
285                 CacheAuthorizationResource.ENABLED.marshallAsAttribute(authorization, writer);
286                 this.writeListAsAttribute(writer, Attribute.ROLES, authorization, ModelKeys.ROLES);
287                 writer.writeEndElement();
288             }
289 
290             writer.writeEndElement();
291         }
292 
293         if (cache.get(ModelKeys.LOADER).isDefined()) {
294             for (Property clusterLoaderEntry : cache.get(ModelKeys.LOADER).asPropertyList()) {
295                 ModelNode loader = clusterLoaderEntry.getValue();
296                 writer.writeStartElement(Element.LOADER.getLocalName());
297                 // write identifier before other attributes
298                 ModelNode name = new ModelNode();
299                 name.get(ModelKeys.NAME).set(clusterLoaderEntry.getName());
300                 LoaderResource.NAME.marshallAsAttribute(name, false, writer);
301                 this.writeRequired(writer, Attribute.CLASS, loader, ModelKeys.CLASS);
302                 this.writeLoaderAttributes(writer, loader);
303                 this.writeStoreProperties(writer, loader);
304                 writer.writeEndElement();
305             }
306         }
307 
308         if (cache.get(ModelKeys.CLUSTER_LOADER).isDefined()) {
309             for (Property clusterLoaderEntry : cache.get(ModelKeys.CLUSTER_LOADER).asPropertyList()) {
310                 ModelNode loader = clusterLoaderEntry.getValue();
311                 writer.writeStartElement(Element.CLUSTER_LOADER.getLocalName());
312                 // write identifier before other attributes
313                 ModelNode name = new ModelNode();
314                 name.get(ModelKeys.NAME).set(clusterLoaderEntry.getName());
315                 ClusterLoaderResource.NAME.marshallAsAttribute(name, false, writer);
316                 this.writeOptional(writer, Attribute.REMOTE_TIMEOUT, loader, ModelKeys.REMOTE_TIMEOUT);
317                 this.writeLoaderAttributes(writer, loader);
318                 this.writeStoreProperties(writer, loader);
319                 writer.writeEndElement();
320             }
321         }
322 
323         if (cache.get(ModelKeys.STORE).isDefined()) {
324             for (Property storeEntry : cache.get(ModelKeys.STORE).asPropertyList()) {
325                 ModelNode store = storeEntry.getValue();
326                 writer.writeStartElement(Element.STORE.getLocalName());
327                 // write identifier before other attributes
328                 ModelNode name = new ModelNode();
329                 name.get(ModelKeys.NAME).set(storeEntry.getName());
330                 StoreResource.NAME.marshallAsAttribute(name, false, writer);
331                 this.writeRequired(writer, Attribute.CLASS, store, ModelKeys.CLASS);
332                 this.writeStoreAttributes(writer, store);
333                 this.writeStoreWriteBehind(writer, store);
334                 this.writeStoreProperties(writer, store);
335                 writer.writeEndElement();
336             }
337         }
338 
339         if (cache.get(ModelKeys.FILE_STORE).isDefined()) {
340             for (Property fileStoreEntry : cache.get(ModelKeys.FILE_STORE).asPropertyList()) {
341                 ModelNode store = fileStoreEntry.getValue();
342                 writer.writeStartElement(Element.FILE_STORE.getLocalName());
343                 // write identifier before other attributes
344                 ModelNode name = new ModelNode();
345                 name.get(ModelKeys.NAME).set(fileStoreEntry.getName());
346                 FileStoreResource.NAME.marshallAsAttribute(name, false, writer);
347                 this.writeOptional(writer, Attribute.MAX_ENTRIES, store, ModelKeys.MAX_ENTRIES);
348                 this.writeOptional(writer, Attribute.RELATIVE_TO, store, ModelKeys.RELATIVE_TO);
349                 this.writeOptional(writer, Attribute.PATH, store, ModelKeys.PATH);
350                 this.writeStoreAttributes(writer, store);
351                 this.writeStoreWriteBehind(writer, store);
352                 this.writeStoreProperties(writer, store);
353                 writer.writeEndElement();
354             }
355         }
356 
357         if (cache.get(ModelKeys.STRING_KEYED_JDBC_STORE).isDefined()) {
358             for (Property stringKeyedJDBCStoreEntry : cache.get(ModelKeys.STRING_KEYED_JDBC_STORE).asPropertyList()) {
359                 ModelNode store = stringKeyedJDBCStoreEntry.getValue();
360                 writer.writeStartElement(Element.STRING_KEYED_JDBC_STORE.getLocalName());
361                 // write identifier before other attributes
362                 ModelNode name = new ModelNode();
363                 name.get(ModelKeys.NAME).set(stringKeyedJDBCStoreEntry.getName());
364                 StringKeyedJDBCStoreResource.NAME.marshallAsAttribute(name, false, writer);
365                 this.writeJdbcStoreAttributes(writer, store);
366                 this.writeStoreWriteBehind(writer, store);
367                 this.writeStoreProperties(writer, store);
368                 this.writeJDBCStoreTable(writer, Element.STRING_KEYED_TABLE, store, ModelKeys.STRING_KEYED_TABLE);
369                 writer.writeEndElement();
370             }
371         }
372 
373         if (cache.get(ModelKeys.BINARY_KEYED_JDBC_STORE).isDefined()) {
374             for (Property binaryKeyedJDBCStoreEntry : cache.get(ModelKeys.BINARY_KEYED_JDBC_STORE).asPropertyList()) {
375                 ModelNode store = binaryKeyedJDBCStoreEntry.getValue();
376                 writer.writeStartElement(Element.BINARY_KEYED_JDBC_STORE.getLocalName());
377                 // write identifier before other attributes
378                 ModelNode name = new ModelNode();
379                 name.get(ModelKeys.NAME).set(binaryKeyedJDBCStoreEntry.getName());
380                 BinaryKeyedJDBCStoreResource.NAME.marshallAsAttribute(name, false, writer);
381                 this.writeJdbcStoreAttributes(writer, store);
382                 this.writeStoreWriteBehind(writer, store);
383                 this.writeStoreProperties(writer, store);
384                 this.writeJDBCStoreTable(writer, Element.BINARY_KEYED_TABLE, store, ModelKeys.BINARY_KEYED_TABLE);
385                 writer.writeEndElement();
386             }
387         }
388 
389         if (cache.get(ModelKeys.MIXED_KEYED_JDBC_STORE).isDefined()) {
390             for (Property mixedKeyedJDBCStoreEntry : cache.get(ModelKeys.MIXED_KEYED_JDBC_STORE).asPropertyList()) {
391                 ModelNode store = mixedKeyedJDBCStoreEntry.getValue();
392                 writer.writeStartElement(Element.MIXED_KEYED_JDBC_STORE.getLocalName());
393                 // write identifier before other attributes
394                 ModelNode name = new ModelNode();
395                 name.get(ModelKeys.NAME).set(mixedKeyedJDBCStoreEntry.getName());
396                 MixedKeyedJDBCStoreResource.NAME.marshallAsAttribute(name, false, writer);
397                 this.writeJdbcStoreAttributes(writer, store);
398                 this.writeStoreWriteBehind(writer, store);
399                 this.writeStoreProperties(writer, store);
400                 this.writeJDBCStoreTable(writer, Element.STRING_KEYED_TABLE, store, ModelKeys.STRING_KEYED_TABLE);
401                 this.writeJDBCStoreTable(writer, Element.BINARY_KEYED_TABLE, store, ModelKeys.BINARY_KEYED_TABLE);
402                 writer.writeEndElement();
403             }
404         }
405 
406         if (cache.get(ModelKeys.REMOTE_STORE).isDefined()) {
407             for (Property remoteStoreEntry : cache.get(ModelKeys.REMOTE_STORE).asPropertyList()) {
408                 ModelNode store = remoteStoreEntry.getValue();
409                 writer.writeStartElement(Element.REMOTE_STORE.getLocalName());
410                 // write identifier before other attributes
411                 ModelNode name = new ModelNode();
412                 name.get(ModelKeys.NAME).set(remoteStoreEntry.getName());
413                 RemoteStoreResource.NAME.marshallAsAttribute(name, false, writer);
414                 this.writeOptional(writer, Attribute.CACHE, store, ModelKeys.CACHE);
415                 this.writeOptional(writer, Attribute.HOTROD_WRAPPING, store, ModelKeys.HOTROD_WRAPPING);
416                 this.writeOptional(writer, Attribute.RAW_VALUES, store, ModelKeys.RAW_VALUES);
417                 this.writeOptional(writer, Attribute.SOCKET_TIMEOUT, store, ModelKeys.SOCKET_TIMEOUT);
418                 this.writeOptional(writer, Attribute.TCP_NO_DELAY, store, ModelKeys.TCP_NO_DELAY);
419                 this.writeStoreAttributes(writer, store);
420                 this.writeStoreWriteBehind(writer, store);
421                 this.writeStoreProperties(writer, store);
422                 for (ModelNode remoteServer: store.require(ModelKeys.REMOTE_SERVERS).asList()) {
423                     writer.writeStartElement(Element.REMOTE_SERVER.getLocalName());
424                     writer.writeAttribute(Attribute.OUTBOUND_SOCKET_BINDING.getLocalName(), remoteServer.get(ModelKeys.OUTBOUND_SOCKET_BINDING).asString());
425                     writer.writeEndElement();
426                 }
427                 writer.writeEndElement();
428             }
429         }
430 
431         if (cache.get(ModelKeys.REST_STORE).isDefined()) {
432             for (Property restStoreEntry : cache.get(ModelKeys.REST_STORE).asPropertyList()) {
433                 ModelNode store = restStoreEntry.getValue();
434                 writer.writeStartElement(Element.REST_STORE.getLocalName());
435                 // write identifier before other attributes
436                 ModelNode name = new ModelNode();
437                 name.get(ModelKeys.NAME).set(restStoreEntry.getName());
438                 RestStoreResource.NAME.marshallAsAttribute(name, false, writer);
439                 this.writeOptional(writer, Attribute.APPEND_CACHE_NAME_TO_PATH, store, ModelKeys.APPEND_CACHE_NAME_TO_PATH);
440                 this.writeOptional(writer, Attribute.PATH, store, ModelKeys.PATH);
441 
442                 this.writeStoreAttributes(writer, store);
443                 this.writeStoreWriteBehind(writer, store);
444                 this.writeStoreProperties(writer, store);
445 
446                 if (store.hasDefined(ModelKeys.CONNECTION_POOL)) {
447                     ModelNode pool = store.get(ModelKeys.CONNECTION_POOL);
448                     writer.writeStartElement(Element.CONNECTION_POOL.getLocalName());
449                     this.writeOptional(writer, Attribute.CONNECTION_TIMEOUT, pool, ModelKeys.CONNECTION_TIMEOUT);
450                     this.writeOptional(writer, Attribute.MAX_CONNECTIONS_PER_HOST, pool, ModelKeys.MAX_CONNECTIONS_PER_HOST);
451                     this.writeOptional(writer, Attribute.MAX_TOTAL_CONNECTIONS, pool, ModelKeys.MAX_TOTAL_CONNECTIONS);
452                     this.writeOptional(writer, Attribute.BUFFER_SIZE, pool, ModelKeys.BUFFER_SIZE);
453                     this.writeOptional(writer, Attribute.SOCKET_TIMEOUT, pool, ModelKeys.SOCKET_TIMEOUT);
454                     this.writeOptional(writer, Attribute.TCP_NO_DELAY, pool, ModelKeys.TCP_NO_DELAY);
455                     writer.writeEndElement();
456                 }
457 
458                 for (ModelNode remoteServer: store.require(ModelKeys.REMOTE_SERVERS).asList()) {
459                     writer.writeStartElement(Element.REMOTE_SERVER.getLocalName());
460                     writer.writeAttribute(Attribute.OUTBOUND_SOCKET_BINDING.getLocalName(), remoteServer.get(ModelKeys.OUTBOUND_SOCKET_BINDING).asString());
461                     writer.writeEndElement();
462                 }
463 
464                 writer.writeEndElement();
465             }
466         }
467 
468         if (cache.get(ModelKeys.INDEXING).isDefined()|| cache.get(ModelKeys.INDEXING_PROPERTIES).isDefined()){
469             writer.writeStartElement(Element.INDEXING.getLocalName());
470             CacheResource.INDEXING.marshallAsAttribute(cache, writer);
471             CacheResource.INDEXING_AUTO_CONFIG.marshallAsAttribute(cache, writer);
472             CacheResource.INDEXING_PROPERTIES.marshallAsElement(cache,writer);
473             writer.writeEndElement();
474         }
475 
476         if (cache.get(ModelKeys.BACKUP).isDefined()) {
477             writer.writeStartElement(Element.BACKUPS.getLocalName());
478             for (Property property : cache.get(ModelKeys.BACKUP).asPropertyList()) {
479                 writer.writeStartElement(Element.BACKUP.getLocalName());
480                 writer.writeAttribute(Attribute.SITE.getLocalName(), property.getName());
481                 ModelNode backup = property.getValue();
482                 BackupSiteResource.FAILURE_POLICY.marshallAsAttribute(backup, writer);
483                 BackupSiteResource.STRATEGY.marshallAsAttribute(backup, writer);
484                 BackupSiteResource.REPLICATION_TIMEOUT.marshallAsAttribute(backup, writer);
485                 BackupSiteResource.ENABLED.marshallAsAttribute(backup, writer);
486                 if (backup.hasDefined(ModelKeys.TAKE_BACKUP_OFFLINE_AFTER_FAILURES)
487                         || backup.hasDefined(ModelKeys.TAKE_BACKUP_OFFLINE_MIN_WAIT)) {
488                     writer.writeStartElement(Element.TAKE_OFFLINE.getLocalName());
489                     BackupSiteResource.TAKE_OFFLINE_AFTER_FAILURES.marshallAsAttribute(backup, writer);
490                     BackupSiteResource.TAKE_OFFLINE_MIN_WAIT.marshallAsAttribute(backup, writer);
491                     writer.writeEndElement();
492                 }
493                 if (backup.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME).isDefined()) {
494                     ModelNode stateTransfer = backup.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME);
495                     if (stateTransfer.hasDefined(ModelKeys.CHUNK_SIZE)
496                           || stateTransfer.hasDefined(ModelKeys.TIMEOUT)
497                           || stateTransfer.hasDefined(ModelKeys.MAX_RETRIES)
498                           || stateTransfer.hasDefined(ModelKeys.WAIT_TIME)) {
499                        writer.writeStartElement(Element.STATE_TRANSFER.getLocalName());
500                        BackupSiteStateTransferResource.STATE_TRANSFER_CHUNK_SIZE.marshallAsAttribute(stateTransfer, writer);
501                        BackupSiteStateTransferResource.STATE_TRANSFER_TIMEOUT.marshallAsAttribute(stateTransfer, writer);
502                        BackupSiteStateTransferResource.STATE_TRANSFER_MAX_RETRIES.marshallAsAttribute(stateTransfer, writer);
503                        BackupSiteStateTransferResource.STATE_TRANSFER_WAIT_TIME.marshallAsAttribute(stateTransfer, writer);
504                        writer.writeEndElement();
505                     }
506                 }
507                 writer.writeEndElement();
508             }
509             writer.writeEndElement();
510         }
511 
512         if (cache.get(ModelKeys.REMOTE_CACHE).isDefined() || cache.get(ModelKeys.REMOTE_SITE).isDefined()) {
513             writer.writeStartElement(Element.BACKUP_FOR.getLocalName());
514             CacheResource.REMOTE_CACHE.marshallAsAttribute(cache, writer);
515             CacheResource.REMOTE_SITE.marshallAsAttribute(cache, writer);
516             writer.writeEndElement();
517         }
518 
519         if (cache.get(ModelKeys.LEVELDB_STORE).isDefined()) {
520             for (Property levelDbStoreEntry : cache.get(ModelKeys.LEVELDB_STORE).asPropertyList()) {
521                 ModelNode store = levelDbStoreEntry.getValue();
522                 writer.writeStartElement(Element.LEVELDB_STORE.getLocalName());
523                 // write identifier before other attributes
524                 ModelNode name = new ModelNode();
525                 name.get(ModelKeys.NAME).set(levelDbStoreEntry.getName());
526                 LevelDBStoreResource.NAME.marshallAsAttribute(name, false, writer);
527                 this.writeOptional(writer, Attribute.RELATIVE_TO, store, ModelKeys.RELATIVE_TO);
528                 this.writeOptional(writer, Attribute.PATH, store, ModelKeys.PATH);
529                 this.writeOptional(writer, Attribute.BLOCK_SIZE, store, ModelKeys.BLOCK_SIZE);
530                 this.writeOptional(writer, Attribute.CACHE_SIZE, store, ModelKeys.CACHE_SIZE);
531                 this.writeOptional(writer, Attribute.CLEAR_THRESHOLD, store, ModelKeys.CLEAR_THRESHOLD);
532                 this.writeStoreAttributes(writer, store);
533                 this.writeStoreExpiration(writer, store);
534                 this.writeStoreCompression(writer, store);
535                 this.writeStoreImplementation(writer, store);
536                 this.writeStoreProperties(writer, store);
537                 writer.writeEndElement();
538             }
539         }
540 
541     }

543     private void writeListAsAttribute(XMLExtendedStreamWriter writer, Attribute attribute, ModelNode node, String key) throws XMLStreamException {
544         if (node.hasDefined(key)) {
545             StringBuilder result = new StringBuilder() ;
546             ModelNode list = node.get(key);
547             if (list.isDefined() && list.getType() == ModelType.LIST) {
548                 List<ModelNode> nodeList = list.asList();
549                 for (int i = 0; i < nodeList.size(); i++) {
550                     result.append(nodeList.get(i).asString());
551                     if (i < nodeList.size()-1) {
552                         result.append(" ");
553                     }
554                 }
555                 writer.writeAttribute(attribute.getLocalName(), result.toString());
556             }
557         }
558     }
