29  /**
30   * * Test to use with a profiler to profile replication.  To be used in conjunction with ProfileSlaveTest.
31   * <p/>
32   * Typical usage pattern:
33   * <p/>
34   * 1.  Start a single test method in ProfileSlaveTest.  This will block until you kill it. 2.  Start the corresponding
35   * test in this class, with the same name, in a different JVM, and attached to a profiler. 3.  Profile away!
36   * <p/>
37   *
38   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)

104    protected void init() {
105       long startTime = System.currentTimeMillis();
106       log.warn("Starting init() phase");
107       fqns.clear();
108       for (int i = 0; i < MAX_OVERALL_NODES; i++) {
109          Fqn fqn;
110          do {
111             fqn = FqnGenerator.createRandomFqn(MAX_DEPTH);
112          }
113          while (fqns.contains(fqn));
114 
115          if (i % 100 == 0) log.warn("Generated " + i + " fqns");
116          fqns.add(fqn);
117       }
118       System.gc();
119       long duration = System.currentTimeMillis() - startTime;
120       log.warn("Finished init() phase.  " + printDuration(duration));
121    }

123    protected void startup() {
124       long startTime = System.currentTimeMillis();
125       log.warn("Starting cache");
126       cache.start();
127       long duration = System.currentTimeMillis() - startTime;
128       log.warn("Started cache.  " + printDuration(duration));
129    }

132    private void warmup() throws InterruptedException {
133       long startTime = System.currentTimeMillis();
134       TaskRunner runner = new TaskRunner(NUM_THREADS);
135       log.warn("Starting warmup");
136       // creates all the Fqns since this can be expensive and we don't really want to measure this (for now)
137       for (final Fqn fqn : fqns) {
138          runner.execute(new Runnable() {
139             public void run() {
140                try {
141                   // this will create the necessary nodes.
142                   cache.put(fqn, "key", Collections.emptyMap());
143                }
144                catch (Exception e) {
145                   log.warn("Caught Exception", e);
146                }
147             }
148          });
149       }
150 
151       // loop through WARMUP_LOOPS gets and puts for JVM optimisation
152       for (int i = 0; i < WARMUP_LOOPS; i++) {
153          runner.execute(new Runnable() {
154             public void run() {
155                try {
156                   Fqn fqn = Generator.getRandomElement(fqns);
157                   DummyTransactionManager.getInstance().begin();
158                   cache.get(fqn, "key");
159                   DummyTransactionManager.getInstance().commit();
160                   DummyTransactionManager.getInstance().begin();
161                   cache.put(fqn, "key", "Value");
162                   DummyTransactionManager.getInstance().commit();
163                   DummyTransactionManager.getInstance().begin();
164                   cache.remove(fqn, "key");
165                   DummyTransactionManager.getInstance().commit();
166                }
167                catch (Exception e) {
168                   log.warn("Caught Exception", e);
169                }
170             }
171 
172          });
173       }
174 
175       runner.stop();
176 
177       long duration = System.currentTimeMillis() - startTime;
178       log.warn("Finished warmup.  " + printDuration(duration));
179       //cache.removeNode(Fqn.ROOT);
180 //      cache.stop();
181 
182       startup();
183    }

185    private void doTest() throws Exception {
186       TaskRunner runner = new TaskRunner(NUM_THREADS);
187 
188       log.warn("Starting test");
189       int i;
190       long print = NUM_OPERATIONS / 10;
191 
192       AtomicLong durationPuts = new AtomicLong();
193       AtomicLong durationGets = new AtomicLong();
194       AtomicLong durationRemoves = new AtomicLong();
195 
196       long stElapsed = System.nanoTime();
197       for (i = 0; i < NUM_OPERATIONS; i++) {
198          MyRunnable r = null;
199          switch (i % 3) {
200             case 0:
201                r = new Putter(i, durationPuts);
202                break;
203             case 1:
204                r = new Getter(i, durationGets);
205                break;
206             case 2:
207                r = new Remover(i, durationRemoves);
208                break;
209          }
210          if (i % print == 0)
211             log.warn("processing iteration " + i);
212          runner.execute(r);
213 //         if (USE_SLEEP) TestingUtil.sleepRandom(MAX_RANDOM_SLEEP_MILLIS);
214          if (USE_SLEEP) TestingUtil.sleepThread(MAX_RANDOM_SLEEP_MILLIS);
215       }
216       log.warn("Finished generating runnables; awaiting executor completion");
217       // wait for executors to complete!
218       runner.stop();
219 
220       // wait up to 1 sec for each call?
221       long elapsedTimeNanos = System.nanoTime() - stElapsed;
222 
223       log.warn("Finished test.  " + printDuration((long) toMillis(elapsedTimeNanos)));
224       log.warn("Throughput: " + ((double) NUM_OPERATIONS * 1000 / toMillis(elapsedTimeNanos)) + " operations per second (roughly equal numbers of PUT, GET and REMOVE)");
225       log.warn("Average GET time: " + printAvg(durationGets.get()));
226       log.warn("Average PUT time: " + printAvg(durationPuts.get()));
227       log.warn("Average REMOVE time: " + printAvg(durationRemoves.get()));
228    }

230    private String printAvg(long totalNanos) {
231       double nOps = (double) (NUM_OPERATIONS / 3);
232       double avg = ((double) totalNanos) / nOps;
233       double avgMicros = avg / 1000;
234       return avgMicros + " Âµs";
235    }

237    private double toMillis(long nanos) {
238       return ((double) nanos / (double) 1000000);
239    }

241    enum Mode {
242       PUT, GET, REMOVE
243    }
244 
245    private abstract class MyRunnable implements Runnable {
246       int id;
247       Mode mode;
248       AtomicLong duration;
249 
250       public void run() {
251          Fqn fqn = Generator.getRandomElement(fqns);
252          long d = 0, st;
253          try {
254             switch (mode) {
255                case PUT:
256                   Object value = Generator.getRandomString();
257 
258                   st = System.nanoTime();
259                   cache.put(fqn, "key", value);
260                   d = System.nanoTime() - st;
261                   break;
262                case GET:
263                   st = System.nanoTime();
264                   cache.get(fqn, "key");
265                   d = System.nanoTime() - st;
266                   break;
267                case REMOVE:
268                   st = System.nanoTime();
269                   cache.remove(fqn, "key");
270                   d = System.nanoTime() - st;
271                   break;
272             }
273          }
274          catch (Exception e) {
275             d = 0;
276          }
277          duration.getAndAdd(d);
278 
279       }
280 
281    }
282 
283    private class Putter extends MyRunnable {
284       private Putter(int id, AtomicLong duration) {
285          this.id = id;
286          this.duration = duration;
287          mode = Mode.PUT;
288       }
289    }
290 
291    private class Getter extends MyRunnable {
292       private Getter(int id, AtomicLong duration) {
293          this.id = id;
294          this.duration = duration;
295          mode = Mode.GET;
296       }
297    }
298 
299    private class Remover extends MyRunnable {
300       private Remover(int id, AtomicLong duration) {
301          this.id = id;
302          this.duration = duration;
303          mode = Mode.REMOVE;
304       }
305    }
306 
307    protected String printDuration(long duration) {
308       if (duration > 2000) {
309          double dSecs = ((double) duration / (double) 1000);
310          return "Duration: " + dSecs + " seconds";
311       } else {
312          return "Duration: " + duration + " millis";
313       }
314    }
