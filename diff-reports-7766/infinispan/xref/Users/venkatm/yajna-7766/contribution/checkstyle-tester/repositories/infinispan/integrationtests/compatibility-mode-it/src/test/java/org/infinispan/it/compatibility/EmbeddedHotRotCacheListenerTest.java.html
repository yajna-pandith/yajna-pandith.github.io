43     public void testLoadingAndStoringEventsHotRod1() {
44        Cache<String, String> embedded = cacheFactory.getEmbeddedCache();
45        RemoteCache<String, String> remote = cacheFactory.getHotRodCache();
46  
47        TestCacheListener l = new TestCacheListener();
48        embedded.addListener(l);
49  
50        assertTrue(l.created.isEmpty());
51        assertTrue(l.removed.isEmpty());
52        assertTrue(l.modified.isEmpty());
53        assertTrue(l.visited.isEmpty());
54  
55        remote.put("k", "v");
56        assertEquals(1, l.createdCounter);
57        assertEquals("v", l.created.get("k"));
58        assertTrue(l.removed.isEmpty());
59        assertTrue(l.modified.isEmpty());
60        assertTrue(l.visited.isEmpty());
61  
62        remote.put("key", "value");
63        assertEquals(2, l.createdCounter);
64        assertTrue(l.removed.isEmpty());
65        assertTrue(l.modified.isEmpty());
66        assertTrue(l.visited.isEmpty());
67  
68        remote.put("key", "modifiedValue");
69        assertEquals(2, l.createdCounter);
70        assertTrue(l.removed.isEmpty());
71        assertEquals(1, l.modifiedCounter);
72        assertEquals("modifiedValue", l.modified.get("key"));
73        assertTrue(l.visited.isEmpty());
74  
75        remote.replace("k", "replacedValue");
76        assertEquals(2, l.createdCounter);
77        assertTrue(l.removed.isEmpty());
78        assertEquals(2, l.modifiedCounter);
79        assertEquals("replacedValue", l.modified.get("k"));
80        assertTrue(l.visited.isEmpty());
81  
82        //resetting so don't have to type "== 2" etc. all over again
83        l.reset();
84  
85        Set<String> keys = new HashSet<>();
86        keys.add("k");
87        keys.add("key");
88        Map<String, String> results = remote.getAll(keys);
89        assertTrue(l.created.isEmpty());
90        assertTrue(l.removed.isEmpty());
91        assertTrue(l.modified.isEmpty());
92        assertEquals(2, l.visitedCounter);
93        assertEquals("replacedValue", l.visited.get("k"));
94        assertEquals("modifiedValue", l.visited.get("key"));
95  
96        l.reset();
97  
98        remote.remove("key");
99        assertTrue(l.created.isEmpty());
100       assertEquals(1, l.removedCounter);
101       assertEquals("modifiedValue", l.removed.get("key"));
102       assertTrue(l.modified.isEmpty());
103 
104       l.reset();
105 
106       String value = remote.get("k");
107       assertTrue(l.created.isEmpty());
108       assertTrue(l.removed.isEmpty());
109       assertTrue(l.modified.isEmpty());
110       assertEquals(1, l.visitedCounter);
111       assertEquals("replacedValue", l.visited.get("k"));
112 
113       l.reset();
114    }

155    public void testLoadingAndStoringWithVersionEventsHotRod() {
156       Cache<String, String> embedded = cacheFactory.getEmbeddedCache();
157       RemoteCache<String, String> remote = cacheFactory.getHotRodCache();
158 
159       TestCacheListener l = new TestCacheListener();
160       embedded.addListener(l);
161 
162       remote.put("key", "value");
163       VersionedValue oldVersionedValue = remote.getVersioned("key");
164       assertEquals("value", oldVersionedValue.getValue());
165       assertEquals(1, l.createdCounter);
166       assertTrue(l.modified.isEmpty());
167       assertTrue(l.removed.isEmpty());
168       assertEquals(1, l.visitedCounter);
169       assertEquals("value", l.visited.get("key"));
170 
171       remote.put("key2", "value2");
172       remote.put("key", "outOfVersionValue");
173       VersionedValue newVersionedValue = remote.getVersioned("key2");
174 
175       l.reset();
176 
177       remote.removeWithVersion("key", oldVersionedValue.getVersion());
178       assertTrue(l.created.isEmpty());
179       assertTrue(l.modified.isEmpty());
180       assertTrue(l.removed.isEmpty());
181       assertEquals(1, l.visitedCounter);
182 
183       l.reset();
184 
185       remote.removeWithVersion("key2", newVersionedValue.getVersion());
186       assertTrue(l.created.isEmpty());
187       assertTrue(l.modified.isEmpty());
188       assertEquals(1, l.removedCounter);
189       assertEquals("value2", l.removed.get("key2"));
190       assertEquals(1, l.visitedCounter);
191 
192       remote.put("newKey", "willBeOutOfDate");
193       VersionedValue oldVersionedValueToBeReplaced = remote.getVersioned("newKey");
194       remote.put("newKey", "changedValue");
195 
196       l.reset();
197 
198       remote.replaceWithVersion("newKey", "tryingToChangeButShouldNotSucceed", oldVersionedValueToBeReplaced.getVersion());
199       assertTrue(l.created.isEmpty());
200       assertTrue(l.modified.isEmpty());
201       assertTrue(l.removed.isEmpty());
202       assertTrue(l.visited.isEmpty());
203 
204       remote.put("newKey2", "willBeSuccessfullyChanged");
205       VersionedValue newVersionedValueToBeReplaced = remote.getVersioned("newKey2");
206 
207       l.reset();
208 
209       remote.replaceWithVersion("newKey2", "successfulChange", newVersionedValueToBeReplaced.getVersion());
210       assertTrue(l.created.isEmpty());
211       assertEquals(1, l.modifiedCounter);
212       assertEquals("successfulChange", l.modified.get("newKey2"));
213       assertTrue(l.removed.isEmpty());
214       assertTrue(l.visited.isEmpty());
215 
216       l.reset();
217    }

219    public void testLoadingAndStoringAsyncEventsHotRod() throws InterruptedException, ExecutionException, TimeoutException {
220       Cache<String, String> embedded = cacheFactory.getEmbeddedCache();
221       RemoteCache<String, String> remote = cacheFactory.getHotRodCache();
222 
223       TestCacheListener l = new TestCacheListener();
224       embedded.addListener(l);
225 
226       NotifyingFuture future = remote.putAsync("k", "v");
227       future.get(60, TimeUnit.SECONDS);
228       assertEquals(1, l.createdCounter);
229       assertEquals("v", l.created.get("k"));
230       assertTrue(l.removed.isEmpty());
231       assertTrue(l.modified.isEmpty());
232       assertTrue(l.visited.isEmpty());
233 
234       NotifyingFuture future2 = remote.putAsync("key", "value");
235       future2.get(60, TimeUnit.SECONDS);
236       assertEquals(2, l.createdCounter);
237       assertTrue(l.removed.isEmpty());
238       assertTrue(l.modified.isEmpty());
239       assertTrue(l.visited.isEmpty());
240 
241       NotifyingFuture future3 = remote.putAsync("key", "modifiedValue");
242       future3.get(60, TimeUnit.SECONDS);
243       assertEquals(2, l.createdCounter);
244       assertTrue(l.removed.isEmpty());
245       assertEquals(1, l.modifiedCounter);
246       assertEquals("modifiedValue", l.modified.get("key"));
247       assertTrue(l.visited.isEmpty());
248 
249       NotifyingFuture future4 = remote.replaceAsync("k", "replacedValue");
250       future4.get(60, TimeUnit.SECONDS);
251       assertEquals(2, l.createdCounter);
252       assertTrue(l.removed.isEmpty());
253       assertEquals(2, l.modifiedCounter);
254       assertEquals("replacedValue", l.modified.get("k"));
255       assertTrue(l.visited.isEmpty());

258       l.reset();
259

262       assertTrue(l.created.isEmpty());

266 
267       l.reset();
268 
269       NotifyingFuture future6 = remote.getAsync("k");

271       assertTrue(l.created.isEmpty());

273       assertTrue(l.modified.isEmpty());
274       assertEquals(1, l.visitedCounter);
275       assertEquals("replacedValue", l.visited.get("k"));
