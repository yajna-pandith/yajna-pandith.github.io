37     /**
38      * This test is to verify then when a new node joins and a cluster listener is installed after the cluster listener
39      * request is finished that it finds it.
40      *
41      * Node 1, 2 & 3 exists
42      * Node 2 adds cluster listener - knows to send to listener to Node 1
43      * Node 4 starts up
44      * Node 4 asks Node 1 for listeners (gets none)
45      * Node 1 receives Node 2 listener
46      *
47      * Test needs to verify in this case that Nod3 3 gets the listener from Node 2
48      */
49     @Test
50     public void testMemberJoinsWhileClusterListenerInstalled() throws TimeoutException, InterruptedException,
51                                                                       ExecutionException {
52        Cache<Object, String> cache0 = cache(0, CACHE_NAME);
53        final Cache<Object, String> cache1 = cache(1, CACHE_NAME);
54  
55        CheckPoint checkPoint = new CheckPoint();
56        waitUntilListenerInstalled(cache0, checkPoint);
57        // We don't want this blocking
58        checkPoint.triggerForever("post_add_listener_release_" + cache0);
59  
60        final ClusterListener clusterListener = new ClusterListener();
61        Future<Void> future = fork(new Callable<Void>() {
62  
63           @Override
64           public Void call() throws Exception {
65  
66              cache1.addListener(clusterListener);
67              return null;
68           }
69        });
70  
71        // Now wait until the listener is about to be installed on cache1
72        checkPoint.awaitStrict("pre_add_listener_invoked_" + cache0, 10, TimeUnit.SECONDS);
73  
74        // First we add the new node, but block the dist exec execution
75        log.info("Adding a new node ..");
76        addClusterEnabledCacheManager(builderUsed);
77        log.info("Added a new node");
78  
79        // Now wait for cache3 to come up fully
80        waitForClusterToForm(CACHE_NAME);
81  
82        Cache<Object, String> cache3 = cache(3, CACHE_NAME);
83  
84        // Finally let the listener be added
85        checkPoint.triggerForever("pre_add_listener_release_" + cache0);
86        future.get(10, TimeUnit.SECONDS);
87  
88        MagicKey key = new MagicKey(cache3);
89        verifySimpleInsertion(cache3, key, FIRST_VALUE, null, clusterListener, FIRST_VALUE);
90     }

261    /**
262     * The premise is the same as {@link AbstractClusterListenerDistAddListenerTest#testNodeJoiningAndStateNodeDiesWithExistingClusterListener}.
263     * This also has the twist of the fact that the node who dies is also has the cluster listener.  This test makes sure
264     * that the subsequent node asked for cluster listeners hasn't yet got the view change and still has the cluster
265     * listener in it.  Also the requesting node should have the view change before installing.
266     * @throws TimeoutException
267     * @throws InterruptedException
268     * @throws ExecutionException
269     */
270    @Test(enabled = false, description = "Test may not be doable, check TODO in test")
271    public void testNodeJoiningAndStateNodeDiesWhichHasClusterListener() throws TimeoutException,
272                                                                                InterruptedException,
273                                                                                ExecutionException {
274       Cache<Object, String> cache0 = cache(0, CACHE_NAME);
275       Cache<Object, String> cache1 = cache(1, CACHE_NAME);
276       Cache<Object, String> cache2 = cache(2, CACHE_NAME);
277 
278       int initialCache0ListenerSize = cache0.getAdvancedCache().getListeners().size();
279       int initialCache1ListenerSize = cache1.getAdvancedCache().getListeners().size();
280       int initialCache2ListenerSize = cache2.getAdvancedCache().getListeners().size();
281 
282       ClusterListener clusterListener = new ClusterListener();
283       cache0.addListener(clusterListener);
284 
285       assertEquals(cache0.getAdvancedCache().getListeners().size(), initialCache0ListenerSize + 1);
286       assertEquals(cache1.getAdvancedCache().getListeners().size(), initialCache1ListenerSize + 1);
287       assertEquals(cache2.getAdvancedCache().getListeners().size(), initialCache2ListenerSize + 1);
288 
289       // Make sure cache0 will  be the one will get the cluster listener request
290       assertEquals(manager(0).getAddress(), manager(0).getMembers().get(0));
291 
292       CheckPoint checkPoint = new CheckPoint();
293 
294       waitUntilRequestingListeners(cache0, checkPoint);
295       checkPoint.triggerForever("post_cluster_listeners_release_" + cache0);
296       waitUntilViewChangeOccurs(manager(1), "manager1", checkPoint);
297 
298       // We let the first view change occur just fine on cache1 (this will be the addition of cache3).
299       // What we want to block is the second one which is the removal of cache0
300       checkPoint.trigger("pre_view_listener_release_" + "manager1");
301 
302       // First we add the new node, but block the dist exec execution
303       log.info("Adding a new node ..");
304       addClusterEnabledCacheManager(builderUsed);
305       log.info("Added a new node");
306 
307       waitUntilViewChangeOccurs(manager(3), "manager3", checkPoint);
308       // We don't want to block the view listener change on cache3
309       checkPoint.trigger("pre_view_listener_release_" + "manager3");
310 
311       Future<Cache<Object, String>> future = fork(new Callable<Cache<Object, String>>() {
312          @Override
313          public Cache<Object, String> call() throws Exception {
314             return cache(3, CACHE_NAME);
315          }
316       });
317 
318       // Wait for view change to occur on cache1 for the addition of cache3
319       // Note we haven't triggered the view change for cache1 for the following removal yet
320       checkPoint.awaitStrict("post_view_listener_invoked_" + "manager1", 10, TimeUnit.SECONDS);
321       // Wait for the cluster listener request to come into cache0 which is from cache3
322       checkPoint.awaitStrict("pre_cluster_listeners_invoked_" + cache0, 10, TimeUnit.SECONDS);
323 
324       // Now we kill cache0 while it is processing the request from cache3, which will in turn force it to ask cache1
325       log.info("Killing node 0 ..");
326       TestingUtil.killCacheManagers(manager(0));
327       cacheManagers.remove(0);
328       log.info("Node 0 killed");
329 
330       // TODO: need a away to verify that the response was sent back to cache3 before releasing the next line.  However
331       // with no reference to cache I don't think this is possible.  If this can be fixed then test can be reenabled
332       Cache<Object, String> cache3 = future.get(10, TimeUnit.SECONDS);
333 
334       // Now we can finally let the view change complete on cache1
335       checkPoint.triggerForever("pre_view_listener_release_" + "manager1");
336 
337       // Now wait for cache3 to come up fully
338       TestingUtil.blockUntilViewsReceived(60000, false, cache1, cache2);
339       TestingUtil.waitForRehashToComplete(cache1, cache2);
340 
341       MagicKey key = new MagicKey(cache3);
342       cache3.put(key, FIRST_VALUE);
343 
344       assertEquals(cache1.getAdvancedCache().getListeners().size(), initialCache1ListenerSize);
345       assertEquals(cache2.getAdvancedCache().getListeners().size(), initialCache2ListenerSize);
346 
347       // Since we can't get a reliable start count, make sure no RemoteClusterListener is present which is added for
348       // a cluster listener
349       for (Object listener : cache3.getAdvancedCache().getListeners()) {
350          assertFalse(listener instanceof RemoteClusterListener);
351       }
352    }

354    protected void waitUntilRequestingListeners(final Cache<?, ?> cache, final CheckPoint checkPoint) {
355       StateProvider sp = TestingUtil.extractComponent(cache, StateProvider.class);
356       final Answer<Object> forwardedAnswer = AdditionalAnswers.delegatesTo(sp);
357       StateProvider mockProvider = mock(StateProvider.class, withSettings().defaultAnswer(forwardedAnswer));
358       doAnswer(new Answer() {
359          @Override
360          public Object answer(InvocationOnMock invocation) throws Throwable {
361             // Wait for main thread to sync up
362             checkPoint.trigger("pre_cluster_listeners_invoked_" + cache);
363             // Now wait until main thread lets us through
364             checkPoint.awaitStrict("pre_cluster_listeners_release_" + cache, 10, TimeUnit.SECONDS);
365 
366             try {
367                return forwardedAnswer.answer(invocation);
368             } finally {
369                // Wait for main thread to sync up
370                checkPoint.trigger("post_cluster_listeners_invoked_" + cache);
371                // Now wait until main thread lets us through
372                checkPoint.awaitStrict("post_cluster_listeners_release_" + cache, 10, TimeUnit.SECONDS);
373             }
374          }
375       }).when(mockProvider).getClusterListenersToInstall();
376       TestingUtil.replaceComponent(cache, StateProvider.class, mockProvider, true);
377    }
