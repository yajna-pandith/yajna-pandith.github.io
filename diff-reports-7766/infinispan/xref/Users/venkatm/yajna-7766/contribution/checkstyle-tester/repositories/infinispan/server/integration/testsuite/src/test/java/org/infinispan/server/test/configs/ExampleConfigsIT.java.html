130     @WithRunningServer({@RunningServer(name = "hotrod-rolling-upgrade-2"),@RunningServer(name = "hotrod-rolling-upgrade-1")})
131     public void testHotRodRollingUpgrades() throws Exception {
132         // Target node
133         MBeanServerConnectionProvider provider1;
134         // Source node
135         MBeanServerConnectionProvider provider2;
136 
137         RemoteInfinispanMBeans s2 = createRemotes("hotrod-rolling-upgrade-2", "local", DEFAULT_CACHE_NAME);
138         final RemoteCache<Object, Object> c2 = createCache(s2);
139 
140         c2.put("key1", "value1");
141         assertEquals("value1", c2.get("key1"));
142 
143         for (int i = 0; i < 50; i++) {
144             c2.put("keyLoad" + i, "valueLoad" + i);
145         }
146 
147         controller.start("hotrod-rolling-upgrade-1");
148 
149         RemoteInfinispanMBeans s1 = createRemotes("hotrod-rolling-upgrade-1", "local", DEFAULT_CACHE_NAME);
150         final RemoteCache<Object, Object> c1 = createCache(s1);
151 
152         assertEquals("Can't access etries stored in source node (target's RemoteCacheStore).", "value1", c1.get("key1"));
153 
154         provider1 = new MBeanServerConnectionProvider(s1.server.getHotrodEndpoint().getInetAddress().getHostName(),
155                                                       SERVER1_MGMT_PORT);
156         provider2 = new MBeanServerConnectionProvider(s2.server.getHotrodEndpoint().getInetAddress().getHostName(),
157                                                       SERVER2_MGMT_PORT);
158 
159         final ObjectName rollMan = new ObjectName("jboss.infinispan:type=Cache," + "name=\"default(local)\","
160                 + "manager=\"local\"," + "component=RollingUpgradeManager");
161 
162         invokeOperation(provider2, rollMan.toString(), "recordKnownGlobalKeyset", new Object[]{}, new String[]{});
163 
164         invokeOperation(provider1, rollMan.toString(), "synchronizeData", new Object[]{"hotrod"},
165                 new String[]{"java.lang.String"});
166 
167         invokeOperation(provider1, rollMan.toString(), "disconnectSource", new Object[]{"hotrod"},
168                 new String[]{"java.lang.String"});
169 
170         // is source (RemoteCacheStore) really disconnected?
171         c2.put("disconnected", "source");
172         assertEquals("Can't obtain value from cache1 (source node).", "source", c2.get("disconnected"));
173         assertNull("Source node entries should NOT be accessible from target node (after RCS disconnection)",
174                 c1.get("disconnected"));
175 
176         // all entries migrated?
177         assertEquals("Entry was not successfully migrated.", "value1", c1.get("key1"));
178         for (int i = 0; i < 50; i++) {
179             assertEquals("Entry was not successfully migrated.", "valueLoad" + i, c1.get("keyLoad" + i));
180         }
181     }

184     @WithRunningServer({@RunningServer(name = "rest-rolling-upgrade-2"),@RunningServer(name = "rest-rolling-upgrade-1")})
185     public void testRestRollingUpgrades() throws Exception {
186         final int PORT_OFFSET = 100;
187         // target node
188         MBeanServerConnectionProvider provider1;
189         // Source node
190         MBeanServerConnectionProvider provider2;
191 
192         controller.start("rest-rolling-upgrade-2");
193         try {
194             RemoteInfinispanMBeans s2 = createRemotes("rest-rolling-upgrade-2", "local", DEFAULT_CACHE_NAME);
195             RESTHelper.addServer(s2.server.getRESTEndpoint().getInetAddress().getHostName(), s2.server.getRESTEndpoint().getContextPath());
196 
197             post(fullPathKey(0, DEFAULT_CACHE_NAME, "key1", PORT_OFFSET), "data", "text/html");
198             get(fullPathKey(0, DEFAULT_CACHE_NAME, "key1", PORT_OFFSET), "data");
199 
200             for (int i = 0; i < 50; i++) {
201                 post(fullPathKey(0, DEFAULT_CACHE_NAME, "keyLoad" + i, PORT_OFFSET), "valueLoad" + i, "text/html");
202             }
203 
204             controller.start("rest-rolling-upgrade-1");
205 
206             RemoteInfinispanMBeans s1 = createRemotes("rest-rolling-upgrade-1", "local", DEFAULT_CACHE_NAME);
207             RESTHelper.addServer(s1.server.getRESTEndpoint().getInetAddress().getHostName(), s1.server.getRESTEndpoint().getContextPath());
208 
209             get(fullPathKey(1, DEFAULT_CACHE_NAME, "key1", 0), "data");
210 
211             provider1 = new MBeanServerConnectionProvider(s1.server.getRESTEndpoint().getInetAddress().getHostName(),
212                                                           SERVER1_MGMT_PORT);
213             provider2 = new MBeanServerConnectionProvider(s2.server.getRESTEndpoint().getInetAddress().getHostName(),
214                                                           SERVER2_MGMT_PORT);
215 
216             final ObjectName rollMan = new ObjectName("jboss.infinispan:type=Cache," + "name=\"default(local)\","
217                                                             + "manager=\"local\"," + "component=RollingUpgradeManager");
218 
219             invokeOperation(provider2, rollMan.toString(), "recordKnownGlobalKeyset", new Object[]{}, new String[]{});
220 
221             invokeOperation(provider1, rollMan.toString(), "synchronizeData", new Object[]{"rest"},
222                             new String[]{"java.lang.String"});
223 
224             invokeOperation(provider1, rollMan.toString(), "disconnectSource", new Object[]{"rest"},
225                             new String[]{"java.lang.String"});
226 
227             post(fullPathKey(0, DEFAULT_CACHE_NAME, "disconnected", PORT_OFFSET), "source", "application/text");
228 
229             //Source node entries should NOT be accessible from target node
230             get(fullPathKey(1, DEFAULT_CACHE_NAME, "disconnected", 0), HttpServletResponse.SC_NOT_FOUND);
231 
232             //All remaining entries migrated?
233             for (int i = 0; i < 50; i++) {
234                 get(fullPathKey(1, DEFAULT_CACHE_NAME, "keyLoad" + i, 0), "valueLoad" + i);
235             }
236         } finally {
237             if (controller.isStarted("rest-rolling-upgrade-1")) {
238                 controller.stop("rest-rolling-upgrade-1");
239             }
240             if (controller.isStarted("rest-rolling-upgrade-2")) {
241                 controller.stop("rest-rolling-upgrade-2");
242             }
243         }

248     public void testCompatibilityModeConfig() throws Exception {
249         MemcachedClient memcachedClient = null;
250         CloseableHttpClient restClient = null;
251         try {
252             RemoteInfinispanMBeans s1 = createRemotes("standalone-compatibility-mode", "local", DEFAULT_CACHE_NAME);
253             RemoteCache<Object, Object> s1Cache = createCache(s1);
254             restClient = HttpClients.createDefault();
255             String restUrl = "http://" + s1.server.getHotrodEndpoint().getInetAddress().getHostName() + ":8080"
256                     + s1.server.getRESTEndpoint().getContextPath() + "/" + DEFAULT_CACHE_NAME;
257             memcachedClient = new MemcachedClient(s1.server.getMemcachedEndpoint().getInetAddress().getHostName(), s1.server
258                     .getMemcachedEndpoint().getPort());
259             String key = "1";
260 
261             // 1. Put with Hot Rod
262             assertEquals(null, s1Cache.withFlags(Flag.FORCE_RETURN_VALUE).put(key, "v1".getBytes()));
263             assertArrayEquals("v1".getBytes(), (byte[]) s1Cache.get(key));
264 
265             // 2. Get with REST
266             HttpGet get = new HttpGet(restUrl + "/" + key);
267             HttpResponse getResponse = restClient.execute(get);
268             assertEquals(HttpServletResponse.SC_OK, getResponse.getStatusLine().getStatusCode());
269             assertArrayEquals("v1".getBytes(), EntityUtils.toByteArray(getResponse.getEntity()));
270 
271             // 3. Get with Memcached
272             assertArrayEquals("v1".getBytes(), readWithMemcachedAndDeserialize(key, memcachedClient));
273 
274             key = "2";
275 
276             // 1. Put with REST
277             HttpPut put = new HttpPut(restUrl + "/" + key);
278             put.setEntity(new ByteArrayEntity("<hey>ho</hey>".getBytes(), ContentType.APPLICATION_OCTET_STREAM));
279             HttpResponse putResponse = restClient.execute(put);
280             assertEquals(HttpServletResponse.SC_OK, putResponse.getStatusLine().getStatusCode());
281 
282             // 2. Get with Hot Rod
283             assertArrayEquals("<hey>ho</hey>".getBytes(), (byte[]) s1Cache.get(key));
284 
285             // 3. Get with Memcached
286             assertArrayEquals("<hey>ho</hey>".getBytes(), readWithMemcachedAndDeserialize(key, memcachedClient));
287         } finally {
288             if (restClient != null) {
289                 restClient.close();
290             }
291             if (memcachedClient != null) {
292                 memcachedClient.close();
293             }
294         }
295     }

451     private void verifyTopologyHinting(String container1, String container2, String container3, String manager, String cache) {
452         RemoteInfinispanMBeans s1 = createRemotes(container1, manager, cache);
453         RemoteInfinispanMBeans s2 = createRemotes(container2, manager, cache);
454         RemoteInfinispanMBeans s3 = createRemotes(container3, manager, cache);
455 
456         RemoteCache<Object, Object> s1Cache = createCache(s1);
457         RemoteCache<Object, Object> s2Cache = createCache(s2);
458         RemoteCache<Object, Object> s3Cache = createCache(s3);
459 
460         assertEquals(3, s1.manager.getClusterSize());
461         assertEquals(3, s2.manager.getClusterSize());
462         assertEquals(3, s3.manager.getClusterSize());
463         int total_elements = 0;
464         s1Cache.clear();
465         s2Cache.clear();
466         s3Cache.clear();
467 
468         long s0Entries = 0;
469         long s1Entries = 0;
470         long s2Entries = 0;
471         List<String> s1Bulk = new ArrayList<String>();
472         List<String> s2Bulk = new ArrayList<String>();
473 
474         // By using topology information we divide our 3 nodes into 2 groups and generate enough elements so there
475         // is at least 1 element in each group and at least 5 elements total,
476         // and keep track of elements that went to server 2 and 3
477         while (s0Entries == 0 || s1Entries == 0 || s2Entries == 0 || total_elements < 5) {
478             s1Cache.put("machine" + total_elements, "machine");
479 
480             if (s1Entries + 1 == s2.cache.getNumberOfEntries()) {
481                 s1Bulk.add("machine" + total_elements);
482             }
483             if (s2Entries + 1 == s3.cache.getNumberOfEntries()) {
484                 s2Bulk.add("machine" + total_elements);
485             }
486 
487             total_elements++;
488             s1Entries = s2.cache.getNumberOfEntries();
489             s2Entries = s3.cache.getNumberOfEntries();
490             s0Entries = s1.cache.getNumberOfEntries();
491             if (total_elements > 10)
492                 break; // in case something goes wrong - do not cycle forever
493         }
494 
495         assertTrue("Unexpected number of entries in server1: " + s0Entries, s0Entries > 0);
496         assertTrue("Unexpected number of entries in server2: " + s1Entries, s1Entries > 0);

640     private void doPutGetRemove(final RemoteInfinispanMBeans sMain, final RemoteInfinispanMBeans sRemoteStore) {
641         assertEquals(0, sMain.cache.getNumberOfEntries());
642         assertEquals(0, sRemoteStore.cache.getNumberOfEntries());
643         RemoteCache<Object, Object> sMainCache = createCache(sMain);
644         RemoteCache<Object, Object> sRemoteStoreCache = createCache(sRemoteStore);
645 
646         for (int i = 0; i < 1100; i++) {
647             sMainCache.put("key" + i, "value" + i);
648         }
649         assertTrue(sMain.cache.getNumberOfEntries() <= 1000);
650         eventually(new Condition() {
651             @Override
652             public boolean isSatisfied() throws Exception {
653                 log.debug("Num entries: Main cache: " + sMain.cache.getNumberOfEntries() + " Remote store: "
654                         + sRemoteStore.cache.getNumberOfEntries() + " Total: "
655                         + (sMain.cache.getNumberOfEntries() + sRemoteStore.cache.getNumberOfEntries()));
656                 return sMain.cache.getNumberOfEntries() + sRemoteStore.cache.getNumberOfEntries() == 1100;
657             }
658         }, 10000);
659 
660         for (int i = 0; i < 1100; i++) {
661             assertNotNull(sMainCache.get("key" + i));
662             sMainCache.remove("key" + i);
663             assertNull(sMainCache.get("key" + i));
664         }
665         assertEquals(0, sMain.cache.getNumberOfEntries());
666         assertEquals(0, sRemoteStore.cache.getNumberOfEntries());
667         sMainCache.clear();
668         sRemoteStoreCache.clear();
669     }
