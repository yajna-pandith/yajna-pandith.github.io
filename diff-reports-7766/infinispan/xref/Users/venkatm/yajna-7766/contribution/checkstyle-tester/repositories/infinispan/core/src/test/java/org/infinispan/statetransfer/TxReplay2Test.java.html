45  /**
46   * Tests that a transaction is replayed only once if the commit command is received twice.
47   *
48   * @author Dan Berindei
49   * @since 7.0
50   */
51  @Test(groups = "functional", testName = "statetransfer.TxReplay2Test")
52  public class TxReplay2Test extends MultipleCacheManagersTest {
53     private static final String VALUE = "value";
54  
55     ControlledConsistentHashFactory consistentHashFactory = new ControlledConsistentHashFactory(0, 1, 2);
56  
57     public void testReplay() throws Exception {
58        final StateSequencer sequencer = new StateSequencer();
59        sequencer.logicalThread("tx", "tx:before_prepare_replay", "tx:resume_prepare_replay", "tx:mark_tx_completed");
60        sequencer.logicalThread("sim", "sim:before_extra_commit", "sim:during_extra_commit", "sim:after_extra_commit");
61        sequencer.order("tx:before_prepare_replay", "sim:before_extra_commit");
62        sequencer.order("sim:during_extra_commit", "tx:resume_prepare_replay");
63        sequencer.order("sim:after_extra_commit", "tx:mark_tx_completed");
64  
65        final Object key = "key";
66        assertEquals(Arrays.asList(address(0), address(1), address(2)), advancedCache(0).getDistributionManager().locate(key));
67        Cache<Object, Object> primaryOwnerCache = cache(0);
68        final Cache<Object, Object> newBackupOwnerCache = cache(3);
69        final CountingInterceptor newBackupCounter = CountingInterceptor.inject(newBackupOwnerCache);
70        final CountingInterceptor primaryCounter = CountingInterceptor.inject(primaryOwnerCache);
71        final CountingInterceptor oldBackup2Counter = CountingInterceptor.inject(cache(2));
72  
73        advanceOnInterceptor(sequencer, newBackupOwnerCache, CallInterceptor.class,
74              matchCommand(PrepareCommand.class).matchCount(0).build())
75              .before("tx:before_prepare_replay", "tx:resume_prepare_replay");
76        advanceOnInterceptor(sequencer, newBackupOwnerCache, TransactionSynchronizerInterceptor.class,
77              matchCommand(CommitCommand.class).matchCount(1).build())
78              .before("sim:during_extra_commit");
79        advanceOnInboundRpc(sequencer, newBackupOwnerCache, matchCommand(TxCompletionNotificationCommand.class).build())
80              .before("tx:mark_tx_completed");
81  
82        final DummyTransactionManager transactionManager = (DummyTransactionManager) tm(0);
83        transactionManager.begin();
84        primaryOwnerCache.put(key, VALUE);
85  
86        final DummyTransaction transaction = transactionManager.getTransaction();
87        TransactionTable transactionTable0 = TestingUtil.getTransactionTable(primaryOwnerCache);
88        final GlobalTransaction gtx = transactionTable0.getLocalTransaction(transaction).getGlobalTransaction();
89        transaction.runPrepare();
90        assertEquals("Wrong transaction status before killing backup owner.",
91              Status.STATUS_PREPARED, transaction.getStatus());
92  
93        // Now, we kill cache(1). the transaction is prepared in cache(1) and it should be transferred to cache(2)
94        killMember(1);
95  
96        final int currentTopologyId = TestingUtil.extractComponentRegistry(primaryOwnerCache).getStateTransferManager().getCacheTopology().getTopologyId();
97        Future<Object> secondCommitFuture = fork(new Callable<Object>() {
98           @Override
99           public Object call() throws Exception {
100             // Wait for the commit command to block replaying the prepare on the new backup
101             sequencer.advance("sim:before_extra_commit");
102             // And try to run another commit command
103             CommitCommand command = new CommitCommand(newBackupOwnerCache.getName(), gtx);
104             command.setTopologyId(currentTopologyId);
105             CommandsFactory cf = TestingUtil.extractCommandsFactory(newBackupOwnerCache);
106             cf.initializeReplicableCommand(command, true);
107             try {
108                command.perform(null);
109             } catch (Throwable throwable) {
110                throw new CacheException(throwable);
111             }
112 
113             sequencer.advance("sim:after_extra_commit");
114             return null;
115          }
116       });
117 
118       checkIfTransactionExists(newBackupOwnerCache);
119       assertEquals("Wrong transaction status after killing backup owner.",
120             Status.STATUS_PREPARED, transaction.getStatus());
121       transaction.runCommit(false);
122 
123       secondCommitFuture.get(10, SECONDS);
124 
125       assertNoTransactions();
126 
127       assertEquals("Wrong number of prepares!", 2, newBackupCounter.numberPrepares.get());
128       assertEquals("Wrong number of commits!", 2, newBackupCounter.numberCommits.get());
129       assertEquals("Wrong number of rollbacks!", 0, newBackupCounter.numberRollbacks.get());
130 
131       assertEquals("Wrong number of prepares!", 2, oldBackup2Counter.numberPrepares.get());
132       assertEquals("Wrong number of commits!", 1, oldBackup2Counter.numberCommits.get());
133       assertEquals("Wrong number of rollbacks!", 0, oldBackup2Counter.numberRollbacks.get());
134 
135       // We only count remote commands, and there shouldn't be any on the primary/originator
136       assertEquals("Wrong number of prepares!", 0, primaryCounter.numberPrepares.get());
137       assertEquals("Wrong number of commits!", 0, primaryCounter.numberCommits.get());
138       assertEquals("Wrong number of rollbacks!", 0, primaryCounter.numberRollbacks.get());
139 
140       checkKeyInDataContainer(key);
141    }

143    @Override
144    protected void createCacheManagers() throws Throwable {
145       ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
146       builder.transaction()
147             .useSynchronization(false)
148             .transactionManagerLookup(new DummyTransactionManagerLookup())
149             .recovery().disable();
150       builder.clustering()
151             .hash().numOwners(3).numSegments(1).consistentHashFactory(consistentHashFactory)
152             .stateTransfer().fetchInMemoryState(true);
153       createClusteredCaches(4, builder);
154    }

156    private void checkKeyInDataContainer(Object key) {
157       for (Cache<Object, Object> cache : caches()) {
158          DataContainer container = cache.getAdvancedCache().getDataContainer();
159          InternalCacheEntry entry = container.get(key);
160          assertNotNull("Cache '" + address(cache) + "' does not contain key!", entry);
161          assertEquals("Cache '" + address(cache) + "' has wrong value!", VALUE, entry.getValue());
162       }
163    }

165    private void checkIfTransactionExists(Cache<Object, Object> cache) {
166       TransactionTable table = TestingUtil.extractComponent(cache, TransactionTable.class);
167       assertFalse("Expected a remote transaction.", table.getRemoteTransactions().isEmpty());
168    }

170    private static class CountingInterceptor extends CommandInterceptor {
171       private static Log log = LogFactory.getLog(CountingInterceptor.class);
172 
173       //counters
174       private final AtomicInteger numberPrepares = new AtomicInteger(0);
175       private final AtomicInteger numberCommits = new AtomicInteger(0);
176       private final AtomicInteger numberRollbacks = new AtomicInteger(0);
177 
178       @Override
179       public Object visitPrepareCommand(TxInvocationContext ctx, PrepareCommand command) throws Throwable {
180          if (!ctx.isOriginLocal()) {
181             log.debugf("Received remote prepare for transaction %s", command.getGlobalTransaction());
182             numberPrepares.incrementAndGet();
183          }
184          return invokeNextInterceptor(ctx, command);
185       }
186 
187       @Override
188       public Object visitCommitCommand(TxInvocationContext ctx, CommitCommand command) throws Throwable {
189          if (!ctx.isOriginLocal()) {
190             log.debugf("Received remote commit for transaction %s", command.getGlobalTransaction());
191             numberCommits.incrementAndGet();
192          }
193          return invokeNextInterceptor(ctx, command);
194       }
195 
196       @Override
197       public Object visitRollbackCommand(TxInvocationContext ctx, RollbackCommand command) throws Throwable {
198          if (!ctx.isOriginLocal()) {
199             log.debugf("Received remote rollback for transaction %s", command.getGlobalTransaction());
200             numberRollbacks.incrementAndGet();
201          }
202          return invokeNextInterceptor(ctx, command);
203       }
204 
205       public static CountingInterceptor inject(Cache cache) {
206          InterceptorChain chain = TestingUtil.extractComponent(cache, InterceptorChain.class);
207          if (chain.containsInterceptorType(CountingInterceptor.class)) {
208             return (CountingInterceptor) chain.getInterceptorsWithClass(CountingInterceptor.class).get(0);
209          }
210          CountingInterceptor interceptor = new CountingInterceptor();
211          chain.addInterceptorBefore(interceptor, CallInterceptor.class);
212          return interceptor;
213       }
214    }
