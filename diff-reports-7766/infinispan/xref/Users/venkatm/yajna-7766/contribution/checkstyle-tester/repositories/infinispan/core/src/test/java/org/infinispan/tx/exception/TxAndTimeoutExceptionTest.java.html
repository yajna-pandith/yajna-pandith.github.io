34     @Override
35     protected EmbeddedCacheManager createCacheManager() throws Exception {
36        ConfigurationBuilder config = getDefaultStandaloneCacheConfig(true);
37        config
38           .transaction().lockingMode(LockingMode.PESSIMISTIC)
39           .locking().useLockStriping(false).lockAcquisitionTimeout(1000);
40        EmbeddedCacheManager cm = TestCacheManagerFactory.createCacheManager(config);
41        cache = cm.getCache();
42        return cm;
43     }

82     private void assertExpectedBehavior(CacheOperation op) throws Exception {
83        LockManager lm = TestingUtil.extractLockManager(cache);
84        TransactionTable txTable = TestingUtil.getTransactionTable(cache);
85        TransactionManager tm = cache.getAdvancedCache().getTransactionManager();
86        tm.begin();
87        cache.put("k1", "v1");
88        Transaction k1LockOwner = tm.suspend();
89        assertTrue(lm.isLocked("k1"));
90  
91        assertEquals(1, txTable.getLocalTxCount());
92        tm.begin();
93        cache.put("k2", "v2");
94        assertTrue(lm.isLocked("k2"));
95        assertEquals(2, txTable.getLocalTxCount());
96        assert tm.getTransaction() != null;
97        try {
98           op.execute();
99           fail("TimeoutException expected.");
100       } catch (TimeoutException e) {
101          //expected
102       }
103 
104       //make sure that locks acquired by that tx were released even before the transaction is rolled back, the tx object
105       //was marked for rollback
106       Transaction transaction = tm.getTransaction();
107       assertNotNull(transaction);
108       assertEquals(Status.STATUS_MARKED_ROLLBACK, transaction.getStatus());
109       assertFalse(lm.isLocked("k2"));
110       assertTrue(lm.isLocked("k1"));
111       try {
112          cache.put("k3", "v3");
113          fail("IllegalStateException expected.");
114       } catch (IllegalStateException e) {
115          //expected
116       }
117       assertEquals(2, txTable.getLocalTxCount());
118 
119       //now the TM is expected to rollback the tx
120       tm.rollback();
121       assertEquals(1, txTable.getLocalTxCount());
122 
123       tm.resume(k1LockOwner);
124       tm.commit();
125 
126       //now test that the other tx works as expected
127       assertEquals(0, txTable.getLocalTxCount());
128       assertEquals("v1", cache.get("k1"));
129       assertFalse(lm.isLocked("k1"));
130       assertEquals(0, txTable.getLocalTxCount());
131    }
