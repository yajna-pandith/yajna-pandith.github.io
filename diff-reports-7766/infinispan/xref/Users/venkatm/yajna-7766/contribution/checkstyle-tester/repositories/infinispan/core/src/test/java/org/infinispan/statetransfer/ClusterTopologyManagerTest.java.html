214    public void testClusterRecoveryWithRebalance() throws Exception {
215       // Compute the merge coordinator by sorting the JGroups addresses, the same way MERGE2/3 do
216       List<Address> members = new ArrayList<Address>(manager(0).getMembers());
217       Collections.sort(members);
218       Address mergeCoordAddress = members.get(0);
219       log.debugf("The merge coordinator will be %s", mergeCoordAddress);
220       EmbeddedCacheManager mergeCoordManager = manager(mergeCoordAddress);
221       int mergeCoordIndex = cacheManagers.indexOf(mergeCoordManager);
222 
223       // create the partitions
224       log.debugf("Splitting the cluster in three");
225       d1.setDiscardAll(true);
226       d2.setDiscardAll(true);
227       d3.setDiscardAll(true);
228 
229       // wait for the coordinator to be separated (don't care about the others)
230       TestingUtil.blockUntilViewsReceived(30000, false, c1);
231       TestingUtil.blockUntilViewsReceived(30000, false, c2);
232       TestingUtil.blockUntilViewsReceived(30000, false, c3);
233       TestingUtil.waitForRehashToComplete(c1);
234       TestingUtil.waitForRehashToComplete(c2);
235       TestingUtil.waitForRehashToComplete(c3);
236 
237       // Disable DISCARD *only* on the merge coordinator
238       if (mergeCoordIndex == 0) d1.setDiscardAll(false);
239       if (mergeCoordIndex == 1) d2.setDiscardAll(false);
240       if (mergeCoordIndex == 2) d3.setDiscardAll(false);
241 
242       int viewIdAfterSplit = mergeCoordManager.getTransport().getViewId();
243       final CheckPoint checkpoint = new CheckPoint();
244       blockRebalanceStart(mergeCoordManager, checkpoint, 2);
245 
246       final EmbeddedCacheManager cm4 = addClusterEnabledCacheManager(defaultConfig,
247             new TransportFlags().withFD(true).withMerge(true));
248       blockRebalanceStart(cm4, checkpoint, 2);
249       // Force the initialization of the transport
250       cm4.getCache(OTHER_CACHE_NAME);
251       Future<Cache<Object,Object>> cacheFuture = fork(new Callable<Cache<Object, Object>>() {
252          @Override
253          public Cache<Object, Object> call() throws Exception {
254             return cm4.getCache(CACHE_NAME);
255          }
256       });
257 
258       log.debugf("Waiting for the REBALANCE_START command to reach the merge coordinator");
259       checkpoint.awaitStrict("rebalance_" + Arrays.asList(mergeCoordAddress, cm4.getAddress()), 10, TimeUnit.SECONDS);
260 
261       // merge the partitions
262       log.debugf("Merging the cluster partitions");
263       d1.setDiscardAll(false);
264       d2.setDiscardAll(false);
265       d3.setDiscardAll(false);
266 
267       // wait for the JGroups merge
268       long startTime = System.currentTimeMillis();
269       TestingUtil.blockUntilViewsReceived(30000, cacheManagers);
270       TestingUtil.waitForRehashToComplete(caches(CACHE_NAME));
271 
272       // unblock the REBALANCE_START command
273       log.debugf("Unblocking the REBALANCE_START command on the coordinator");
274       checkpoint.triggerForever("merge");
275 
276       // wait for the 4th cache to finish joining
277       Cache<Object, Object> c4 = cacheFuture.get(30, TimeUnit.SECONDS);
278       TestingUtil.waitForRehashToComplete(c1, c2, c3, c4);
279 
280       long endTime = System.currentTimeMillis();
281       log.debugf("Merge took %s", Util.prettyPrintTime(endTime - startTime));
282       assert endTime - startTime < 30000 : "Merge took too long: " + Util.prettyPrintTime(endTime - startTime);
283 
284       // Check that another node can join
285       ConfigurationBuilder defaultConfig = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
286       EmbeddedCacheManager cm5 = addClusterEnabledCacheManager(defaultConfig,
287             new TransportFlags().withFD(true).withMerge(true));
288       Cache<Object, Object> c5 = cm5.getCache(CACHE_NAME);
289       TestingUtil.blockUntilViewsReceived(30000, true, c1, c2, c3, c4, c5);
290       TestingUtil.waitForRehashToComplete(c1, c2, c3, c4, c5);
291    }

293    protected void blockRebalanceStart(final EmbeddedCacheManager manager, final CheckPoint checkpoint, final int numMembers)
294          throws InterruptedException {
295       final LocalTopologyManager localTopologyManager = TestingUtil.extractGlobalComponent(manager,
296             LocalTopologyManager.class);
297       LocalTopologyManager spyLocalTopologyManager = spy(localTopologyManager);
298       doAnswer(new Answer<Object>() {
299          @Override
300          public Object answer(InvocationOnMock invocation) throws Throwable {
301             CacheTopology topology = (CacheTopology) invocation.getArguments()[1];
302             List<Address> members = topology.getMembers();
303             checkpoint.trigger("rebalance_" + members);
304             if (members.size() == numMembers) {
305                log.debugf("Blocking the REBALANCE_START command with members %s on %s", members, manager.getAddress());
306                checkpoint.awaitStrict("merge", 30, TimeUnit.SECONDS);
307             }
308             return invocation.callRealMethod();
309          }
310       }).when(spyLocalTopologyManager).handleRebalance(eq(CACHE_NAME), any(CacheTopology.class), anyInt(),
311             any(Address.class));
312       TestingUtil.replaceComponent(manager, LocalTopologyManager.class, spyLocalTopologyManager, true);
313    }

316       // Block the GET_STATUS command on node 2
317       final LocalTopologyManager localTopologyManager2 = TestingUtil.extractGlobalComponent(manager(1),
318             LocalTopologyManager.class);
319       final CheckPoint checkpoint = new CheckPoint();
320       LocalTopologyManager spyLocalTopologyManager2 = spy(localTopologyManager2);
321       final CacheTopology initialTopology = localTopologyManager2.getCacheTopology(CACHE_NAME);
322       doAnswer(new Answer<Object>() {
323          @Override
324          public Object answer(InvocationOnMock invocation) throws Throwable {
325             int viewId = (Integer) invocation.getArguments()[0];
326             checkpoint.trigger("GET_STATUS_" + viewId);
327             log.debugf("Blocking the GET_STATUS command on the new coordinator");
328             checkpoint.awaitStrict("3 left", 10, TimeUnit.SECONDS);
329             return invocation.callRealMethod();
330          }
331       }).when(spyLocalTopologyManager2).handleStatusRequest(anyInt());
332       doAnswer(new Answer<Object>() {
333          @Override
334          public Object answer(InvocationOnMock invocation) throws Throwable {
335             CacheTopology topology = (CacheTopology) invocation.getArguments()[1];
336             if (topology.getRebalanceId() == initialTopology.getRebalanceId() + 1) {
337                log.debugf("Discarding CH update command %s", topology);
338                return null;
339             }
340             return invocation.callRealMethod();
341          }
342       }).when(spyLocalTopologyManager2).handleTopologyUpdate(eq(CACHE_NAME), any(CacheTopology.class),
343             any(AvailabilityMode.class), anyInt(), any(Address.class));
344       doAnswer(new Answer<Object>() {
345          @Override
346          public Object answer(InvocationOnMock invocation) throws Throwable {
347             CacheTopology topology = (CacheTopology) invocation.getArguments()[1];
348             if (topology.getRebalanceId() == initialTopology.getRebalanceId() + 2) {
349                log.debugf("Discarding rebalance command %s", topology);
350                return null;
351             }
352             return invocation.callRealMethod();
353          }
354       }).when(spyLocalTopologyManager2).handleRebalance(eq(CACHE_NAME), any(CacheTopology.class), anyInt(),
355             any(Address.class));
356       TestingUtil.replaceComponent(manager(1), LocalTopologyManager.class, spyLocalTopologyManager2, true);
357 
358       // Node 1 (the coordinator) dies. Node 2 becomes coordinator and tries to call GET_STATUS
359       log.debugf("Killing coordinator");
360       manager(0).stop();
361       TestingUtil.blockUntilViewsReceived(30000, false, manager(1), manager(2));
362 
363       // Wait for the GET_STATUS command and stop node 3 abruptly
364       int viewId = manager(1).getTransport().getViewId();
365       checkpoint.awaitStrict("GET_STATUS_" + viewId, 10, TimeUnit.SECONDS);
366       d3.setDiscardAll(true);
367       manager(2).stop();
368       TestingUtil.blockUntilViewsReceived(30000, false, manager(1));
369       checkpoint.triggerForever("3 left");
370 
371       // Wait for node 2 to install a view with only itself and unblock the GET_STATUS command
372       TestingUtil.waitForRehashToComplete(c2);
373    }

375    public void testAbruptLeaveAfterGetStatus2() throws TimeoutException, InterruptedException {
376       // Similar to testAbruptLeaveAfterGetStatus, but also test that delayed CacheTopologyControlCommands
377       // are handled properly.
378       // After node 2 becomes the coordinator and the GET_STATUS command is unblocked, it normally installs these topologies:
379       // 1. The recovered topology with 2 and 3 as members (topologyId = initial topologyId + 1, rebalanceId = initial rebalanceId + 1)
380       // 2. A topology starting the rebalance with 2 and 3 (topologyId = initial topologyId + 2, rebalanceId = initial rebalanceId + 2)
381       // 3. A topology with 2 as the only member, but still with a pending CH (topologyId = initialTopologyId + 3, rebalanceId = initial rebalanceId + 2)
382       // 4. A topology ending the rebalance (topologyId = initialTopologyId + 4, rebalanceId = initial rebalanceId + 2)
383       // Sometimes node 2 can confirm the rebalance before receiving the topology in step 3, in which case step 3 is skipped.
384       // We discard the topologies from steps 1 and 2, to test that the topology update in step 3 is enough to start and finish the rebalance.
385 
386       // Block the GET_STATUS command on node 2
387       final LocalTopologyManager localTopologyManager2 = TestingUtil.extractGlobalComponent(manager(1),
388             LocalTopologyManager.class);
389       final CheckPoint checkpoint = new CheckPoint();

391       final CacheTopology initialTopology = localTopologyManager2.getCacheTopology(CACHE_NAME);
392       doAnswer(new Answer<Object>() {
393          @Override
394          public Object answer(InvocationOnMock invocation) throws Throwable {
395             int viewId = (Integer) invocation.getArguments()[0];
396             checkpoint.trigger("GET_STATUS_" + viewId);
397             log.debugf("Blocking the GET_STATUS command on the new coordinator");
398             checkpoint.awaitStrict("3 left", 10, TimeUnit.SECONDS);
399             return invocation.callRealMethod();
400          }
401       }).when(spyLocalTopologyManager2).handleStatusRequest(anyInt());
402 
403       // Delay the first CH_UPDATE after the merge so that it arrives after
404       doAnswer(new Answer<Object>() {
405          @Override
406          public Object answer(InvocationOnMock invocation) throws Throwable {
407             CacheTopology topology = (CacheTopology) invocation.getArguments()[1];
408             if (topology.getRebalanceId() == initialTopology.getRebalanceId() + 1) {
409                log.debugf("Discarding CH update command %s", topology);
410                return null;
411             }
412             return invocation.callRealMethod();
413          }
414       }).when(spyLocalTopologyManager2).handleTopologyUpdate(eq(CACHE_NAME), any(CacheTopology.class),
415             any(AvailabilityMode.class), anyInt(), any(Address.class));
416       doAnswer(new Answer<Object>() {
417          @Override
418          public Object answer(InvocationOnMock invocation) throws Throwable {
419             CacheTopology topology = (CacheTopology) invocation.getArguments()[1];
420             if (topology.getRebalanceId() == initialTopology.getRebalanceId() + 2) {
421                log.debugf("Discarding rebalance command %s", topology);
422                return null;
423             }
424             return invocation.callRealMethod();
425          }
426       }).when(spyLocalTopologyManager2).handleRebalance(eq(CACHE_NAME), any(CacheTopology.class), anyInt(),
427             any(Address.class));
428       TestingUtil.replaceComponent(manager(1), LocalTopologyManager.class, spyLocalTopologyManager2, true);
429 
430       // Node 1 (the coordinator) dies. Node 2 becomes coordinator and tries to call GET_STATUS
431       log.debugf("Killing coordinator");

433       TestingUtil.blockUntilViewsReceived(30000, false, manager(1), manager(2));
434
