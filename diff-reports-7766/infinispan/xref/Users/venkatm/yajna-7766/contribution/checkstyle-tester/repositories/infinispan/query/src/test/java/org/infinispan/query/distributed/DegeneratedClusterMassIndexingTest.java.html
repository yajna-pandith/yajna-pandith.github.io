9   import org.infinispan.query.Search;
10  import org.infinispan.query.SearchManager;
11  import org.infinispan.query.queries.faceting.Car;

13  import org.testng.annotations.Test;
14  
15  import static org.junit.Assert.assertEquals;
16  
17  /**
18   * @author anistor@redhat.com
19   * @since 6.1
20   */
21  @Test(groups = "functional", testName = "query.distributed.DegeneratedClusterMassIndexingTest")
22  public class DegeneratedClusterMassIndexingTest extends MultipleCacheManagersTest {
23  
24     @Override
25     protected void createCacheManagers() throws Throwable {
26        ConfigurationBuilder cfg = getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, true);
27        cfg.indexing()
28              .index(Index.ALL)
29              .addProperty("default.directory_provider", "ram")
30              .addProperty("lucene_version", "LUCENE_CURRENT");
31  
32        addClusterEnabledCacheManager(cfg);
33        waitForClusterToForm();
34     }
35  
36     public void testReindexing() throws Exception {
37        Cache cache = cache(0).getAdvancedCache().withFlags(Flag.SKIP_INDEXING);
38  
39        SearchManager searchManager = Search.getSearchManager(cache);
40        Query query = searchManager.buildQueryBuilderForClass(Car.class)
41              .get()
42              .keyword()
43              .onField("make")
44              .matching("ford")
45              .createQuery();
46  
47        cache.put("car1", new Car("ford", "white", 300));
48        cache.put("car2", new Car("ford", "blue", 300));
49        cache.put("car3", new Car("ford", "red", 300));
50  
51        // ensure these were not indexed
52        assertEquals(0, searchManager.getQuery(query, Car.class).getResultSize());
53  
54        //reindex
55        searchManager.getMassIndexer().start();
56  
57        // check that the indexing is complete immediately
58        assertEquals(3, searchManager.getQuery(query, Car.class).getResultSize());
59     }
60  }
