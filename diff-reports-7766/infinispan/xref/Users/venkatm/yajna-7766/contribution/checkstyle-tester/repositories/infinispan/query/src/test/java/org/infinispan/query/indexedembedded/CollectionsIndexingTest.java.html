27  @Test(groups = "functional", testName = "query.indexedembedded.CollectionsIndexingTest")
28  public class CollectionsIndexingTest extends SingleCacheManagerTest {
29  
30     private SearchManager qf;
31  
32     protected EmbeddedCacheManager createCacheManager() throws Exception {
33        ConfigurationBuilder cfg = getDefaultStandaloneCacheConfig(true);
34        cfg
35           .indexing()
36              .index(Index.ALL)
37              .addProperty("default.directory_provider", "ram")
38              .addProperty("lucene_version", "LUCENE_CURRENT");
39        return TestCacheManagerFactory.createCacheManager(cfg);
40     }
41  
42     @BeforeClass
43     public void prepareSearchFactory() throws Exception {
44        qf = Search.getSearchManager(cache);
45     }
46     
47     @AfterMethod
48     public void cleanupData() {
49        cache.clear();
50     }
51     
52     @Test
53     public void searchOnEmptyIndex() throws ParseException {
54        QueryParser queryParser = TestQueryHelperFactory.createQueryParser("countryName");
55        Query query = queryParser.parse("Italy");
56        List<Object> list = qf.getQuery(query, Country.class, City.class).list();
57        assertEquals(0, list.size());
58     }
59     
60     @Test
61     public void searchOnAllTypes() throws ParseException {
62        QueryParser queryParser = TestQueryHelperFactory.createQueryParser("countryName");
63        Query query = queryParser.parse("Italy");
64        Country italy = new Country();
65        italy.countryName = "Italy";
66        cache.put("IT", italy);
67        List<Object> list = qf.getQuery(query, Country.class, City.class).list();
68        assertEquals(1, list.size());
69        list = qf.getQuery(query).list();
70        assertEquals(1, list.size());
71        list = qf.getQuery( new MatchAllDocsQuery() ).list();
72        assertEquals(1, list.size());
73     }
74     
75     @Test
76     public void searchOnSimpleField() throws ParseException {
77        QueryParser queryParser = TestQueryHelperFactory.createQueryParser("countryName");
78        Query query = queryParser.parse("Italy");
79        Country italy = new Country();
80        italy.countryName = "Italy";
81        cache.put("IT", italy);
82        List<Object> list = qf.getQuery(query, Country.class, City.class).list();
83        assertEquals(1, list.size());
84     }
85     
86     @Test
87     public void searchOnEmbeddedField() throws ParseException {
88        QueryParser queryParser = TestQueryHelperFactory.createQueryParser("cities.name");
89        Query query = queryParser.parse("Newcastle");
90        
91        Country uk = new Country();
92        City london = new City();
93        london.name = "London";
94        City newcastle = new City();
95        newcastle.name = "Newcastle";
96        uk.countryName = "United Kingdom";
97        uk.cities.add(newcastle);
98        uk.cities.add(london);
99        
100       //verify behaviour on multiple insertions as well:
101       cache.put("UK", uk);
102       cache.put("UK", uk);
103       cache.put("UK", uk);
104       List<Object> list = qf.getQuery(query, Country.class, City.class).list();
105       assertEquals(1, list.size());
106       assertTrue(uk == list.get(0));
107    }
