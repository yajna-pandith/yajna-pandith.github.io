30  @Test(groups = "functional", testName = "statetransfer.MergeDuringReplaceTest")
31  @CleanupAfterMethod
32  public class MergeDuringReplaceTest extends MultipleCacheManagersTest {
33  
34     private DISCARD[] discard;
35  
36     @Override
37     protected void createCacheManagers() throws Throwable {
38        ConfigurationBuilder defaultConfig = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false);
39        createClusteredCaches(3, defaultConfig, new TransportFlags().withFD(true).withMerge(true));
40  
41        DISCARD d1 = TestingUtil.getDiscardForCache(cache(0));
42        d1.setExcludeItself(true);
43        DISCARD d2 = TestingUtil.getDiscardForCache(cache(1));
44        d2.setExcludeItself(true);
45        DISCARD d3 = TestingUtil.getDiscardForCache(cache(2));
46        d3.setExcludeItself(true);
47        discard = new DISCARD[]{d1, d2, d3};
48     }
49  
50     public void testMergeDuringReplace() throws Exception {
51        final String key = "myKey";
52        final String value = "myValue";
53  
54        cache(0).put(key, value);
55  
56        ConsistentHash ch = cache(0).getAdvancedCache().getComponentRegistry()
57              .getStateTransferManager().getCacheTopology().getCurrentCH();
58        List<Address> members = new ArrayList<Address>(ch.getMembers());
59        List<Address> owners = ch.locateOwners(key);
60        members.removeAll(owners);
61        int nonOwner = ch.getMembers().indexOf(members.get(0));
62        final Cache<Object, Object> c = cache(nonOwner);
63  
64        List<Cache<Object, Object>> partition1 = caches();
65        partition1.remove(c);
66  
67        ControlledRpcManager controlledRpcManager = new ControlledRpcManager(c.getAdvancedCache().getRpcManager());
68        TestingUtil.replaceComponent(c, RpcManager.class, controlledRpcManager, true);
69  
70        controlledRpcManager.blockBefore(ReplaceCommand.class);
71  
72        Future<Boolean> future = fork(new Callable<Boolean>() {
73           @Override
74           public Boolean call() throws Exception {
75              return c.replace(key, value, "myNewValue");
76           }
77        });
78  
79        discard[nonOwner].setDiscardAll(true);
80  
81        // wait for the partitions to form
82        TestingUtil.blockUntilViewsReceived(30000, false, partition1.get(0), partition1.get(1));
83        TestingUtil.blockUntilViewsReceived(30000, false, c);
84        TestingUtil.waitForRehashToComplete(partition1.get(0), partition1.get(1));
85        TestingUtil.waitForRehashToComplete(c);
86  
87        controlledRpcManager.stopBlocking();
88  
89        // Since the non owner didn't have the value before the split it can't do the replace correctly
90        assertEquals(future.get(10, TimeUnit.SECONDS), Boolean.FALSE);
91     }
