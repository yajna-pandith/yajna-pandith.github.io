21  @Test(testName = "container.versioning.VersionedReplStateTransferTest", groups = "functional")
22  @CleanupAfterMethod
23  public class VersionedReplStateTransferTest extends MultipleCacheManagersTest {
24  
25     ConfigurationBuilder builder;
26  
27     @Override
28     protected void createCacheManagers() throws Throwable {
29        builder = TestCacheManagerFactory.getDefaultCacheConfiguration(true);
30  
31        builder
32              .clustering()
33                 .cacheMode(CacheMode.REPL_SYNC)
34              .versioning()
35                 .enable()
36                 .scheme(VersioningScheme.SIMPLE)
37              .locking()
38                 .isolationLevel(IsolationLevel.REPEATABLE_READ)
39                 .writeSkewCheck(true)
40              .transaction()
41                 .lockingMode(LockingMode.OPTIMISTIC)
42                 .syncCommitPhase(true);
43  
44        amendConfig(builder);
45  
46        createCluster(builder, 2);
47        waitForClusterToForm();
48     }
49  
50     protected void amendConfig(ConfigurationBuilder builder) {
51     }
52  
53     public void testStateTransfer() throws Exception {
54        Cache<Object, Object> cache0 = cache(0);
55        Cache<Object, Object> cache1 = cache(1);
56  
57        cache0.put("hello", "world");
58  
59        assertEquals("world", cache0.get("hello"));
60        assertEquals("world", cache1.get("hello"));
61  
62        tm(1).begin();
63        assertEquals("world", cache1.get("hello"));
64        Transaction t = tm(1).suspend();
65  
66        // create a cache2
67        addClusterEnabledCacheManager(builder);
68        Cache<Object, Object> cache2 = cache(2);
69  
70        assertEquals("world", cache2.get("hello"));
71  
72        cacheManagers.get(0).stop();
73        cacheManagers.remove(0);
74        TestingUtil.waitForRehashToComplete(caches());
75  
76        // Cause a write skew
77        cache2.put("hello", "new world");
78  
79        tm(1).resume(t);
80        cache1.put("hello", "world2");
81  
82        try {
83           tm(1).commit();
84           fail("Write skew check should fail");
85        } catch (RollbackException expected) {
86           // Expected
87        }
88  
89        assertEquals("new world", cache1.get("hello"));
90        assertEquals("new world", cache2.get("hello"));
91     }
