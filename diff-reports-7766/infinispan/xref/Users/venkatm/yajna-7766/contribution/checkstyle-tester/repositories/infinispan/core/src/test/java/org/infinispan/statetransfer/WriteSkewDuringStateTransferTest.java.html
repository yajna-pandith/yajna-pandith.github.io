59     /*
60     Replicated TX cache with WSC, A, B are in cluster, C is joining
61     0. The current CH already contains A and B as owners, C is joining (is not primary owner of anything yet).
62  B is primary owner of K=V.
63     1. A sends PrepareCommand to B and C with put(K, V) (V is null on all nodes); //A has already received the
64  rebalance_start
65     2. C receives PrepareCommand and responds with no versions (it is not primary owner); //C has already received the
66  rebalance_start
67     3. topology changes on B - primary ownership of K is transferred to C; //B has already received the ch_update
68     4. B receives PrepareCommand, responds without K's version (it is not primary)
69     5. B forwards the Prepare to C as it sees that the command has lower topology ID
70     6. C responds to B's prepare with version of K; //at this point, C has received the ch_update
71     7. K version is not added to B's response, B responds to A
72     8. A finds out that topology has changed, forwards prepare to C; //A has received the ch_update
73     9. C responds to C's prepare with version of K
74     10. A receives C's response, but the versions are not added to transaction
75     11. A sends out CommitCommand missing version of K
76     12. all nodes record K=V without version as usual ImmortalCacheEntry
77     */

80      * See description above or ISPN-3738
81      */
82     public void testVersionsAfterStateTransfer() throws Exception {
83        assertClusterSize("Wrong cluster size", 2);
84        final Object key = "key1";
85        assertKeyOwnership(key, cache(1), cache(0));
86        final int currentTopologyId = currentTopologyId(cache(0));
87  
88        final ControlledRpcManager nodeARpcManager = replaceRpcManager(cache(0));
89        final NodeController nodeAController = setNodeControllerIn(cache(0));
90        setInitialPhaseForNodeA(nodeAController, currentTopologyId);
91        final NodeController nodeBController = setNodeControllerIn(cache(1));
92        setInitialPhaseForNodeB(nodeBController, currentTopologyId);
93        final NewNode nodeC = addNode(currentTopologyId);
94  
95        //node A thinks that node B is the primary owner. Node B is blocking the prepare command until it thinks that
96        //node C is the primary owner
97        nodeAController.topologyManager.waitToBlock(BlockingLocalTopologyManager.LatchType.CONSISTENT_HASH_UPDATE);
98        nodeARpcManager.blockAfter(VersionedPrepareCommand.class);
99        //node C thinks that node B is the primary owner.
100       //nodeC.controller.topologyManager.waitToBlock(BlockingLocalTopologyManager.LatchType.CONSISTENT_HASH_UPDATE);
101 
102       //after this waiting phase, node A thinks that node B is the primary owner, node B thinks that node C is the
103       // primary owner and node C thinks that node B is the primary owner
104       //lets execute the transaction...
105 
106       Future<Object> tx = executeTransaction(cache(0), key);
107 
108       //it waits until all nodes has replied. then, we change the topology ID and let it collect the responses.
109       nodeARpcManager.waitForCommandToBlock();
110       nodeAController.topologyManager.stopBlocking(BlockingLocalTopologyManager.LatchType.CONSISTENT_HASH_UPDATE);
111       awaitForTopology(currentTopologyId + 2, cache(0));
112 
113       nodeARpcManager.stopBlocking();
114       assertNull("Wrong put() return value.", tx.get());
115 
116       nodeAController.topologyManager.stopBlockingAll();
117       nodeBController.topologyManager.stopBlockingAll();
118       nodeC.controller.topologyManager.stopBlockingAll();
119 
120       nodeC.joinerFuture.get();
121 
122       awaitForTopology(currentTopologyId + 2, cache(0));
123       awaitForTopology(currentTopologyId + 2, cache(1));
124       awaitForTopology(currentTopologyId + 2, cache(2));
125 
126       assertKeyVersionInDataContainer(key, cache(1), cache(2));
127       cache(0).put(key, "v2");
128    }

179    private NewNode addNode(final int currentTopologyId) {
180       final NewNode newNode = new NewNode();
181       ConfigurationBuilder builder = configuration();
182       newNode.controller = new NodeController();
183       newNode.controller.interceptor = new ControlledCommandInterceptor();
184       builder.customInterceptors().addInterceptor().index(0).interceptor(newNode.controller.interceptor);
185       EmbeddedCacheManager embeddedCacheManager = addClusterEnabledCacheManager(builder);
186       newNode.controller.topologyManager = replaceTopologyManager(embeddedCacheManager);
187       newNode.controller.interceptor.addAction(new Action() {
188          @Override
189          public boolean isApplicable(InvocationContext context, VisitableCommand command) {
190             return !context.isOriginLocal() && command instanceof PrepareCommand;
191          }
192 
193          @Override
194          public void before(InvocationContext context, VisitableCommand command, Cache cache) {
195             log.tracef("Before: command=%s. origin=%s", command, context.getOrigin());
196             if (context.getOrigin().equals(address(cache(1)))) {
197                //from node B, i.e, it is forwarded. it needs to wait until the topology changes
198                try {
199                   cache.getAdvancedCache().getComponentRegistry().getStateTransferLock().waitForTopology(currentTopologyId + 2,
200                                                                                                          10, TimeUnit.SECONDS);
201                } catch (InterruptedException e) {
202                   Thread.currentThread().interrupt();
203                }
204             }
205          }
206 
207          @Override
208          public void after(InvocationContext context, VisitableCommand command, Cache cache) {
209             log.tracef("After: command=%s. origin=%s", command, context.getOrigin());
210             if (context.getOrigin().equals(address(cache(0)))) {
211                newNode.controller.topologyManager.stopBlocking(BlockingLocalTopologyManager.LatchType.CONSISTENT_HASH_UPDATE);
212             }
213          }
214       });
215 
216       newNode.controller.topologyManager.startBlocking(BlockingLocalTopologyManager.LatchType.CONSISTENT_HASH_UPDATE);
217       newNode.joinerFuture = fork(new Callable<Void>() {
218          @Override
219          public Void call() throws Exception {
220             waitForClusterToForm();
221             return null;
222          }
223       });
224       return newNode;
225    }

340    public static class ControlledCommandInterceptor extends BaseCustomInterceptor {
341 
342       private final List<Action> actionList;
343 
344       public ControlledCommandInterceptor(Cache<Object, Object> cache) {
345          actionList = new ArrayList<Action>(3);
346          this.cache = cache;
347          this.cacheConfiguration = cache.getCacheConfiguration();
348          this.embeddedCacheManager = cache.getCacheManager();
349          cache.getAdvancedCache().addInterceptor(this, 0);
350       }
351 
352       public ControlledCommandInterceptor() {
353          actionList = new ArrayList<Action>(3);
354       }
355 
356       public void addAction(Action action) {
357          actionList.add(action);
358       }
359 
360       @Override
361       protected Object handleDefault(InvocationContext ctx, VisitableCommand command) throws Throwable {
362          List<Action> actions = extractActions(ctx, command);
363          if (actions.isEmpty()) {
364             return invokeNextInterceptor(ctx, command);
365          }
366          for (Action action : actions) {
367             action.before(ctx, command, cache);
368          }
369          Object retVal = invokeNextInterceptor(ctx, command);
370          for (Action action : actions) {
371             action.after(ctx, command, cache);
372          }
373          return retVal;
374       }
375 
376       private List<Action> extractActions(InvocationContext context, VisitableCommand command) {
377          if (actionList.isEmpty()) {
378             return Collections.emptyList();
379          }
380          List<Action> actions = new ArrayList<Action>(actionList.size());
381          for (Action action : actionList) {
382             if (action.isApplicable(context, command)) {
383                actions.add(action);
384             }
385          }
386          return actions;
387       }
388    }
