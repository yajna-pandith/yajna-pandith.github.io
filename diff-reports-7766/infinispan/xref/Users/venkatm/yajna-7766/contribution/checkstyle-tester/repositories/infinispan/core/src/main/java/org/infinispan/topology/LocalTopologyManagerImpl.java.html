95     @Override
96     public CacheTopology join(String cacheName, CacheJoinInfo joinInfo, CacheTopologyHandler stm,
97           PartitionHandlingManager phm) throws Exception {
98        log.debugf("Node %s joining cache %s", transport.getAddress(), cacheName);
99  
100       // For Total Order caches, we must not move the topology updates to another thread
101       ExecutorService topologyUpdatesExecutor = joinInfo.isTotalOrder() ? withinThreadExecutor : asyncTransportExecutor;
102       LocalCacheStatus cacheStatus = new LocalCacheStatus(joinInfo, stm, phm, topologyUpdatesExecutor);
103       runningCaches.put(cacheName, cacheStatus);
104 
105       long timeout = joinInfo.getTimeout();
106       long endTime = timeService.expectedEndTime(timeout, TimeUnit.MILLISECONDS);
107       // Pretend the join is using up a thread from the topology updates completion service.
108       // This ensures that the initial topology and the GET_CACHE_LISTENERS request will happen on this thread,
109       // and other topology updates are only handled after we call backgroundTaskFinished(null)
110       cacheStatus.getTopologyUpdatesCompletionService().continueTaskInBackground();
111       try {
112          while (true) {
113             int viewId = transport.getViewId();
114             try {
115                ReplicableCommand command = new CacheTopologyControlCommand(cacheName,
116                        CacheTopologyControlCommand.Type.JOIN, transport.getAddress(), joinInfo, viewId);
117                CacheStatusResponse initialStatus = (CacheStatusResponse) executeOnCoordinator(command, timeout);
118                // Ignore null responses, that's what the current coordinator returns if is shutting down
119                if (initialStatus != null) {
120                   doHandleTopologyUpdate(cacheName, initialStatus.getCacheTopology(), initialStatus.getAvailabilityMode(),
121                         viewId, transport.getCoordinator(), cacheStatus);
122                   doHandleStableTopologyUpdate(cacheName, initialStatus.getStableTopology(), viewId, cacheStatus);
123                   return initialStatus.getCacheTopology();
124                }
125             } catch (Exception e) {
126                log.debugf(e, "Error sending join request for cache %s to coordinator", cacheName);
127                if (timeService.isTimeExpired(endTime)) {
128                   throw e;
129                }
130                // TODO Add some configuration for this, or use a fraction of state transfer timeout
131                Thread.sleep(100);
132             }
133          }
134       } finally {
135          cacheStatus.getTopologyUpdatesCompletionService().backgroundTaskFinished(null);
136       }
137    }

153    @Override
154    public void confirmRebalance(String cacheName, int topologyId, int rebalanceId, Throwable throwable) {
155       // Note that if the coordinator changes again after we sent the command, we will get another
156       // query for the status of our running caches. So we don't need to retry if the command failed.
157       ReplicableCommand command = new CacheTopologyControlCommand(cacheName,
158             CacheTopologyControlCommand.Type.REBALANCE_CONFIRM, transport.getAddress(),
159             topologyId, rebalanceId, throwable, transport.getViewId());
160       try {
161          executeOnCoordinatorAsync(command);
162       } catch (Exception e) {
163          log.debugf(e, "Error sending the rebalance completed notification for cache %s to the coordinator",
164                cacheName);
165       }
166    }

169    @Override
170    public ManagerStatusResponse handleStatusRequest(int viewId) {
171       Map<String, CacheStatusResponse> caches = new HashMap<String, CacheStatusResponse>();
172       synchronized (runningCaches) {
173          for (Map.Entry<String, LocalCacheStatus> e : runningCaches.entrySet()) {
174             String cacheName = e.getKey();
175             LocalCacheStatus cacheStatus = runningCaches.get(cacheName);
176             AvailabilityMode availabilityMode = cacheStatus.getPartitionHandlingManager() != null ?
177                     cacheStatus.getPartitionHandlingManager().getAvailabilityMode() : null;
178             caches.put(e.getKey(), new CacheStatusResponse(cacheStatus.getJoinInfo(),
179                     cacheStatus.getCurrentTopology(), cacheStatus.getStableTopology(),
180                     availabilityMode));
181          }
182       }
183 
184       boolean rebalancingEnabled = true;
185       // Avoid adding a direct dependency to the ClusterTopologyManager
186       ReplicableCommand command = new CacheTopologyControlCommand(null,
187             CacheTopologyControlCommand.Type.POLICY_GET_STATUS, transport.getAddress(),
188             transport.getViewId());
189       try {
190          gcr.wireDependencies(command);
191          rebalancingEnabled = (Boolean) ((SuccessfulResponse) command.perform(null)).getResponseValue();
192       } catch (Throwable t) {
193          log.warn("Failed to obtain the rebalancing status", t);
194       }
195       log.debugf("Sending cluster status response for view %d", viewId);
196       return new ManagerStatusResponse(caches, rebalancingEnabled);
197    }

199    @Override
200    public void handleTopologyUpdate(final String cacheName, final CacheTopology cacheTopology,
201          final AvailabilityMode availabilityMode, final int viewId, final Address sender) throws InterruptedException {
202       if (!running) {
203          log.tracef("Ignoring consistent hash update %s for cache %s, the local cache manager is not running",
204                cacheTopology.getTopologyId(), cacheName);
205          return;
206       }
207 
208       final LocalCacheStatus cacheStatus = runningCaches.get(cacheName);
209       if (cacheStatus == null) {
210          log.tracef("Ignoring consistent hash update %s for cache %s that doesn't exist locally",
211                cacheTopology.getTopologyId(), cacheName);
212          return;
213       }
214 
215       cacheStatus.getTopologyUpdatesCompletionService().submit(new Runnable() {
216          @Override
217          public void run() {
218             doHandleTopologyUpdate(cacheName, cacheTopology, availabilityMode, viewId, sender, cacheStatus);
219          }
220       }, null);
221       // Clear any finished tasks from the completion service's queue to avoid leaks
222       List<? extends Future<Void>> futures = cacheStatus.getTopologyUpdatesCompletionService().drainCompletionQueue();
223       boolean interrupted = false;
224       for (Future<Void> future : futures) {
225          // These will not block since they are already completed
226          try {
227             future.get();
228          } catch (InterruptedException e) {
229             // This shouldn't happen as each task is complete
230             interrupted = true;
231          } catch (ExecutionException e) {
232             log.topologyUpdateError(cacheTopology.getTopologyId(), e.getCause());
233          }
234       }
235       if (interrupted) {
236          Thread.currentThread().interrupt();
237       }
238    }

240    protected void doHandleTopologyUpdate(String cacheName, CacheTopology cacheTopology,
241          AvailabilityMode availabilityMode, int viewId, Address sender, LocalCacheStatus cacheStatus) {
242       try {
243          waitForView(viewId);
244       } catch (InterruptedException e) {
245          // Shutting down, ignore the exception and the rebalance
246          return;
247       }
248 
249       synchronized (cacheStatus) {
250          CacheTopology existingTopology = cacheStatus.getCurrentTopology();
251          if (existingTopology != null && cacheTopology.getTopologyId() <= existingTopology.getTopologyId()) {
252             log.debugf("Ignoring late consistent hash update for cache %s, current topology is %s: %s",
253                   cacheName, existingTopology.getTopologyId(), cacheTopology);
254             return;
255          }
256 
257          CacheTopologyHandler handler = cacheStatus.getHandler();
258          resetLocalTopologyBeforeRebalance(cacheName, cacheTopology, existingTopology, handler);
259 
260          if (!updateCacheTopology(cacheName, cacheTopology, sender, cacheStatus))
261             return;
262 
263          ConsistentHash unionCH = null;
264          if (cacheTopology.getPendingCH() != null) {
265             unionCH = cacheStatus.getJoinInfo().getConsistentHashFactory().union(cacheTopology.getCurrentCH(),
266                   cacheTopology.getPendingCH());
267          }
268 
269          CacheTopology unionTopology = new CacheTopology(cacheTopology.getTopologyId(), cacheTopology.getRebalanceId(),
270                cacheTopology.getCurrentCH(), cacheTopology.getPendingCH(), unionCH, cacheTopology.getActualMembers());
271          unionTopology.logRoutingTableInformation();
272 
273          boolean updateAvailabilityModeFirst = availabilityMode != AvailabilityMode.AVAILABLE;
274          if (updateAvailabilityModeFirst) {
275             if (cacheStatus.getPartitionHandlingManager() != null && availabilityMode != null) {
276                cacheStatus.getPartitionHandlingManager().setAvailabilityMode(availabilityMode);
277             }
278          }
279          if ((existingTopology == null || existingTopology.getRebalanceId() != cacheTopology.getRebalanceId()) && unionCH != null) {
280             // This CH_UPDATE command was sent after a REBALANCE_START command, but arrived first.
281             // We will start the rebalance now and ignore the REBALANCE_START command when it arrives.
282             log.tracef("This topology update has a pending CH, starting the rebalance now");
283             handler.rebalance(unionTopology);
284          } else {
285             handler.updateConsistentHash(unionTopology);
286          }
287 
288          if (!updateAvailabilityModeFirst) {
289             if (cacheStatus.getPartitionHandlingManager() != null && availabilityMode != null) {
290                cacheStatus.getPartitionHandlingManager().setAvailabilityMode(availabilityMode);
291             }
292          }
293       }
294    }

296    private boolean updateCacheTopology(String cacheName, CacheTopology cacheTopology, Address sender,
297          LocalCacheStatus cacheStatus) {
298       // Block if there is a status request in progress
299       // The caller should have already acquired the cacheStatus monitor
300       synchronized (runningCaches) {
301          if (!sender.equals(transport.getCoordinator())) {
302             log.debugf("Ignoring topology %d from old coordinator %s", cacheTopology.getTopologyId(), sender);
303             return false;
304          }
305       }
306 
307       log.debugf("Updating local topology for cache %s: %s", cacheName, cacheTopology);
308       cacheStatus.setCurrentTopology(cacheTopology);
309       return true;
310    }

312    private void resetLocalTopologyBeforeRebalance(String cacheName, CacheTopology newCacheTopology,
313          CacheTopology oldCacheTopology, CacheTopologyHandler handler) {
314       boolean newRebalance = newCacheTopology.getPendingCH() != null;
315       if (newRebalance) {
316          // The initial topology doesn't need a reset because we are guaranteed not to be a member
317          if (oldCacheTopology == null)
318             return;
319 
320          // We only need a reset if we missed a topology update
321          if (newCacheTopology.getTopologyId() == oldCacheTopology.getTopologyId() + 1)
322             return;
323 
324          // We have missed a topology update, and that topology might have removed some of our segments.
325          // If this rebalance adds those same segments, we need to remove the old data/inbound transfers first.
326          // This can happen when the coordinator changes, either because the old one left or because there was a merge,
327          // and the rebalance after merge arrives before the merged topology update.
328          if (newCacheTopology.getRebalanceId() != oldCacheTopology.getRebalanceId()) {
329             // The currentCH changed, we need to install a "reset" topology with the new currentCH first
330             CacheTopology resetTopology = new CacheTopology(newCacheTopology.getTopologyId() - 1,
331                   newCacheTopology.getRebalanceId() - 1, newCacheTopology.getCurrentCH(), null, newCacheTopology.getActualMembers());
332             log.debugf("Installing fake cache topology %s for cache %s", resetTopology, cacheName);
333             handler.updateConsistentHash(resetTopology);
334          }
335       }
336    }

339    public void handleStableTopologyUpdate(final String cacheName, final CacheTopology newStableTopology,
340          final int viewId) {
341       final LocalCacheStatus cacheStatus = runningCaches.get(cacheName);
342       if (cacheStatus != null) {
343          cacheStatus.getTopologyUpdatesCompletionService().submit(new Runnable() {
344             @Override
345             public void run() {
346                doHandleStableTopologyUpdate(cacheName, newStableTopology, viewId, cacheStatus);
347             }
348          }, null);

350    }

352    protected void doHandleStableTopologyUpdate(String cacheName, CacheTopology newStableTopology, int viewId,
353          LocalCacheStatus cacheStatus) {
354       synchronized (cacheStatus) {
355          CacheTopology stableTopology = cacheStatus.getStableTopology();
356          if (stableTopology == null || stableTopology.getTopologyId() < newStableTopology.getTopologyId()) {
357             log.tracef("Updating stable topology for cache %s: %s", cacheName, newStableTopology);
358             cacheStatus.setStableTopology(newStableTopology);
359          }
360       }
361    }

364    public void handleRebalance(final String cacheName, final CacheTopology cacheTopology, final int viewId,
365          final Address sender) throws InterruptedException {
366       if (!running) {
367          log.debugf("Ignoring rebalance request %s for cache %s, the local cache manager is not running",
368                cacheTopology.getTopologyId(), cacheName);
369          return;
370       }
371 
372       final LocalCacheStatus cacheStatus = runningCaches.get(cacheName);
373       if (cacheStatus == null) {
374          log.tracef("Ignoring rebalance %s for cache %s that doesn't exist locally",
375                cacheTopology.getTopologyId(), cacheName);
376          return;
377       }
378 
379       cacheStatus.getTopologyUpdatesCompletionService().submit(new Runnable() {
380          @Override
381          public void run() {
382             doHandleRebalance(viewId, cacheStatus, cacheTopology, cacheName, sender);
383          }
384       }, null);
385    }

387    protected void doHandleRebalance(int viewId, LocalCacheStatus cacheStatus, CacheTopology cacheTopology, String cacheName, Address sender) {
388       try {
389          waitForView(viewId);
390       } catch (InterruptedException e) {
391          // Shutting down, ignore the exception and the rebalance
392          return;
393       }
394 
395       synchronized (cacheStatus) {
396          CacheTopology existingTopology = cacheStatus.getCurrentTopology();
397          if (existingTopology != null && cacheTopology.getTopologyId() <= existingTopology.getTopologyId()) {
398             // Start rebalance commands are sent asynchronously to the entire cluster
399             // So it's possible to receive an old one on a joiner after the joiner has already become a member.
400             log.debugf("Ignoring old rebalance for cache %s, current topology is %s: %s", cacheName,
401                   existingTopology.getTopologyId(), cacheTopology);
402             return;
403          }
404 
405          if (!updateCacheTopology(cacheName, cacheTopology, sender, cacheStatus))
406             return;
407 
408          log.debugf("Starting local rebalance for cache %s, topology = %s", cacheName, cacheTopology);
409          cacheTopology.logRoutingTableInformation();
410          cacheStatus.setCurrentTopology(cacheTopology);
411 
412          CacheTopologyHandler handler = cacheStatus.getHandler();
413          resetLocalTopologyBeforeRebalance(cacheName, cacheTopology, existingTopology, handler);
414 
415          ConsistentHash unionCH = cacheStatus.getJoinInfo().getConsistentHashFactory().union(
416                cacheTopology.getCurrentCH(), cacheTopology.getPendingCH());
417          CacheTopology newTopology = new CacheTopology(cacheTopology.getTopologyId(), cacheTopology
418                .getRebalanceId(),
419                cacheTopology.getCurrentCH(), cacheTopology.getPendingCH(), unionCH, cacheTopology.getActualMembers());
420          handler.rebalance(newTopology);
421       }
422    }

505    private Object executeOnCoordinator(ReplicableCommand command, long timeout) throws Exception {
506       Response response;
507       if (transport.isCoordinator()) {
508          try {
509             if (log.isTraceEnabled()) log.tracef("Attempting to execute command on self: %s", command);
510             gcr.wireDependencies(command);
511             response = (Response) command.perform(null);
512          } catch (Throwable t) {
513             throw new CacheException("Error handling join request", t);
514          }
515       } else {
516          // this node is not the coordinator
517          Address coordinator = transport.getCoordinator();
518          Map<Address, Response> responseMap = transport.invokeRemotely(Collections.singleton(coordinator),
519                command, ResponseMode.SYNCHRONOUS, timeout, null, DeliverOrder.NONE, false);
520          response = responseMap.get(coordinator);
521       }
522       if (response == null || !response.isSuccessful()) {
523          Throwable exception = response instanceof ExceptionResponse
524                ? ((ExceptionResponse)response).getException() : null;
525          throw new CacheException("Bad response received from coordinator: " + response, exception);
526       }
527       return ((SuccessfulResponse) response).getResponseValue();
528    }

530    private void executeOnCoordinatorAsync(final ReplicableCommand command) throws Exception {
531       // if we are the coordinator, the execution is actually synchronous
532       if (transport.isCoordinator()) {
533          asyncTransportExecutor.submit(new Callable<Object>() {
534             @Override
535             public Object call() throws Exception {
536                if (log.isTraceEnabled()) log.tracef("Attempting to execute command on self: %s", command);
537                gcr.wireDependencies(command);
538                try {
539                   return command.perform(null);
540                } catch (Throwable t) {
541                   log.errorf(t, "Failed to execute ReplicableCommand %s on coordinator async: %s", command, t.getMessage());
542                   throw new Exception(t);
543                }
544             }
545          });
546       } else {
547          Address coordinator = transport.getCoordinator();
548          // ignore the responses
549          transport.invokeRemotely(Collections.singleton(coordinator), command,
550                                   ResponseMode.ASYNCHRONOUS_WITH_SYNC_MARSHALLING, 0, null, DeliverOrder.NONE, false);
551       }
552    }

554    private Map<Address, Object> executeOnClusterSync(final ReplicableCommand command, final int timeout,
555                                                      boolean totalOrder, boolean distributed)
556          throws Exception {
557       // first invoke remotely
558 
559       if (totalOrder) {
560          Map<Address, Response> responseMap = transport.invokeRemotely(null, command,
561                ResponseMode.SYNCHRONOUS_IGNORE_LEAVERS,
562                timeout, null, DeliverOrder.TOTAL, distributed);
563          Map<Address, Object> responseValues = new HashMap<Address, Object>(transport.getMembers().size());
564          for (Map.Entry<Address, Response> entry : responseMap.entrySet()) {
565             Address address = entry.getKey();
566             Response response = entry.getValue();
567             if (!response.isSuccessful()) {
568                Throwable cause = response instanceof ExceptionResponse ? ((ExceptionResponse) response).getException() : null;
569                throw new CacheException("Unsuccessful response received from node " + address + ": " + response, cause);
570             }
571             responseValues.put(address, ((SuccessfulResponse) response).getResponseValue());
572          }
573          return responseValues;
574       }
575 
576       Future<Map<Address, Response>> remoteFuture = asyncTransportExecutor.submit(new Callable<Map<Address, Response>>() {
577          @Override
578          public Map<Address, Response> call() throws Exception {
579             return transport.invokeRemotely(null, command,
580                   ResponseMode.SYNCHRONOUS_IGNORE_LEAVERS, timeout, null, DeliverOrder.NONE, false);
581          }
582       });
583 
584       // invoke the command on the local node
585       gcr.wireDependencies(command);
586       Response localResponse;
587       try {
588          if (log.isTraceEnabled()) log.tracef("Attempting to execute command on self: %s", command);
589          localResponse = (Response) command.perform(null);
590       } catch (Throwable throwable) {
591          throw new Exception(throwable);
592       }
593       if (!localResponse.isSuccessful()) {
594          throw new CacheException("Unsuccessful local response");
595       }
596 
597       // wait for the remote commands to finish
598       Map<Address, Response> responseMap = remoteFuture.get(timeout, TimeUnit.MILLISECONDS);
599 
600       // parse the responses
601       Map<Address, Object> responseValues = new HashMap<Address, Object>(transport.getMembers().size());
602       for (Map.Entry<Address, Response> entry : responseMap.entrySet()) {
603          Address address = entry.getKey();
604          Response response = entry.getValue();
605          if (!response.isSuccessful()) {
606             Throwable cause = response instanceof ExceptionResponse ? ((ExceptionResponse) response).getException() : null;
607             throw new CacheException("Unsuccessful response received from node " + address + ": " + response, cause);
608          }
609          responseValues.put(address, ((SuccessfulResponse) response).getResponseValue());
610       }
611 
612       responseValues.put(transport.getAddress(), ((SuccessfulResponse) localResponse).getResponseValue());
613 
614       return responseValues;
615    }
616 
617    private int getGlobalTimeout() {
618       // TODO Rename setting to something like globalRpcTimeout
619       return (int) gcr.getGlobalConfiguration().transport().distributedSyncTimeout();
620    }
621 }
