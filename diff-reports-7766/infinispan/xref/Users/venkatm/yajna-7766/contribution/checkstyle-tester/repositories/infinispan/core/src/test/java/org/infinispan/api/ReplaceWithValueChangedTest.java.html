13  /**
14   * Test the condition described here: {@link org.infinispan.interceptors.distribution.TxDistributionInterceptor#ignorePreviousValueOnBackup}.
15   *
16   * @author Mircea Markus
17   * @since 5.2
18   */
19  @Test (groups = "functional", testName = "api.ReplaceWithValueChangedTest")
20  public class ReplaceWithValueChangedTest extends MultipleCacheManagersTest {
21  
22     @Override
23     protected void createCacheManagers() throws Throwable {
24        createClusteredCaches(2, getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true));
25     }
26  
27     public void testReplace1() throws Throwable {
28        Object k1 = getKeyForCache(0);
29        cache(0).put(k1, "v1");
30        tm(0).begin();
31        assertEquals("v1", cache(0).replace(k1, "v2"));
32        Transaction suspendedTx = tm(0).suspend();
33  
34        cache(0).remove(k1);
35        assertNull(cache(0).get(k1));
36        assertNull(cache(1).get(k1));
37  
38        log.trace("Here it begins");
39        suspendedTx.commit();
40  
41        assertEquals("v2", cache(0).get(k1));
42        assertEquals("v2", cache(1).get(k1));
43     }
44  
45     public void testReplace2() throws Throwable {
46        Object k1 = getKeyForCache(0);
47        cache(0).put(k1, "v1");
48        tm(0).begin();
49        assertEquals("v1", cache(0).replace(k1, "v2"));
50        Transaction suspendedTx = tm(0).suspend();
51  
52        cache(0).put(k1, "v3");
53        assertEquals(cache(0).get(k1), "v3");
54        assertEquals(cache(1).get(k1), "v3");
55  
56        suspendedTx.commit();
57  
58        assertEquals("v2", cache(0).get(k1));
59        assertEquals("v2", cache(1).get(k1));
60     }
61  
62     public void testPutIfAbsent() throws Throwable {
63        Object k1 = getKeyForCache(0);
64  
65        tm(0).begin();
66        assertNull(cache(0).putIfAbsent(k1, "v1"));
67        Transaction suspendedTx = tm(0).suspend();
68  
69        cache(0).put(k1, "v2");
70        assertEquals(cache(0).get(k1), "v2");
71        assertEquals(cache(1).get(k1), "v2");
72  
73        suspendedTx.commit();
74  
75        assertEquals("v1", cache(0).get(k1));
76        assertEquals("v1", cache(1).get(k1));
77     }
78  
79     public void testConditionalRemove() throws Throwable {
80        Object k1 = getKeyForCache(0);
81        cache(0).put(k1, "v1");
82        tm(0).begin();
83        assertTrue(cache(0).remove(k1, "v1"));
84        Transaction suspendedTx = tm(0).suspend();
85  
86        cache(0).put(k1, "v2");
87        assertEquals(cache(0).get(k1), "v2");
88        assertEquals(cache(1).get(k1), "v2");
89  
90        log.trace("here it is");
91        suspendedTx.commit();
92  
93        assertNull(cache(0).get(k1));
94        assertNull(cache(1).get(k1));
95     }
