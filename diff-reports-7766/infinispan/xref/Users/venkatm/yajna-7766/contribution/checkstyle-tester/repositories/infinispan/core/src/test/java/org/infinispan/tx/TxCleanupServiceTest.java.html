30  @CleanupAfterMethod
31  @Test(groups = "functional", testName = "tx.TxCleanupServiceTest")
32  public class TxCleanupServiceTest extends MultipleCacheManagersTest {
33     private static final int TX_COUNT = 1;
34     private ConfigurationBuilder dcc;
35     private ControlledConsistentHashFactory consistentHashFactory;
36  
37     @Override
38     protected void createCacheManagers() throws Throwable {
39        dcc = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
40        dcc.transaction().transactionManagerLookup(new DummyTransactionManagerLookup());
41        consistentHashFactory = new ControlledConsistentHashFactory(1);
42        dcc.clustering().hash().numOwners(1).numSegments(1).consistentHashFactory(consistentHashFactory);
43        createCluster(dcc, 2);
44        waitForClusterToForm();
45     }
46  
47     public void testTransactionStateNotLost() throws Throwable {
48        final ControlledCommandFactory ccf = ControlledCommandFactory.registerControlledCommandFactory(cache(1), CommitCommand.class);
49        ccf.gate.close();
50  
51        final Map<Object, DummyTransaction> keys2Tx = new HashMap<Object, DummyTransaction>(TX_COUNT);
52  
53        int viewId = advancedCache(0).getRpcManager().getTransport().getViewId();
54  
55        log.tracef("ViewId before %s", viewId);
56  
57        //fork it into another thread as this is going to block in commit
58        Future<Object> future = fork(new Callable<Object>() {
59           @Override
60           public Object call() throws Exception {
61              for (int i = 0; i < TX_COUNT; i++) {
62                 Object k = getKeyForCache(1);
63                 tm(0).begin();
64                 cache(0).put(k, k);
65                 DummyTransaction transaction = ((DummyTransactionManager) tm(0)).getTransaction();
66                 keys2Tx.put(k, transaction);
67                 tm(0).commit();
68              }
69              return null;
70           }
71        });
72  
73        //now wait for all the commits to block
74        eventually(new Condition() {
75           @Override
76           public boolean isSatisfied() throws Exception {
77              return ccf.blockTypeCommandsReceived.get() == TX_COUNT;
78           }
79        });
80  
81        log.tracef("Viewid middle %s", viewId);
82  
83  
84        //now add a one new member
85        consistentHashFactory.setOwnerIndexes(2);
86        addClusterEnabledCacheManager(dcc);
87        waitForClusterToForm();
88  
89  
90        viewId = advancedCache(0).getRpcManager().getTransport().getViewId();
91  
92        log.tracef("Viewid after before %s", viewId);
93  
94  
95        final Map<Object, DummyTransaction> migratedTx = new HashMap<Object, DummyTransaction>(TX_COUNT);
96        for (Object key : keys2Tx.keySet()) {
97           if (keyMapsToNode(key, 2)) {
98              migratedTx.put(key, keys2Tx.get(key));
99           }
100       }
101 
102       log.tracef("Number of migrated tx is %s", migratedTx.size());
103       assertEquals(TX_COUNT, migratedTx.size());
104 
105       eventually(new Condition() {
106          @Override
107          public boolean isSatisfied() throws Exception {
108             return TestingUtil.getTransactionTable(cache(2)).getRemoteTxCount() == migratedTx.size();
109          }
110       });
111 
112       log.trace("Releasing the gate");
113       ccf.gate.open();
114 
115       future.get(10, TimeUnit.SECONDS);
116 
117       eventually(new Condition() {
118          @Override
119          public boolean isSatisfied() throws Exception {
120             return TestingUtil.getTransactionTable(cache(2)).getRemoteTxCount() == 0;
121          }
122       });
123 
124 
125       eventually(new Condition() {
126          @Override
127          public boolean isSatisfied() throws Exception {
128             boolean allZero = true;
129             for (int i = 0; i < 3; i++) {
130                TransactionTable tt = TestingUtil.getTransactionTable(cache(i));
131 //               assertEquals("For cache " + i, 0, tt.getLocalTxCount());
132 //               assertEquals("For cache " + i, 0, tt.getRemoteTxCount());
133                int local = tt.getLocalTxCount();
134                int remote = tt.getRemoteTxCount();
135                log.tracef("For cache %i , localTxCount=%s, remoteTxCount=%s", i, local, remote);
136                log.tracef(String.format("For cache %s , localTxCount=%s, remoteTxCount=%s", i, local, remote));
137                allZero = allZero && (local == 0);
138                allZero = allZero && (remote == 0);
139             }
140             return allZero;
141          }
142       });
143 
144       for (Object key : keys2Tx.keySet()) {
145          assertNotLocked(key);
146          assertEquals(key, cache(0).get(key));
147       }
148    }
149 
150    private boolean keyMapsToNode(Object key, int nodeIndex) {
151       Address owner = owner(key);
152       return owner.equals(address(nodeIndex));
153    }
154 
155    private Address owner(Object key) {
156       return advancedCache(0).getDistributionManager().getConsistentHash().locatePrimaryOwner(key);
157    }
