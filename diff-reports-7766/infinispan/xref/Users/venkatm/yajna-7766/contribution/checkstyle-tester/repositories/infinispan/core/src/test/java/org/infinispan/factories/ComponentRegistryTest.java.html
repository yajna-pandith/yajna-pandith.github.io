23  /**
24   * Test the concurrent lookup of components for ISPN-2796.
25   *
26   * @author Dan Berindei
27   * @since 5.2
28   */
29  @Test(groups = "unit", testName = "factories.ComponentRegistryTest")
30  public class ComponentRegistryTest extends AbstractInfinispanTest {
31     private GlobalComponentRegistry gcr;
32     private ComponentRegistry cr1;
33     private ComponentRegistry cr2;
34     private TestDelayFactory.Control control;
35  
36     @BeforeMethod
37     public void setUp() throws InterruptedException, ExecutionException {
38        GlobalConfiguration gc = new GlobalConfigurationBuilder().build();
39        Configuration c = new ConfigurationBuilder().build();
40        Set<String> cachesSet = new HashSet<String>();
41        EmbeddedCacheManager cm = mock(EmbeddedCacheManager.class);
42        AdvancedCache cache = mock(AdvancedCache.class);
43  
44        gcr = new GlobalComponentRegistry(gc, cm, cachesSet);
45        cr1 = new ComponentRegistry("cache", c, cache, gcr, ComponentRegistryTest.class.getClassLoader());
46        cr2 = new ComponentRegistry("cache", c, cache, gcr, ComponentRegistryTest.class.getClassLoader());
47  
48        control = new TestDelayFactory.Control();
49        gcr.registerComponent(control, TestDelayFactory.Control.class);
50     }
51  
52     public void testSingleThreadLookup() throws InterruptedException, ExecutionException {
53        control.unblock();
54  
55        TestDelayFactory.Component c1 = cr1.getOrCreateComponent(TestDelayFactory.Component.class);
56        assertNotNull(c1);
57  
58        TestDelayFactory.Component c2 = cr1.getOrCreateComponent(TestDelayFactory.Component.class);
59        assertNotNull(c2);
60     }
61  
62     public void testConcurrentLookupSameComponentRegistry() throws InterruptedException, ExecutionException {
63  
64        Future<Object> future = fork(new Callable<Object>() {
65           @Override
66           public Object call() throws Exception {
67              return cr1.getOrCreateComponent(TestDelayFactory.Component.class);
68           }
69        });
70  
71        // getComponent doesn't wait for the getOrCreateComponent call on the forked thread to finish
72        // It returns null instead, but that's ok because getComponent doesn't guarantee anything
73        Thread.sleep(500);
74        assertNull(cr1.getComponent(TestDelayFactory.Component.class));
75  
76        control.unblock();
77        assertNotNull(future.get());
78        // now that getOrCreateComponent has finished, getComponent works as well
79        assertNotNull(cr1.getComponent(TestDelayFactory.Component.class));
80     }
81  
82     public void testConcurrentLookupDifferentComponentRegistries() throws InterruptedException, ExecutionException {
83  
84        Future<Object> future = fork(new Callable<Object>() {
85           @Override
86           public Object call() throws Exception {
87              return cr1.getOrCreateComponent(TestDelayFactory.Component.class);
88           }
89        });
90  
91        Thread.sleep(500);
92        assertNotNull(cr2.getOrCreateComponent(TestDelayFactory.Component.class));
93  
94        control.unblock();
95        assertNotNull(future.get());
96     }
97  }
