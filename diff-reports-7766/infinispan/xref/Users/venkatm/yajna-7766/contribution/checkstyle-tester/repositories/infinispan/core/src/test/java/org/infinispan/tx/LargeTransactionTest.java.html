23  @Test(testName = "tx.LargeTransactionTest", groups = "functional")
24  public class LargeTransactionTest extends MultipleCacheManagersTest {
25     private static final Log log = LogFactory.getLog(LargeTransactionTest.class);
26  
27     protected void createCacheManagers() throws Throwable {
28        ConfigurationBuilder c = new ConfigurationBuilder();
29        c.clustering().cacheMode(CacheMode.REPL_SYNC)
30              .sync().replTimeout(30000)
31              .transaction().transactionMode(TransactionMode.TRANSACTIONAL)
32              .syncCommitPhase(true).syncRollbackPhase(true)
33              .invocationBatching().enable()
34              .locking().lockAcquisitionTimeout(60000).useLockStriping(false);
35  
36        EmbeddedCacheManager container = TestCacheManagerFactory.createClusteredCacheManager(c);
37        container.start();
38        registerCacheManager(container);
39        container.startCaches(CacheContainer.DEFAULT_CACHE_NAME, "TestCache");
40        Cache cache1 = container.getCache("TestCache");
41        assert cache1.getCacheConfiguration().clustering().cacheMode().equals(CacheMode.REPL_SYNC);
42        cache1.start();
43  
44        container = TestCacheManagerFactory.createClusteredCacheManager(c);
45        container.start();
46        registerCacheManager(container);
47        container.startCaches(CacheContainer.DEFAULT_CACHE_NAME, "TestCache");
48        Cache cache2 = container.getCache("TestCache");
49        assert cache2.getCacheConfiguration().clustering().cacheMode().equals(CacheMode.REPL_SYNC);
50     }
51  
52     public void testLargeTx() throws Exception {
53        Cache cache1 = cache(0, "TestCache");
54        Cache cache2 = cache(1, "TestCache");
55        TransactionManager tm = TestingUtil.getTransactionManager(cache1);
56        tm.begin();
57        for (int i = 0; i < 200; i++)
58           cache1.put("key" + i, "value" + i);
59        log.trace("___________ before commit");
60        tm.commit();
61  
62        for (int i = 0; i < 200; i++) {
63           assert cache2.get("key" + i).equals("value"+i);
64        }
65     }
66  
67     public void testSinglePutInTx() throws Exception {
68        Cache cache1 = cache(0, "TestCache");
69        Cache cache2 = cache(1, "TestCache");
70        TransactionManager tm = TestingUtil.getTransactionManager(cache1);
71  
72        tm.begin();
73        cache1.put("key", "val");
74        log.trace("___________ before commit");
75        tm.commit();
76  
77        assert cache2.get("key").equals("val");
78     }
79  
80     public void testSimplePutNoTx() {
81        Cache cache1 = cache(0, "TestCache");
82        Cache cache2 = cache(1, "TestCache");
83        cache1.put("key", "val");
84        assert cache2.get("key").equals("val");
85     }
