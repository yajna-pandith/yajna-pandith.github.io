32  /**
33   * Test that entries in a shared store are not touched in any way during state transfer.
34   *
35   * @author Dan Berindei
36   */
37  @Test(testName = "distribution.rehash.SharedStoreInvalidationDuringRehashTest", groups = "functional")
38  @CleanupAfterMethod
39  public class SharedStoreInvalidationDuringRehashTest extends MultipleCacheManagersTest {

41     private static final Log log = LogFactory.getLog(SharedStoreInvalidationDuringRehashTest.class);
42  
43     private static final int NUM_KEYS = 20;
44     private static final String TEST_CACHE_NAME = "testCache";
45  
46     private final Map<Integer, AtomicInteger> invalidationCounts = CollectionFactory.makeConcurrentMap();
47     private final Map<Integer, AtomicInteger> l1InvalidationCounts = CollectionFactory.makeConcurrentMap();

51        // cacheManagers started one after another in test()
52     }

54     private void addNewCacheManagerAndWaitForRehash(int index, boolean preload) {
55        EmbeddedCacheManager cacheManager = addClusterEnabledCacheManager(getDefaultClusteredCacheConfig(
56              CacheMode.DIST_SYNC, false));
57        Configuration config = buildCfg(index, true, preload);
58        cacheManager.defineConfiguration(TEST_CACHE_NAME, config);
59        log.debugf("\n\nstarted CacheManager #%d", getCacheManagers().size() - 1);
60        waitForClusterToForm(TEST_CACHE_NAME);
61     }
62  
63     private Configuration buildCfg(final int index, boolean clustered, boolean preload) {
64        ConfigurationBuilder cb = new ConfigurationBuilder();
65        cb.persistence().passivation(false);
66        cb.customInterceptors().addInterceptor().index(0).interceptor(new BaseCustomInterceptor() {
67           @Override
68           public Object visitInvalidateCommand(InvocationContext ctx, InvalidateCommand invalidateCommand) throws Throwable {
69              incrementCounter(invalidationCounts, index, invalidateCommand.getKeys().length);
70              return invokeNextInterceptor(ctx, invalidateCommand);
71           }
72  
73           @Override
74           public Object visitInvalidateL1Command(InvocationContext ctx, InvalidateL1Command invalidateL1Command) throws Throwable {
75              incrementCounter(l1InvalidationCounts, index, invalidateL1Command.getKeys().length);
76              return invokeNextInterceptor(ctx, invalidateL1Command);
77           }
78        });
79  
80        DummyInMemoryStoreConfigurationBuilder dummyCB = cb.persistence().addStore(DummyInMemoryStoreConfigurationBuilder.class);
81        dummyCB.debug(true).preload(preload).shared(true).purgeOnStartup(false);
82        dummyCB.storeName(SharedStoreInvalidationDuringRehashTest.class.getSimpleName());
83  
84        if (clustered) {
85           cb.clustering().l1().disable();
86           cb.clustering().cacheMode(CacheMode.DIST_SYNC);
87           cb.clustering().hash().numOwners(1); // one owner!
88  
89           cb.clustering().stateTransfer().fetchInMemoryState(true);
90           cb.clustering().hash().groups().enabled();
91        }
92        return cb.build(true);
93     }
94  
95     private void incrementCounter(Map<Integer, AtomicInteger> counterMap, int index, int delta) {
96        AtomicInteger counter = counterMap.get(index);
97        if (counter == null) {
98           counter = new AtomicInteger(0);
99           counterMap.put(index, counter);
100       }
101       counter.addAndGet(delta);
102    }
103 
104    private int getCounter(Map<Integer, AtomicInteger> counterMap, int index) {
105       AtomicInteger counter = counterMap.get(index);
106       return counter != null ? counter.get() : 0;
107    }
108 
109    private int getSum(Map<Integer, AtomicInteger> counterMap) {
110       int sum = 0;
111       for (AtomicInteger c : counterMap.values()) {
112          sum += c.get();
113       }
114       return sum;
115    }

117    public void testRehashWithPreload() {
118       doTest(true);
119    }
120 
121    public void testRehashWithoutPreload() {
122       doTest(false);
123    }

125    private void doTest(boolean preload) {
126 
127       // start a cluster that uses this cache store
128       // add 1st member
129       addNewCacheManagerAndWaitForRehash(0, preload);
130 
131       // insert the data and test that it's in the store
132       insertTestData();
133       printCacheContents();
134       printStoreContents();
135       checkContentAndInvalidations(preload);
136 
137       // stop 1st member
138       killMember(0);
139 
140       // re-add 1st member
141       addNewCacheManagerAndWaitForRehash(0, preload);
142       printCacheContents();
143       printStoreContents();
144       checkContentAndInvalidations(preload);
145 
146       // add 2nd member
147       addNewCacheManagerAndWaitForRehash(1, preload);
148       printCacheContents();
149       printStoreContents();
150       checkContentAndInvalidations(preload);
151 
152       // add 3rd member
153       addNewCacheManagerAndWaitForRehash(2, preload);
154       printCacheContents();
155       printStoreContents();
156       checkContentAndInvalidations(preload);
157    }

159    private void insertTestData() {
160       Cache<String, String> cache = manager(0).getCache(TEST_CACHE_NAME);
161       for (int i = 0; i < NUM_KEYS; i++) {
162          cache.put("key" + i, Integer.toString(i));
163       }
164 
165       log.debugf("Added %d entries to test cache", NUM_KEYS);
166    }
167 
168    private void checkContentAndInvalidations(boolean preload) {
169       int clusterSize = getCacheManagers().size();
170       int joiner = clusterSize - 1;
171 
172       for (int i = 0; i < clusterSize; i++) {
173          Cache<String, String> testCache = manager(i).getCache(TEST_CACHE_NAME);
174          DistributionManager dm = testCache.getAdvancedCache().getDistributionManager();
175          DataContainer dataContainer = testCache.getAdvancedCache().getDataContainer();
176 
177          for (int j = 0; j < NUM_KEYS; j++) {
178             String key = "key" + j;
179             if (!dm.getLocality(key).isLocal()) {
180                assertFalse("Key '" + key + "' is not owned by node " + address(i) + " but it still appears there",
181                      dataContainer.containsKey(key));
182             } else if (preload) {
183                assertTrue("Key '" + key + "' is owned by node " + address(i) + " but it does not appear there",
184                      dataContainer.containsKey(key));
185             }
186          }
187       }
188 
189       DummyInMemoryStore store = (DummyInMemoryStore) TestingUtil.getFirstLoader(cache(0, TEST_CACHE_NAME));
190       for (int i = 0; i < NUM_KEYS; i++) {
191          String key = "key" + i;
192          assertTrue("Key " + key + " is missing from the shared store", store.keySet().contains(key));
193       }
194 
195       log.debugf("Invalidations: %s, L1 invalidations: %s", invalidationCounts, l1InvalidationCounts);
196       int joinerSize = advancedCache(joiner, TEST_CACHE_NAME).getDataContainer().size();
197       if (preload) {
198          // L1 is disabled, so no InvalidateL1Commands
199          assertEquals(0, getSum(l1InvalidationCounts));
200 
201          // The joiner has preloaded the entire store, and the entries not owned have been invalidated
202          assertEquals(NUM_KEYS - joinerSize, getCounter(invalidationCounts, joiner));
203 
204          // The other nodes have invalidated the entries moved to the joiner
205          if (clusterSize > 1) {
206             assertEquals(NUM_KEYS, getSum(invalidationCounts));
207          }
208       } else {
209          // L1 is disabled, so no InvalidateL1Commands
210          assertEquals(0, getSum(l1InvalidationCounts));
211 
212          // No entries to invalidate on the joiner
213          assertEquals(0, getCounter(invalidationCounts, joiner));
214 
215          // The other nodes have invalidated the entries moved to the joiner
216          if (clusterSize > 1) {
217             assertEquals(joinerSize, getSum(invalidationCounts));
218          }
219       }
220 
221       // Reset stats for the next check
222       store.clearStats();
223       invalidationCounts.clear();
224       l1InvalidationCounts.clear();
225    }

227    private void printCacheContents() {
228       log.debugf("%d cache managers: %s", getCacheManagers().size(), getCacheManagers());
229       for (int i = 0; i < getCacheManagers().size(); i++) {
230          Cache<String, String> testCache = manager(i).getCache(TEST_CACHE_NAME);
231          Set<String> keySet = testCache.keySet();
232          log.debugf("Cache %s has %d keys: %s", address(i), keySet.size(), keySet);
233       }
234    }

236    private void printStoreContents() {
237       DummyInMemoryStore store = (DummyInMemoryStore) TestingUtil.getFirstLoader(cache(0, TEST_CACHE_NAME));
238       Set<Object> keySet = store.keySet();
239       log.debugf("Shared store has %d keys: %s", keySet.size(), keySet);
240    }
