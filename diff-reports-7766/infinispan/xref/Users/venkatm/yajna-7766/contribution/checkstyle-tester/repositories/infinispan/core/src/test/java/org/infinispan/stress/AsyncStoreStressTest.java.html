89     // Lock container that mimics per-key locking produced by the cache.
90     // This per-key lock holder provides guarantees that the final expected
91     // state has not been affected by ordering issues such as this:
92     //
93     // (Thread-200:) Enqueuing modification Store{storedEntry=
94     // ImmortalCacheEntry{key=key165168, value=ImmortalCacheValue {value=60483}}}
95     // (Thread-194:) Enqueuing modification Store{storedEntry=
96     // ImmortalCacheEntry{key=key165168, value=ImmortalCacheValue {value=61456}}}
97     // (Thread-194:) Expected state updated with key=key165168, value=61456
98     // (Thread-200:) Expected state updated with key=key165168, value=60483
99     private LockContainer locks = new ReentrantPerEntryLockContainer(32, AnyEquivalence.getInstance());

102       Map<String, KeyValuePair<AdvancedAsyncCacheLoader, AdvancedAsyncCacheWriter>> stores = new TreeMap<String, KeyValuePair<AdvancedAsyncCacheLoader, AdvancedAsyncCacheWriter>>();

146    @Test(dataProvider = "readWriteRemove")
147    public void testReadWriteRemove(int capacity, int numKeys,
148          int readerThreads, int writerThreads, int removerThreads) throws Exception {
149       System.out.printf("Testing independent read/write/remove performance " +
150             "with capacity %d, keys %d, readers %d, writers %d, removers %d\n",
151             capacity, numKeys, readerThreads, writerThreads, removerThreads);
152 
153       generateKeyList(numKeys);
154 
155       Map<String, KeyValuePair<AdvancedAsyncCacheLoader, AdvancedAsyncCacheWriter>> stores = createAsyncStores();
156       try {
157          for (Map.Entry<String, KeyValuePair<AdvancedAsyncCacheLoader, AdvancedAsyncCacheWriter>> e : stores.entrySet()) {
158             mapTestReadWriteRemove(e.getKey(), e.getValue().getKey(), e.getValue().getValue(), numKeys,
159                   readerThreads, writerThreads, removerThreads);
160             e.setValue(null);
161          }
162       } finally {
163          for (Iterator<KeyValuePair<AdvancedAsyncCacheLoader, AdvancedAsyncCacheWriter>> it = stores.values().iterator(); it.hasNext(); ) {
164             KeyValuePair<AdvancedAsyncCacheLoader, AdvancedAsyncCacheWriter> store = it.next();
165             try {
166                store.getKey().stop();
167                store.getValue().stop();
168                it.remove();
169             } catch (Exception ex) {
170                log.error("Failed to stop cache store", ex);
171             }
172          }
173       }
174       assertTrue("Not all stores were properly shut down", stores.isEmpty());
175    }

177    private void mapTestReadWriteRemove(String name, AdvancedCacheLoader loader, AdvancedCacheWriter writer,
178          int numKeys, int readerThreads, int writerThreads, int removerThreads) throws Exception {
179       DummyInMemoryStore delegate = (DummyInMemoryStore) ((DelegatingCacheLoader) loader).undelegate();
180       try {
181          // warm up for 1 second
182          System.out.printf("[store=%s] Warming up\n", name);
183          runMapTestReadWriteRemove(name, loader, writer, readerThreads, writerThreads, removerThreads, 1000);
184 
185          // real test
186          System.out.printf("[store=%s] Testing...\n", name);
187          TotalStats perf = runMapTestReadWriteRemove(name, loader, writer, readerThreads, writerThreads, removerThreads, RUNNING_TIME);
188 
189          // Wait until the cache store contains the expected state
190          System.out.printf("[store=%s] Verify contents\n", name);
191          delegate.blockUntilCacheStoreContains(expectedState.keySet(), 60000);
192 
193          System.out.printf("Container %-12s  ", name);
194          System.out.printf("Ops/s %10.2f  ", perf.getTotalOpsPerSec());
195          System.out.printf("Gets/s %10.2f  ", perf.getOpsPerSec("GET"));
196          System.out.printf("Puts/s %10.2f  ", perf.getOpsPerSec("PUT"));
197          System.out.printf("Removes/s %10.2f  ", perf.getOpsPerSec("REMOVE"));
198          System.out.printf("HitRatio %10.2f  ", perf.getTotalHitRatio() * 100);
199          System.out.printf("Size %10d  ", PersistenceUtil.count(loader, null));
200          double stdDev = computeStdDev(loader, numKeys);
201          System.out.printf("StdDev %10.2f\n", stdDev);
202       } finally {
203          // Clean up state, expected state and keys
204          expectedState.clear();
205          delegate.clear();
206       }
207    }

209    private TotalStats runMapTestReadWriteRemove(String name, final AdvancedCacheLoader loader, final AdvancedCacheWriter cWriter, int numReaders, int numWriters,
210          int numRemovers, final long runningTimeout) throws Exception {
211       latch = new CountDownLatch(1);
212       final TotalStats perf = new TotalStats();
213       List<Thread> threads = new LinkedList<Thread>();
214 
215       for (int i = 0; i < numReaders; i++) {
216          Thread reader = new WorkerThread("worker-" + name + "-get-" + i, runningTimeout, perf, readOperation(loader));
217          threads.add(reader);
218       }
219 
220       for (int i = 0; i < numWriters; i++) {
221          Thread writer = new WorkerThread("worker-" + name + "-put-" + i, runningTimeout, perf, writeOperation(cWriter));
222          threads.add(writer);
223       }
224 
225       for (int i = 0; i < numRemovers; i++) {
226          Thread remover = new WorkerThread("worker-" + name + "-remove-" + i, runningTimeout, perf, removeOperation(cWriter));
227          threads.add(remover);
228       }
229 
230       for (Thread t : threads)
231          t.start();
232       latch.countDown();
233 
234       for (Thread t : threads)
235          t.join();
236 
237       return perf;
238    }

240    private void generateKeyList(int numKeys) {
241       // without this we keep getting OutOfMemoryErrors
242       keys = null;
243       keys = new ArrayList<String>(numKeys * LOOP_FACTOR);
244       for (int i = 0; i < numKeys * LOOP_FACTOR; i++) {
245          keys.add("key" + nextIntGaussian(numKeys));
246       }
247    }
248 
249    private int nextIntGaussian(int numKeys) {
250       double gaussian = RANDOM.nextGaussian();
251       if (gaussian < -3 || gaussian > 3)
252          return nextIntGaussian(numKeys);
253 
254       return (int) Math.abs((gaussian + 3) * numKeys / 6);
255    }

257    private void waitForStart() {

265    private Operation<String, Integer> readOperation(final AdvancedCacheLoader store) {
266       return new Operation<String, Integer>("GET") {
267          @Override
268          public boolean call(String key, long run) {
269             MarshalledEntry me = store.load(key);
270             if (trace)
271                log.tracef("Loaded key=%s, value=%s", key, me != null ? me.getValue() : "null");
272             return me != null;
273          }
274       };
275    }
276 
277    private Operation<String, Integer> writeOperation(final AdvancedCacheWriter store) {
278       return new Operation<String, Integer>("PUT") {
279          @Override
280          public boolean call(final String key, long run) {
281             final int value = (int) run;
282             final InternalCacheEntry entry =
283                   entryFactory.create(key, value, new EmbeddedMetadata.Builder().build());
284             // Store acquiring locks and catching exceptions
285             boolean result = withStore(key, new Callable<Boolean>() {
286                @Override
287                public Boolean call() throws Exception {
288                   store.write(marshalledEntry(entry, marshaller));
289                   expectedState.put(key, entry);
290                   if (trace)
291                      log.tracef("Expected state updated with key=%s, value=%s", key, value);
292                   return true;
293                }
294             });
295             return result;
296          }
297       };
298    }
299 
300    private Operation<String, Integer> removeOperation(final AdvancedCacheWriter store) {
301       return new Operation<String, Integer>("REMOVE") {
302          @Override
303          public boolean call(final String key, long run) {
304             // Remove acquiring locks and catching exceptions
305             boolean result = withStore(key, new Callable<Boolean>() {
306                @Override
307                public Boolean call() throws Exception {
308                   boolean removed = store.delete(key);
309                   if (removed) {
310                      expectedState.remove(key);
311                      if (trace)
312                         log.tracef("Expected state removed key=%s", key);
313                   }
314                   return true;
315                }
316             });
317             return result;
318          }
319       };
320    }

322    private boolean withStore(String key, Callable<Boolean> call) {
323       Lock lock = null;
324       boolean result = false;
325       try {
326          lock = locks.acquireLock(Thread.currentThread(), key, 30, TimeUnit.SECONDS);
327          if (lock != null) {
328             result = call.call().booleanValue();
329          }
330       } catch (PersistenceException e) {
331          e.printStackTrace();
332          result = false;
333       } catch (InterruptedException e) {
334          e.printStackTrace();
335          result = false;
336       } finally {
337          if (lock == null) return false;
338          else {
339             lock.unlock();
340             return result;
341          }
342       }
343    }

345    private double computeStdDev(AdvancedCacheLoader store, int numKeys) throws PersistenceException {
346       // The keys closest to the mean are suposed to be accessed more often
347       // So we score each map by the standard deviation of the keys in the map
348       // at the end of the test
349       double variance = 0;
350       Set<Object> keys = PersistenceUtil.toKeySet(store, null);
351       for (Object key : keys) {
352          double value = Integer.parseInt(((String )key).substring(3));
353          variance += (value - numKeys / 2) * (value - numKeys / 2);
354       }
355       return sqrt(variance / keys.size());
356    }

358    private class WorkerThread extends Thread {
359       private final long runningTimeout;
360       private final TotalStats perf;
361       private Operation<String, Integer> op;
362 
363       public WorkerThread(String name, long runningTimeout, TotalStats perf, Operation<String, Integer> op) {
364          super(name);
365          this.runningTimeout = runningTimeout;
366          this.perf = perf;
367          this.op = op;
368       }
369 
370       public void run() {
371          waitForStart();
372          long startMilis = System.currentTimeMillis();
373          long endMillis = startMilis + runningTimeout;
374          int keyIndex = RANDOM.nextInt(keys.size());
375          long runs = 0;
376          long missCount = 0;
377          while ((runs & 0x3FFF) != 0 || System.currentTimeMillis() < endMillis) {
378             boolean hit = op.call(keys.get(keyIndex), runs);
379             if (!hit) missCount++;
380             keyIndex++;
381             runs++;
382             if (keyIndex >= keys.size()) {
383                keyIndex = 0;
384             }
385          }
386          perf.addStats(op.getName(), runs, System.currentTimeMillis() - startMilis, missCount);
387       }
388    }

407    private static class TotalStats {
408       private ConcurrentHashMap<String, OpStats> statsMap = new ConcurrentHashMap<String, OpStats>();
409 
410       public void addStats(String opName, long opCount, long runningTime, long missCount) {
411          OpStats s = new OpStats(opName, opCount, runningTime, missCount);
412          OpStats old = statsMap.putIfAbsent(opName, s);
413          boolean replaced = old == null;
414          while (!replaced) {
415             old = statsMap.get(opName);
416             s = new OpStats(old, opCount, runningTime, missCount);
417             replaced = statsMap.replace(opName, old, s);
418          }
419       }
420 
421       public double getOpsPerSec(String opName) {
422          OpStats s = statsMap.get(opName);
423          if (s == null) return 0;
424          return s.opCount * 1000. / s.runningTime * s.threadCount;
425       }
426 
427       public double getTotalOpsPerSec() {
428          long totalOpCount = 0;
429          long totalRunningTime = 0;
430          long totalThreadCount = 0;
431          for (Map.Entry<String, OpStats> e : statsMap.entrySet()) {
432             OpStats s = e.getValue();
433             totalOpCount += s.opCount;
434             totalRunningTime += s.runningTime;
435             totalThreadCount += s.threadCount;
436          }
437          return totalOpCount * 1000. / totalRunningTime * totalThreadCount;
438       }
439 
440       public double getHitRatio(String opName) {
441          OpStats s = statsMap.get(opName);
442          if (s == null) return 0;
443          return 1 - 1. * s.missCount / s.opCount;
444       }
445 
446       public double getTotalHitRatio() {
447          long totalOpCount = 0;
448          long totalMissCount = 0;
449          for (Map.Entry<String, OpStats> e : statsMap.entrySet()) {
450             OpStats s = e.getValue();
451             totalOpCount += s.opCount;
452             totalMissCount += s.missCount;
453          }
454          return 1 - 1. * totalMissCount / totalOpCount;
455       }
456    }

458    private static class OpStats {
459       public final String opName;
460       public final int threadCount;
461       public final long opCount;
462       public final long runningTime;
463       public final long missCount;
464 
465       private OpStats(String opName, long opCount, long runningTime, long missCount) {
466          this.opName = opName;
467          this.threadCount = 1;
468          this.opCount = opCount;
469          this.runningTime = runningTime;
470          this.missCount = missCount;
471       }
472 
473       private OpStats(OpStats base, long opCount, long runningTime, long missCount) {
474          this.opName = base.opName;
475          this.threadCount = base.threadCount + 1;
476          this.opCount = base.opCount + opCount;
477          this.runningTime = base.runningTime + runningTime;
478          this.missCount = base.missCount + missCount;
479       }
480    }
