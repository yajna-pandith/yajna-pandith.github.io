121       assert !cache.keySet().contains(key2);
122       assert cache.values().contains(value);

125

127       assert !cache.containsKey(key2);
128       assert cache.isEmpty();
129       assert !cache.keySet().contains(key1);

162             col.addAll(newObjCol);

399    public void testKeyValueEntryCollections() {
400       String key1 = "1", value1 = "one", key2 = "2", value2 = "two", key3 = "3", value3 = "three";
401       Map<String, String> m = new HashMap<String, String>();
402       m.put(key1, value1);
403       m.put(key2, value2);
404       m.put(key3, value3);
405       cache.putAll(m);
406       assert 3 == cache.size() && 3 == cache.keySet().size() && 3 == cache.values().size() && 3 == cache.entrySet().size();
407 
408       Set expKeys = new HashSet();
409       expKeys.add(key1);
410       expKeys.add(key2);
411       expKeys.add(key3);
412 
413       Set expValues = new HashSet();
414       expValues.add(value1);
415       expValues.add(value2);
416       expValues.add(value3);
417 
418       Set expKeyEntries = ObjectDuplicator.duplicateSet(expKeys);
419       Set expValueEntries = ObjectDuplicator.duplicateSet(expValues);
420 
421       Set<Object> keys = cache.keySet();
422       for (Object key : keys) {
423          assert expKeys.remove(key);
424       }
425       assert expKeys.isEmpty() : "Did not see keys " + expKeys + " in iterator!";
426 
427       Collection<Object> values = cache.values();
428       for (Object value : values) {
429          assert expValues.remove(value);
430       }
431       assert expValues.isEmpty() : "Did not see keys " + expValues + " in iterator!";
432 
433       Set<Map.Entry<Object, Object>> entries = cache.entrySet();
434       for (Map.Entry entry : entries) {
435          assert expKeyEntries.remove(entry.getKey());
436          assert expValueEntries.remove(entry.getValue());
437       }
438       assert expKeyEntries.isEmpty() : "Did not see keys " + expKeyEntries + " in iterator!";
439       assert expValueEntries.isEmpty() : "Did not see keys " + expValueEntries + " in iterator!";
440    }
