40     public void testPrimaryOwnerGoesDownBeforeBackupRaisesEvent() throws InterruptedException, TimeoutException,
41                                                                     ExecutionException, BrokenBarrierException {
42        final Cache<Object, String> cache0 = cache(0, CACHE_NAME);
43        Cache<Object, String> cache1 = cache(1, CACHE_NAME);
44        Cache<Object, String> cache2 = cache(2, CACHE_NAME);
45  
46        ClusterListener clusterListener = new ClusterListener();
47        cache0.addListener(clusterListener);
48  
49        // Now we want to block the outgoing put to the backup owner
50        RpcManager rpcManager = TestingUtil.extractComponent(cache1, RpcManager.class);
51        ControlledRpcManager controlledRpcManager = new ControlledRpcManager(rpcManager);
52        controlledRpcManager.blockBefore(PutKeyValueCommand.class);
53        TestingUtil.replaceComponent(cache1, RpcManager.class, controlledRpcManager, true);
54  
55        final MagicKey key = new MagicKey(cache1, cache2);
56        Future<String> future = fork(new Callable<String>() {
57           @Override
58           public String call() throws Exception {
59              return cache0.put(key, FIRST_VALUE);
60           }
61        });
62  
63        // Wait until the primary owner has sent the put command successfully to  backup
64        controlledRpcManager.waitForCommandToBlock(10, TimeUnit.SECONDS);
65  
66        // Kill the cache now
67        TestingUtil.killCacheManagers(cache1.getCacheManager());
68  
69        // This should return null normally, but since it was retried it returns it's own value :(
70        // Maybe some day this can work properly
71        assertNull(future.get(10, TimeUnit.SECONDS));
72  
73        // We should have received an event that was marked as retried
74        assertEquals(clusterListener.events.size(), 1);
75        CacheEntryEvent<Object, String> event = clusterListener.events.get(0);
76        // Since it was a retry but the backup got the write the event isn't a CREATE!!
77        assertEquals(event.getType(), Event.Type.CACHE_ENTRY_CREATED);
78        CacheEntryCreatedEvent<Object, String> modEvent = (CacheEntryCreatedEvent<Object, String>)event;
79        assertTrue(modEvent.isCommandRetried());
80        assertEquals(modEvent.getKey(), key);
81        assertEquals(modEvent.getValue(), FIRST_VALUE);
82     }

84     public void testPrimaryOwnerGoesDownAfterBackupRaisesEvent() throws InterruptedException, TimeoutException,
85                                                                     ExecutionException, BrokenBarrierException {
86        final Cache<Object, String> cache0 = cache(0, CACHE_NAME);
87        Cache<Object, String> cache1 = cache(1, CACHE_NAME);
88        Cache<Object, String> cache2 = cache(2, CACHE_NAME);
89  
90        ClusterListener clusterListener = new ClusterListener();
91        cache0.addListener(clusterListener);
92  
93        CyclicBarrier barrier = new CyclicBarrier(2);
94        BlockingInterceptor blockingInterceptor = new BlockingInterceptor(barrier, PutKeyValueCommand.class, true, false);
95        cache2.getAdvancedCache().addInterceptorBefore(blockingInterceptor, EntryWrappingInterceptor.class);
96  
97        final MagicKey key = new MagicKey(cache1, cache2);
98        Future<String> future = fork(new Callable<String>() {
99           @Override
100          public String call() throws Exception {
101             return cache0.put(key, FIRST_VALUE);
102          }
103       });
104 
105       // Wait until the primary owner has sent the put command successfully to  backup
106       barrier.await(10, TimeUnit.SECONDS);
107 
108       // Remove the interceptor so the next command can proceed properly
109       cache2.getAdvancedCache().removeInterceptor(BlockingInterceptor.class);
110 
111       // Kill the cache now - note this will automatically unblock the fork thread
112       TestingUtil.killCacheManagers(cache1.getCacheManager());
113 
114       // Unblock the command
115       barrier.await(10, TimeUnit.SECONDS);
116 
117       // This should return null normally, but since it was retried it returns it's own value :(
118       // Maybe some day this can work properly
119       assertEquals(future.get(10, TimeUnit.SECONDS), FIRST_VALUE);
120 
121       assertEquals(clusterListener.events.size(), 2);
122       CacheEntryEvent<Object, String> event = clusterListener.events.get(0);
123       assertEquals(event.getType(), Event.Type.CACHE_ENTRY_CREATED);
124       CacheEntryCreatedEvent<Object, String> createEvent = (CacheEntryCreatedEvent<Object, String>)event;
125       assertFalse(createEvent.isCommandRetried());
126       assertEquals(createEvent.getKey(), key);
127       assertEquals(createEvent.getValue(), FIRST_VALUE);
128 
129       // We should have received an event that was marked as retried
130       event = clusterListener.events.get(1);
131       // Since it was a retry but the backup got the write the event isn't a CREATE!!
132       assertEquals(event.getType(), Event.Type.CACHE_ENTRY_MODIFIED);
133       CacheEntryModifiedEvent<Object, String> modEvent = (CacheEntryModifiedEvent<Object, String>)event;
134       assertTrue(modEvent.isCommandRetried());
135       assertEquals(modEvent.getKey(), key);
136       assertEquals(modEvent.getValue(), FIRST_VALUE);
137    }
