328    public void testSkip() throws Exception {
329       String filePath = "skip.txt";
330       writeToFile(filePath, "abcde" + "fghij" + "klmno" + "pqrst" + "uvwxy" + "z", 5);
331 
332       InputStream in = fs.getInput(filePath);
333       try {
334          long skipped = in.skip(2); // skip inside current chunk
335          assertEquals(skipped, 2);
336          assertEquals((char)in.read(), 'c');
337 
338          skipped = in.skip(2);  // skip to end of chunk
339          assertEquals(skipped, 2);
340          assertEquals((char)in.read(), 'f');
341 
342          skipped = in.skip(6); // skip into next chunk
343          assertEquals(skipped, 6);
344          assertEquals((char)in.read(), 'm');
345 
346          skipped = in.skip(9);      // skip _over_ next chunk
347          assertEquals(skipped, 9);
348          assertEquals((char)in.read(), 'w');
349 
350          skipped = in.skip(-1);  // negative skip
351          assertEquals(skipped, 0);
352          assertEquals((char)in.read(), 'x');
353 
354          skipped = in.skip(10);  // skip beyond EOF
355          assertEquals(skipped, 2);
356          assertEquals(in.read(), -1);
357       } finally {
358          in.close();
359       }
360    }

775    private void writeToFile(String filePath, String text, boolean append, Integer chunkSize) throws IOException {
776       OutputStream out = chunkSize == null

778          : fs.getOutput(filePath, append, chunkSize);

801    private static byte[] toBytes(InputStream is) throws IOException {
802       ByteArrayOutputStream buffer = new ByteArrayOutputStream();
803       int nRead;
804       byte[] data = new byte[16384];
805 
806       while ((nRead = is.read(data, 0, data.length)) != -1) {
807          buffer.write(data, 0, nRead);
808       }
809 
810       buffer.flush();
811       return buffer.toByteArray();
812    }
