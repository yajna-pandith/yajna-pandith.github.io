1   package org.infinispan.client.hotrod.impl.operations;
2   
3   import net.jcip.annotations.Immutable;
4   import org.infinispan.client.hotrod.Flag;
5   import org.infinispan.client.hotrod.impl.VersionedOperationResponse;
6   import org.infinispan.client.hotrod.impl.protocol.Codec;
7   import org.infinispan.client.hotrod.impl.protocol.HeaderParams;
8   import org.infinispan.client.hotrod.impl.transport.Transport;
9   import org.infinispan.client.hotrod.impl.transport.TransportFactory;
10  
11  import java.util.concurrent.atomic.AtomicInteger;
12  
13  /**
14   * Implements "removeIfUnmodified" operation as defined by
15   * <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod protocol specification</a>.
16   *
17   * @author Mircea.Markus@jboss.com
18   * @since 4.1
19   */
20  @Immutable
21  public class RemoveIfUnmodifiedOperation extends AbstractKeyOperation<VersionedOperationResponse> {
22  
23     private final long version;
24  
25     public RemoveIfUnmodifiedOperation(Codec codec, TransportFactory transportFactory,
26              byte[] key, byte[] cacheName, AtomicInteger topologyId, Flag[] flags, long version) {
27        super(codec, transportFactory, key, cacheName, topologyId, flags);
28        this.version = version;
29     }
30  
31     @Override
32     protected VersionedOperationResponse executeOperation(Transport transport) {
33        // 1) write header
34        HeaderParams params = writeHeader(transport, REMOVE_IF_UNMODIFIED_REQUEST);
35  
36        //2) write message body
37        transport.writeArray(key);
38        transport.writeLong(version);
39        transport.flush();
40  
41        //process response and return
42        return returnVersionedOperationResponse(transport, params);
43     }
44  }
