119     @Override
120     protected void populateModel(final OperationContext context, final ModelNode operation, final Resource resource) {
121 
122         final ModelNode model = resource.getModel();
123 
124         // handle the basic add() operation parameters
125         populate(operation, model);
126 
127         // add a step to initialize an *optional* TRANSPORT parameter
128         if (operation.hasDefined(ModelKeys.TRANSPORT)) {
129             // create an ADD operation to add the transport=TRANSPORT child
130             ModelNode addTransport = operation.get(ModelKeys.TRANSPORT).clone();
131 
132             addTransport.get(OPERATION_NAME).set(ModelDescriptionConstants.ADD);
133             ModelNode transportAddress = operation.get(OP_ADDR).clone();
134             transportAddress.add(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
135             transportAddress.protect();
136             addTransport.get(OP_ADDR).set(transportAddress);
137 
138             // execute the operation using the transport handler
139             context.addStep(addTransport, TransportResourceDefinition.TRANSPORT_ADD, OperationContext.Stage.MODEL, true);
140         }
141 
142         // add steps to initialize *optional* PROTOCOL parameters
143         if (operation.hasDefined(ModelKeys.PROTOCOLS)) {
144 
145             List<ModelNode> protocols = operation.get(ModelKeys.PROTOCOLS).asList();
146             // because we use stage IMMEDIATE when creating protocols, unless we reverse the order
147             // of the elements in the LIST, they will get applied in reverse order - the last step
148             // added gets executed first
149 
150             for (int i = protocols.size()-1; i >= 0; i--) {
151                 ModelNode protocol = protocols.get(i);
152 
153                 // create an ADD operation to add the protocol=* child
154                 ModelNode addProtocol = protocol.clone();
155                 addProtocol.get(OPERATION_NAME).set(ModelKeys.ADD_PROTOCOL);
156                 // add-protocol is a stack operation
157                 ModelNode protocolAddress = operation.get(OP_ADDR).clone();
158                 protocolAddress.protect();
159                 addProtocol.get(OP_ADDR).set(protocolAddress);
160 
161                 // execute the operation using the transport handler
162                 context.addStep(addProtocol, ProtocolResourceDefinition.PROTOCOL_ADD_HANDLER, OperationContext.Stage.MODEL, true);
163             }
164         }
165     }

177     protected void installRuntimeServices(OperationContext context, ModelNode operation, ModelNode model, ServiceVerificationHandler verificationHandler, List<ServiceController<?>> newControllers) throws OperationFailedException {
178 
179         final PathAddress address = PathAddress.pathAddress(operation.get(OP_ADDR));
180         final String name = address.getLastElement().getValue();
181 
182         // check that we have enough information to create a stack
183         protocolStackSanityCheck(name, model);
184 
185         // we need to preserve the order of the protocols as maintained by PROTOCOLS
186         // pick up the ordered protocols here as a List<Property> where property is <name, ModelNode>
187         List<Property> orderedProtocols = getOrderedProtocolPropertyList(model);
188 
189         // pick up the transport here and its values
190         ModelNode resolvedValue = null;
191         ModelNode transport = model.get(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
192         final String type = (resolvedValue = TransportResourceDefinition.TYPE.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
193         final boolean  shared = TransportResourceDefinition.SHARED.resolveModelAttribute(context, transport).asBoolean();
194         final String machine = (resolvedValue = TransportResourceDefinition.MACHINE.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
195         final String rack = (resolvedValue = TransportResourceDefinition.RACK.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
196         final String site = (resolvedValue = TransportResourceDefinition.SITE.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
197         final String timerExecutor = (resolvedValue = TransportResourceDefinition.TIMER_EXECUTOR.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
198         final String threadFactory = (resolvedValue = TransportResourceDefinition.THREAD_FACTORY.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
199         final String diagnosticsSocketBinding = (resolvedValue = TransportResourceDefinition.DIAGNOSTICS_SOCKET_BINDING.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
200         final String defaultExecutor = (resolvedValue = TransportResourceDefinition.DEFAULT_EXECUTOR.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
201         final String oobExecutor = (resolvedValue = TransportResourceDefinition.OOB_EXECUTOR.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
202         final String transportSocketBinding = (resolvedValue = TransportResourceDefinition.SOCKET_BINDING.resolveModelAttribute(context, transport)).isDefined() ? resolvedValue.asString() : null;
203 
204         // set up the transport
205         Transport transportConfig = new Transport(type);
206         transportConfig.setShared(shared);
207         transportConfig.setTopology(site, rack, machine);
208         initProtocolProperties(context, transport, transportConfig);
209 
210         Relay relayConfig = null;
211         List<Map.Entry<String, Injector<ChannelFactory>>> stacks = new LinkedList<Map.Entry<String, Injector<ChannelFactory>>>();
212         if (model.hasDefined(ModelKeys.RELAY)) {
213             final ModelNode relay = model.get(ModelKeys.RELAY, ModelKeys.RELAY_NAME);
214             final String siteName = RelayResourceDefinition.SITE.resolveModelAttribute(context, relay).asString();
215             relayConfig = new Relay(siteName);
216             initProtocolProperties(context, relay, relayConfig);
217             if (relay.hasDefined(ModelKeys.REMOTE_SITE)) {
218                 List<RemoteSiteConfiguration> remoteSites = relayConfig.getRemoteSites();
219                 for (Property remoteSiteProperty : relay.get(ModelKeys.REMOTE_SITE).asPropertyList()) {
220                     final String remoteSiteName = remoteSiteProperty.getName();
221                     final ModelNode remoteSite = remoteSiteProperty.getValue();
222                     final String clusterName = RemoteSiteResourceDefinition.CLUSTER.resolveModelAttribute(context, remoteSite)
223                             .asString();
224                     final String stack = RemoteSiteResourceDefinition.STACK.resolveModelAttribute(context, remoteSite).asString();
225                     final InjectedValue<ChannelFactory> channelFactory = new InjectedValue<ChannelFactory>();
226                     remoteSites.add(new RemoteSite(remoteSiteName, clusterName, channelFactory));
227                     stacks.add(new AbstractMap.SimpleImmutableEntry<String, Injector<ChannelFactory>>(stack, channelFactory));
228                 }
229             }
230         }
231 
232         Sasl saslConfig = null;
233         if (model.hasDefined(ModelKeys.SASL)) {
234             final ModelNode sasl = model.get(ModelKeys.SASL, ModelKeys.SASL_NAME);
235             final String clusterRole = (resolvedValue = SaslResourceDefinition.CLUSTER_ROLE.resolveModelAttribute(context, sasl)).isDefined() ? resolvedValue.asString() : null;
236             final String securityRealm = SaslResourceDefinition.SECURITY_REALM.resolveModelAttribute(context, sasl).asString();
237             final String mech = SaslResourceDefinition.MECH.resolveModelAttribute(context, sasl).asString();
238             saslConfig = new Sasl(securityRealm, mech, clusterRole);
239             initProtocolProperties(context, sasl, saslConfig);
240         }
241 
242         // set up the protocol stack Protocol objects
243         ProtocolStack stackConfig = new ProtocolStack(name, transportConfig, relayConfig, saslConfig);
244         List<Map.Entry<Protocol, String>> protocolSocketBindings = new ArrayList<Map.Entry<Protocol, String>>(orderedProtocols.size());
245         for (Property protocolProperty : orderedProtocols) {
246             ModelNode protocol = protocolProperty.getValue();
247             final String protocolType = (resolvedValue = ProtocolResourceDefinition.TYPE.resolveModelAttribute(context, protocol)).isDefined() ? resolvedValue.asString() : null;
248             Protocol protocolConfig = new Protocol(protocolType);
249             initProtocolProperties(context, protocol, protocolConfig);
250             stackConfig.getProtocols().add(protocolConfig);
251             final String protocolSocketBinding = (resolvedValue = ProtocolResourceDefinition.SOCKET_BINDING.resolveModelAttribute(context, protocol)).isDefined() ? resolvedValue.asString() : null;
252             protocolSocketBindings.add(new AbstractMap.SimpleImmutableEntry<Protocol, String>(protocolConfig, protocolSocketBinding));
253         }
254 
255         // install the default channel factory service
256         ServiceController<ChannelFactory> cfsController = installChannelFactoryService(context.getServiceTarget(),
257                         name, diagnosticsSocketBinding, defaultExecutor, oobExecutor, timerExecutor, threadFactory,
258                         transportSocketBinding, protocolSocketBindings, transportConfig, stackConfig, stacks, saslConfig, verificationHandler);
259         if (newControllers != null) {
260             newControllers.add(cfsController);
261         }
262     }

275     protected ServiceController<ChannelFactory> installChannelFactoryService(ServiceTarget target,
276                                                                              String name,
277                                                                              String diagnosticsSocketBinding,
278                                                                              String defaultExecutor,
279                                                                              String oobExecutor,
280                                                                              String timerExecutor,
281                                                                              String threadFactory,
282                                                                              String transportSocketBinding,
283                                                                              List<Map.Entry<Protocol, String>> protocolSocketBindings,
284                                                                              Transport transportConfig,
285                                                                              ProtocolStack stackConfig,
286                                                                              List<Map.Entry<String, Injector<ChannelFactory>>> stacks,
287                                                                              Sasl sasl,
288                                                                              ServiceVerificationHandler verificationHandler) {
289 
290         // create the channel factory service builder
291         ServiceBuilder<ChannelFactory> builder = target
292                 .addService(ChannelFactoryService.getServiceName(name), new ChannelFactoryService(stackConfig))
293                 .addDependency(ProtocolDefaultsService.SERVICE_NAME, ProtocolDefaults.class, stackConfig.getDefaultsInjector())
294                 .addDependency(MBeanServerService.SERVICE_NAME, MBeanServer.class, stackConfig.getMBeanServerInjector())
295                 .addDependency(ServerEnvironmentService.SERVICE_NAME, ServerEnvironment.class, stackConfig.getEnvironmentInjector())
296                 .setInitialMode(ServiceController.Mode.ON_DEMAND)
297         ;
298         // add transport dependencies
299         addSocketBindingDependency(builder, transportSocketBinding, transportConfig.getSocketBindingInjector());
300 
301         for (Map.Entry<Protocol, String> entry: protocolSocketBindings) {
302             addSocketBindingDependency(builder, entry.getValue(), entry.getKey().getSocketBindingInjector());
303         }
304 
305         // add remaining dependencies
306         addSocketBindingDependency(builder, diagnosticsSocketBinding, transportConfig.getDiagnosticsSocketBindingInjector());
307         addExecutorDependency(builder, defaultExecutor, transportConfig.getDefaultExecutorInjector());
308         addExecutorDependency(builder, oobExecutor, transportConfig.getOOBExecutorInjector());
309         if (timerExecutor != null) {
310             builder.addDependency(ThreadsServices.executorName(timerExecutor), ScheduledExecutorService.class, transportConfig.getTimerExecutorInjector());
311         }
312         if (threadFactory != null) {
313             builder.addDependency(ThreadsServices.threadFactoryName(threadFactory), ThreadFactory.class, transportConfig.getThreadFactoryInjector());
314         }
315         if (sasl != null) {
316             builder.addDependency(SecurityRealm.ServiceUtil.createServiceName(sasl.getRealm()), SecurityRealm.class, sasl.getSecurityRealmInjector());
317         }
318         for (Map.Entry<String, Injector<ChannelFactory>> entry: stacks) {
319             builder.addDependency(ChannelFactoryService.getServiceName(entry.getKey()), ChannelFactory.class, entry.getValue());
320         }
321         return builder.install();
322     }

324     private void initProtocolProperties(OperationContext context, ModelNode protocol, Protocol protocolConfig) throws OperationFailedException {
325 
326         Map<String, String> properties = protocolConfig.getProperties();
327         // properties are a child resource of protocol
328         if (protocol.hasDefined(ModelKeys.PROPERTY)) {
329             for (Property property : protocol.get(ModelKeys.PROPERTY).asPropertyList()) {
330                 // the format of the property elements
331                 //  "property" => {
332                 //       "relative-to" => {"value" => "fred"},
333                 //   }
334                 String propertyName = property.getName();
335                 String propertyValue = PropertyResourceDefinition.VALUE.resolveModelAttribute(context, property.getValue()).asString();
336                 if (propertyValue != null && !propertyValue.isEmpty()) {
337                    properties.put(propertyName, propertyValue);
338                 }
339             }
340         }
341     }

343     public static List<Property> getOrderedProtocolPropertyList(ModelNode stack) {
344         ModelNode orderedProtocols = new ModelNode();
345 
346         // check for the empty ordering list
347         if  (!stack.hasDefined(ModelKeys.PROTOCOLS)) {
348             return null;
349         }
350         // PROTOCOLS is a list of protocol names only, reflecting the order in which protocols were added to the stack
351         List<ModelNode> protocolOrdering = stack.get(ModelKeys.PROTOCOLS).asList();
352 
353         // now construct an ordered list of the full protocol model nodes
354         ModelNode unorderedProtocols = stack.get(ModelKeys.PROTOCOL);
355         for (ModelNode protocolName : protocolOrdering) {
356             ModelNode protocolModel = unorderedProtocols.get(protocolName.asString());
357             orderedProtocols.add(protocolName.asString(), protocolModel);
358         }
359         return orderedProtocols.asPropertyList();
360     }

374     /*
375      * A check that we have the minimal configuration required to create a protocol stack.
376      */
377     private void protocolStackSanityCheck(String stackName, ModelNode model) throws OperationFailedException {
378 
379          ModelNode transport = model.get(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
380          if (!transport.isDefined()) {
381             throw JGroupsMessages.MESSAGES.transportNotDefined(stackName);
382          }
383 
384          List<Property> protocols = getOrderedProtocolPropertyList(model);
385          if ( protocols == null || !(protocols.size() > 0)) {
386              throw JGroupsMessages.MESSAGES.protocolListNotDefined(stackName);
387          }
388     }

664     static class Protocol implements ProtocolConfiguration {
665         private final String name;
666         private final InjectedValue<SocketBinding> socketBinding = new InjectedValue<SocketBinding>();
667         private final Map<String, String> properties = new HashMap<String, String>();
668         final Class<?> protocolClass;
669 
670         Protocol(final String name) {
671             this.name = name;
672             PrivilegedAction<Class<?>> action = new PrivilegedAction<Class<?>>() {
673                 @Override
674                 public Class<?> run() {
675                     try {
676                         return Protocol.this.getClass().getClassLoader().loadClass(org.jgroups.conf.ProtocolConfiguration.protocol_prefix + '.' + name);
677                     } catch (ClassNotFoundException e) {
678                         throw new IllegalStateException(e);
679                     }
680                 }
681             };
682             this.protocolClass = AccessController.doPrivileged(action);
683         }
684 
685         @Override
686         public String getName() {
687             return this.name;
688         }
689 
690         @Override
691         public boolean hasProperty(final String property) {
692             PrivilegedAction<Field> action = new PrivilegedAction<Field>() {
693                 @Override
694                 public Field run() {
695                     return getField(Protocol.this.protocolClass, property);
696                 }
697             };
698             return AccessController.doPrivileged(action) != null;
699         }
700 
701         static Field getField(Class<?> targetClass, String property) {
702             try {
703                 return targetClass.getDeclaredField(property);
704             } catch (NoSuchFieldException e) {
705                 Class<?> superClass = targetClass.getSuperclass();
706                 return (superClass != null) && org.jgroups.stack.Protocol.class.isAssignableFrom(superClass) ? getField(superClass, property) : null;
707             }
708         }
709 
710         @Override
711         public Map<String, String> getProperties() {
712             return this.properties;
713         }
714 
715         Injector<SocketBinding> getSocketBindingInjector() {
716             return this.socketBinding;
717         }
718 
719         @Override
720         public SocketBinding getSocketBinding() {
721             return this.socketBinding.getOptionalValue();
722         }
