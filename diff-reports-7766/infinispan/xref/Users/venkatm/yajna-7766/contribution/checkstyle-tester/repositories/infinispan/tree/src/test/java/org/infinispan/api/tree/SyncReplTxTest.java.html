26  @Test(groups = "functional", testName = "api.tree.SyncReplTxTest")
27  public class SyncReplTxTest extends MultipleCacheManagersTest {
28     private TreeCache<Object, Object> cache1, cache2;
29  
30     protected void createCacheManagers() throws Throwable {
31        ConfigurationBuilder cb = getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, true);
32        cb.invocationBatching().enable();
33  
34        createClusteredCaches(2, "replSync", cb);
35  
36        Cache c1 = cache(0, "replSync");
37        Cache c2 = cache(1, "replSync");
38  
39        cache1 = new TreeCacheImpl<Object, Object>(c1);
40        cache2 = new TreeCacheImpl<Object, Object>(c2);
41     }
42  
43     private TransactionManager beginTransaction(Cache cache) throws NotSupportedException, SystemException {
44        TransactionManager mgr = cache.getAdvancedCache().getTransactionManager();
45        mgr.begin();
46        return mgr;
47     }
48  
49     public void testBasicOperation() throws SystemException, NotSupportedException, HeuristicMixedException, HeuristicRollbackException, RollbackException {
50        assertClusterSize("Should only be 2  caches in the cluster!!!", 2);
51  
52        Fqn f = Fqn.fromString("/test/data");
53        String k = "key", v = "value";
54  
55        assertNull("Should be null", cache1.getRoot().getChild(f));
56        assertNull("Should be null", cache2.getRoot().getChild(f));
57  
58        Node<Object, Object> node = cache1.getRoot().addChild(f);
59  
60        assertNotNull("Should not be null", node);
61  
62        TransactionManager tm = beginTransaction(cache1.getCache());
63        node.put(k, v);
64        tm.commit();
65  
66        assertEquals(v, node.get(k));
67        assertEquals(v, cache1.get(f, k));
68        assertEquals("Should have replicated", v, cache2.get(f, k));
69     }
