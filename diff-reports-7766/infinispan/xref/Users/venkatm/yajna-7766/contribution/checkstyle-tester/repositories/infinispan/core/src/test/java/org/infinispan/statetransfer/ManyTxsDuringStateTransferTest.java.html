34  @Test(testName = "lock.ManyTxsDuringStateTransferTest", groups = "functional")
35  @CleanupAfterMethod
36  public class ManyTxsDuringStateTransferTest extends MultipleCacheManagersTest {
37     public static final String CACHE_NAME = "testCache";
38     private static final int NUM_TXS = 20;
39  
40     @Override
41     protected void createCacheManagers() throws Throwable {
42        ConfigurationBuilder defaultBuilder = new ConfigurationBuilder();
43        addClusterEnabledCacheManager(getGlobalConfigurationBuilder(), defaultBuilder);
44        addClusterEnabledCacheManager(getGlobalConfigurationBuilder(), defaultBuilder);
45        waitForClusterToForm();
46     }
47  
48     private GlobalConfigurationBuilder getGlobalConfigurationBuilder() {
49        GlobalConfigurationBuilder globalBuilder = GlobalConfigurationBuilder.defaultClusteredBuilder();
50        BlockingThreadPoolExecutorFactory threadPoolFactory = new BlockingThreadPoolExecutorFactory(1, 1, 0, Thread.NORM_PRIORITY);
51        globalBuilder.transport().remoteCommandThreadPool().threadPoolFactory(threadPoolFactory);
52        return globalBuilder;
53     }
54  
55     public void testManyTxs() throws Throwable {
56        ConfigurationBuilder cfg = TestCacheManagerFactory.getDefaultCacheConfiguration(true);
57        cfg.clustering().cacheMode(CacheMode.DIST_SYNC)
58              .stateTransfer().awaitInitialTransfer(false)
59              .transaction().lockingMode(LockingMode.OPTIMISTIC);
60        manager(0).defineConfiguration(CACHE_NAME, cfg.build());
61        manager(1).defineConfiguration(CACHE_NAME, cfg.build());
62  
63        final CheckPoint checkpoint = new CheckPoint();
64        final AdvancedCache<Object, Object> cache0 = advancedCache(0, CACHE_NAME);
65        final TransactionManager tm0 = cache0.getTransactionManager();
66  
67        // Block state request commands on cache 0
68        StateProvider stateProvider = TestingUtil.extractComponent(cache0, StateProvider.class);
69        StateProvider spyProvider = spy(stateProvider);
70        doAnswer(new Answer<Object>() {
71           @Override
72           public Object answer(InvocationOnMock invocation) throws Throwable {
73              Object[] arguments = invocation.getArguments();
74              Address source = (Address) arguments[0];
75              int topologyId = (Integer) arguments[1];
76              Object result = invocation.callRealMethod();
77              checkpoint.trigger("post_get_transactions_" + topologyId + "_from_" + source);
78              checkpoint.awaitStrict("resume_get_transactions_" + topologyId + "_from_" + source, 10, SECONDS);
79              return result;
80           }
81        }).when(spyProvider).getTransactionsForSegments(any(Address.class), anyInt(), anySetOf(Integer.class));
82        TestingUtil.replaceComponent(cache0, StateProvider.class, spyProvider, true);
83  
84        // Start cache 1, but the tx data request will be blocked on cache 0
85        StateTransferManager stm0 = TestingUtil.extractComponent(cache0, StateTransferManager.class);
86        int initialTopologyId = stm0.getCacheTopology().getTopologyId();
87        int rebalanceTopologyId = initialTopologyId + 1;
88        AdvancedCache<Object, Object> cache1 = advancedCache(1, CACHE_NAME);
89        checkpoint.awaitStrict("post_get_transactions_" + rebalanceTopologyId + "_from_" + address(1), 10, SECONDS);
90  
91        // Start many transaction on cache 0, which will block on cache 1
92        Future<Object>[] futures = new Future[NUM_TXS];
93        for (int i = 0; i < NUM_TXS; i++) {
94           // The rollback command should be invoked on cache 1 and it should block until the tx is created there
95           final int ii = i;
96           futures[i] = fork(new Callable<Object>() {
97              @Override
98              public Object call() throws Exception {
99                 tm0.begin();
100                cache0.put("testkey" + ii, "v" + ii);
101                tm0.commit();
102                return null;
103             }
104          });
105       }
106 
107       // Wait for all (or at least most of) the txs to be replicated to cache 1
108       Thread.sleep(1000);
109 
110       // Let cache 1 receive the tx from cache 0.
111       checkpoint.trigger("resume_get_transactions_" + rebalanceTopologyId + "_from_" + address(1));
112       TestingUtil.waitForRehashToComplete(caches(CACHE_NAME));
113 
114       // Wait for the txs to finish and check the results
115       DataContainer dataContainer0 = TestingUtil.extractComponent(cache0, DataContainer.class);
116       DataContainer dataContainer1 = TestingUtil.extractComponent(cache1, DataContainer.class);
117       for (int i = 0; i < NUM_TXS; i++) {
118          futures[i].get(10, SECONDS);
119          assertEquals("v" + i, dataContainer0.get("testkey" + i).getValue());
120          assertEquals("v" + i, dataContainer1.get("testkey" + i).getValue());
121       }
122 
123       // Check for stale locks
124       final TransactionTable tt0 = TestingUtil.extractComponent(cache0, TransactionTable.class);
125       final TransactionTable tt1 = TestingUtil.extractComponent(cache1, TransactionTable.class);
126       eventually(new Condition() {
127          @Override
128          public boolean isSatisfied() throws Exception {
129             return tt0.getLocalTxCount() == 0 && tt1.getRemoteTxCount() == 0;
130          }
131       });
132    }
