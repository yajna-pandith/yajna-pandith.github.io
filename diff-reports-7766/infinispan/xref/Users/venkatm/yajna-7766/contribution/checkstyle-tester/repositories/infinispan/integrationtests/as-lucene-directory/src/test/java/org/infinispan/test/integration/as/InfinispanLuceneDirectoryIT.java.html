101    @Test
102    public void testQParserLucene4() throws IOException, ParseException {
103       Document document = buildLuceneDoc("field1", "The quick brown fox jumped over the lazy dog");
104       index(document);
105       Query matchingQuery = buildQuery("field1:box~");
106       Query nonMatchingQuery = buildQuery("-field1:over");
107       IndexSearcher indexSearcher = openSearcher();
108 
109       assertEquals(1, indexSearcher.search(matchingQuery, 1).totalHits);
110       assertEquals(0, indexSearcher.search(nonMatchingQuery, 1).totalHits);
111    }
112 
113    @Test
114    public void testGrouping() throws IOException {
115       Document doc1 = buildLuceneDoc("field1", "value1");
116       Document doc2 = buildLuceneDoc("field1", "value1");
117       Document doc3 = buildLuceneDoc("field1", "value2");
118       index(doc1, doc2, doc3);
119       GroupingSearch groupingSearch = new GroupingSearch("field1");
120       TopGroups<Object> topGroups = groupingSearch.search(openSearcher(), new MatchAllDocsQuery(), 0, 10);
121 
122       assertEquals(3, topGroups.totalHitCount);
123       assertEquals(2, topGroups.groups.length);
124    }

126    @Test
127    public void testFaceting() throws Exception {
128       Directory taxonomyDirectory =
129             DirectoryBuilder.newDirectoryInstance(cache, cache, cache, "taxonomy").create();
130       DirectoryTaxonomyWriter tw = new DirectoryTaxonomyWriter(taxonomyDirectory);
131       FacetsConfig cfg = new FacetsConfig();
132       Document doc1 = new Document();
133       Document doc2 = new Document();
134       Document doc3 = new Document();
135       doc1.add(new FacetField("category", "c2"));
136       doc2.add(new FacetField("category", "c2"));
137       doc3.add(new FacetField("category", "c1"));
138       index(cfg.build(tw, doc1), cfg.build(tw, doc2), cfg.build(tw, doc3));
139       tw.close();
140       DirectoryTaxonomyReader tr = new DirectoryTaxonomyReader(taxonomyDirectory);
141       FacetsCollector fc = new FacetsCollector();
142       FacetsCollector.search(openSearcher(), new MatchAllDocsQuery(), 10, fc);
143       Facets facets = new FastTaxonomyFacetCounts(tr, cfg, fc);
144       FacetResult category = facets.getTopChildren(10, "category");
145 
146       assertEquals(2, category.childCount);
147 
148    }

150    private Query buildQuery(String q) throws ParseException {
151       QueryParser queryParser = new QueryParser("field1", new WhitespaceAnalyzer());
152       return queryParser.parse(q);
153    }
154 
155    private IndexSearcher openSearcher() throws IOException {
156       return new IndexSearcher(DirectoryReader.open(directory));
157    }
158 
159    private Terms terms(String field) throws IOException {
160       DirectoryReader reader = DirectoryReader.open(directory);
161       AtomicReaderContext readerContext = reader.getContext().leaves().iterator().next();
162       return readerContext.reader().terms(field);
163    }
164 
165    private void index(Document... documents) throws IOException {
166       IndexWriterConfig iwc = new IndexWriterConfig(LUCENE_VERSION, new WhitespaceAnalyzer());
167       IndexWriter indexWriter = new IndexWriter(directory, iwc);
168       for (Document doc : documents) {
169          indexWriter.addDocument(doc);
170       }
171       indexWriter.close();
172    }
173 
174    private Document buildLuceneDoc(String fieldName, String contents) {
175       Document document = new Document();
176       FieldType fieldType = new FieldType();
177       fieldType.setIndexed(true);
178       document.add(new Field(fieldName, contents, fieldType));
179       return document;
180    }
