88     private String greatestDist(SortedMap<Integer, Address> pos, int hashSpace) {
89        // calc distances between entries 0 and n - 1 first.
90        int largest = 0;
91        int lastPos = lastEntry(pos).getKey();
92        int firstPos = -1;
93        for (int currentPos: pos.keySet()) {
94           if (firstPos == -1) firstPos = currentPos;
95           largest = Math.max(largest, Math.abs(currentPos - lastPos));
96           lastPos = currentPos;
97        }
98  
99        return String.valueOf(largest);
100    }
101 
102    private String smallestDist(SortedMap<Integer, Address> pos, int hashSpace) {
103       // calc distances between entries 0 and n - 1 first.
104       int smallest = Integer.MAX_VALUE;
105       int lastPos = lastEntry(pos).getKey();
106       int firstPos = -1;
107       for (int currentPos: pos.keySet()) {
108          if (firstPos == -1) firstPos = currentPos;
109          smallest = Math.min(smallest, Math.abs(currentPos - lastPos));
110          lastPos = currentPos;
111       }
112       return String.valueOf(smallest);
113    }
114 
115    private String meanDist(SortedMap<Integer, Address> pos, int hashSpace) {
116       // calc distances between entries 0 and n - 1 first.
117       int totalDist = 0;
118       int lastPos = lastEntry(pos).getKey();
119       int firstPos = -1;
120       for (int currentPos: pos.keySet()) {
121          if (firstPos == -1) firstPos = currentPos;         
122          totalDist += Math.abs(currentPos - lastPos);
123          lastPos = currentPos;
124       }
125 
126       return String.valueOf(totalDist / pos.size());
127    }
128 
129    private Map.Entry<Integer, Address> lastEntry(SortedMap<Integer, Address> m) {
130       Map.Entry<Integer, Address> last = null;
131       for (Map.Entry<Integer, Address> e: m.entrySet()) last = e;
132       return last;
133    }

160    private void perform(Collection<HashFunction> functions, Set<Object> objectKeys, Set<String> stringKeys, Set<byte[]> byteArrayKeys, boolean warmup) {
161 
162       if (!warmup)
163          System.out.printf("%s %s %s %s%n", padString("Function Impl", 25), padString("String keys", 18), padString("Byte array keys", 18), padString("Object keys", 18));
164 
165       for (HashFunction f : functions) {
166          long oRes = 0, sRes = 0, bRes = 0;
167          DescriptiveStatistics oStats = new DescriptiveStatistics();
168          DescriptiveStatistics sStats = new DescriptiveStatistics();
169          DescriptiveStatistics bStats = new DescriptiveStatistics();
170 
171          long st = System.currentTimeMillis();
172          for (Object o : objectKeys) captureStats(f.hash(o.hashCode()), oStats);
173          oRes = System.currentTimeMillis() - st;
174 
175          st = System.currentTimeMillis();
176          for (String s : stringKeys) captureStats(f.hash(s), sStats);
177          sRes = System.currentTimeMillis() - st;
178 
179          st = System.currentTimeMillis();
180          for (byte[] b : byteArrayKeys) captureStats(f.hash(b), bStats);
181          bRes = System.currentTimeMillis() - st;
182 
183          if (!warmup) {
184             System.out.printf("%s %s %s %s%n",
185                     padString(f.functionName(), 25),
186                     padString(prettyPrintTime(sRes), 18),
187                     padString(prettyPrintTime(bRes), 18),
188                     padString(prettyPrintTime(oRes), 18)
189             );
190             System.out.printf("%s %s %s %s%n",
191                     padString("  mean", 25),
192                     padDouble(sStats.getMean()),
193                     padDouble(bStats.getMean()),
194                     padDouble(oStats.getMean())
195             );
196             System.out.printf("%s %s %s %s%n",
197                     padString("  median", 25),
198                     padDouble(sStats.getPercentile(50.0)),

201             );

206                     padDouble(oStats.getStandardDeviation())

215          }
216       }

224 abstract class HashFunction {
225    abstract int hash(byte[] payload);
226 
227    public int hash(String payload) {
228       return hash(payload.getBytes(Charset.forName("UTF-8")));
229    }
230 
231    public int hash(int hashcode) {
232       byte[] b = new byte[4];
233       b[0] = (byte) hashcode;
234       b[1] = (byte) (hashcode >> 8);
235       b[2] = (byte) (hashcode >> 16);
236       b[3] = (byte) (hashcode >> 24);
237       return hash(b);
238    }
239 
240    abstract String functionName();
241 }

243 class SuperFastHash extends HashFunction {
244 
245    @Override
246    public int hash(byte[] data) {
247       if (data == null || data.length == 0) return 0;
248       int len = data.length;
249       int hash = len;
250       int rem = len & 3;
251       len >>= 2;
252       int tmp;
253       int offset = 0;
254       /* Main loop */
255       for (; len > 0; len--) {
256          hash += get16bits(data, offset);
257          tmp = (get16bits(data, offset + 2) << 11) ^ hash;
258          hash = (hash << 16) ^ tmp;
259          offset += 4;
260          hash += hash >> 11;
261       }
262 
263       /* Handle end cases */
264       switch (rem) {
265          case 3:
266             hash += get16bits(data, offset);
267             hash ^= hash << 16;
268             hash ^= data[2] << 18;
269             hash += hash >> 11;
270             break;
271          case 2:
272             hash += get16bits(data, offset);
273             hash ^= hash << 11;
274             hash += hash >> 17;
275             break;
276          case 1:
277             hash += data[0];
278             hash ^= hash << 10;
279             hash += hash >> 1;
280       }
281 
282       /* Force "avalanching" of final 127 bits */
283       hash ^= hash << 3;
284       hash += hash >> 5;
285       hash ^= hash << 4;
286       hash += hash >> 17;
287       hash ^= hash << 25;
288       hash += hash >> 6;
289 
290       return hash;
291    }
292 
293    private int get16bits(byte[] bytes, int offset) {
294       short s = bytes[offset];
295       s &= bytes[offset + 1] << 8;
296       return s;
297    }
298 
299    @Override
300    public String functionName() {
301       return "SuperFastHash";
302    }
303 }
