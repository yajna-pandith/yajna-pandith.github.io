91     private void doTest(final TestWriteOperation op) throws Exception {
92        // Test scenario:
93        // cache0 is the only member in the cluster, cache1 joins
94        // Key k is in the cache, and is transferred to cache1
95        // A user operation/tx also modifies key k
96        // The user tx must update k even if it doesn't find key k in the data container.
97        final AdvancedCache<Object, Object> cache0 = advancedCache(0);
98        final String key = "key";
99  
100       // Prepare for replace/remove: put a previous value in cache0
101       final Object previousValue = op.getPreviousValue();
102       if (previousValue != null) {
103          cache0.put(key, previousValue);
104          assertEquals(previousValue, cache0.get(key));
105          log.tracef("Previous value inserted: %s = %s", key, previousValue);
106       }
107 
108       int preJoinTopologyId = cache0.getComponentRegistry().getStateTransferManager().getCacheTopology().getTopologyId();
109 
110       // Block any state response commands on cache0
111       CheckPoint checkPoint = new CheckPoint();
112       ControlledRpcManager blockingRpcManager0 = blockStateResponseCommand(cache0);
113 
114       // Block the rebalance confirmation on cache0
115       blockRebalanceConfirmation(manager(0), checkPoint);
116 
117       // Start the joiner
118       log.tracef("Starting the cache on the joiner");
119       ConfigurationBuilder c = getConfigurationBuilder();
120       c.clustering().stateTransfer().awaitInitialTransfer(false);
121       addClusterEnabledCacheManager(c);
122 
123       final AdvancedCache<Object,Object> cache1 = advancedCache(1);
124       int rebalanceTopologyId = preJoinTopologyId + 1;
125 
126       // Wait for the write CH to contain the joiner everywhere
127       eventually(new Condition() {
128          @Override
129          public boolean isSatisfied() throws Exception {
130             return cache0.getRpcManager().getMembers().size() == 2 &&
131                   cache1.getRpcManager().getMembers().size() == 2;
132          }
133       });
134 
135       // Every PutKeyValueCommand will be blocked before committing the entry on cache1
136       CyclicBarrier beforeCommitCache1Barrier = new CyclicBarrier(2);
137       BlockingInterceptor blockingInterceptor1 = new BlockingInterceptor(beforeCommitCache1Barrier,
138             op.getCommandClass(), true, false);
139       cache1.addInterceptorAfter(blockingInterceptor1, EntryWrappingInterceptor.class);
140 
141       // Wait for cache0 to collect the state to send to cache1 (including our previous value).
142       blockingRpcManager0.waitForCommandToBlock();
143 
144       // Put/Replace/Remove from cache0 with cache0 as primary owner, cache1 will become a backup owner for the retry
145       // The put command will be blocked on cache1 just before committing the entry.
146       Future<Object> future = fork(new Callable<Object>() {
147          @Override
148          public Object call() throws Exception {
149             return op.perform(cache0, key);
150          }
151       });
152 
153       // Wait for the entry to be wrapped on cache1
154       beforeCommitCache1Barrier.await(10, TimeUnit.SECONDS);
155 
156       // Allow the state to be applied on cache1 (writing the old value for our entry)
157       blockingRpcManager0.stopBlocking();
158 
159       // Wait for cache1 to finish applying the state, but don't allow the rebalance confirmation to be processed.
160       // (It would change the topology and it would trigger a retry for the command.)
161       checkPoint.awaitStrict("pre_rebalance_confirmation_" + rebalanceTopologyId + "_from_" + address(1), 10, SECONDS);
162 
163       // Now allow the command to commit on cache1
164       beforeCommitCache1Barrier.await(10, TimeUnit.SECONDS);
165 
166       // Wait for the command to finish and check that it didn't fail
167       Object result = future.get(10, TimeUnit.SECONDS);
168       assertEquals(op.getReturnValue(), result);
169       log.tracef("%s operation is done", op);
170 
171       // Allow the rebalance confirmation to proceed and wait for the topology to change everywhere
172       checkPoint.trigger("resume_rebalance_confirmation_" + rebalanceTopologyId + "_from_" + address(0));
173       checkPoint.trigger("resume_rebalance_confirmation_" + rebalanceTopologyId + "_from_" + address(1));
174       TestingUtil.waitForRehashToComplete(cache0, cache1);
175 
176       // Check the value on all the nodes
177       assertEquals(op.getValue(), cache0.get(key));
178       assertEquals(op.getValue(), cache1.get(key));
179    }

181    private ControlledRpcManager blockStateResponseCommand(final Cache cache) throws InterruptedException {
182       RpcManager rpcManager = TestingUtil.extractComponent(cache, RpcManager.class);
183       ControlledRpcManager controlledRpcManager = new ControlledRpcManager(rpcManager);
184       controlledRpcManager.blockBefore(StateResponseCommand.class);
185       TestingUtil.replaceComponent(cache, RpcManager.class, controlledRpcManager, true);
186       return controlledRpcManager;
187    }
188 
189    private void blockRebalanceConfirmation(final EmbeddedCacheManager manager, final CheckPoint checkPoint)
190          throws Exception {
191       ClusterTopologyManager ctm = TestingUtil.extractGlobalComponent(manager, ClusterTopologyManager.class);
192       ClusterTopologyManager spyManager = spy(ctm);
193       doAnswer(new Answer<Object>() {
194          @Override
195          public Object answer(InvocationOnMock invocation) throws Throwable {
196             Object[] arguments = invocation.getArguments();
197             Address source = (Address) arguments[1];
198             int topologyId = (Integer) arguments[2];
199             checkPoint.trigger("pre_rebalance_confirmation_" + topologyId + "_from_" + source);
200             checkPoint.awaitStrict("resume_rebalance_confirmation_" + topologyId + "_from_" + source, 10, SECONDS);
201             return invocation.callRealMethod();
202          }
203       }).when(spyManager).handleRebalanceCompleted(anyString(), any(Address.class), anyInt(), any(Throwable.class),
204             anyInt());
205       TestingUtil.replaceComponent(manager, ClusterTopologyManager.class, spyManager, true);
206    }
