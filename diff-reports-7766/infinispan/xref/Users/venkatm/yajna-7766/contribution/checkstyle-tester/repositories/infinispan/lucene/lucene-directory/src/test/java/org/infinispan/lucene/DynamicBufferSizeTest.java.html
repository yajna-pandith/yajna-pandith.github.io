18  /**
19   * The chunk size might be changed after a segment has been created.
20   * This test verifies an existing index is still readable.
21   *
22   * @author Sanne Grinovero
23   * @since 4.1
24   */
25  @Test(groups = "functional", testName = "lucene.DynamicBufferSizeTest")
26  public class DynamicBufferSizeTest extends SingleCacheManagerTest {
27  
28     @Override
29     protected EmbeddedCacheManager createCacheManager() throws Exception {
30        ConfigurationBuilder builder = TestCacheManagerFactory.getDefaultCacheConfiguration(true);
31        builder.clustering().cacheMode(CacheMode.LOCAL)
32              .invocationBatching().enable();
33        return TestCacheManagerFactory.createCacheManager(builder);
34     }
35  
36     @Test
37     public void roundingTest() {
38        FileMetadata m = new FileMetadata(10);
39        AssertJUnit.assertEquals(0, m.getNumberOfChunks());
40        m.setSize(10);
41        AssertJUnit.assertEquals(1, m.getNumberOfChunks());
42        m.setSize(11);
43        AssertJUnit.assertEquals(2, m.getNumberOfChunks());
44        m = new FileMetadata(11);
45        m.setSize(11);
46        AssertJUnit.assertEquals(1, m.getNumberOfChunks());
47        m.setSize(22);
48        AssertJUnit.assertEquals(2, m.getNumberOfChunks());
49        m.setSize(31);
50        m = new FileMetadata(10);
51        m.setSize(31);
52        AssertJUnit.assertEquals(4, m.getNumberOfChunks());
53     }
54  
55     @Test
56     public void testReadingFromDifferentlySizedBuffers() throws IOException {
57        cache = cacheManager.getCache();
58        Directory dirA = DirectoryBuilder.newDirectoryInstance(cache, cache, cache, "indexName").chunkSize(7).create();
59        writeTextToIndex(dirA, 0, "hi from node A");
60        Directory dirB = DirectoryBuilder.newDirectoryInstance(cache, cache, cache, "indexName").chunkSize(8).create();
61        assertTextIsFoundInIds(dirB, "hi", 0);
62        writeTextToIndex(dirB, 1, "index B is sharing the same index but using a differently sized chunk size");
63        assertTextIsFoundInIds(dirA, "size", 1);
64     }
65  
66     @Test
67     public void testFileMetaData() {
68        FileMetadata data1 = new FileMetadata(1024);
69        FileMetadata data2 = new FileMetadata(2048);
70        FileMetadata data3 = new FileMetadata(1024);
71  
72        FileMetadata data4 = data1;
73  
74  
75        assert !data1.equals(new FileCacheKey("testIndex", "testFile"));
76        AssertJUnit.assertNotNull(data1);
77        assert data1.equals(data4);
78  
79        assert data1.equals(data3);
80  
81        data3.setSize(2048);
82        assert !data1.equals(data3);
83  
84        assert !data1.equals(data2);
85  
86        AssertJUnit.assertEquals("FileMetadata{ size=" + data1.getSize() + '}', data1.toString());
87     }
