314    public void testConcurrentMoveSiblings() throws Exception {
315       // tests a tree structure as such:
316       // /a/x, a/y, /b, /c
317       // N threads try to move /a/x from /a to /b and /a/y from /a to /c
318       final int N = 5;
319       final Fqn X = Fqn.fromString("/x"), Y = Fqn.fromString("/y");
320 
321       // set up the initial structure.
322       Node<Object, Object> rootNode = treeCache.getRoot();
323       Node<Object, Object> nodeA = rootNode.addChild(A);
324       nodeA.addChild(X);
325       nodeA.addChild(Y);
326       Node<Object, Object> nodeB = rootNode.addChild(B);
327       Node<Object, Object> nodeC = rootNode.addChild(C);
328 
329       Callable<Object>[] movers = new Callable[N];
330       for (int i = 0; i < N; i++) {
331          final Fqn source = Fqn.fromRelativeFqn(A, i % 2 == 0 ? X : Y);
332          final Fqn dest = i % 2 == 0 ? B : C;
333          movers[i] = new Callable<Object>() {
334             public Object call() {
335                try {
336                   treeCache.move(source, dest);
337                } catch (Exception e) {
338                   // expected on some of the threads, in optimistic mode
339                }
340                return null;
341             }
342          };
343       }
344 
345       runConcurrently(movers);
346 
347       log.info("Tree: " + TreeStructureSupport.printTree(treeCache, true));
348       assertNoLocks();
349       // Any of the move operations may fail, so just check that each node exists in exactly one place
350       assertTrue(nodeA.getChildrenNames().contains("x") ^ nodeB.getChildrenNames().contains("x"));
351       assertTrue(nodeA.getChildrenNames().contains("y") ^ nodeC.getChildrenNames().contains("y"));
352    }

392    public void testConcurrentMoveSameNode() throws Exception {
393       // set up the initial structure
394       // /a, /b, /c
395       // one thread tries to move /c under /a, another tries to move /c under /b
396       Node<Object, Object> rootNode = treeCache.getRoot();
397       final Fqn FQN_A = A, FQN_B = B, FQN_C = C;
398       Node nodeA = rootNode.addChild(FQN_A);
399       Node nodeB = rootNode.addChild(FQN_B);
400       Node nodeC = rootNode.addChild(FQN_C);
401 
402       final CountDownLatch nodeReadLatch = new CountDownLatch(1);
403       final CountDownLatch nodeMovedLatch = new CountDownLatch(1);
404 
405       // tries to move C under B right after another thread already moved C under A
406       Callable<Object> moveCtoB = new Callable<Object>() {
407          public Object call() throws Exception {
408             tm().begin();
409             try {
410                // ensure we already 'see' node C in this tx. this is what actually triggers the issue.
411                assertEquals(asSet("a", "b", "c"), treeCache.getRoot().getChildrenNames());
412                assertEquals(Collections.emptySet(), treeCache.getNode(FQN_C).getChildrenNames());
413                nodeReadLatch.countDown();
414 
415                nodeMovedLatch.await();
416                treeCache.move(FQN_C, FQN_B);
417                tm().commit(); //this is expected to fail
418                fail("Transaction should have failed");
419             } catch (Exception e) {
420                if (tm().getTransaction() != null) {
421                   // the TX is most likely rolled back already, but we attempt a rollback just in case it isn't
422                   try {
423                      tm().rollback();
424                   } catch (SystemException e1) {
425                      log.error("Failed to rollback", e1);
426                   }
427                }
428             }
429             return null;
430          }
431       };
432 
433       // moves C under A successfully
434       Callable<Object> moveCtoA = new Callable<Object>() {
435          public Object call() throws Exception {
436             tm().begin();
437             try {
438                try {
439                   nodeReadLatch.await();
440                   treeCache.move(FQN_C, FQN_A);
441                   tm().commit();
442                } finally {
443                   nodeMovedLatch.countDown();
444                }
445             } catch (Exception e) {
446                if (tm().getTransaction() != null) {
447                   // the TX is most likely rolled back already, but we attempt a rollback just in case it isn't
448                   try {
449                      tm().rollback();
450                   } catch (SystemException e1) {
451                      log.error("Failed to rollback", e1);
452                   }
453                }
454                throw e;
455             }
456             return null;
457          }
458       };
459 
460       runConcurrently(moveCtoB, moveCtoA);
461 
462       log.trace("Tree: " + TreeStructureSupport.printTree(treeCache, true));
463       assertNoLocks();
464       assertFalse(nodeC.isValid());
465       assertEquals(asSet("a", "b"), rootNode.getChildrenNames());
466       assertEquals(asSet("c"), nodeA.getChildrenNames());
467       assertEquals(Collections.emptySet(), nodeB.getChildrenNames());
468    }

487    /**
488     * Looks up the CacheEntry stored in transactional context corresponding to this AtomicMap.  If this AtomicMap
489     * has yet to be touched by the current transaction, this method will return a null.
490     * @return
491     */
492    protected CacheEntry lookupEntryFromCurrentTransaction(TransactionTable transactionTable, TransactionManager transactionManager, Object key) {
493       // Prior to 5.1, this used to happen by grabbing any InvocationContext in ThreadLocal.  Since ThreadLocals
494       // can no longer be relied upon in 5.1, we need to grab the TransactionTable and check if an ongoing
495       // transaction exists, peeking into transactional state instead.
496       try {
497          LocalTransaction localTransaction = transactionTable.getLocalTransaction(transactionManager.getTransaction());
498 
499          // The stored localTransaction could be null, if this is the first call in a transaction.  In which case
500          // we know that there is no transactional state to refer to - i.e., no entries have been looked up as yet.
501          return localTransaction == null ? null : localTransaction.lookupEntry(key);
502       } catch (SystemException e) {
503          return null;
504       }
505    }

507    protected boolean isNodeLocked(Fqn fqn, boolean includeData) {
508       TransactionManager tm = cache.getAdvancedCache().getTransactionManager();
509       TransactionTable tt = cache.getAdvancedCache().getComponentRegistry().getComponent(TransactionTable.class);
510       CacheEntry structure = lookupEntryFromCurrentTransaction(tt, tm, new NodeKey(fqn, NodeKey.Type.STRUCTURE));
511       CacheEntry data = lookupEntryFromCurrentTransaction(tt, tm, new NodeKey(fqn, NodeKey.Type.DATA));
512       return structure != null && data != null && structure.isChanged() && (!includeData || data.isChanged());
513    }
