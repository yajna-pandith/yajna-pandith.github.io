165    /**
166     * Encodes up to three bytes of the array <var>source</var>
167     * and writes the resulting four Base64 bytes to <var>destination</var>.
168     * The source and destination arrays can be manipulated
169     * anywhere along their length by specifying
170     * <var>srcOffset</var> and <var>destOffset</var>.
171     * This method does not check to make sure your arrays
172     * are large enough to accomodate <var>srcOffset</var> + 3 for
173     * the <var>source</var> array or <var>destOffset</var> + 4 for
174     * the <var>destination</var> array.
175     * The actual number of significant bytes in your array is
176     * given by <var>numSigBytes</var>.
177     *
178     * @param source the array to convert
179     * @param srcOffset the index where conversion begins
180     * @param numSigBytes the number of significant bytes in your array
181     * @param destination the array to hold the conversion
182     * @param destOffset the index where output will be put
183     * @return the <var>destination</var> array
184     * @since 1.3
185     */
186    private static byte[] encode3to4(byte[] source, int srcOffset, int numSigBytes, byte[] destination, int destOffset) {
187       //           1         2         3
188       // 01234567890123456789012345678901 Bit position
189       // --------000000001111111122222222 Array position from threeBytes
190       // --------|    ||    ||    ||    | Six bit groups to index ALPHABET
191       //          >>18  >>12  >> 6  >> 0  Right shift necessary
192       //                0x3f  0x3f  0x3f  Additional AND
193 
194       // Create buffer with zero-padding if there are only one or two
195       // significant bytes passed in the array.
196       // We have to shift left 24 in order to flush out the 1's that appear
197       // when Java treats a value as negative that is cast from a byte to an int.
198       int inBuff = (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8) : 0) | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16) : 0)
199             | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24) : 0);
200 
201       switch (numSigBytes) {
202          case 3:
203             destination[destOffset] = ALPHABET[(inBuff >>> 18)];
204             destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
205             destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
206             destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
207             return destination;
208 
209          case 2:
210             destination[destOffset] = ALPHABET[(inBuff >>> 18)];
211             destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
212             destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
213             destination[destOffset + 3] = EQUALS_SIGN;
214             return destination;
215 
216          case 1:
217             destination[destOffset] = ALPHABET[(inBuff >>> 18)];
218             destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
219             destination[destOffset + 2] = EQUALS_SIGN;
220             destination[destOffset + 3] = EQUALS_SIGN;
221             return destination;
222 
223          default:
224             return destination;
225       } // end switch
226    } // end encode3to4

243    /**
244     * Serializes an object and returns the Base64-encoded
245     * version of that serialized object. If the object
246     * cannot be serialized or there is another error,
247     * the method will return <tt>null</tt>.
248     * <p>
249     * Valid options:<pre>
250     *   GZIP: gzip-compresses object before encoding it.
251     *   DONT_BREAK_LINES: don't break lines at 76 characters
252     *     <i>Note: Technically, this makes your encoding non-compliant.</i>
253     * </pre>
254     * <p>
255     * Example: <code>encodeObject( myObj, Base64.GZIP )</code> or
256     * <p>
257     * Example: <code>encodeObject( myObj, Base64.GZIP | Base64.DONT_BREAK_LINES )</code>
258     *
259     * @param serializableObject The object to encode
260     * @param options Specified options
261     * @return The Base64-encoded object
262     * @see Base64#GZIP
263     * @see Base64#DONT_BREAK_LINES
264     * @since 2.0
265     */
266    public static String encodeObject(java.io.Serializable serializableObject, int options) {
267       // Streams
268       java.io.ByteArrayOutputStream baos = null;
269       java.io.OutputStream b64os = null;
270       java.io.ObjectOutputStream oos = null;
271       java.util.zip.GZIPOutputStream gzos = null;
272 
273       // Isolate options
274       int gzip = (options & GZIP);
275       int dontBreakLines = (options & DONT_BREAK_LINES);
276 
277       try {
278          // ObjectOutputStream -> (GZIP) -> Base64 -> ByteArrayOutputStream
279          baos = new java.io.ByteArrayOutputStream();
280          b64os = new Base64.OutputStream(baos, ENCODE | dontBreakLines);
281 
282          // GZip?
283          if (gzip == GZIP) {
284             gzos = new java.util.zip.GZIPOutputStream(b64os);
285             oos = new java.io.ObjectOutputStream(gzos);
286          } // end if: gzip
287          else oos = new java.io.ObjectOutputStream(b64os);
288 
289          oos.writeObject(serializableObject);
290       } // end try
291       catch (java.io.IOException e) {
292          e.printStackTrace();
293          return null;
294       } // end catch
295       finally {
296          Util.close(oos, gzos, b64os, baos);
297       } // end finally
298 
299       // Return value according to relevant encoding.
300       try {
301          return new String(baos.toByteArray(), PREFERRED_ENCODING);
302       } // end try
303       catch (java.io.UnsupportedEncodingException uue) {
304          return new String(baos.toByteArray());
305       } // end catch
306 
307    } // end encode

360    /**
361     * Encodes a byte array into Base64 notation.
362     * <p>
363     * Valid options:<pre>
364     *   GZIP: gzip-compresses object before encoding it.
365     *   DONT_BREAK_LINES: don't break lines at 76 characters
366     *     <i>Note: Technically, this makes your encoding non-compliant.</i>
367     * </pre>
368     * <p>
369     * Example: <code>encodeBytes( myData, Base64.GZIP )</code> or
370     * <p>
371     * Example: <code>encodeBytes( myData, Base64.GZIP | Base64.DONT_BREAK_LINES )</code>
372     *
373     *
374     * @param source The data to convert
375     * @param off Offset in array where conversion should begin
376     * @param len Length of data to convert
377     * @param options Specified options
378     * @see Base64#GZIP
379     * @see Base64#DONT_BREAK_LINES
380     * @return the text node
381     * @since 2.0
382     */
383    public static String encodeBytes(byte[] source, int off, int len, int options) {
384       // Isolate options
385       int dontBreakLines = (options & DONT_BREAK_LINES);
386       int gzip = (options & GZIP);
387 
388       // Compress?
389       if (gzip == GZIP) {
390          java.io.ByteArrayOutputStream baos = null;
391          java.util.zip.GZIPOutputStream gzos = null;
392          Base64.OutputStream b64os = null;
393 
394          try {
395             // GZip -> Base64 -> ByteArray
396             baos = new java.io.ByteArrayOutputStream();
397             b64os = new Base64.OutputStream(baos, ENCODE | dontBreakLines);
398             gzos = new java.util.zip.GZIPOutputStream(b64os);
399 
400             gzos.write(source, off, len);
401             gzos.close();
402          } // end try
403          catch (java.io.IOException e) {
404             e.printStackTrace();
405             return null;
406          } // end catch
407          finally {
408             try {
409                gzos.close();
410             } catch (Exception e) {
411             }
412             try {
413                b64os.close();
414             } catch (Exception e) {
415             }
416             try {
417                baos.close();
418             } catch (Exception e) {
419             }
420          } // end finally
421 
422          // Return value according to relevant encoding.
423          try {
424             return new String(baos.toByteArray(), PREFERRED_ENCODING);
425          } // end try
426          catch (java.io.UnsupportedEncodingException uue) {
427             return new String(baos.toByteArray());
428          } // end catch
429       } // end if: compress
430 
431       // Else, don't compress. Better not to use streams at all then.
432       else {
433          // Convert option to boolean in way that code likes it.
434          boolean breakLines = dontBreakLines == 0;
435 
436          int len43 = len * 4 / 3;
437          byte[] outBuff = new byte[(len43) // Main 4:3
438                + ((len % 3) > 0 ? 4 : 0) // Account for padding
439                + (breakLines ? (len43 / MAX_LINE_LENGTH) : 0)]; // New lines
440          int d = 0;
441          int e = 0;
442          int len2 = len - 2;
443          int lineLength = 0;
444          for (; d < len2; d += 3, e += 4) {
445             encode3to4(source, d + off, 3, outBuff, e);
446 
447             lineLength += 4;
448             if (breakLines && lineLength == MAX_LINE_LENGTH) {
449                outBuff[e + 4] = NEW_LINE;
450                e++;
451                lineLength = 0;
452             } // end if: end of line
453          } // en dfor: each piece of array
454 
455          if (d < len) {
456             encode3to4(source, d + off, len - d, outBuff, e);
457             e += 4;
458          } // end if: some padding needed
459 
460          // Return value according to relevant encoding.
461          try {
462             return new String(outBuff, 0, e, PREFERRED_ENCODING);
463          } // end try
464          catch (java.io.UnsupportedEncodingException uue) {
465             return new String(outBuff, 0, e);
466          } // end catch
467 
468       } // end else: don't compress
469 
470    } // end encodeBytes

474    /**
475     * Decodes four bytes from array <var>source</var>
476     * and writes the resulting bytes (up to three of them)
477     * to <var>destination</var>.
478     * The source and destination arrays can be manipulated
479     * anywhere along their length by specifying
480     * <var>srcOffset</var> and <var>destOffset</var>.
481     * This method does not check to make sure your arrays
482     * are large enough to accomodate <var>srcOffset</var> + 4 for
483     * the <var>source</var> array or <var>destOffset</var> + 3 for
484     * the <var>destination</var> array.
485     * This method returns the actual number of bytes that
486     * were converted from the Base64 encoding.
487     *
488     *
489     * @param source the array to convert
490     * @param srcOffset the index where conversion begins
491     * @param destination the array to hold the conversion
492     * @param destOffset the index where output will be put
493     * @return the number of decoded bytes converted
494     * @since 1.3
495     */
496    private static int decode4to3(byte[] source, int srcOffset, byte[] destination, int destOffset) {
497       // Example: Dk==
498       if (source[srcOffset + 2] == EQUALS_SIGN) {
499          // Two ways to do the same thing. Don't know which way I like best.
500          //int outBuff =   ( ( DECODABET[ source[ srcOffset    ] ] << 24 ) >>>  6 )
501          //              | ( ( DECODABET[ source[ srcOffset + 1] ] << 24 ) >>> 12 );
502          int outBuff = ((DECODABET[source[srcOffset]] & 0xFF) << 18) | ((DECODABET[source[srcOffset + 1]] & 0xFF) << 12);
503 
504          destination[destOffset] = (byte) (outBuff >>> 16);
505          return 1;
506       }
507 
508       // Example: DkL=
509       else if (source[srcOffset + 3] == EQUALS_SIGN) {
510          // Two ways to do the same thing. Don't know which way I like best.
511          //int outBuff =   ( ( DECODABET[ source[ srcOffset     ] ] << 24 ) >>>  6 )
512          //              | ( ( DECODABET[ source[ srcOffset + 1 ] ] << 24 ) >>> 12 )
513          //              | ( ( DECODABET[ source[ srcOffset + 2 ] ] << 24 ) >>> 18 );
514          int outBuff = ((DECODABET[source[srcOffset]] & 0xFF) << 18) | ((DECODABET[source[srcOffset + 1]] & 0xFF) << 12)
515                | ((DECODABET[source[srcOffset + 2]] & 0xFF) << 6);
516 
517          destination[destOffset] = (byte) (outBuff >>> 16);
518          destination[destOffset + 1] = (byte) (outBuff >>> 8);
519          return 2;
520       }
521 
522       // Example: DkLE
523       else {
524          try {
525             // Two ways to do the same thing. Don't know which way I like best.
526             //int outBuff =   ( ( DECODABET[ source[ srcOffset     ] ] << 24 ) >>>  6 )

548    /**
549     * Very low-level access to decoding ASCII characters in
550     * the form of a byte array. Does not support automatically
551     * gunzipping or any other "fancy" features.
552     *
553     * @param source The Base64 encoded data
554     * @param off    The offset of where to begin decoding
555     * @param len    The length of characters to decode
556     * @return decoded data
557     * @since 1.3
558     */
559    public static byte[] decode(byte[] source, int off, int len) {
560       int len34 = len * 3 / 4;
561       byte[] outBuff = new byte[len34]; // Upper limit on size of output
562       int outBuffPosn = 0;
563 
564       byte[] b4 = new byte[4];
565       int b4Posn = 0;
566       int i;
567       byte sbiCrop;
568       byte sbiDecode;
569       for (i = off; i < off + len; i++) {
570          sbiCrop = (byte) (source[i] & 0x7f); // Only the low seven bits
571          sbiDecode = DECODABET[sbiCrop];
572 
573          if (sbiDecode >= WHITE_SPACE_ENC) // White space, Equals sign or better
574          {
575             if (sbiDecode >= EQUALS_SIGN_ENC) {
576                b4[b4Posn++] = sbiCrop;
577                if (b4Posn > 3) {
578                   outBuffPosn += decode4to3(b4, 0, outBuff, outBuffPosn);
579                   b4Posn = 0;
580 
581                   // If that was the equals sign, break out of 'for' loop
582                   if (sbiCrop == EQUALS_SIGN)
583                      break;
584                } // end if: quartet built
585 
586             } // end if: equals sign or better
587 
588          } // end if: white space, equals sign or better
589          else {
590             throw new IllegalStateException("Bad Base64 input character at " + i + ": " + source[i] + "(decimal)");
591          } // end else:
592       } // each input character
593 
594       byte[] out = new byte[outBuffPosn];
595       System.arraycopy(outBuff, 0, out, 0, outBuffPosn);
596       return out;
597    } // end decode

599    /**
600     * Decodes data from Base64 notation, automatically
601     * detecting gzip-compressed data and decompressing it.
602     *
603     * @param s the string to decode
604     * @return the decoded data
605     * @since 1.4
606     */
607    public static byte[] decode(String s) {
608       byte[] bytes;
609       try {
610          bytes = s.getBytes(PREFERRED_ENCODING);
611       } // end try
612       catch (java.io.UnsupportedEncodingException uee) {
613          bytes = s.getBytes();
614       } // end catch
615       //</change>
616 
617       // Decode
618       bytes = decode(bytes, 0, bytes.length);
619 
620       // Check to see if it's gzip-compressed
621       // GZIP Magic Two-Byte Number: 0x8b1f (35615)
622       if (bytes != null && bytes.length >= 4) {
623 
624          int head = (bytes[0] & 0xff) | ((bytes[1] << 8) & 0xff00);
625          if (java.util.zip.GZIPInputStream.GZIP_MAGIC == head) {
626             java.io.ByteArrayInputStream bais = null;
627             java.util.zip.GZIPInputStream gzis = null;
628             java.io.ByteArrayOutputStream baos = null;
629             byte[] buffer = new byte[2048];
630             int length;
631 
632             try {
633                baos = new java.io.ByteArrayOutputStream();
634                bais = new java.io.ByteArrayInputStream(bytes);
635                gzis = new java.util.zip.GZIPInputStream(bais);
636 
637                while ((length = gzis.read(buffer)) >= 0) {
638                   baos.write(buffer, 0, length);
639                } // end while: reading input
640 
641                // No error? Get new bytes.
642                bytes = baos.toByteArray();
643 
644             } // end try
645             catch (java.io.IOException e) {
646                // Just return originally-decoded bytes
647             } // end catch
648             finally {
649                try {
650                   baos.close();
651                } catch (Exception e) {
652                }
653                try {
654                   gzis.close();
655                } catch (Exception e) {
656                }
657                try {
658                   bais.close();
659                } catch (Exception e) {
660                }
661             } // end finally
662 
663          } // end if: gzipped
664       } // end if: bytes.length >= 2
665 
666       return bytes;
667    } // end decode

852     * and encode/decode to/from Base64 notation on the fly.

856     */
857    public static class InputStream extends java.io.FilterInputStream {
858       private boolean encode; // Encoding or decoding
859       private int position; // Current position in the buffer
860       private byte[] buffer; // Small buffer holding converted data
861       private int bufferLength; // Length of buffer (3 or 4)
862       private int numSigBytes; // Number of meaningful bytes in the buffer
863       private int lineLength;
864       private boolean breakLines; // Break lines at less than 80 characters

866       /**
867        * Constructs a {@link Base64.InputStream} in DECODE mode.
868        *
869        * @param in the <tt>java.io.InputStream</tt> from which to read data.

876       /**
877        * Constructs a {@link Base64.InputStream} in
878        * either ENCODE or DECODE mode.
879        * <p>
880        * Valid options:<pre>
881        *   ENCODE or DECODE: Encode or Decode as data is read.
882        *   DONT_BREAK_LINES: don't break lines at 76 characters
883        *     (only meaningful when encoding)
884        *     <i>Note: Technically, this makes your encoding non-compliant.</i>
885        * </pre>
886        * <p>
887        * Example: <code>new Base64.InputStream( in, Base64.DECODE )</code>
888        *
889        *
890        * @param in the <tt>java.io.InputStream</tt> from which to read data.
891        * @param options Specified options
892        * @see Base64#ENCODE
893        * @see Base64#DECODE
894        * @see Base64#DONT_BREAK_LINES
895        * @since 2.0
896        */
897       public InputStream(java.io.InputStream in, int options) {
898          super(in);
899          this.breakLines = (options & DONT_BREAK_LINES) != DONT_BREAK_LINES;
900          this.encode = (options & ENCODE) == ENCODE;
901          this.bufferLength = encode ? 4 : 3;
902          this.buffer = new byte[bufferLength];
903          this.position = -1;
904          this.lineLength = 0;
905       } // end constructor
906 
907       /**
908        * Reads enough of the input stream to convert
909        * to/from Base64 and returns the next byte.
910        *
911        * @return next byte
912        * @since 1.3
913        */
914       @Override
915       public int read() throws java.io.IOException {
916          // Do we need to get data?
917          if (position < 0) {
918             if (encode) {
919                byte[] b3 = new byte[3];
920                int numBinaryBytes = 0;
921                for (int i = 0; i < 3; i++) {
922                   try {
923                      int b = in.read();
924 
925                      // If end of stream, b is -1.
926                      if (b >= 0) {
927                         b3[i] = (byte) b;
928                         numBinaryBytes++;
929                      } // end if: not end of stream
930 
931                   } // end try: read
932                   catch (java.io.IOException e) {
933                      // Only a problem if we got no data at all.
934                      if (i == 0)
935                         throw e;
936 
937                   } // end catch
938                } // end for: each needed input byte
939 
940                if (numBinaryBytes > 0) {
941                   encode3to4(b3, 0, numBinaryBytes, buffer, 0);
942                   position = 0;
943                   numSigBytes = 4;
944                } // end if: got data
945                else {
946                   return -1;
947                } // end else
948             } // end if: encoding
949 
950             // Else decoding
951             else {
952                byte[] b4 = new byte[4];
953                int i;
954                for (i = 0; i < 4; i++) {
955                   // Read four "meaningful" bytes:
956                   int b;
957                   do {
958                      b = in.read();
959                   }
960                   while (b >= 0 && DECODABET[b & 0x7f] <= WHITE_SPACE_ENC);
961 
962                   if (b < 0)
963                      break; // Reads a -1 if end of stream
964 
965                   b4[i] = (byte) b;
966                } // end for: each needed input byte
967 
968                if (i == 4) {
969                   numSigBytes = decode4to3(b4, 0, buffer, 0);
970                   position = 0;
971                } // end if: got four characters
972                else if (i == 0) {
973                   return -1;
974                } // end else if: also padded correctly
975                else {
976                   // Must have broken out from above.
977                   throw new java.io.IOException("Improperly padded Base64 input.");
978                } // end
979 
980             } // end else: decode
981          } // end else: get data
982 
983          // Got data?
984          if (position >= 0) {
985             // End of relevant data?
986             if (/*!encode &&*/position >= numSigBytes)
987                return -1;
988 
989             if (encode && breakLines && lineLength >= MAX_LINE_LENGTH) {
990                lineLength = 0;
991                return '\n';
992             } // end if
993             else {
994                lineLength++; // This isn't important when decoding
995                // but throwing an extra "if" seems
996                // just as wasteful.
997 
998                int b = buffer[position++];
999 
1000                if (position >= bufferLength)
1001                   position = -1;
1002 
1003                return b & 0xFF; // This is how you "cast" a byte that's
1004                // intended to be unsigned.
1005             } // end else
1006          } // end if: position >= 0
1007 
1008          // Else error
1009          else {
1010             // When JDK1.4 is more accepted, use an assertion here.
1011             throw new java.io.IOException("Error in Base64 code reading stream.");
1012          } // end else
1013       } // end read
1014 
1015       /**
1016        * Calls {@link #read()} repeatedly until the end of stream
1017        * is reached or <var>len</var> bytes are read.
1018        * Returns number of bytes read into array or -1 if
1019        * end of stream is encountered.
1020        *
1021        * @param dest array to hold values
1022        * @param off offset for array
1023        * @param len max number of bytes to read into array
1024        * @return bytes read into array or -1 if end of stream is encountered.
1025        * @since 1.3
1026        */
1027       @Override
1028       public int read(byte[] dest, int off, int len) throws java.io.IOException {
1029          int i;
1030          int b;
1031          for (i = 0; i < len; i++) {
1032             b = read();
1033

1035             //    return -1;

1038                dest[off + i] = (byte) b;
1039             else if (i == 0)
1040                return -1;
1041             else break; // Out of 'for' loop
1042          } // end for: each byte read
1043          return i;
1044       } // end read

1112        * converting to/from Base64 notation.
1113        * When encoding, bytes are buffered three
1114        * at a time before the output stream actually
1115        * gets a write() call.
1116        * When decoding, bytes are buffered four
1117        * at a time.
1118        *
1119        * @param theByte the byte to write
1120        * @since 1.3
1121        */
1122       @Override
1123       public void write(int theByte) throws java.io.IOException {
1124          // Encoding suspended?
1125          if (suspendEncoding) {
1126             super.out.write(theByte);
1127             return;
1128          } // end if: supsended
1129 
1130          // Encode?
1131          if (encode) {
1132             buffer[position++] = (byte) theByte;
1133             if (position >= bufferLength) // Enough to encode.
1134             {
1135                out.write(encode3to4(b4, buffer, bufferLength));
1136 
1137                lineLength += 4;
1138                if (breakLines && lineLength >= MAX_LINE_LENGTH) {
1139                   out.write(NEW_LINE);
1140                   lineLength = 0;
1141                } // end if: end of line
1142 
1143                position = 0;
1144             } // end if: enough to output
1145          } // end if: encoding
1146 
1147          // Else, Decoding
1148          else {
1149             // Meaningful Base64 character?
1150             if (DECODABET[theByte & 0x7f] > WHITE_SPACE_ENC) {
1151                buffer[position++] = (byte) theByte;
1152                if (position >= bufferLength) // Enough to output.
1153                {

1155                   out.write(b4, 0, len);
1156                   //out.write( Base64.decode4to3( buffer ) );
1157                   position = 0;
1158                } // end if: enough to output
1159             } // end if: meaningful base64 character
1160             else if (DECODABET[theByte & 0x7f] != WHITE_SPACE_ENC) {
1161                throw new java.io.IOException("Invalid character in Base64 data.");
1162             } // end else: not white space either
1163          } // end else: decoding
