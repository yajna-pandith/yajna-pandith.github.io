117    @Test
118    public void verifyNodeLeavesAfterSendingBackSomeData() throws TimeoutException, InterruptedException, ExecutionException {
119       Cache<Object, String> cache0 = cache(0, CACHE_NAME);
120       Cache<Object, String> cache1 = cache(1, CACHE_NAME);
121 
122       Map<Object, String> values = new HashMap<Object, String>();
123       int chunkSize = cache0.getCacheConfiguration().clustering().stateTransfer().chunkSize();
124       // Now insert 10 more values than the chunk size into the node we will kill
125       for (int i = 0; i < chunkSize + 10; ++i) {
126          MagicKey key = new MagicKey(cache1);
127          cache1.put(key, key.toString());
128          values.put(key, key.toString());
129       }
130 
131       CheckPoint checkPoint = new CheckPoint();
132       // Let the first request come through fine
133       checkPoint.trigger("pre_send_response_released");
134       waitUntilSendingResponse(cache1, checkPoint);
135 
136       final EntryRetriever<Object, String> retriever = cache0.getAdvancedCache().getComponentRegistry().getComponent(
137             EntryRetriever.class);
138 
139       final BlockingQueue<Map.Entry<Object, String>> returnQueue = new LinkedBlockingQueue<Map.Entry<Object, String>>();
140       Future<Void> future = fork(new Callable<Void>() {
141          @Override
142          public Void call() throws Exception {
143             Iterator<CacheEntry<Object, String>> iter = retriever.retrieveEntries(new NoOpFilterConverterWithDependencies<Object, String>(), null, null, null);
144             while (iter.hasNext()) {
145                Map.Entry<Object, String> entry = iter.next();
146                returnQueue.add(entry);
147             }
148             return null;
149          }
150       });
151 
152       // Now wait for them to send back first results
153       checkPoint.awaitStrict("post_send_response_invoked", 10, TimeUnit.SECONDS);
154 
155       // We should get a value now, note all values are currently residing on cache1 as primary
156       Map.Entry<Object, String> value = returnQueue.poll(10, TimeUnit.SECONDS);
157 
158       // Now kill the cache - we should recover
159       killMember(1, CACHE_NAME);
160 
161       future.get(10, TimeUnit.SECONDS);
162 
163       for (Map.Entry<Object, String> entry : values.entrySet()) {
164          assertTrue("Entry wasn't found:" + entry, returnQueue.contains(entry) || entry.equals(value));
165       }
166    }

281    @Test
282    public void testLocallyForcedEntryRetriever() {
283       Cache<Object, String> cache0 = cache(0, CACHE_NAME);
284       Cache<Object, String> cache1 = cache(1, CACHE_NAME);
285       Cache<Object, String> cache2 = cache(2, CACHE_NAME);
286 
287       Map<Object, String> values = new HashMap<Object, String>();
288       for (int i = 0; i < 501; ++i) {
289          switch (i % 3) {
290             case 0:
291                MagicKey key = new MagicKey(cache0);
292                cache0.put(key, key.toString());
293                values.put(key, key.toString());
294                break;
295             case 1:
296                // Force it so only cache0 has it's primary owned keys
297                key = new MagicKey(cache1, cache2);
298                cache1.put(key, key.toString());
299                break;
300             case 2:
301                // Force it so only cache0 has it's primary owned keys
302                key = new MagicKey(cache2, cache1);
303                cache2.put(key, key.toString());
304                break;
305             default:
306                fail("Unexpected switch case!");
307          }
308       }
309 
310       final EntryRetriever<Object, String> retriever = cache0.getAdvancedCache().getComponentRegistry().getComponent(
311             EntryRetriever.class);
312 
313       int count = 0;
314       try (CloseableIterator<CacheEntry<Object, String>> iter = retriever.retrieveEntries(null, null,
315                                                                             EnumSet.of(Flag.CACHE_MODE_LOCAL), null)) {
316          while (iter.hasNext()) {
317             CacheEntry<Object, String> entry = iter.next();
318             String cacheValue = cache0.get(entry.getKey());
319             assertNotNull(cacheValue);
320             assertEquals(cacheValue, entry.getValue());
321             count++;
322          }
323       }
324 
325       assertEquals(values.size(), count);
326    }

328    private Map<Integer, Set<Map.Entry<Object, String>>> generateEntriesPerSegment(ConsistentHash hash, Iterable<Map.Entry<Object, String>> entries) {
329       Map<Integer, Set<Map.Entry<Object, String>>> returnMap = new HashMap<Integer, Set<Map.Entry<Object, String>>>();
330 
331       for (Map.Entry<Object, String> value : entries) {
332          int segment = hash.getSegment(value.getKey());
333          Set<Map.Entry<Object, String>> set = returnMap.get(segment);
334          if (set == null) {
335             set = new HashSet<Map.Entry<Object, String>>();
336             returnMap.put(segment, set);
337          }
338          set.add(new ImmortalCacheEntry(value.getKey(), value.getValue()));
339       }
340       return returnMap;
341    }

343    protected RpcManager waitUntilSendingResponse(final Cache<?, ?> cache, final CheckPoint checkPoint) {
344       RpcManager rpc = TestingUtil.extractComponent(cache, RpcManager.class);
345       final Answer<Object> forwardedAnswer = AdditionalAnswers.delegatesTo(rpc);
346       RpcManager mockManager = mock(RpcManager.class, withSettings().defaultAnswer(forwardedAnswer));
347       doAnswer(new Answer() {
348          @Override
349          public Object answer(InvocationOnMock invocation) throws Throwable {
350             // Wait for main thread to sync up
351             checkPoint.trigger("pre_send_response_invoked");
352             // Now wait until main thread lets us through
353             checkPoint.awaitStrict("pre_send_response_released", 10, TimeUnit.SECONDS);
354 
355             try {
356                return forwardedAnswer.answer(invocation);
357             } finally {
358                // Wait for main thread to sync up
359                checkPoint.trigger("post_send_response_invoked");
360                // Now wait until main thread lets us through
361                checkPoint.awaitStrict("post_send_response_released", 10, TimeUnit.SECONDS);
362             }
363          }
364       }).when(mockManager).invokeRemotely(anyCollectionOf(Address.class), any(EntryRequestCommand.class),
365                                           any(RpcOptions.class));
366       TestingUtil.replaceComponent(cache, RpcManager.class, mockManager, true);
367       return rpc;
368    }

397    protected DataContainer waitUntilDataContainerWillBeIteratedOn(final Cache<?, ?> cache, final CheckPoint checkPoint) {
398       DataContainer rpc = TestingUtil.extractComponent(cache, DataContainer.class);
399       final Answer<Object> forwardedAnswer = AdditionalAnswers.delegatesTo(rpc);
400       DataContainer mocaContainer = mock(DataContainer.class, withSettings().defaultAnswer(forwardedAnswer));
401       final AtomicInteger invocationCount = new AtomicInteger();
402       doAnswer(new Answer() {
403          @Override
404          public Object answer(InvocationOnMock invocation) throws Throwable {
405             boolean waiting = false;
406             if (invocationCount.getAndIncrement() == 0) {
407                waiting = true;
408                // Wait for main thread to sync up
409                checkPoint.trigger("pre_iterator_invoked");
410                // Now wait until main thread lets us through
411                checkPoint.awaitStrict("pre_iterator_released", 10, TimeUnit.SECONDS);
412             }
413 
414             try {
415                return forwardedAnswer.answer(invocation);
416             } finally {
417                invocationCount.getAndDecrement();
418                if (waiting) {
419                   // Wait for main thread to sync up
420                   checkPoint.trigger("post_iterator_invoked");
421                   // Now wait until main thread lets us through
422                   checkPoint.awaitStrict("post_iterator_released", 10, TimeUnit.SECONDS);
423                }
424             }
425          }
426       }).when(mocaContainer).iterator();
427       TestingUtil.replaceComponent(cache, DataContainer.class, mocaContainer, true);
428       return rpc;
