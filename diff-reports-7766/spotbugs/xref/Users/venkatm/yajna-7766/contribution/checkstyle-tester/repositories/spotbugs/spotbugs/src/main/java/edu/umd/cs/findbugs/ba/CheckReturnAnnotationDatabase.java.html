233         addMethodAnnotation(c.getName(), "submit",

240                 "(Ljava/lang/Runnable;Ljava/lang/Object;)Ljava/util/concurrent/Future;", false,
241                 CheckReturnValueAnnotation.CHECK_RETURN_VALUE_MEDIUM_BAD_PRACTICE);

244     @Override
245     public CheckReturnValueAnnotation getResolvedAnnotation(Object o, boolean getMinimal) {
246         if (!(o instanceof XMethod)) {
247             return null;
248         }
249         XMethod m = (XMethod) o;
250         if (m.getName().startsWith("access$")) {
251             return null;
252         } else if (Const.CONSTRUCTOR_NAME.equals(m.getName())) {
253             try {
254                 if (throwableClass != null && Repository.instanceOf(m.getClassName(), throwableClass)) {
255                     return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_VERY_HIGH;
256                 }
257             } catch (ClassNotFoundException e) {
258                 AnalysisContext.reportMissingClass(e);
259             }
260             if ("java.lang.Thread".equals(m.getClassName())) {
261                 return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_VERY_HIGH;
262             }
263             try {
264 
265                 if (threadClass != null && Repository.instanceOf(m.getClassName(), threadClass)) {
266                     return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_LOW;
267                 }
268             } catch (ClassNotFoundException e) {
269                 AnalysisContext.reportMissingClass(e);
270             }
271         } else if ("equals".equals(m.getName()) && "(Ljava/lang/Object;)Z".equals(m.getSignature()) && !m.isStatic()) {
272             return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_MEDIUM;
273         } else if (m.getSignature().endsWith(")Ljava/lang/String;")
274                 && ("java.lang.StringBuffer".equals(m.getClassName()) || "java.lang.StringBuilder".equals(m.getClassName()))) {
275             return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_MEDIUM;
276         }
277         if (!AnalysisContext.currentAnalysisContext().isApplicationClass(m.getClassDescriptor())) {
278             // https://github.com/spotbugs/spotbugs/issues/429
279             // BuildCheckReturnAnnotationDatabase does not visit non-application classes,
280             // so we need to check package info dynamically
281 
282             CheckReturnValueAnnotation packageDefault = packageInfoCache.computeIfAbsent(m.getPackageName(),
283                     this::parsePackage);
284             if (packageDefault != null) {
285                 return packageDefault;
286             }
287         }
288         return super.getResolvedAnnotation(o, getMinimal);
289     }

291     private static final ClassDescriptor CHECK_RETURN_NULL_SPOTBUGS = DescriptorFactory
292             .createClassDescriptor(CheckReturnValue.class);
293     private static final ClassDescriptor CHECK_RETURN_NULL_JSR305 = DescriptorFactory
294             .createClassDescriptor("javax/annotation/CheckReturnValue");
295     private static final ClassDescriptor CAN_IGNORE_RETURN_VALUE = DescriptorFactory
296             .createClassDescriptor("com/google/errorprone/annotations/CanIgnoreReturnValue");

298     private final Map<String, CheckReturnValueAnnotation> packageInfoCache = new HashMap<>();
299 
300     /**
301      * Try to find default {@link CheckReturnValueAnnotation} for methods inside of target class.
302      *
303      */
304     @CheckForNull
305     private CheckReturnValueAnnotation parsePackage(@DottedClassName String packageName) {
306         String className = ClassName.toSlashedClassName(packageName) + "/package-info";
307         ClassDescriptor descriptor = DescriptorFactory.createClassDescriptor(className);
308         XClass clazz;
309         try {
310             clazz = descriptor.getXClass();
311         } catch (ResourceNotFoundException e) {
312             // no annotation on package
313             return null;
314         } catch (CheckedAnalysisException e) {
315             // ignore unexpected error to keep backward compatibility
316             if (FindBugs2.DEBUG) {
317                 e.printStackTrace();
318             }
319             return null;
320         }
321 
322         AnnotationValue annotation = clazz.getAnnotation(CHECK_RETURN_NULL_SPOTBUGS);
323         if (annotation != null) {
324             Confidence confidence = (Confidence) annotation.getValue("confidence");
325             if (confidence != null) {
326                 return CheckReturnValueAnnotation.parse(confidence.name());
327             } else {
328                 // use default name
329                 return CheckReturnValueAnnotation.parse(Confidence.MEDIUM.name());
330             }
331         }
332 
333         annotation = clazz.getAnnotation(CHECK_RETURN_NULL_JSR305);
334         if (annotation != null) {
335             When when = (When) annotation.getValue("when");
336             if (when != null) {
337                 return CheckReturnValueAnnotation.createFor(when);
338             } else {
339                 // use default value
340                 return CheckReturnValueAnnotation.createFor(When.ALWAYS);
341             }
342         }
343 
344         annotation = clazz.getAnnotation(CAN_IGNORE_RETURN_VALUE);
345         if (annotation != null) {
346             return CheckReturnValueAnnotation.createFor(When.NEVER);
347         }
348 
349         return null;
350     }
