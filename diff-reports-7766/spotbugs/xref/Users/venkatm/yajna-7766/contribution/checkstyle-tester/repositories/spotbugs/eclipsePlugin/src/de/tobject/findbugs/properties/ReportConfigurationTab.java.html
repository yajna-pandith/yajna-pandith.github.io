88      public ReportConfigurationTab(TabFolder parent, FindbugsPropertyPage page, int style) {
89          super(parent, style);
90          this.propertyPage = page;
91          setLayout(new GridLayout());
92  
93          TabItem tabDetector = new TabItem(parent, SWT.NONE);
94          tabDetector.setText(getMessage("property.reportConfigurationTab"));
95          tabDetector.setControl(this);
96          tabDetector.setToolTipText("Configure bugs reported to the UI");

98          Composite rankAndPrioGroup = new Composite(this, SWT.NONE);
99          rankAndPrioGroup.setLayout(new GridLayout(2, false));

101         createRankGroup(rankAndPrioGroup);
102         createPriorityGroup(rankAndPrioGroup);
103 
104         createBugCategoriesGroup(rankAndPrioGroup, page.getProject());
105         createBugSeverityGroup(rankAndPrioGroup);
106     }

108     private void createBugSeverityGroup(Composite parent) {
109         IPreferenceStore store = propertyPage.getPreferenceStore();
110         MarkerSeverity[] markerSeverities = MarkerSeverity.values();
111 
112         Group rankGroup = new Group(parent, SWT.NONE);
113         rankGroup.setLayout(new GridLayout(2, false));
114         rankGroup.setText("Mark bugs with ... rank as:");
115         rankGroup.setLayoutData(new GridData(SWT.BEGINNING, SWT.TOP, true, true));
116 
117         Label bugSeverityLabel = new Label(rankGroup, SWT.NONE);
118         bugSeverityLabel.setText("Scariest:");
119 
120         scariestRankCombo = new Combo(rankGroup, SWT.DROP_DOWN | SWT.READ_ONLY);
121         for (MarkerSeverity markerSeverity : markerSeverities) {
122             scariestRankCombo.add(markerSeverity.name());
123         }
124         initialScariestRank = MarkerSeverity.get(store.getString(FindBugsConstants.RANK_SCARIEST_MARKER_SEVERITY));
125         scariestRankCombo.setText(initialScariestRank.name());
126 
127         bugSeverityLabel = new Label(rankGroup, SWT.NONE);
128         bugSeverityLabel.setText("Scary:");
129 
130         scaryRankCombo = new Combo(rankGroup, SWT.DROP_DOWN | SWT.READ_ONLY);
131         for (MarkerSeverity markerSeverity : markerSeverities) {
132             scaryRankCombo.add(markerSeverity.name());
133         }
134         initialScaryRank = MarkerSeverity.get(store.getString(FindBugsConstants.RANK_SCARY_MARKER_SEVERITY));
135         scaryRankCombo.setText(initialScaryRank.name());
136 
137         bugSeverityLabel = new Label(rankGroup, SWT.NONE);
138         bugSeverityLabel.setText("Troubling:");
139 
140         troublingRankCombo = new Combo(rankGroup, SWT.DROP_DOWN | SWT.READ_ONLY);

142             troublingRankCombo.add(markerSeverity.name());

159     private void createPriorityGroup(Composite parent) {
160         Composite prioGroup = new Composite(parent, SWT.NONE);
161         prioGroup.setLayout(new GridLayout(2, false));
162 
163         Label minPrioLabel = new Label(prioGroup, SWT.NONE);
164         minPrioLabel.setText(getMessage("property.minPriority"));
165         minPrioLabel.setLayoutData(new GridData(SWT.BEGINNING, SWT.CENTER, false, false));
166 
167         minPriorityCombo = new Combo(prioGroup, SWT.DROP_DOWN | SWT.READ_ONLY);
168         minPriorityCombo.add(ProjectFilterSettings.HIGH_PRIORITY);
169         minPriorityCombo.add(ProjectFilterSettings.MEDIUM_PRIORITY);
170         minPriorityCombo.add(ProjectFilterSettings.LOW_PRIORITY);
171         minPriorityCombo.setText(propertyPage.getOriginalUserPreferences().getFilterSettings().getMinPriority());
172         minPriorityCombo.setLayoutData(new GridData(SWT.BEGINNING, SWT.CENTER, false, false));
173         minPriorityCombo.addSelectionListener(new SelectionAdapter() {
174             @Override
175             public void widgetSelected(SelectionEvent event) {
176                 String data = minPriorityCombo.getText();
177                 getCurrentProps().getFilterSettings().setMinPriority(data);
178             }
179         });
180

182     }

185     private void createRankGroup(Composite parent) {
186         Composite prioGroup = new Composite(parent, SWT.NONE);
187         prioGroup.setLayout(new GridLayout(2, false));
188 
189         Label minRankLabel = new Label(prioGroup, SWT.NONE);
190         minRankLabel.setText(getMessage("property.minRank") + System.getProperty("line.separator")
191                 + getMessage("property.minRank.line2"));
192         minRankLabel.setLayoutData(new GridData(SWT.BEGINNING, SWT.CENTER, false, false));
193 
194         minRankSlider = new Scale(prioGroup, SWT.DROP_DOWN | SWT.READ_ONLY);
195         minRankSlider.setLayoutData(new GridData(SWT.BEGINNING, SWT.CENTER, true, false));
196         minRankSlider.addSelectionListener(new SelectionAdapter() {
197             @Override
198             public void widgetSelected(SelectionEvent event) {
199                 int rank = minRankSlider.getSelection();
200                 getCurrentProps().getFilterSettings().setMinRank(rank);
201                 updateRankValueLabel();
202             }
203         });
204         minRankSlider.setMinimum(BugRanker.VISIBLE_RANK_MIN);
205         minRankSlider.setMaximum(BugRanker.VISIBLE_RANK_MAX);
206         minRankSlider.setSelection(getCurrentProps().getFilterSettings().getMinRank());
207         minRankSlider.setIncrement(1);
208         minRankSlider.setPageIncrement(5);
209         Label dummyLabel = new Label(prioGroup, SWT.NONE);
210         dummyLabel.setLayoutData(new GridData(SWT.CENTER, SWT.CENTER, false, false));
211 
212         rankValueLabel = new Label(prioGroup, SWT.NONE);
213         rankValueLabel.setLayoutData(new GridData(SWT.CENTER, SWT.CENTER, true, false));
214         updateRankValueLabel();
215     }
216 
217     private void updateRankValueLabel() {
218         int rank = minRankSlider.getSelection();
219         BugRankCategory category = BugRankCategory.getRank(rank);
220         String label = category.toString();
221         rankValueLabel.setText(rank + " (" + label + ")");
222     }

235     /**
236      * Build list of bug categories to be enabled or disabled. Populates
237      * chkEnableBugCategoryList and bugCategoryList fields.
238      *
239      * @param parent
240      *            control checkboxes should be added to
241      * @param project
242      *            the project being configured
243      */
244     private void createBugCategoriesGroup(Composite parent, final IProject project) {
245         Group checkBoxGroup = new Group(parent, SWT.SHADOW_ETCHED_OUT);
246         checkBoxGroup.setText(getMessage("property.categoriesGroup"));
247         checkBoxGroup.setLayout(new GridLayout(1, true));
248         checkBoxGroup.setLayoutData(new GridData(SWT.BEGINNING, SWT.TOP, true, true));
249 
250         List<String> bugCategoryList = new LinkedList<>(DetectorFactoryCollection.instance().getBugCategories());
251         chkEnableBugCategoryList = new LinkedList<>();
252         ProjectFilterSettings origFilterSettings = propertyPage.getOriginalUserPreferences().getFilterSettings();
253         for (String category : bugCategoryList) {
254             Button checkBox = new Button(checkBoxGroup, SWT.CHECK);
255             checkBox.setText(I18N.instance().getBugCategoryDescription(category));
256             checkBox.setSelection(origFilterSettings.containsCategory(category));
257             GridData layoutData = new GridData();
258             layoutData.horizontalIndent = 10;
259             checkBox.setLayoutData(layoutData);
260 
261             // Every time a checkbox is clicked, rebuild the detector factory
262             // table
263             // to show only relevant entries
264             checkBox.addListener(SWT.Selection, new Listener() {
265                 @Override
266                 public void handleEvent(Event e) {
267                     syncSelectedCategories();
268                 }
269             });
270             checkBox.setData(category);
271             chkEnableBugCategoryList.add(checkBox);
272         }
273     }

275     /**
276      * Synchronize selected bug category checkboxes with the current user
277      * preferences.
278      */
279     protected void syncSelectedCategories() {
280         ProjectFilterSettings filterSettings = getCurrentProps().getFilterSettings();
281         for (Button checkBox : chkEnableBugCategoryList) {
282             String category = (String) checkBox.getData();
283             if (checkBox.getSelection()) {
284                 filterSettings.addCategory(category);
285             } else {
286                 filterSettings.removeCategory(category);
287             }
288         }
289         propertyPage.getVisibleDetectors().clear();
290     }

318     public boolean isMarkerSeveritiesChanged() {
319         if (isMarkerSeveritiesChanged(FindBugsConstants.RANK_SCARIEST_MARKER_SEVERITY, initialScariestRank)) {
320             return true;
321         }
322         if (isMarkerSeveritiesChanged(FindBugsConstants.RANK_SCARY_MARKER_SEVERITY, initialScaryRank)) {
323             return true;
324         }
325         if (isMarkerSeveritiesChanged(FindBugsConstants.RANK_TROUBLING_MARKER_SEVERITY, initialTroublingRank)) {
326             return true;
327         }
328         if (isMarkerSeveritiesChanged(FindBugsConstants.RANK_OFCONCERN_MARKER_SEVERITY, initialOfConcernRank)) {
329             return true;
330         }
331         return false;
332 
333     }
334 
335     private boolean isMarkerSeveritiesChanged(String propertyName, MarkerSeverity marker) {
336         IPreferenceStore store = propertyPage.getPreferenceStore();
337         return !store.getString(propertyName).equals(marker.name());
338 
339     }

341     void refreshUI(UserPreferences prefs) {
342         IPreferenceStore store = propertyPage.getPreferenceStore();
343         scariestRankCombo.setText(MarkerSeverity.get(store.getString(FindBugsConstants.RANK_SCARIEST_MARKER_SEVERITY)).name());
344         scaryRankCombo.setText(MarkerSeverity.get(store.getString(FindBugsConstants.RANK_SCARY_MARKER_SEVERITY)).name());
345         troublingRankCombo.setText(MarkerSeverity.get(store.getString(FindBugsConstants.RANK_TROUBLING_MARKER_SEVERITY)).name());
346         ofConcernRankCombo.setText(MarkerSeverity.get(store.getString(FindBugsConstants.RANK_OFCONCERN_MARKER_SEVERITY)).name());
347 
348         ProjectFilterSettings filterSettings = prefs.getFilterSettings();
349         minRankSlider.setSelection(filterSettings.getMinRank());
350         updateRankValueLabel();
351         minPriorityCombo.setText(filterSettings.getMinPriority());
352         for (Button checkBox : chkEnableBugCategoryList) {
353             checkBox.setSelection(filterSettings.containsCategory((String) checkBox.getData()));
354         }
355         syncSelectedCategories();
356     }

362     public void performOk() {
363         IPreferenceStore store = propertyPage.getPreferenceStore();
364         String scariest = scariestRankCombo.getText();
365         store.setValue(FindBugsConstants.RANK_SCARIEST_MARKER_SEVERITY, scariest);
366 
367         String scary = scaryRankCombo.getText();
368         store.setValue(FindBugsConstants.RANK_SCARY_MARKER_SEVERITY, scary);
369 
370         String troubling = troublingRankCombo.getText();
371         store.setValue(FindBugsConstants.RANK_TROUBLING_MARKER_SEVERITY, troubling);
372 
373         String ofConcern = ofConcernRankCombo.getText();
374         store.setValue(FindBugsConstants.RANK_OFCONCERN_MARKER_SEVERITY, ofConcern);
375 
376     }
