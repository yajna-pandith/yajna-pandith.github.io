64  	@Test
65  	public void testCompleteGroupWithinTimeout() throws InterruptedException {
66  		QueueChannel replyChannel = new QueueChannel();
67  		Message<?> message1 = createMessage(3, "ABC", 3, 1, replyChannel, null);
68  		Message<?> message2 = createMessage(5, "ABC", 3, 2, replyChannel, null);
69  		Message<?> message3 = createMessage(7, "ABC", 3, 3, replyChannel, null);
70  		CountDownLatch latch = new CountDownLatch(3);
71  		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
72  				message1, latch));
73  		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
74  				message2, latch));
75  		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
76  				message3, latch));
77  
78  		assertThat(latch.await(10, TimeUnit.SECONDS)).isTrue();
79  
80  		assertThat(latch.getCount()).isEqualTo(0L);
81  		Message<?> reply = replyChannel.receive(2000);
82  		assertThat(reply).isNotNull();
83  		assertThat(105).isEqualTo(reply.getPayload());
84  	}

108 	@Test
109 	public void testShouldNotSendPartialResultOnTimeoutByDefault() throws InterruptedException {
110 
111 		QueueChannel discardChannel = new QueueChannel();
112 		this.aggregator.setDiscardChannel(discardChannel);
113 		QueueChannel replyChannel = new QueueChannel();
114 		Message<?> message = createMessage(3, "ABC", 2, 1, replyChannel, null);
115 		CountDownLatch latch = new CountDownLatch(1);
116 		AggregatorTestTask task = new AggregatorTestTask(this.aggregator,
117 				message, latch);
118 		this.taskExecutor.execute(task);
119 
120 		assertThat(latch.await(10, TimeUnit.SECONDS)).isTrue();
121 
122 		assertThat(latch
123 				.getCount()).as("Task should have completed within timeout").isEqualTo(0);
124 		Message<?> reply = replyChannel.receive(10);
125 		assertThat(reply).as("No message should have been sent normally").isNull();
126 		this.store.expireMessageGroups(-10000);
127 		Message<?> discardedMessage = discardChannel.receive(10000);
128 		assertThat(discardedMessage).as("A message should have been discarded").isNotNull();
129 		assertThat(discardedMessage).isEqualTo(message);
130 	}

157 	@Test
158 	public void testMultipleGroupsSimultaneously() throws InterruptedException {
159 		QueueChannel replyChannel1 = new QueueChannel();
160 		QueueChannel replyChannel2 = new QueueChannel();
161 		Message<?> message1 = createMessage(3, "ABC", 3, 1, replyChannel1, null);
162 		Message<?> message2 = createMessage(5, "ABC", 3, 2, replyChannel1, null);
163 		Message<?> message3 = createMessage(7, "ABC", 3, 3, replyChannel1, null);
164 		Message<?> message4 = createMessage(11, "XYZ", 3, 1, replyChannel2,
165 				null);
166 		Message<?> message5 = createMessage(13, "XYZ", 3, 2, replyChannel2,
167 				null);
168 		Message<?> message6 = createMessage(17, "XYZ", 3, 3, replyChannel2,
169 				null);
170 		CountDownLatch latch = new CountDownLatch(6);
171 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
172 				message1, latch));
173 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
174 				message6, latch));
175 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
176 				message2, latch));
177 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
178 				message5, latch));
179 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
180 				message3, latch));
181 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
182 				message4, latch));
183 
184 		assertThat(latch.await(10, TimeUnit.SECONDS)).isTrue();
185 
186 		@SuppressWarnings("unchecked")
187 		Message<Integer> reply1 = (Message<Integer>) replyChannel1.receive(1000);
188 		assertThat(reply1).isNotNull();
189 		assertThat(reply1.getPayload()).isEqualTo(105);
190 		@SuppressWarnings("unchecked")
191 		Message<Integer> reply2 = (Message<Integer>) replyChannel2.receive(1000);
192 		assertThat(reply2).isNotNull();
193 		assertThat(reply2.getPayload()).isEqualTo(2431);
194 	}

200 	public void testTrackedCorrelationIdsCapacityAtLimit() {
201 		QueueChannel replyChannel = new QueueChannel();

203 		// this.aggregator.setTrackedCorrelationIdCapacity(3);

211 		this.aggregator.handleMessage(createMessage(4, 3, 1, 1, replyChannel,
212 				null));

214 		// next message with same correlation ID is discarded
215 		this.aggregator.handleMessage(createMessage(2, 1, 1, 1, replyChannel,
216 				null));
217 		assertThat(discardChannel.receive(1000).getPayload()).isEqualTo(2);
218 	}

254 	@Test
255 	public void testAdditionalMessageAfterCompletion() throws InterruptedException {
256 		QueueChannel replyChannel = new QueueChannel();
257 		Message<?> message1 = createMessage(3, "ABC", 3, 1, replyChannel, null);
258 		Message<?> message2 = createMessage(5, "ABC", 3, 2, replyChannel, null);
259 		Message<?> message3 = createMessage(7, "ABC", 3, 3, replyChannel, null);
260 		Message<?> message4 = createMessage(7, "ABC", 3, 3, replyChannel, null);
261 		CountDownLatch latch = new CountDownLatch(4);
262 
263 		this.aggregator.setReleaseStrategy(new SequenceSizeReleaseStrategy());
264 
265 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
266 				message1, latch));
267 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
268 				message2, latch));
269 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
270 				message3, latch));
271 		this.taskExecutor.execute(new AggregatorTestTask(this.aggregator,
272 				message4, latch));
273 
274 		assertThat(latch.await(10, TimeUnit.SECONDS)).isTrue();
275 
276 		Message<?> reply = replyChannel.receive(10000);
277 		assertThat(reply).as("A message should be aggregated").isNotNull();
278 		assertThat(reply.getPayload()).isEqualTo(105);
279 	}

282 	private static Message<?> createMessage(Object payload,
283 			Object correlationId, int sequenceSize, int sequenceNumber,
284 			MessageChannel replyChannel, String predefinedId) {
285 
286 		MessageBuilder<Object> builder = MessageBuilder.withPayload(payload)
287 				.setCorrelationId(correlationId).setSequenceSize(sequenceSize)
288 				.setSequenceNumber(sequenceNumber)
289 				.setReplyChannel(replyChannel);
290 		if (predefinedId != null) {
291 			builder.setHeader(MessageHeaders.ID, predefinedId);
292 		}
293 		return builder.build();
294 	}

297 	private static class AggregatorTestTask implements Runnable {
298 
299 		private final MessageHandler aggregator;
300 
301 		private final Message<?> message;
302 
303 		private Exception exception;
304 
305 		private final CountDownLatch latch;
306 
307 		AggregatorTestTask(MessageHandler aggregator, Message<?> message,
308 				CountDownLatch latch) {
309 			this.aggregator = aggregator;
310 			this.message = message;
311 			this.latch = latch;
312 		}
313 
314 		public Exception getException() {
315 			return this.exception;
316 		}
317 
318 		@Override
319 		public void run() {
320 			try {
321 				this.aggregator.handleMessage(message);
322 			}
323 			catch (Exception e) {
324 				this.exception = e;
325 			}
326 			finally {
327 				this.latch.countDown();
328 			}
329 		}
330 
331 	}

334 	private class MultiplyingProcessor implements MessageGroupProcessor {
335 
336 		MultiplyingProcessor() {
337 			super();
338 		}
339 
340 		@Override
341 		public Object processMessageGroup(MessageGroup group) {
342 			Integer product = 1;
343 			for (Message<?> message : group.getMessages()) {
344 				product *= (Integer) message.getPayload();
345 			}
346 			return product;
347 		}
348 
349 	}
