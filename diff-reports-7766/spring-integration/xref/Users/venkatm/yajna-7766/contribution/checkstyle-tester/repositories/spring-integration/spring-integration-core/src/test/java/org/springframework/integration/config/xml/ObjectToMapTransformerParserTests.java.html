70  	@SuppressWarnings("unchecked")
71  	@Test
72  	public void testObjectToSpelMapTransformer() {
73  		Employee employee = this.buildEmployee();
74  		StandardEvaluationContext context = new StandardEvaluationContext(employee);
75  		context.addPropertyAccessor(new MapAccessor());
76  		ExpressionParser parser = new SpelExpressionParser();
77  
78  		Message<Employee> message = MessageBuilder.withPayload(employee).build();
79  		directInput.send(message);
80  
81  		Message<Map<String, Object>> outputMessage = (Message<Map<String, Object>>) output.receive();
82  		Map<String, Object> transformedMap = outputMessage.getPayload();
83  		assertThat(outputMessage.getPayload()).isNotNull();
84  		for (String key : transformedMap.keySet()) {
85  			Expression expression = parser.parseExpression(key);
86  			Object valueFromTheMap = transformedMap.get(key);
87  			Object valueFromExpression = expression.getValue(context);
88  			assertThat(valueFromExpression).isEqualTo(valueFromTheMap);
89  		}
90  	}

103 	@Test
104 	public void testObjectToNotFlattenedMapTransformer() {
105 		Employee employee = this.buildEmployee();
106 
107 		Message<Employee> message = MessageBuilder.withPayload(employee).build();
108 		nestedInput.send(message);
109 
110 		@SuppressWarnings("unchecked")
111 		Message<Map<String, Object>> outputMessage = (Message<Map<String, Object>>) nestedOutput.receive(1000);
112 		Map<String, Object> transformedMap = outputMessage.getPayload();
113 		assertThat(outputMessage.getPayload()).isNotNull();
114 
115 		assertThat(transformedMap.get("companyName")).isEqualTo(employee.getCompanyName());
116 		assertThat(transformedMap.get("companyAddress")).isInstanceOf(Map.class);
117 		assertThat(transformedMap.get("departments")).isInstanceOf(List.class);
118 	}
