76  		MessageChannel input = (MessageChannel) context.getBean("aggregatorWithReferenceInput");

78  		List<Message<?>> outboundMessages = new ArrayList<>();

82  
83  		outboundMessages.forEach(input::send);

90  		Object mbf = context.getBean(IntegrationUtils.INTEGRATION_MESSAGE_BUILDER_FACTORY_BEAN_NAME);
91  		Object handler = context.getBean("aggregatorWithReference.handler");
92  		assertThat(TestUtils.getPropertyValue(handler, "outputProcessor.messageBuilderFactory")).isSameAs(mbf);
93  		assertThat(TestUtils.getPropertyValue(handler, "releaseLockBeforeSend", Boolean.class)).isTrue();
94  	}

152 	public void testPropertyAssignment() {
153 		EventDrivenConsumer endpoint = (EventDrivenConsumer) context.getBean("completelyDefinedAggregator");
154 		ReleaseStrategy releaseStrategy = (ReleaseStrategy) context.getBean("releaseStrategy");
155 		CorrelationStrategy correlationStrategy = (CorrelationStrategy) context.getBean("correlationStrategy");
156 		Object consumer = new DirectFieldAccessor(endpoint).getPropertyValue("handler");
157 		assertThat(consumer).isInstanceOf(AggregatingMessageHandler.class);
158 		DirectFieldAccessor accessor = new DirectFieldAccessor(consumer);
159 		Object handlerMethods = new DirectFieldAccessor(new DirectFieldAccessor(new DirectFieldAccessor(accessor
160 				.getPropertyValue("outputProcessor")).getPropertyValue("processor")).getPropertyValue("delegate"))
161 				.getPropertyValue("handlerMethods");
162 		assertThat(handlerMethods).isNotNull();
163 		Object handlerMethod = new DirectFieldAccessor(new DirectFieldAccessor(new DirectFieldAccessor(accessor
164 				.getPropertyValue("outputProcessor")).getPropertyValue("processor")).getPropertyValue("delegate"))
165 				.getPropertyValue("handlerMethod");
166 		assertThat(handlerMethod.toString().contains("createSingleMessageFromGroup")).isTrue();
167 		assertThat(accessor.getPropertyValue("releaseStrategy"))
168 				.as("The AggregatorEndpoint is not injected with the appropriate ReleaseStrategy instance")
169 				.isEqualTo(releaseStrategy);
170 		assertThat(accessor.getPropertyValue("correlationStrategy"))
171 				.as("The AggregatorEndpoint is not injected with the appropriate CorrelationStrategy instance")
172 				.isEqualTo(correlationStrategy);
173 		assertThat(accessor.getPropertyValue("outputChannelName"))
174 				.as("The AggregatorEndpoint is not injected with the appropriate output channel")
175 				.isEqualTo("outputChannel");
176 		assertThat(accessor.getPropertyValue("discardChannelName"))
177 				.as("The AggregatorEndpoint is not injected with the appropriate discard channel")
178 				.isEqualTo("discardChannel");
179 		assertThat(TestUtils.getPropertyValue(consumer, "messagingTemplate.sendTimeout"))
180 				.as("The AggregatorEndpoint is not set with the appropriate timeout value").isEqualTo(86420000L);
181 		assertThat(accessor.getPropertyValue("sendPartialResultOnExpiry"))
182 				.as("The AggregatorEndpoint is not configured with the appropriate 'send partial results on timeout' " +
183 						"flag")
184 				.isEqualTo(true);
185 		assertThat(TestUtils.getPropertyValue(consumer, "expireGroupsUponTimeout", Boolean.class)).isFalse();
186 		assertThat(TestUtils.getPropertyValue(consumer, "expireGroupsUponCompletion", Boolean.class)).isTrue();
187 		assertThat(TestUtils.getPropertyValue(consumer, "minimumTimeoutForEmptyGroups")).isEqualTo(123L);
188 		assertThat(TestUtils.getPropertyValue(consumer, "groupTimeoutExpression", Expression.class)
189 				.getExpressionString()).isEqualTo("456");
190 		assertThat(TestUtils.getPropertyValue(consumer, "lockRegistry"))
191 				.isSameAs(this.context.getBean(LockRegistry.class));
192 		assertThat(TestUtils.getPropertyValue(consumer, "taskScheduler")).isSameAs(this.context.getBean("scheduler"));
193 		assertThat(TestUtils.getPropertyValue(consumer, "messageStore")).isSameAs(this.context.getBean("store"));
194 		assertThat(TestUtils.getPropertyValue(consumer, "order")).isEqualTo(5);
195 		assertThat(TestUtils.getPropertyValue(consumer, "forceReleaseAdviceChain")).isNotNull();
196 		assertThat(TestUtils.getPropertyValue(consumer, "popSequence", Boolean.class)).isFalse();
197 	}

237 	@Test
238 	public void testAggregatorWithPojoReleaseStrategy() {
239 		MessageChannel input = this.context.getBean("aggregatorWithPojoReleaseStrategyInput", MessageChannel.class);
240 		EventDrivenConsumer endpoint = this.context.getBean("aggregatorWithPojoReleaseStrategy", EventDrivenConsumer.class);
241 		ReleaseStrategy releaseStrategy =
242 				TestUtils.getPropertyValue(endpoint, "handler.releaseStrategy", ReleaseStrategy.class);
243 		assertThat(releaseStrategy instanceof MethodInvokingReleaseStrategy).isTrue();
244 		MessagingMethodInvokerHelper methodInvokerHelper =
245 				TestUtils.getPropertyValue(releaseStrategy, "adapter.delegate", MessagingMethodInvokerHelper.class);
246 		Object handlerMethods = TestUtils.getPropertyValue(methodInvokerHelper, "handlerMethods");
247 		assertThat(handlerMethods).isNotNull();
248 		Object handlerMethod = TestUtils.getPropertyValue(methodInvokerHelper, "handlerMethod");
249 		assertThat(handlerMethod.toString().contains("checkCompleteness")).isTrue();
250 		input.send(createMessage(1L, "correlationId", 4, 0, null));
251 		input.send(createMessage(2L, "correlationId", 4, 1, null));
252 		input.send(createMessage(3L, "correlationId", 4, 2, null));
253 		PollableChannel outputChannel = (PollableChannel) context.getBean("outputChannel");
254 		Message<?> reply = outputChannel.receive(0);
255 		assertThat(reply).isNull();
256 		input.send(createMessage(5L, "correlationId", 4, 3, null));
257 		reply = outputChannel.receive(0);
258 		assertThat(reply).isNotNull();
259 		assertThat(reply.getPayload()).isEqualTo(11L);
260 	}

327 	private static <T> Message<T> createMessage(T payload, Object correlationId, int sequenceSize, int sequenceNumber,
328 			MessageChannel outputChannel) {
329 		return MessageBuilder.withPayload(payload).setCorrelationId(correlationId).setSequenceSize(sequenceSize)
330 				.setSequenceNumber(sequenceNumber).setReplyChannel(outputChannel).build();
331 	}
