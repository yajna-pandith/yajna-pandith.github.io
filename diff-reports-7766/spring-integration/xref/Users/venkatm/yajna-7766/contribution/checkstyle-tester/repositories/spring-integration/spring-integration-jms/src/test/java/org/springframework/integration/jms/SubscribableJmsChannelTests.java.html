92  	@Test
93  	public void queueReference() throws Exception {
94  		final CountDownLatch latch = new CountDownLatch(2);
95  		final List<Message<?>> receivedList1 = Collections.synchronizedList(new ArrayList<Message<?>>());
96  		MessageHandler handler1 = message -> {
97  			receivedList1.add(message);
98  			latch.countDown();
99  		};
100 		final List<Message<?>> receivedList2 = Collections.synchronizedList(new ArrayList<Message<?>>());
101 		MessageHandler handler2 = message -> {
102 			receivedList2.add(message);
103 			latch.countDown();
104 		};
105 		JmsChannelFactoryBean factoryBean = new JmsChannelFactoryBean(true);
106 		factoryBean.setConnectionFactory(this.connectionFactory);
107 		factoryBean.setDestination(this.queue);
108 		factoryBean.setBeanFactory(mock(BeanFactory.class));
109 		factoryBean.afterPropertiesSet();
110 		SubscribableJmsChannel channel = (SubscribableJmsChannel) factoryBean.getObject();
111 		channel.afterPropertiesSet();
112 		channel.start();
113 		channel.subscribe(handler1);
114 		channel.subscribe(handler2);
115 		channel.send(new GenericMessage<String>("foo"));
116 		channel.send(new GenericMessage<String>("bar"));
117 		latch.await(TIMEOUT, TimeUnit.MILLISECONDS);
118 		assertThat(receivedList1.size()).isEqualTo(1);
119 		assertThat(receivedList1.get(0)).isNotNull();
120 		assertThat(receivedList1.get(0).getPayload()).isEqualTo("foo");
121 		assertThat(receivedList2.size()).isEqualTo(1);
122 		assertThat(receivedList2.get(0)).isNotNull();
123 		assertThat(receivedList2.get(0).getPayload()).isEqualTo("bar");
124 		channel.stop();
125 	}

248 		BeanDefinitionBuilder builder = BeanDefinitionBuilder.genericBeanDefinition(JmsChannelFactoryBean.class);
249 		builder.addConstructorArgValue(true);
250 		builder.addPropertyValue("connectionFactory", this.connectionFactory);
251 		builder.addPropertyValue("destinationName", "dynamicQueue");
252 		builder.addPropertyValue("pubSubDomain", false);
253 		StaticApplicationContext context = new StaticApplicationContext();
254 		context.registerBeanDefinition("channel", builder.getBeanDefinition());
255 		SubscribableJmsChannel channel = context.getBean("channel", SubscribableJmsChannel.class);
256 		assertThat(channel.isRunning()).isFalse();
257 		context.refresh();
258 		assertThat(channel.isRunning()).isTrue();
259 		context.stop();
260 		assertThat(channel.isRunning()).isFalse();
261 		context.close();

307 		verifyLogReceived(logList);

310 	private List<String> insertMockLoggerInListener(
311 			SubscribableJmsChannel channel) {
312 		AbstractMessageListenerContainer container = TestUtils.getPropertyValue(
313 				channel, "container", AbstractMessageListenerContainer.class);
314 		Log logger = mock(Log.class);
315 		final ArrayList<String> logList = new ArrayList<String>();
316 		doAnswer(invocation -> {
317 			String message = invocation.getArgument(0);
318 			if (message.startsWith("Dispatcher has no subscribers")) {
319 				logList.add(message);
320 			}
321 			return null;
322 		}).when(logger).warn(anyString(), any(Exception.class));
323 		when(logger.isWarnEnabled()).thenReturn(true);
324 		Object listener = container.getMessageListener();
325 		DirectFieldAccessor dfa = new DirectFieldAccessor(listener);
326 		dfa.setPropertyValue("logger", logger);
327 		return logList;
328 	}
329 
330 	private void verifyLogReceived(final List<String> logList) {
331 		assertThat(logList.size() > 0).as("Failed to get expected exception").isTrue();
332 		boolean expectedExceptionFound = false;
333 		while (logList.size() > 0) {
334 			String message = logList.remove(0);
335 			assertThat(message).as("Failed to get expected exception").isNotNull();
336 			if (message.startsWith("Dispatcher has no subscribers")) {
337 				expectedExceptionFound = true;
338 				assertThat(message).isEqualTo("Dispatcher has no subscribers for jms-channel 'noSubscribersChannel'.");
339 				break;
340 			}
341 		}
342 		assertThat(expectedExceptionFound).as("Failed to get expected exception").isTrue();
343 	}

345 	/**
346 	 * Blocks until the listener container has subscribed; if the container does not support
347 	 * this test, or the caching mode is incompatible, true is returned. Otherwise blocks
348 	 * until timeout milliseconds have passed, or the consumer has registered.
349 	 * @see DefaultMessageListenerContainer#isRegisteredWithDestination()
350 	 * @param timeout Timeout in milliseconds.
351 	 * @return True if a subscriber has connected or the container/attributes does not support
352 	 * the test. False if a valid container does not have a registered consumer within
353 	 * timeout milliseconds.
354 	 */
355 	private static boolean waitUntilRegisteredWithDestination(SubscribableJmsChannel channel, long timeout) {
356 		AbstractMessageListenerContainer container =
357 				(AbstractMessageListenerContainer) new DirectFieldAccessor(channel).getPropertyValue("container");
358 		if (container instanceof DefaultMessageListenerContainer) {
359 			DefaultMessageListenerContainer listenerContainer =
360 				(DefaultMessageListenerContainer) container;
361 			if (listenerContainer.getCacheLevel() != DefaultMessageListenerContainer.CACHE_CONSUMER) {
362 				return true;
363 			}
364 			while (timeout > 0) {
365 				if (listenerContainer.isRegisteredWithDestination()) {
366 					return true;
367 				}
368 				try {
369 					Thread.sleep(100);
370 				}
371 				catch (InterruptedException e) {
372 
373 				}
374 				timeout -= 100;
375 			}
376 			return false;
377 		}
378 		return true;
379 	}
