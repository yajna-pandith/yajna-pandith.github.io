53  @ContextConfiguration
54  @RunWith(SpringJUnit4ClassRunner.class)
55  @DirtiesContext(classMode = ClassMode.AFTER_EACH_TEST_METHOD)
56  public class ChatMessageInboundChannelAdapterParserTests {
57  
58  	@Autowired
59  	private ApplicationContext context;
60  
61  	@Autowired
62  	private QueueChannel xmppInbound;
63  
64  	@Autowired
65  	private MessageChannel autoChannel;
66  
67  	@Autowired
68  	@Qualifier("autoChannel.adapter")
69  	private ChatMessageListeningEndpoint autoChannelAdapter;
70  
71  	@Test
72  	@SuppressWarnings({"rawtypes", "unchecked"})
73  	public void testInboundAdapter() {
74  		ChatMessageListeningEndpoint adapter = context.getBean("xmppInboundAdapter", ChatMessageListeningEndpoint.class);
75  		MessageChannel errorChannel = (MessageChannel) TestUtils.getPropertyValue(adapter, "errorChannel");
76  		assertThat(errorChannel).isEqualTo(context.getBean("errorChannel"));
77  		assertThat(adapter.isAutoStartup()).isFalse();
78  		QueueChannel channel = (QueueChannel) TestUtils.getPropertyValue(adapter, "outputChannel");
79  		assertThat(channel.getComponentName()).isEqualTo("xmppInbound");
80  		XMPPConnection connection = (XMPPConnection) TestUtils.getPropertyValue(adapter, "xmppConnection");
81  		assertThat(context.getBean("testConnection")).isSameAs(connection);
82  		Object stanzaFilter = context.getBean("stanzaFilter");
83  		assertThat(TestUtils.getPropertyValue(adapter, "stanzaFilter")).isSameAs(stanzaFilter);
84  		assertThat(TestUtils.getPropertyValue(adapter, "payloadExpression.expression")).isEqualTo("#root");
85  		adapter.start();
86  		Map asyncRecvListeners = TestUtils.getPropertyValue(connection, "asyncRecvListeners", Map.class);
87  		assertThat(asyncRecvListeners.size()).isEqualTo(6);
88  		Object lastListener = asyncRecvListeners.values().stream().reduce((first, second) -> second).get();
89  		assertThat(TestUtils.getPropertyValue(lastListener, "packetFilter")).isSameAs(stanzaFilter);
90  		adapter.stop();
91  	}
92  
93  	@Test
94  	public void testInboundAdapterUsageWithHeaderMapper() throws Exception {
95  		XMPPConnection xmppConnection = Mockito.mock(XMPPConnection.class);
96  
97  		ChatMessageListeningEndpoint adapter = context.getBean("xmppInboundAdapter", ChatMessageListeningEndpoint.class);
98  
99  		Field xmppConnectionField = ReflectionUtils.findField(ChatMessageListeningEndpoint.class, "xmppConnection");
100 		xmppConnectionField.setAccessible(true);
101 		ReflectionUtils.setField(xmppConnectionField, adapter, xmppConnection);
102 
103 		StanzaListener stanzaListener = TestUtils.getPropertyValue(adapter, "stanzaListener", StanzaListener.class);
104 
105 		Message message = new Message();
106 		message.setBody("hello");
107 		message.setTo(JidCreate.from("oleg"));
108 		JivePropertiesManager.addProperty(message, "foo", "foo");
109 		JivePropertiesManager.addProperty(message, "bar", "bar");
110 		stanzaListener.processStanza(message);
111 		org.springframework.messaging.Message<?> siMessage = xmppInbound.receive(0);
112 		assertThat(siMessage.getHeaders().get("foo")).isEqualTo("foo");
113 		assertThat(siMessage.getHeaders().get("xmpp_to")).isEqualTo("oleg");
114 	}
115 
116 	@Test
117 	public void testAutoChannel() {
118 		assertThat(TestUtils.getPropertyValue(autoChannelAdapter, "outputChannel")).isSameAs(autoChannel);
119 	}
