46  	@Test
47  	public void adapterWithMessageSelector() {
48  		ClassPathXmlApplicationContext context = new ClassPathXmlApplicationContext(
49  				"jmsInboundWithMessageSelector.xml", this.getClass());
50  		PollableChannel output = (PollableChannel) context.getBean("output2");
51  		Message<?> message = output.receive(timeoutOnReceive);
52  		MessageHistory history = MessageHistory.read(message);
53  		assertThat(history).isNotNull();
54  		Properties componentHistoryRecord = TestUtils.locateComponentInHistory(history, "messageDrivenAdapter", 0);
55  		assertThat(componentHistoryRecord).isNotNull();
56  		JmsMessageDrivenEndpoint endpoint =  context.getBean("messageDrivenAdapter", JmsMessageDrivenEndpoint.class);
57  		assertThat(componentHistoryRecord.get("type")).isEqualTo("jms:message-driven-channel-adapter");
58  		assertThat(message).as("message should not be null").isNotNull();
59  		assertThat(message.getPayload()).isEqualTo("test [with selector: TestProperty = 'foo']");
60  		endpoint.stop();
61  		context.close();
62  	}

164 	@Test
165 	public void testAdapterWithContainerClass() {
166 		ClassPathXmlApplicationContext context = new ClassPathXmlApplicationContext(
167 				"jmsInboundWithContainerClass.xml", this.getClass());
168 		JmsMessageDrivenEndpoint adapter = context.getBean("adapterWithIdleConsumerLimit.adapter", JmsMessageDrivenEndpoint.class);
169 		MessageChannel channel = context.getBean("adapterWithIdleConsumerLimit", MessageChannel.class);
170 		assertThat(TestUtils.getPropertyValue(adapter, "listener.gatewayDelegate.requestChannel")).isSameAs(channel);
171 		adapter.start();
172 		FooContainer container = TestUtils.getPropertyValue(adapter, "listenerContainer", FooContainer.class);
173 		assertThat(context.getBean("adapterWithIdleConsumerLimit.container")).isSameAs(container);
174 		assertThat(new DirectFieldAccessor(container).getPropertyValue("idleConsumerLimit")).isEqualTo(33);
175 		assertThat(new DirectFieldAccessor(container).getPropertyValue("cacheLevel")).isEqualTo(3);
176 		assertThat(TestUtils.getPropertyValue(container, "messageListener"))
177 				.isSameAs(context.getBean("adapterWithIdleConsumerLimit.listener"));
178 		adapter.stop();
179 
180 		adapter = context.getBean("adapterWithIdleConsumerLimit2.adapter", JmsMessageDrivenEndpoint.class);
181 		channel = context.getBean("adapterWithIdleConsumerLimit2", MessageChannel.class);
182 		assertThat(TestUtils.getPropertyValue(adapter, "listener.gatewayDelegate.requestChannel")).isSameAs(channel);
183 		adapter.start();
184 		container = TestUtils.getPropertyValue(adapter, "listenerContainer", FooContainer.class);
185 		assertThat(context.getBean("adapterWithIdleConsumerLimit2.container")).isSameAs(container);
186 		assertThat(new DirectFieldAccessor(container).getPropertyValue("idleConsumerLimit")).isEqualTo(33);
187 		assertThat(new DirectFieldAccessor(container).getPropertyValue("cacheLevel")).isEqualTo(3);
188 		adapter.stop();
189 
190 		adapter = context.getBean("org.springframework.integration.jms.JmsMessageDrivenEndpoint#0", JmsMessageDrivenEndpoint.class);
191 		channel = context.getBean("in", MessageChannel.class);
192 		assertThat(TestUtils.getPropertyValue(adapter, "listener.gatewayDelegate.requestChannel")).isSameAs(channel);
193 		adapter.start();
194 		container = TestUtils.getPropertyValue(adapter, "listenerContainer", FooContainer.class);
195 		assertThat(context.getBean("org.springframework.integration.jms.JmsMessageDrivenEndpoint#0.container"))
196 				.isSameAs(container);
197 		assertThat(new DirectFieldAccessor(container).getPropertyValue("idleConsumerLimit")).isEqualTo(33);
198 		assertThat(new DirectFieldAccessor(container).getPropertyValue("cacheLevel")).isEqualTo(3);
199 		adapter.stop();
200 		context.close();
