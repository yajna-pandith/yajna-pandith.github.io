17  package org.springframework.integration.config.xml;
18  
19  import org.w3c.dom.Element;
20  
21  import org.springframework.beans.BeanMetadataElement;
22  import org.springframework.beans.factory.config.RuntimeBeanReference;
23  import org.springframework.beans.factory.support.AbstractBeanDefinition;
24  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
25  import org.springframework.beans.factory.support.RootBeanDefinition;
26  import org.springframework.beans.factory.xml.AbstractBeanDefinitionParser;
27  import org.springframework.beans.factory.xml.ParserContext;
28  import org.springframework.integration.config.ExpressionFactoryBean;
29  import org.springframework.integration.config.IdempotentReceiverAutoProxyCreatorInitializer;
30  import org.springframework.integration.handler.ExpressionEvaluatingMessageProcessor;
31  import org.springframework.integration.handler.advice.IdempotentReceiverInterceptor;
32  import org.springframework.integration.metadata.SimpleMetadataStore;
33  import org.springframework.integration.selector.MetadataStoreSelector;
34  import org.springframework.util.StringUtils;
35  
36  /**
37   * Parser for the &lt;idempotent-receiver/&gt; element.
38   *
39   * @author Artem Bilan
40   * @author Gary Russell
41   *
42   * @since 4.1
43   */
44  public class IdempotentReceiverInterceptorParser extends AbstractBeanDefinitionParser {
45  
46  	@Override // NOSONAR complexity
47  	protected AbstractBeanDefinition parseInternal(Element element, ParserContext parserContext) { // NOSONAR
48  		Object source = parserContext.extractSource(element);
49  
50  		String selector = element.getAttribute("selector");
51  		boolean hasSelector = StringUtils.hasText(selector);
52  		String store = element.getAttribute("metadata-store");
53  		boolean hasStore = StringUtils.hasText(store);
54  		String keyStrategy = element.getAttribute("key-strategy");
55  		boolean hasKeyStrategy = StringUtils.hasText(keyStrategy);
56  		String keyExpression = element.getAttribute("key-expression");
57  		boolean hasKeyExpression = StringUtils.hasText(keyExpression);
58  		String valueStrategy = element.getAttribute("value-strategy");
59  		boolean hasValueStrategy = StringUtils.hasText(valueStrategy);
60  		String valueExpression = element.getAttribute("value-expression");
61  		boolean hasValueExpression = StringUtils.hasText(valueExpression);
62  
63  		String endpoints = element.getAttribute("endpoint");
64  
65  		if (!hasSelector && !(hasKeyStrategy || hasKeyExpression)) {
66  			parserContext.getReaderContext().error("One of the 'selector', 'key-strategy' or 'key-expression' " +
67  					"attributes must be provided", source);
68  		}
69  
70  		if (hasSelector && (hasStore || hasKeyStrategy || hasKeyExpression || hasValueStrategy // NOSONAR complexity
71  				|| hasValueExpression)) {
72  			parserContext.getReaderContext().error("The 'selector' attribute is mutually exclusive with " +
73  					"'metadata-store', 'key-strategy', 'key-expression', 'value-strategy' " +
74  					"or 'value-expression'", source);
75  		}
76  
77  		if (hasKeyStrategy && hasKeyExpression) {
78  			parserContext.getReaderContext().error("The 'key-strategy' and 'key-expression' attributes " +
79  					"are mutually exclusive", source);
80  		}
81  
82  		if (hasValueStrategy && hasValueExpression) {
83  			parserContext.getReaderContext().error("The 'value-strategy' and 'value-expression' attributes " +
84  					"are mutually exclusive", source);
85  		}
86  
87  		if (!StringUtils.hasText(endpoints)) {
88  			parserContext.getReaderContext().error("The 'endpoint' attribute is required", source);
89  		}
90  
91  		BeanMetadataElement selectorBeanDefinition;
92  		if (hasSelector) {
93  			selectorBeanDefinition = new RuntimeBeanReference(selector);
94  		}
95  		else {
96  			BeanDefinitionBuilder selectorBuilder =
97  					BeanDefinitionBuilder.genericBeanDefinition(MetadataStoreSelector.class);
98  			BeanMetadataElement keyStrategyBeanDefinition;
99  			if (hasKeyStrategy) {
100 				keyStrategyBeanDefinition = new RuntimeBeanReference(keyStrategy);
101 			}
102 			else {
103 				keyStrategyBeanDefinition =
104 						BeanDefinitionBuilder.genericBeanDefinition(ExpressionEvaluatingMessageProcessor.class)
105 								.addConstructorArgValue(
106 										BeanDefinitionBuilder.genericBeanDefinition(ExpressionFactoryBean.class).
107 												addConstructorArgValue(keyExpression)
108 												.getBeanDefinition()
109 								)
110 								.getBeanDefinition();
111 			}
112 			selectorBuilder.addConstructorArgValue(keyStrategyBeanDefinition);
113 
114 			BeanMetadataElement valueStrategyBeanDefinition = null;
115 			if (hasValueStrategy) {
116 				valueStrategyBeanDefinition = new RuntimeBeanReference(valueStrategy);
117 			}
118 			else if (hasValueExpression) {
119 				valueStrategyBeanDefinition =
120 						BeanDefinitionBuilder.genericBeanDefinition(ExpressionEvaluatingMessageProcessor.class)
121 								.addConstructorArgValue(
122 										BeanDefinitionBuilder.genericBeanDefinition(ExpressionFactoryBean.class).
123 												addConstructorArgValue(valueExpression)
124 												.getBeanDefinition()
125 								)
126 								.getBeanDefinition();
127 			}
128 			selectorBuilder.addConstructorArgValue(valueStrategyBeanDefinition);
129 
130 			if (hasStore) {
131 				selectorBuilder.addConstructorArgReference(store);
132 			}
133 			else {
134 				selectorBuilder.addConstructorArgValue(new RootBeanDefinition(SimpleMetadataStore.class));
135 			}
136 			selectorBeanDefinition = selectorBuilder.getBeanDefinition();
137 		}
138 
139 		BeanDefinitionBuilder builder = BeanDefinitionBuilder.genericBeanDefinition(IdempotentReceiverInterceptor.class)
140 				.addConstructorArgValue(selectorBeanDefinition);
141 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "throw-exception-on-rejection");
142 		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "discard-channel");
143 		AbstractBeanDefinition interceptorBeanDefinition = builder.getBeanDefinition();
144 		interceptorBeanDefinition.setAttribute(
145 				IdempotentReceiverAutoProxyCreatorInitializer.IDEMPOTENT_ENDPOINTS_MAPPING,
146 				endpoints);
147 		return interceptorBeanDefinition;
148 	}
149 
150 	@Override
151 	protected boolean shouldGenerateIdAsFallback() {
152 		return true;
153 	}
154 
155 	@Override
156 	protected boolean shouldFireEvents() {
157 		return false;
158 	}
