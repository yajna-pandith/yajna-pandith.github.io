301 	@Test
302 	public void testSharedSessionCachedReset() throws Exception {
303 		JSch jsch = spy(new JSch());
304 		Constructor<com.jcraft.jsch.Session> ctor =
305 				com.jcraft.jsch.Session.class.getDeclaredConstructor(JSch.class, String.class, String.class,
306 						int.class);
307 		ctor.setAccessible(true);
308 		com.jcraft.jsch.Session jschSession1 = spy(ctor.newInstance(jsch, "foo", "host", 22));
309 		com.jcraft.jsch.Session jschSession2 = spy(ctor.newInstance(jsch, "foo", "host", 22));
310 
311 		willAnswer(invocation -> {
312 			new DirectFieldAccessor(jschSession1).setPropertyValue("isConnected", true);
313 			return null;
314 		})
315 				.given(jschSession1)
316 				.connect();
317 
318 		willAnswer(invocation -> {
319 			new DirectFieldAccessor(jschSession2).setPropertyValue("isConnected", true);
320 			return null;
321 		})
322 				.given(jschSession2)
323 				.connect();
324 
325 		when(jsch.getSession("foo", "host", 22)).thenReturn(jschSession1, jschSession2);
326 		final ChannelSftp channel1 = spy(new ChannelSftp());
327 		doReturn("channel1").when(channel1).toString();
328 		final ChannelSftp channel2 = spy(new ChannelSftp());
329 		doReturn("channel2").when(channel2).toString();
330 		final ChannelSftp channel3 = spy(new ChannelSftp());
331 		doReturn("channel3").when(channel3).toString();
332 		final ChannelSftp channel4 = spy(new ChannelSftp());
333 		doReturn("channel4").when(channel4).toString();
334 		new DirectFieldAccessor(channel1).setPropertyValue("session", jschSession1);
335 		new DirectFieldAccessor(channel2).setPropertyValue("session", jschSession1);
336 		// Can't use when(session.open()) with a spy
337 		final AtomicInteger n = new AtomicInteger();
338 		doAnswer(invocation -> n.getAndIncrement() == 0 ? channel1 : channel2).when(jschSession1).openChannel("sftp");
339 		doAnswer(invocation -> n.getAndIncrement() < 3 ? channel3 : channel4).when(jschSession2).openChannel("sftp");
340 		DefaultSftpSessionFactory factory = new DefaultSftpSessionFactory(jsch, true);
341 		factory.setHost("host");
342 		factory.setUser("foo");
343 		factory.setPassword("bar");
344 		CachingSessionFactory<LsEntry> cachedFactory = new CachingSessionFactory<LsEntry>(factory);
345 		noopConnect(channel1);
346 		noopConnect(channel2);
347 		noopConnect(channel3);
348 		noopConnect(channel4);
349 		Session<LsEntry> s1 = cachedFactory.getSession();
350 		Session<LsEntry> s2 = cachedFactory.getSession();
351 		assertThat(TestUtils.getPropertyValue(s2, "targetSession.jschSession")).isSameAs(jschSession1);
352 		assertThat(TestUtils.getPropertyValue(s1, "targetSession.channel")).isSameAs(channel1);
353 		assertThat(TestUtils.getPropertyValue(s2, "targetSession.channel")).isSameAs(channel2);
354 		assertThat(TestUtils.getPropertyValue(s2, "targetSession.jschSession"))
355 				.isSameAs(TestUtils.getPropertyValue(s1, "targetSession.jschSession"));
356 		s1.close();
357 		Session<LsEntry> s3 = cachedFactory.getSession();
358 		assertThat(TestUtils.getPropertyValue(s3, "targetSession"))
359 				.isSameAs(TestUtils.getPropertyValue(s1, "targetSession"));
360 		assertThat(TestUtils.getPropertyValue(s3, "targetSession.channel")).isSameAs(channel1);
361 		s3.close();
362 		cachedFactory.resetCache();
363 		verify(jschSession1, never()).disconnect();
364 		s3 = cachedFactory.getSession();
365 		assertThat(TestUtils.getPropertyValue(s3, "targetSession.jschSession")).isSameAs(jschSession2);
366 		assertThat(TestUtils.getPropertyValue(s3, "targetSession"))
367 				.isNotSameAs(TestUtils.getPropertyValue(s1, "targetSession"));
368 		assertThat(TestUtils.getPropertyValue(s3, "targetSession.channel")).isSameAs(channel3);
369 		s2.close();
370 		verify(jschSession1).disconnect();
371 		s2 = cachedFactory.getSession();
372 		assertThat(TestUtils.getPropertyValue(s2, "targetSession.jschSession")).isSameAs(jschSession2);
373 		assertThat(TestUtils.getPropertyValue(s2, "targetSession"))
374 				.isNotSameAs(TestUtils.getPropertyValue(s3, "targetSession"));
375 		assertThat(TestUtils.getPropertyValue(s2, "targetSession.channel")).isSameAs(channel4);
376 		s2.close();
377 		s3.close();
378 		verify(jschSession2, never()).disconnect();
379 		cachedFactory.resetCache();
380 		verify(jschSession2).disconnect();
381 	}

408 
409 				String[] files = new File("remote-test-dir").list();
410 				Vector<LsEntry> sftpEntries = new Vector<>();

412 					LsEntry lsEntry = mock(LsEntry.class);
413 					SftpATTRS attributes = mock(SftpATTRS.class);
