138 	public void errorFlowAndRetries() throws Exception {
139 		delayHandler.setDefaultDelay(10);
140 		delayHandler.setRetryDelay(15);
141 		startDelayerHandler();
142 		Message<?> message = MessageBuilder.withPayload("test")
143 				.setHeader("foo", new AtomicInteger())
144 				.build();
145 		final CountDownLatch latch = new CountDownLatch(1);
146 		final AtomicInteger count = new AtomicInteger();
147 		delayHandler.setDelayedMessageErrorChannel((m, t) -> {
148 			count.incrementAndGet();
149 			int deliveries = StaticMessageHeaderAccessor.getDeliveryAttempt(m).get();
150 			((MessagingException) m.getPayload())
151 					.getFailedMessage()
152 					.getHeaders()
153 					.get("foo", AtomicInteger.class)
154 					.incrementAndGet();
155 			if (deliveries < 3) {
156 				throw new RuntimeException("fail");
157 			}
158 			else if (deliveries == 3) {
159 				return false;
160 			}
161 			else {
162 				latch.countDown();
163 				return true;
164 			}
165 		});
166 		delayHandler.setOutputChannel((m, t) -> {
167 			throw new MessagingException(m);
168 		});
169 		input.send(message);
170 		assertThat(latch.await(10, TimeUnit.SECONDS)).isTrue();
171 		Thread.sleep(50);
172 		assertThat(count.get()).isEqualTo(4);
173 		assertThat(TestUtils.getPropertyValue(this.delayHandler, "deliveries", Map.class).size()).isEqualTo(0);
174 	}

528 	private void waitForLatch(long timeout) {
529 		try {
530 			this.latch.await(timeout, TimeUnit.MILLISECONDS);
531 			if (latch.getCount() != 0) {
532 				throw new TestTimedOutException();
533 			}
534 		}
535 		catch (InterruptedException e) {
536 			throw new RuntimeException("interrupted while waiting for latch");
537 		}
538 	}

541 	private class ResultHandler implements MessageHandler {
542 
543 		private volatile Message<?> lastMessage;
544 
545 		private volatile Thread lastThread;
546 
547 		ResultHandler() {
548 			super();
549 		}
550 
551 		@Override
552 		public void handleMessage(Message<?> message) {
553 			this.lastMessage = message;
554 			this.lastThread = Thread.currentThread();
555 			latch.countDown();
556 		}
557 
558 	}
