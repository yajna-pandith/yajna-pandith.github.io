54  	@Test
55  	public void endpointRegistrationWithInputChannelReference() {
56  		TestApplicationContext context = TestUtils.createTestApplicationContext();
57  		QueueChannel sourceChannel = new QueueChannel();
58  		QueueChannel targetChannel = new QueueChannel();
59  		context.registerChannel("sourceChannel", sourceChannel);
60  		context.registerChannel("targetChannel", targetChannel);
61  		Message<String> message = MessageBuilder.withPayload("test")
62  				.setReplyChannelName("targetChannel").build();
63  		sourceChannel.send(message);
64  		AbstractReplyProducingMessageHandler handler = new AbstractReplyProducingMessageHandler() {
65  
66  			@Override
67  			public Object handleRequestMessage(Message<?> message) {
68  				return message;
69  			}
70  		};
71  		handler.setBeanFactory(context);
72  		handler.afterPropertiesSet();
73  		PollingConsumer endpoint = new PollingConsumer(sourceChannel, handler);
74  		endpoint.setBeanFactory(mock(BeanFactory.class));
75  		context.registerEndpoint("testEndpoint", endpoint);
76  		context.refresh();
77  		Message<?> result = targetChannel.receive(10000);
78  		assertThat(result.getPayload()).isEqualTo("test");
79  		context.close();
80  	}

148 	@Test
149 	public void bothConsumersReceivePublishSubscribeMessage() throws InterruptedException {
150 		TestApplicationContext context = TestUtils.createTestApplicationContext();
151 		PublishSubscribeChannel inputChannel = new PublishSubscribeChannel();
152 		QueueChannel outputChannel1 = new QueueChannel();
153 		QueueChannel outputChannel2 = new QueueChannel();
154 		final CountDownLatch latch = new CountDownLatch(2);
155 		AbstractReplyProducingMessageHandler handler1 = new AbstractReplyProducingMessageHandler() {
156 
157 			@Override
158 			public Object handleRequestMessage(Message<?> message) {
159 				latch.countDown();
160 				return message;
161 			}
162 		};
163 		AbstractReplyProducingMessageHandler handler2 = new AbstractReplyProducingMessageHandler() {
164 
165 			@Override
166 			public Object handleRequestMessage(Message<?> message) {
167 				latch.countDown();
168 				return message;
169 			}
170 		};
171 		context.registerChannel("input", inputChannel);
172 		context.registerChannel("output1", outputChannel1);
173 		context.registerChannel("output2", outputChannel2);
174 		handler1.setOutputChannel(outputChannel1);
175 		handler2.setOutputChannel(outputChannel2);
176 		EventDrivenConsumer endpoint1 = new EventDrivenConsumer(inputChannel, handler1);
177 		EventDrivenConsumer endpoint2 = new EventDrivenConsumer(inputChannel, handler2);
178 		context.registerEndpoint("testEndpoint1", endpoint1);
179 		context.registerEndpoint("testEndpoint2", endpoint2);
180 		context.refresh();
181 		inputChannel.send(new GenericMessage<String>("testing"));
182 		latch.await(500, TimeUnit.MILLISECONDS);
183 		assertThat(latch.getCount()).as("both handlers should have been invoked").isEqualTo(0);
184 		Message<?> message1 = outputChannel1.receive(500);
185 		Message<?> message2 = outputChannel2.receive(500);
186 		context.close();
187 		assertThat(message1).as("both handlers should have replied to the message").isNotNull();
188 		assertThat(message2).as("both handlers should have replied to the message").isNotNull();
189 	}

191 	@Test
192 	public void errorChannelWithFailedDispatch() throws InterruptedException {
193 		TestApplicationContext context = TestUtils.createTestApplicationContext();
194 		QueueChannel errorChannel = new QueueChannel();
195 		QueueChannel outputChannel = new QueueChannel();
196 		context.registerChannel("errorChannel", errorChannel);
197 		CountDownLatch latch = new CountDownLatch(1);
198 		SourcePollingChannelAdapter channelAdapter = new SourcePollingChannelAdapter();
199 		channelAdapter.setSource(new FailingSource(latch));
200 		PollerMetadata pollerMetadata = new PollerMetadata();
201 		pollerMetadata.setTrigger(new PeriodicTrigger(1000));
202 		channelAdapter.setOutputChannel(outputChannel);
203 		context.registerEndpoint("testChannel", channelAdapter);
204 		context.refresh();
205 		latch.await(2000, TimeUnit.MILLISECONDS);
206 		Message<?> message = errorChannel.receive(5000);
207 		context.close();
208 		assertThat(outputChannel.receive(100)).isNull();
209 		assertThat(message).as("message should not be null").isNotNull();
210 		assertThat(message instanceof ErrorMessage).isTrue();
211 		Throwable exception = ((ErrorMessage) message).getPayload();
212 		assertThat(exception.getCause().getMessage()).isEqualTo("intentional test failure");
213 	}

215 	@Test
216 	public void consumerSubscribedToErrorChannel() throws InterruptedException {
217 		TestApplicationContext context = TestUtils.createTestApplicationContext();
218 		QueueChannel errorChannel = new QueueChannel();
219 		context.registerChannel(IntegrationContextUtils.ERROR_CHANNEL_BEAN_NAME, errorChannel);
220 		final CountDownLatch latch = new CountDownLatch(1);
221 		AbstractReplyProducingMessageHandler handler = new AbstractReplyProducingMessageHandler() {
222 
223 			@Override
224 			public Object handleRequestMessage(Message<?> message) {
225 				latch.countDown();
226 				return null;
227 			}
228 		};
229 		PollingConsumer endpoint = new PollingConsumer(errorChannel, handler);
230 		endpoint.setBeanFactory(mock(BeanFactory.class));
231 		context.registerEndpoint("testEndpoint", endpoint);
232 		context.refresh();
233 		errorChannel.send(new ErrorMessage(new RuntimeException("test-exception")));
234 		latch.await(1000, TimeUnit.MILLISECONDS);
235 		assertThat(latch.getCount()).as("handler should have received error message").isEqualTo(0);
236 		context.close();
237 	}

240 	private static class FailingSource implements MessageSource<Object> {
241 
242 		private final CountDownLatch latch;
243 
244 		FailingSource(CountDownLatch latch) {
245 			this.latch = latch;
246 		}
247 
248 		@Override
249 		public Message<Object> receive() {
250 			latch.countDown();
251 			throw new RuntimeException("intentional test failure");
252 		}

254 	}
