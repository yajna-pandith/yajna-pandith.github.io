56  	@Test
57  	void testSend() {
58  		DirectChannel channel = new DirectChannel();
59  		Log logger = spy(TestUtils.getPropertyValue(channel, "logger", Log.class));
60  		when(logger.isDebugEnabled()).thenReturn(true);
61  		new DirectFieldAccessor(channel).setPropertyValue("logger", logger);
62  		ThreadNameExtractingTestTarget target = new ThreadNameExtractingTestTarget();
63  		channel.subscribe(target);
64  		GenericMessage<String> message = new GenericMessage<>("test");
65  		assertThat(channel.send(message)).isTrue();
66  		assertThat(target.threadName).isEqualTo(Thread.currentThread().getName());
67  		DirectFieldAccessor channelAccessor = new DirectFieldAccessor(channel);
68  		UnicastingDispatcher dispatcher = (UnicastingDispatcher) channelAccessor.getPropertyValue("dispatcher");
69  		DirectFieldAccessor dispatcherAccessor = new DirectFieldAccessor(dispatcher);
70  		Object loadBalancingStrategy = dispatcherAccessor.getPropertyValue("loadBalancingStrategy");
71  		assertThat(loadBalancingStrategy instanceof RoundRobinLoadBalancingStrategy).isTrue();
72  		ArgumentCaptor<String> captor = ArgumentCaptor.forClass(String.class);
73  		verify(logger, times(2)).debug(captor.capture());
74  		List<String> logs = captor.getAllValues();
75  		assertThat(logs.size()).isEqualTo(2);
76  		assertThat(logs.get(0)).startsWith("preSend");
77  		assertThat(logs.get(1)).startsWith("postSend");
78  	}

80  	@Test
81  	void testSendPerfOneHandler() {
82  		/*
83  		 *  INT-3308 - used to run 12 million/sec
84  		 *  1. optimize for single handler 20 million/sec
85  		 *  2. Don't iterate over empty datatypes 23 million/sec
86  		 *  3. Don't iterate over empty interceptors 31 million/sec
87  		 *  4. Move single handler optimization to dispatcher 34 million/sec
88  		 *
89  		 *  29 million per second with increment counter in the handler
90  		 */
91  		DirectChannel channel = new DirectChannel();
92  		final AtomicInteger count = new AtomicInteger();
93  		channel.subscribe(message -> count.incrementAndGet());
94  		GenericMessage<String> message = new GenericMessage<>("test");
95  		assertThat(channel.send(message)).isTrue();
96  		for (int i = 0; i < 10000000; i++) {
97  			channel.send(message);
98  		}
99  	}

101 	@Test
102 	void testSendPerfTwoHandlers() {
103 		/*
104 		 *  INT-3308 - used to run 6.4 million/sec
105 		 *  1. Skip empty iterators as above 7.2 million/sec
106 		 *  2. optimize for single handler 6.7 million/sec (small overhead added)
107 		 *  3. remove LB rwlock from UnicastingDispatcher 7.2 million/sec
108 		 *  4. Move single handler optimization to dispatcher 7.3 million/sec
109 		 */
110 		DirectChannel channel = new DirectChannel();
111 		final AtomicInteger count1 = new AtomicInteger();
112 		final AtomicInteger count2 = new AtomicInteger();
113 		channel.subscribe(message -> count1.incrementAndGet());
114 		channel.subscribe(message -> count2.getAndIncrement());
115 		GenericMessage<String> message = new GenericMessage<>("test");
116 		assertThat(channel.send(message)).isTrue();
117 		for (int i = 0; i < 10000000; i++) {
118 			channel.send(message);
119 		}
120 		assertThat(count1.get()).isEqualTo(5000001);
121 		assertThat(count2.get()).isEqualTo(5000000);
122 	}

142 	@Test
143 	void testSendInSeparateThread() throws InterruptedException {
144 		CountDownLatch latch = new CountDownLatch(1);
145 		final DirectChannel channel = new DirectChannel();
146 		ThreadNameExtractingTestTarget target = new ThreadNameExtractingTestTarget(latch);
147 		channel.subscribe(target);
148 		final GenericMessage<String> message = new GenericMessage<>("test");
149 		new Thread((Runnable) () -> channel.send(message), "test-thread").start();
150 		latch.await(1000, TimeUnit.MILLISECONDS);
151 		assertThat(target.threadName).isEqualTo("test-thread");
152 	}

154 	@Test
155 	void testChannelCreationWithBeanDefinitionOverrideTrue() throws Exception {
156 		ClassPathXmlApplicationContext parentContext =
157 				new ClassPathXmlApplicationContext("parent-config.xml", getClass());
158 		MessageChannel parentChannelA = parentContext.getBean("parentChannelA", MessageChannel.class);
159 		MessageChannel parentChannelB = parentContext.getBean("parentChannelB", MessageChannel.class);
160 
161 		ClassPathXmlApplicationContext context = new ClassPathXmlApplicationContext();
162 		context.setAllowBeanDefinitionOverriding(false);
163 		context.setConfigLocations("classpath:org/springframework/integration/channel/channel-override-config.xml");
164 		context.setParent(parentContext);
165 		Method method = ReflectionUtils.findMethod(ClassPathXmlApplicationContext.class, "obtainFreshBeanFactory");
166 		method.setAccessible(true);
167 		method.invoke(context);
168 		assertThat(context.containsBean("channelA")).isFalse();
169 		assertThat(context.containsBean("channelB")).isFalse();
170 		assertThat(context.containsBean("channelC")).isTrue();
171 		assertThat(context.containsBean("channelD")).isTrue();
172 
173 		context.refresh();
174 
175 		PublishSubscribeChannel channelEarly = context.getBean("channelEarly", PublishSubscribeChannel.class);
176 
177 		assertThat(context.containsBean("channelA")).isTrue();
178 		assertThat(context.containsBean("channelB")).isTrue();
179 		assertThat(context.containsBean("channelC")).isTrue();
180 		assertThat(context.containsBean("channelD")).isTrue();
181 		EventDrivenConsumer consumerA = context.getBean("serviceA", EventDrivenConsumer.class);
182 		assertThat(TestUtils.getPropertyValue(consumerA, "inputChannel")).isEqualTo(context.getBean("channelA"));
183 		assertThat(TestUtils.getPropertyValue(consumerA, "handler.outputChannelName")).isEqualTo("channelB");
184 
185 		EventDrivenConsumer consumerB = context.getBean("serviceB", EventDrivenConsumer.class);
186 		assertThat(TestUtils.getPropertyValue(consumerB, "inputChannel")).isEqualTo(context.getBean("channelB"));
187 		assertThat(TestUtils.getPropertyValue(consumerB, "handler.outputChannelName")).isEqualTo("channelC");
188 
189 		EventDrivenConsumer consumerC = context.getBean("serviceC", EventDrivenConsumer.class);
190 		assertThat(TestUtils.getPropertyValue(consumerC, "inputChannel")).isEqualTo(context.getBean("channelC"));
191 		assertThat(TestUtils.getPropertyValue(consumerC, "handler.outputChannelName")).isEqualTo("channelD");
192 
193 		EventDrivenConsumer consumerD = context.getBean("serviceD", EventDrivenConsumer.class);
194 		assertThat(TestUtils.getPropertyValue(consumerD, "inputChannel")).isEqualTo(parentChannelA);
195 		assertThat(TestUtils.getPropertyValue(consumerD, "handler.outputChannelName")).isEqualTo("parentChannelB");
196 
197 		EventDrivenConsumer consumerE = context.getBean("serviceE", EventDrivenConsumer.class);
198 		assertThat(TestUtils.getPropertyValue(consumerE, "inputChannel")).isEqualTo(parentChannelB);
199 
200 		EventDrivenConsumer consumerF = context.getBean("serviceF", EventDrivenConsumer.class);
201 		assertThat(TestUtils.getPropertyValue(consumerF, "inputChannel")).isEqualTo(channelEarly);
202 
203 		context.close();
204 		parentContext.close();
205 	}

208 	private static class ThreadNameExtractingTestTarget implements MessageHandler {
209 
210 		private String threadName;
211 
212 		private final CountDownLatch latch;
213 
214 
215 		ThreadNameExtractingTestTarget() {
216 			this(null);
217 		}
218 
219 		ThreadNameExtractingTestTarget(@Nullable CountDownLatch latch) {
220 			this.latch = latch;
221 		}
222 
223 		@Override
224 		public void handleMessage(Message<?> message) {
225 			this.threadName = Thread.currentThread().getName();
226 			if (this.latch != null) {
227 				this.latch.countDown();
228 			}
229 		}
230 
231 	}
