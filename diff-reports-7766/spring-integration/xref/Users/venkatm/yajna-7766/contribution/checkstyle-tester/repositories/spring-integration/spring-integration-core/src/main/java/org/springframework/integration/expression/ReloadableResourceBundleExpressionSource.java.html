46  /**
47   * {@link ExpressionSource} implementation that accesses resource bundles using specified basenames.
48   * This class uses {@link java.util.Properties} instances as its custom data structure for expressions,
49   * loading them via a {@link org.springframework.util.PropertiesPersister} strategy: The default
50   * strategy is capable of loading properties files with a specific character encoding, if desired.
51   *
52   * @author Juergen Hoeller
53   * @author Mark Fisher
54   * @author Gary Russell
55   * @author Artem Bilan
56   *
57   * @since 2.0

60   * @see #setBasenames
61   * @see #setDefaultEncoding
62   * @see #setFileEncodings
63   * @see #setPropertiesPersister
64   * @see #setResourceLoader
65   * @see org.springframework.util.DefaultPropertiesPersister
66   * @see org.springframework.core.io.DefaultResourceLoader
67   * @see java.util.ResourceBundle
68   */

111 	 * Set a single basename, following the basic ResourceBundle convention of
112 	 * not specifying file extension or language codes, but referring to a Spring
113 	 * resource location: e.g. "META-INF/expressions" for "META-INF/expressions.properties",
114 	 * "META-INF/expressions_en.properties", etc.
115 	 * <p>XML properties files are also supported: .g. "META-INF/expressions" will find
116 	 * and load "META-INF/expressions.xml", "META-INF/expressions_en.xml", etc as well.
117 	 * @param basename the single basename
118 	 * @see #setBasenames
119 	 * @see org.springframework.core.io.ResourceEditor
120 	 * @see java.util.ResourceBundle
121 	 */
122 	public void setBasename(String basename) {
123 		setBasenames(new String[] { basename });
124 	}

126 	/**
127 	 * Set an array of basenames, each following the basic ResourceBundle convention
128 	 * of not specifying file extension or language codes, but referring to a Spring
129 	 * resource location: e.g. "META-INF/expressions" for "META-INF/expressions.properties",
130 	 * "META-INF/expressions_en.properties", etc.
131 	 * <p>XML properties files are also supported: .g. "META-INF/expressions" will find
132 	 * and load "META-INF/expressions.xml", "META-INF/expressions_en.xml", etc as well.
133 	 * <p>The associated resource bundles will be checked sequentially when resolving
134 	 * an expression key. Note that expression definitions in a <i>previous</i> resource
135 	 * bundle will override ones in a later bundle, due to the sequential lookup.
136 	 * @param basenames an array of basenames
137 	 * @see #setBasename
138 	 * @see java.util.ResourceBundle
139 	 */
140 	public void setBasenames(@Nullable String[] basenames) {
141 		if (basenames != null) {
142 			this.basenames = new String[basenames.length];
143 			for (int i = 0; i < basenames.length; i++) {
144 				String basename = basenames[i];
145 				Assert.hasText(basename, "Basename must not be empty");
146 				this.basenames[i] = basename.trim();
147 			}
148 		}
149 		else {
150 			this.basenames = new String[0];
151 		}
152 	}

183 	 * Set whether to fall back to the system Locale if no files for a specific
184 	 * Locale have been found. Default is "true"; if this is turned off, the only
185 	 * fallback will be the default file (e.g. "expressions.properties" for
186 	 * basename "expressions").
187 	 * <p>Falling back to the system Locale is the default behavior of
188 	 * <code>java.util.ResourceBundle</code>. However, this is often not
189 	 * desirable in an application server environment, where the system Locale
190 	 * is not relevant to the application at all: Set this flag to "false"
191 	 * in such a scenario.
192 	 * @param fallbackToSystemLocale true to fall back.
193 	 */
194 	public void setFallbackToSystemLocale(boolean fallbackToSystemLocale) {
195 		this.fallbackToSystemLocale = fallbackToSystemLocale;
196 	}

198 	/**
199 	 * Set the number of seconds to cache loaded properties files.
200 	 * <ul>
201 	 * <li>Default is "-1", indicating to cache forever (just like
202 	 * <code>java.util.ResourceBundle</code>).
203 	 * <li>A positive number will cache loaded properties files for the given
204 	 * number of seconds. This is essentially the interval between refresh checks.
205 	 * Note that a refresh attempt will first check the last-modified timestamp
206 	 * of the file before actually reloading it; so if files don't change, this
207 	 * interval can be set rather low, as refresh attempts will not actually reload.
208 	 * <li>A value of "0" will check the last-modified timestamp of the file on
209 	 * every expression access. <b>Do not use this in a production environment!</b>
210 	 * </ul>
211 	 * @param cacheSeconds The cache seconds.
212 	 */
213 	public void setCacheSeconds(int cacheSeconds) {
214 		this.cacheMillis = (cacheSeconds * 1000);
215 	}

244 	 * Resolves the given key in the retrieved bundle files to an Expression.
245 	 */
246 	@Override
247 	public Expression getExpression(String key, Locale locale) {
248 		String expressionString = getExpressionString(key, locale);
249 		if (expressionString != null) {
250 			return this.parser.parseExpression(expressionString);
251 		}
252 		return null;
253 	}
254 
255 	@Nullable
256 	private String getExpressionString(String key, Locale locale) {
257 		if (this.cacheMillis < 0) {
258 			PropertiesHolder propHolder = getMergedProperties(locale);
259 			return propHolder.getProperty(key);
260 		}
261 		else {
262 			for (String basename : this.basenames) {
263 				List<String> filenames = calculateAllFilenames(basename, locale);
264 				for (String filename : filenames) {
265 					PropertiesHolder propHolder = getProperties(filename);
266 					String result = propHolder.getProperty(key);
267 					if (result != null) {
268 						return result;
269 					}
270 				}
271 			}
272 		}
273 		return null;
274 	}

276 	/**
277 	 * Get a PropertiesHolder that contains the actually visible properties
278 	 * for a Locale, after merging all specified resource bundles.
279 	 * Either fetches the holder from the cache or freshly loads it.
280 	 * <p> Only used when caching resource bundle contents forever, i.e.
281 	 * with cacheSeconds < 0. Therefore, merged properties are always
282 	 * cached forever.
283 	 */
284 	private PropertiesHolder getMergedProperties(Locale locale) {
285 		synchronized (this.cachedMergedProperties) {
286 			PropertiesHolder mergedHolder = this.cachedMergedProperties.get(locale);
287 			if (mergedHolder != null) {
288 				return mergedHolder;
289 			}
290 			Properties mergedProps = new Properties();
291 			mergedHolder = new PropertiesHolder(mergedProps, -1);
292 			for (int i = this.basenames.length - 1; i >= 0; i--) {
293 				List<String> filenames = calculateAllFilenames(this.basenames[i], locale);
294 				for (int j = filenames.size() - 1; j >= 0; j--) {
295 					String filename = filenames.get(j);
296 					PropertiesHolder propHolder = getProperties(filename);
297 					Properties propHolderProperties = propHolder.getProperties();
298 					if (propHolderProperties != null) {
299 						mergedProps.putAll(propHolderProperties);
300 					}
301 				}
302 			}
303 			this.cachedMergedProperties.put(locale, mergedHolder);
304 			return mergedHolder;
305 		}
306 	}

308 	/**
309 	 * Calculate all filenames for the given bundle basename and Locale.
310 	 * Will calculate filenames for the given Locale, the system Locale
311 	 * (if applicable), and the default file.
312 	 * @param basename the basename of the bundle
313 	 * @param locale the locale
314 	 * @return the List of filenames to check
315 	 * @see #setFallbackToSystemLocale
316 	 * @see #calculateFilenamesForLocale
317 	 */
318 	private List<String> calculateAllFilenames(String basename, Locale locale) {
319 		synchronized (this.cachedFilenames) {
320 			Map<Locale, List<String>> localeMap = this.cachedFilenames.get(basename);
321 			if (localeMap != null) {
322 				List<String> filenames = localeMap.get(locale);
323 				if (filenames != null) {
324 					return filenames;
325 				}
326 			}
327 			List<String> filenames = new ArrayList<>(7);
328 			filenames.addAll(calculateFilenamesForLocale(basename, locale));
329 			if (this.fallbackToSystemLocale && !locale.equals(Locale.getDefault())) {
330 				List<String> fallbackFilenames = calculateFilenamesForLocale(basename, Locale.getDefault());
331 				for (String fallbackFilename : fallbackFilenames) {
332 					if (!filenames.contains(fallbackFilename)) {
333 						// Entry for fallback locale that isn't already in filenames list.
334 						filenames.add(fallbackFilename);
335 					}
336 				}
337 			}
338 			filenames.add(basename);
339 			if (localeMap != null) {
340 				localeMap.put(locale, filenames);
341 			}
342 			else {
343 				localeMap = new HashMap<>();
344 				localeMap.put(locale, filenames);
345 				this.cachedFilenames.put(basename, localeMap);
346 			}
347 			return filenames;
348 		}
349 	}

351 	/**
352 	 * Calculate the filenames for the given bundle basename and Locale,
353 	 * appending language code, country code, and variant code.
354 	 * E.g.: basename "expressions", Locale "de_AT_oo" -> "expressions_de_AT_OO",
355 	 * "expressions_de_AT", "expressions_de".
356 	 * <p>Follows the rules defined by {@link java.util.Locale#toString()}.
357 	 * @param basename the basename of the bundle
358 	 * @param locale the locale
359 	 * @return the List of filenames to check
360 	 */
361 	private List<String> calculateFilenamesForLocale(String basename, Locale locale) {
362 		List<String> result = new ArrayList<>(3);
363 		String language = locale.getLanguage();
364 		String country = locale.getCountry();
365 		String variant = locale.getVariant();
366 		StringBuilder temp = new StringBuilder(basename);
367 
368 		temp.append('_');
369 		if (language.length() > 0) {
370 			temp.append(language);
371 			result.add(0, temp.toString());
372 		}
373 
374 		temp.append('_');
375 		if (country.length() > 0) {
376 			temp.append(country);
377 			result.add(0, temp.toString());
378 		}
379 
380 		if (variant.length() > 0 && (language.length() > 0 || country.length() > 0)) {
381 			temp.append('_').append(variant);
382 			result.add(0, temp.toString());
383 		}
384 
385 		return result;
386 	}

389 	/**
390 	 * Get a PropertiesHolder for the given filename, either from the
391 	 * cache or freshly loaded.
392 	 * @param filename the bundle filename (basename + Locale)
393 	 * @return the current PropertiesHolder for the bundle
394 	 */
395 	private PropertiesHolder getProperties(String filename) {
396 		synchronized (this.cachedProperties) {
397 			PropertiesHolder propHolder = this.cachedProperties.get(filename);
398 			if (propHolder != null &&
399 					(propHolder.getRefreshTimestamp() < 0 ||
400 							propHolder.getRefreshTimestamp() > System.currentTimeMillis() - this.cacheMillis)) {
401 				return propHolder;
402 			}
403 			return refreshProperties(filename, propHolder);
404 		}
405 	}
406 
407 	/**
408 	 * Refresh the PropertiesHolder for the given bundle filename.
409 	 * The holder can be <code>null</code> if not cached before, or a timed-out cache entry
410 	 * (potentially getting re-validated against the current last-modified timestamp).
411 	 * @param filename the bundle filename (basename + Locale)
412 	 * @param propHolderArg the current PropertiesHolder for the bundle
413 	 */
414 	private PropertiesHolder refreshProperties(String filename, @Nullable PropertiesHolder propHolderArg) {
415 		PropertiesHolder propHolder = propHolderArg;
416 		long refreshTimestamp = (this.cacheMillis < 0) ? -1 : System.currentTimeMillis();
417 
418 		Resource resource = getResource(filename);
419 
420 		if (resource.exists()) {
421 			long fileTimestamp = -1;
422 			if (this.cacheMillis >= 0) {
423 				// Last-modified timestamp of file will just be read if caching with timeout.
424 				try {
425 					fileTimestamp = resource.lastModified();
426 					if (propHolder != null && propHolder.getFileTimestamp() == fileTimestamp) {
427 						if (LOGGER.isDebugEnabled()) {
428 							LOGGER.debug("Re-caching properties for filename [" + filename
429 									+ "] - file hasn't been modified");
430 						}
431 						propHolder.setRefreshTimestamp(refreshTimestamp);
432 						return propHolder;
433 					}
434 				}
435 				catch (IOException ex) {
436 					// Probably a class path resource: cache it forever.
437 					if (LOGGER.isDebugEnabled()) {
438 						LOGGER.debug(resource
439 								+ " could not be resolved in the file system - assuming that is hasn't changed", ex);
440 					}
441 					fileTimestamp = -1;
442 				}
443 			}
444 			propHolder = load(filename, resource, fileTimestamp);
445 		}
446 
447 		else {
448 			// Resource does not exist.
449 			if (LOGGER.isDebugEnabled()) {
450 				LOGGER.debug("No properties file found for [" + filename + "] - neither plain properties nor XML");
451 			}
452 			// Empty holder representing "not found".
453 			propHolder = new PropertiesHolder();
454 		}
455 
456 		propHolder.setRefreshTimestamp(refreshTimestamp);
457 		this.cachedProperties.put(filename, propHolder);
458 		return propHolder;
459 	}

461 	private Resource getResource(String filename) {
462 		Resource resource = this.resourceLoader.getResource(filename + PROPERTIES_SUFFIX);
463 		if (!resource.exists()) {
464 			resource = this.resourceLoader.getResource(filename + XML_SUFFIX);
465 		}
466 		return resource;
467 	}
468 
469 	private PropertiesHolder load(String filename, Resource resource, long fileTimestamp) {
470 		PropertiesHolder propHolder;
471 		try {
472 			Properties props = loadProperties(resource, filename);
473 			propHolder = new PropertiesHolder(props, fileTimestamp);
474 		}
475 		catch (IOException ex) {
476 			if (LOGGER.isWarnEnabled()) {
477 				LOGGER.warn("Could not parse properties file [" + resource.getFilename() + "]", ex);
478 			}
479 			// Empty holder representing "not valid".
480 			propHolder = new PropertiesHolder();
481 		}
482 		return propHolder;
483 	}

485 	/**
486 	 * Load the properties from the given resource.
487 	 * @param resource the resource to load from
488 	 * @param filename the original bundle filename (basename + Locale)
489 	 * @return the populated Properties instance
490 	 * @throws IOException if properties loading failed
491 	 */
492 	private Properties loadProperties(Resource resource, String filename) throws IOException {
493 		try (InputStream is = resource.getInputStream()) {
494 			Properties props = new Properties();
495 			String resourceFilename = resource.getFilename();
496 			if (resourceFilename != null && resourceFilename.endsWith(XML_SUFFIX)) {
497 				if (LOGGER.isDebugEnabled()) {
498 					LOGGER.debug("Loading properties [" + resourceFilename + "]");
499 				}
500 				this.propertiesPersister.loadFromXml(props, is);
501 			}
502 			else {
503 				loadFromProperties(resource, filename, is, props, resourceFilename);
504 			}
505 			return props;
506 		}
507 	}
508 
509 	private void loadFromProperties(Resource resource, String filename, InputStream is, Properties props,
510 			@Nullable String resourceFilename) throws IOException {
511 
512 		String encoding = null;
513 		if (this.fileEncodings != null) {
514 			encoding = this.fileEncodings.getProperty(filename);
515 		}
516 		if (encoding == null) {
517 			encoding = this.defaultEncoding;
518 		}
519 		if (encoding != null) {
520 			if (LOGGER.isDebugEnabled()) {
521 				LOGGER.debug("Loading properties ["
522 						+ (resourceFilename == null ? resource : resourceFilename)
523 						+ "] with encoding '" + encoding + "'");
524 			}
525 			this.propertiesPersister.load(props, new InputStreamReader(is, encoding));
526 		}
527 		else {
528 			if (LOGGER.isDebugEnabled()) {
529 				LOGGER.debug("Loading properties [" + (resourceFilename == null ? resource : resourceFilename)
530 						+ "]");
531 			}
532 			this.propertiesPersister.load(props, is);
533 		}
534 	}

537 	/**
538 	 * Clear the resource bundle cache.
539 	 * Subsequent resolve calls will lead to reloading of the properties files.
540 	 */
541 	public void clearCache() {
542 		LOGGER.debug("Clearing entire resource bundle cache");
543 		synchronized (this.cachedProperties) {
544 			this.cachedProperties.clear();
545 		}
546 		synchronized (this.cachedMergedProperties) {
547 			this.cachedMergedProperties.clear();
548 		}
549 	}

557 	/**
558 	 * PropertiesHolder for caching.
559 	 * Stores the last-modified timestamp of the source file for efficient
560 	 * change detection, and the timestamp of the last refresh attempt
561 	 * (updated every time the cache entry gets re-validated).
562 	 */
563 	private static final class PropertiesHolder {
564 
565 		private Properties properties;
566 
567 		private long fileTimestamp = -1;
568 
569 		private long refreshTimestamp = -1;
570 
571 		PropertiesHolder() {
572 		}
573 
574 		PropertiesHolder(Properties properties, long fileTimestamp) {
575 			this.properties = properties;
576 			this.fileTimestamp = fileTimestamp;
577 		}
578 
579 		@Nullable
580 		public Properties getProperties() {
581 			return this.properties;
582 		}
583 
584 		public long getFileTimestamp() {
585 			return this.fileTimestamp;
586 		}
587 
588 		public void setRefreshTimestamp(long refreshTimestamp) {
589 			this.refreshTimestamp = refreshTimestamp;
590 		}
591 
592 		public long getRefreshTimestamp() {
593 			return this.refreshTimestamp;
594 		}
595 
596 		@Nullable
597 		public String getProperty(String code) {
598 			if (this.properties == null) {
599 				return null;
600 			}
601 			return this.properties.getProperty(code);
602 		}

604 	}
