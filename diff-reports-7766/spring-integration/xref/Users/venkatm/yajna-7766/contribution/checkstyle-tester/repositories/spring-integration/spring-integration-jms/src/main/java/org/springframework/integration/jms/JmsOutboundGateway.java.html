453 	private Destination determineRequestDestination(Message<?> message, Session session) throws JMSException {
454 		if (this.requestDestination != null) {
455 			return this.requestDestination;
456 		}
457 		if (this.requestDestinationName != null) {
458 			return resolveRequestDestination(this.requestDestinationName, session);
459 		}
460 		if (this.requestDestinationExpressionProcessor != null) {
461 			Object result = this.requestDestinationExpressionProcessor.processMessage(message);
462 			if (result instanceof Destination) {
463 				return (Destination) result;
464 			}
465 			if (result instanceof String) {
466 				return resolveRequestDestination((String) result, session);
467 			}
468 			throw new MessageDeliveryException(message,
469 					"Evaluation of requestDestinationExpression failed " +
470 							"to produce a Destination or destination name. Result was: " + result);
471 		}
472 		throw new MessageDeliveryException(message,
473 				"No requestDestination, requestDestinationName, or requestDestinationExpression has been configured.");
474 	}

482 	private Destination determineReplyDestination(Message<?> message, Session session) throws JMSException {
483 		if (this.replyDestination != null) {
484 			return this.replyDestination;
485 		}
486 		if (this.replyDestinationName != null) {
487 			return resolveReplyDestination(this.replyDestinationName, session);
488 		}
489 		if (this.replyDestinationExpressionProcessor != null) {
490 			Object result = this.replyDestinationExpressionProcessor.processMessage(message);
491 			if (result instanceof Destination) {
492 				return (Destination) result;
493 			}
494 			if (result instanceof String) {

498 					"Evaluation of replyDestinationExpression failed to produce a Destination or destination name. " +

510 	@Override
511 	protected void doInit() {
512 		synchronized (this.initializationMonitor) {
513 			if (this.initialized) {
514 				return;
515 			}
516 			Assert.notNull(this.connectionFactory, "connectionFactory must not be null");
517 			Assert.isTrue(this.requestDestination != null
518 							^ this.requestDestinationName != null
519 							^ this.requestDestinationExpressionProcessor != null,
520 					"Exactly one of 'requestDestination', 'requestDestinationName', " +
521 							"or 'requestDestinationExpression' is required.");
522 			ConversionService conversionService = getConversionService();
523 			if (this.requestDestinationExpressionProcessor != null) {
524 				this.requestDestinationExpressionProcessor.setBeanFactory(getBeanFactory());
525 				if (conversionService != null) {
526 					this.requestDestinationExpressionProcessor.setConversionService(conversionService);
527 				}
528 			}
529 			if (this.replyDestinationExpressionProcessor != null) {
530 				this.replyDestinationExpressionProcessor.setBeanFactory(getBeanFactory());
531 				if (conversionService != null) {
532 					this.replyDestinationExpressionProcessor.setConversionService(conversionService);
533 				}
534 			}
535 			/*
536 			 *  This is needed because there is no way to detect 2 or more gateways using the same reply queue
537 			 *  with no correlation key.
538 			 */
539 			boolean hasAReplyDest = this.replyDestination != null || this.replyDestinationName != null
540 					|| this.replyDestinationExpressionProcessor != null;
541 			if (this.useReplyContainer && (this.correlationKey == null && hasAReplyDest)) {
542 				logger.warn("The gateway cannot use a reply listener container with a specified " +
543 						"destination(Name/Expression) " +
544 						"without a 'correlation-key'; " +
545 						"a container will NOT be used; " +
546 						"to avoid this problem, set the 'correlation-key' attribute; " +
547 						"some consumers, including the Spring Integration <jms:inbound-gateway/>, " +
548 						"support the use of the value 'JMSCorrelationID' " +
549 						"for this purpose. Alternatively, do not specify a reply destination " +
550 						"and a temporary queue will be used for replies.");
551 				this.useReplyContainer = false;
552 			}
553 			if (this.useReplyContainer) {
554 				Assert.state(!"JMSCorrelationID*".equals(this.correlationKey),
555 						"Using an existing 'JMSCorrelationID' mapped from the 'requestMessage' ('JMSCorrelationID*') " +
556 								"can't be used when using a 'reply-container'");
557 				GatewayReplyListenerContainer container = new GatewayReplyListenerContainer();
558 				setContainerProperties(container);
559 				container.afterPropertiesSet();
560 				this.replyContainer = container;
561 				if (isAsync() && this.correlationKey == null) {
562 					logger.warn("'async=true' requires a correlationKey; ignored");
563 					setAsync(false);
564 				}
565 			}
566 			else {
567 				if (isAsync()) {
568 					logger.warn("'async=true' is ignored when a reply container is not being used");
569 					setAsync(false);
570 				}
571 			}
572 			this.initialized = true;
573 		}
574 	}

576 	private void setContainerProperties(GatewayReplyListenerContainer container) {
577 		container.setConnectionFactory(this.connectionFactory);
578 		if (this.replyDestination != null) {
579 			container.setDestination(this.replyDestination);
580 		}
581 		else if (StringUtils.hasText(this.replyDestinationName)) {
582 			container.setDestinationName(this.replyDestinationName);
583 		}
584 		else {
585 			// to be resolved to the TemporaryQueue
586 			container.setDestinationName("");
587 		}
588 		if (this.destinationResolver != null) {
589 			container.setDestinationResolver(this.destinationResolver);
590 		}
591 		container.setPubSubDomain(this.replyPubSubDomain);
592 		if (this.correlationKey != null) {
593 			String messageSelector = this.correlationKey + " LIKE '" + this.gatewayCorrelation + "%'";
594 			container.setMessageSelector(messageSelector);
595 		}
596 		container.setMessageListener(this);
597 		if (this.replyContainerProperties != null) {
598 			if (this.replyContainerProperties.isSessionTransacted() != null) {
599 				container.setSessionTransacted(this.replyContainerProperties.isSessionTransacted());
600 			}
601 			if (this.replyContainerProperties.getCacheLevel() != null) {
602 				container.setCacheLevel(this.replyContainerProperties.getCacheLevel());
603 			}
604 			if (this.replyContainerProperties.getConcurrentConsumers() != null) {
605 				container.setConcurrentConsumers(this.replyContainerProperties.getConcurrentConsumers());
606 			}
607 			if (this.replyContainerProperties.getIdleConsumerLimit() != null) {
608 				container.setIdleConsumerLimit(this.replyContainerProperties.getIdleConsumerLimit());
609 			}
610 			if (this.replyContainerProperties.getIdleTaskExecutionLimit() != null) {
611 				container.setIdleTaskExecutionLimit(this.replyContainerProperties.getIdleTaskExecutionLimit());
612 			}
613 			if (this.replyContainerProperties.getMaxConcurrentConsumers() != null) {

988 	private javax.jms.Message retryableReceiveReply(Session session, Destination replyTo, String messageSelector)
989 			throws JMSException {
990 
991 		Connection consumerConnection = null; //NOSONAR
992 		Session consumerSession = session;
993 		MessageConsumer messageConsumer = null;
994 		JMSException exception;
995 		boolean isTemporaryReplyTo = replyTo instanceof TemporaryQueue || replyTo instanceof TemporaryTopic;
996 		long replyTimeout = isTemporaryReplyTo
997 				? Long.MIN_VALUE
998 				: this.receiveTimeout < 0
999 						? Long.MAX_VALUE
1000 						: System.currentTimeMillis() + this.receiveTimeout;
1001 		try {
1002 			do {
1003 				try {
1004 					messageConsumer = consumerSession.createConsumer(replyTo, messageSelector);
1005 					javax.jms.Message reply = receiveReplyMessage(messageConsumer);

1012 				}
1013 				catch (JMSException e) { // NOSONAR - exception as flow control
1014 					exception = e;
1015 					if (logger.isDebugEnabled()) {

1017 					}

1030 								Thread.sleep(1000);
1031 							}

1035 							}
1036 						}
1037 					}

1043 				return null;
1044 			}

1302 	private void onMessageSync(javax.jms.Message message, String correlationId) {
1303 		try {
1304 			LinkedBlockingQueue<javax.jms.Message> queue = this.replies.get(correlationId);
1305 			if (queue == null) {
1306 				if (this.correlationKey != null) {
1307 					Log debugLogger = LogFactory.getLog("si.jmsgateway.debug");
1308 					if (debugLogger.isDebugEnabled()) {
1309 						Object siMessage = this.messageConverter.fromMessage(message);
1310 						debugLogger.debug("No pending reply for " + siMessage + " with correlationId: "
1311 								+ correlationId + " pending replies: " + this.replies.keySet());
1312 					}
1313 					throw new IllegalStateException("No sender waiting for reply");
1314 				}
1315 				synchronized (this.earlyOrLateReplies) {
1316 					queue = this.replies.get(correlationId);
1317 					if (queue == null) {
1318 						if (logger.isDebugEnabled()) {
1319 							logger.debug("Reply for correlationId " + correlationId + " received early or late");
1320 						}
1321 						this.earlyOrLateReplies.put(correlationId, new TimedReply(message));
1322 					}
1323 				}
1324 			}
1325 			if (queue != null) {
1326 				if (logger.isDebugEnabled()) {
1327 					logger.debug("Received reply with correlationId " + correlationId);
1328 				}
1329 				queue.add(message);
1330 			}
1331 		}
1332 		catch (Exception e) {
1333 			if (logger.isWarnEnabled()) {
1334 				logger.warn("Failed to consume reply with correlationId " + correlationId, e);
1335 			}
1336 		}

1458 			logger.trace("Running late reply reaper");

1460 					JmsOutboundGateway.this.earlyOrLateReplies.entrySet().iterator();
1461 			long now = System.currentTimeMillis();
1462 			long expired = now - (JmsOutboundGateway.this.receiveTimeout * 2);
1463 			while (lateReplyIterator.hasNext()) {

1465 				if (entry.getValue().getTimeStamp() < expired) {

1468 					}
1469 					lateReplyIterator.remove();
1470 				}
1471 			}
1472 			// reschedule myself
1473 			if (JmsOutboundGateway.this.receiveTimeout >= 0) {
1474 				JmsOutboundGateway.this.reaper = getTaskScheduler().schedule(this,
1475 						new Date(now + JmsOutboundGateway.this.receiveTimeout));
1476 			}
