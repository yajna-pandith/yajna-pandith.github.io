111 		final AtomicBoolean doFail = new AtomicBoolean();
112 		AbstractReplyProducingMessageHandler handler = new AbstractReplyProducingMessageHandler() {
113 
114 			@Override
115 			protected Object handleRequestMessage(Message<?> requestMessage) {
116 				if (doFail.get()) {
117 					throw new RuntimeException("qux");
118 				}
119 				return "baz";
120 			}
121 		};
122 		String componentName = "testComponentName";
123 		handler.setComponentName(componentName);
124 		QueueChannel replies = new QueueChannel();
125 		handler.setOutputChannel(replies);
126 		Message<String> message = new GenericMessage<>("Hello, world!");
127 
128 		// no advice
129 		handler.handleMessage(message);
130 		Message<?> reply = replies.receive(1000);
131 		assertThat(reply).isNotNull();
132 		assertThat(reply.getPayload()).isEqualTo("baz");
133 
134 		PollableChannel successChannel = new QueueChannel();
135 		PollableChannel failureChannel = new QueueChannel();
136 		ExpressionEvaluatingRequestHandlerAdvice advice = new ExpressionEvaluatingRequestHandlerAdvice();
137 		advice.setBeanFactory(mock(BeanFactory.class));
138 		advice.setSuccessChannel(successChannel);
139 		advice.setFailureChannel(failureChannel);
140 		advice.setOnSuccessExpressionString("'foo'");
141 		advice.setOnFailureExpressionString("'bar:' + #exception.message");
142 
143 		List<Advice> adviceChain = new ArrayList<Advice>();
144 		adviceChain.add(advice);
145 		final AtomicReference<String> compName = new AtomicReference<>();
146 		adviceChain.add(new AbstractRequestHandlerAdvice() {
147 
148 			@Override
149 			protected Object doInvoke(ExecutionCallback callback, Object target, Message<?> message) {
150 				compName.set(((AbstractReplyProducingMessageHandler.RequestHandler) target).getAdvisedHandler()
151 						.getComponentName());
152 				return callback.execute();
153 			}
154 
155 		});
156 		handler.setAdviceChain(adviceChain);
157 		handler.setBeanFactory(mock(BeanFactory.class));
158 		handler.afterPropertiesSet();
159 
160 		// advice with success
161 		handler.handleMessage(message);
162 		reply = replies.receive(1000);
163 		assertThat(reply).isNotNull();
164 		assertThat(reply.getPayload()).isEqualTo("baz");
165 
166 		assertThat(compName.get()).isEqualTo(componentName);
167 
168 		Message<?> success = successChannel.receive(1000);
169 		assertThat(success).isNotNull();
170 		assertThat(((AdviceMessage<?>) success).getInputMessage().getPayload()).isEqualTo("Hello, world!");
171 		assertThat(success.getPayload()).isEqualTo("foo");
172 
173 		// advice with failure, not trapped
174 		doFail.set(true);
175 		try {
176 			handler.handleMessage(message);
177 			fail("Expected exception");
178 		}
179 		catch (Exception e) {
180 			assertThat(e.getCause().getMessage()).isEqualTo("qux");
181 		}
182 
183 		Message<?> failure = failureChannel.receive(1000);
184 		assertThat(failure).isNotNull();
185 		assertThat(((MessagingException) failure.getPayload()).getFailedMessage().getPayload())
186 				.isEqualTo("Hello, world!");
187 		assertThat(((MessageHandlingExpressionEvaluatingAdviceException) failure.getPayload()).getEvaluationResult())
188 				.isEqualTo("bar:qux");
189 
190 		// advice with failure, trapped
191 		advice.setTrapException(true);
192 		handler.handleMessage(message);
193 		failure = failureChannel.receive(1000);
194 		assertThat(failure).isNotNull();
195 		assertThat(((MessagingException) failure.getPayload()).getFailedMessage().getPayload())
196 				.isEqualTo("Hello, world!");
197 		assertThat(((MessageHandlingExpressionEvaluatingAdviceException) failure.getPayload()).getEvaluationResult())
198 				.isEqualTo("bar:qux");
199 		assertThat(replies.receive(1)).isNull();
200 
201 		// advice with failure, eval is result
202 		advice.setReturnFailureExpressionResult(true);
203 		handler.handleMessage(message);
204 		failure = failureChannel.receive(1000);
205 		assertThat(failure).isNotNull();
206 		assertThat(((MessagingException) failure.getPayload()).getFailedMessage().getPayload())
207 				.isEqualTo("Hello, world!");
208 		assertThat(((MessageHandlingExpressionEvaluatingAdviceException) failure.getPayload()).getEvaluationResult())
209 				.isEqualTo("bar:qux");
210 
211 		reply = replies.receive(1000);
212 		assertThat(reply).isNotNull();
213 		assertThat(reply.getPayload()).isEqualTo("bar:qux");
214

394 			handler.handleMessage(message);
395 			fail("Expected failure");

413

418 			handler.handleMessage(message);
