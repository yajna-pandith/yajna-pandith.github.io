121 		final CountDownLatch latch = new CountDownLatch(1);
122 		final CountDownLatch done = new CountDownLatch(1);
123 		final AtomicReference<ServerSocket> serverSocket = new AtomicReference<>();
124 		this.executor.execute(() -> {
125 			try {
126 				ServerSocket server = ServerSocketFactory.getDefault().createServerSocket(0);
127 				logger.debug(testName.getMethodName() + " starting server for " + server.getLocalPort());
128 				serverSocket.set(server);
129 				latch.countDown();
130 				Socket s = server.accept();
131 				// block so we fill the buffer
132 				done.await(10, TimeUnit.SECONDS);
133 				s.close();
134 			}
135 			catch (Exception e) {
136 				e.printStackTrace();
137 			}
138 		});
139 		assertThat(latch.await(10000, TimeUnit.MILLISECONDS)).isTrue();
140 		TcpNioClientConnectionFactory factory = new TcpNioClientConnectionFactory("localhost",
141 				serverSocket.get().getLocalPort());
142 		factory.setApplicationEventPublisher(nullPublisher);
143 		factory.setSoTimeout(100);
144 		factory.start();
145 		try {
146 			TcpConnection connection = factory.getConnection();
147 			connection.send(MessageBuilder.withPayload(new byte[1000000]).build());
148 		}
149 		catch (MessagingException e) {
150 			assertThat(e.getCause() instanceof SocketTimeoutException)
151 					.as("Expected SocketTimeoutException, got " + e.getClass().getSimpleName() +
152 							":" + e.getMessage()).isTrue();
153 		}
154 		done.countDown();
155 		factory.stop();
156 		serverSocket.get().close();
157 	}

257 		factory.setApplicationEventPublisher(nullPublisher);

259 		Map<SocketChannel, TcpNioConnection> connections = new HashMap<>();
260 		SocketChannel chan1 = mock(SocketChannel.class);

265 		TcpNioConnection conn3 = mock(TcpNioConnection.class);

267 		connections.put(chan2, conn2);
268 		connections.put(chan3, conn3);
269 		final List<Field> fields = new ArrayList<>();
270 		ReflectionUtils.doWithFields(SocketChannel.class, field -> {
271 			field.setAccessible(true);
272 			fields.add(field);
273 		}, field -> field.getName().equals("open"));
274 		Field field = fields.get(0);
275 		// Can't use Mockito because isOpen() is final
276 		ReflectionUtils.setField(field, chan1, true);
277 		ReflectionUtils.setField(field, chan2, true);
278 		ReflectionUtils.setField(field, chan3, true);
279 		Selector selector = mock(Selector.class);
280 		HashSet<SelectionKey> keys = new HashSet<>();
281 		when(selector.selectedKeys()).thenReturn(keys);

284 
285 		ReflectionUtils.setField(field, chan1, false);
286 		factory.processNioSelections(1, selector, null, connections);
287 		assertThat(connections.size()).isEqualTo(3); // interval didn't pass
288 		Thread.sleep(110);
289 		factory.processNioSelections(1, selector, null, connections);
290 		assertThat(connections.size()).isEqualTo(2); // first is closed
291 
292 		ReflectionUtils.setField(field, chan2, false);
293 		factory.processNioSelections(1, selector, null, connections);
294 		assertThat(connections.size()).isEqualTo(2); // interval didn't pass
295 		Thread.sleep(110);
296 		factory.processNioSelections(1, selector, null, connections);
297 		assertThat(connections.size()).isEqualTo(1); // second is closed
298 
299 		ReflectionUtils.setField(field, chan3, false);
300 		factory.processNioSelections(1, selector, null, connections);
301 		assertThat(connections.size()).isEqualTo(1); // interval didn't pass
302 		Thread.sleep(110);
303 		factory.processNioSelections(1, selector, null, connections);
304 		assertThat(connections.size()).isEqualTo(0); // third is closed
305 
306 		assertThat(TestUtils.getPropertyValue(factory, "connections", Map.class).size()).isEqualTo(0);

311 		final ExecutorService exec = Executors.newFixedThreadPool(2);
312 		Future<Object> future = exec.submit(() -> {
313 			SocketChannel channel = mock(SocketChannel.class);
314 			Socket socket = mock(Socket.class);
315 			Mockito.when(channel.socket()).thenReturn(socket);
316 			doAnswer(invocation -> {
317 				ByteBuffer buffer = invocation.getArgument(0);
318 				buffer.position(1);
319 				return 1;
320 			}).when(channel).read(Mockito.any(ByteBuffer.class));
321 			when(socket.getReceiveBufferSize()).thenReturn(1024);
322 			final TcpNioConnection connection = new TcpNioConnection(channel, false, false, nullPublisher, null);
323 			connection.setTaskExecutor(exec);
324 			connection.setPipeTimeout(200);
325 			Method method = TcpNioConnection.class.getDeclaredMethod("doRead");
326 			method.setAccessible(true);
327 			// Nobody reading, should timeout on 6th write.
328 			try {
329 				for (int i = 0; i < 6; i++) {
330 					method.invoke(connection);
331 				}
332 			}
333 			catch (Exception e) {
334 				logger.debug("Expected timeout", e);
335 				throw (Exception) e.getCause();
336 			}
337 			finally {
338 				connection.setPipeTimeout(15);
339 				connection.close();
340 			}
341 			return null;
342 		});
343 		try {
344 			Object o = future.get(10, TimeUnit.SECONDS);
345 			fail("Expected exception, got " + o);
346 		}
347 		catch (ExecutionException e) {
348 			assertThat(e.getCause().getMessage()).isEqualTo("Timed out waiting for buffer space");
349 		}
350 		finally {
351 			exec.shutdownNow();
352 		}
353 	}

856 	private void testMulti(boolean multiAccept) throws InterruptedException, IOException {
857 		CountDownLatch serverReadyLatch = new CountDownLatch(1);
858 		CountDownLatch latch = new CountDownLatch(21);
859 		List<Socket> sockets = new ArrayList<>();
860 		TcpNioServerConnectionFactory server = new TcpNioServerConnectionFactory(0);
861 		try {
862 			List<IpIntegrationEvent> events = Collections.synchronizedList(new ArrayList<>());
863 			List<Message<?>> messages = Collections.synchronizedList(new ArrayList<>());
864 			server.setMultiAccept(multiAccept);
865 			server.setApplicationEventPublisher(e -> {
866 				if (e instanceof TcpConnectionServerListeningEvent) {
867 					serverReadyLatch.countDown();
868 				}
869 				events.add((IpIntegrationEvent) e);
870 				latch.countDown();
871 			});
872 			server.registerListener(m -> {
873 				messages.add(m);
874 				latch.countDown();
875 				return false;
876 			});
877 			server.afterPropertiesSet();
878 			server.start();
879 			assertThat(serverReadyLatch.await(10, TimeUnit.SECONDS)).isTrue();
880 			for (int i = 0; i < 10; i++) {
881 				Socket socket = SocketFactory.getDefault().createSocket("localhost", server.getPort());
882 				socket.getOutputStream().write("foo\r\n".getBytes());
883 				sockets.add(socket);
884 			}
885 			assertThat(latch.await(10, TimeUnit.SECONDS));
886 			assertThat(events).hasSize(11); // server ready + 10 opens
887 			assertThat(messages).hasSize(10);
888 		}
889 		finally {
890 			for (Socket socket : sockets) {
891 				socket.close();
892 			}
893 			server.stop();
894 		}
895 	}
