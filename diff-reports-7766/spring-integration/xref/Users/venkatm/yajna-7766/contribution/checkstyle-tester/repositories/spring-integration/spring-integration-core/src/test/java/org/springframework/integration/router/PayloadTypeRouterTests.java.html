42  	@Test
43  	public void resolveExactMatch() {
44  		QueueChannel stringChannel = new QueueChannel();
45  		QueueChannel integerChannel = new QueueChannel();
46  		DefaultListableBeanFactory beanFactory = new DefaultListableBeanFactory();
47  		beanFactory.registerSingleton("stringChannel", stringChannel);
48  		beanFactory.registerSingleton("integerChannel", integerChannel);
49  
50  		Map<String, String> payloadTypeChannelMap = new ConcurrentHashMap<>();
51  		payloadTypeChannelMap.put(String.class.getName(), "stringChannel");
52  		payloadTypeChannelMap.put(Integer.class.getName(), "integerChannel");
53  		PayloadTypeRouter router = new PayloadTypeRouter();
54  		router.setChannelMappings(payloadTypeChannelMap);
55  		router.setBeanFactory(beanFactory);
56  
57  		Message<String> message1 = new GenericMessage<>("test");
58  		Message<Integer> message2 = new GenericMessage<>(123);
59  		assertThat(router.getChannelKeys(message1).size()).isEqualTo(1);
60  
61  		assertThat(stringChannel.receive(0)).isNull();
62  		router.handleMessage(message1);
63  		assertThat(stringChannel.receive(0)).isEqualTo(message1);
64  
65  		assertThat(router.getChannelKeys(message2).size()).isEqualTo(1);
66  
67  		assertThat(integerChannel.receive(0)).isNull();
68  		router.handleMessage(message2);
69  		assertThat(integerChannel.receive(0)).isEqualTo(message2);
70  
71  		// validate dynamics
72  		QueueChannel newChannel = new QueueChannel();
73  		beanFactory.registerSingleton("newChannel", newChannel);
74  		router.setChannelMapping(String.class.getName(), "newChannel");
75  		assertThat(router.getChannelKeys(message1).size()).isEqualTo(1);
76  
77  		assertThat(newChannel.receive(0)).isNull();
78  		router.handleMessage(message1);
79  		assertThat(newChannel.receive(0)).isEqualTo(message1);
80  
81  		// validate exception is thrown if mappings were removed and
82  		// channelResolutionRequires = true (which is the default)
83  
84  		router.removeChannelMapping(String.class.getName());
85  		router.removeChannelMapping(Integer.class.getName());
86  
87  		router.setResolutionRequired(true);
88  
89  		try {
90  			router.handleMessage(message1);
91  			fail("Exception expected");
92  		}
93  		catch (Exception e) {
94  			// ignore
95  		}
96  	}
