478     private void verifyRuleSet(RuleSet ruleset, int size, Set<RuleViolation> values) {
479 
480         RuleContext context = new RuleContext();
481         Set<RuleViolation> reportedValues = new HashSet<>();
482         context.setReport(new Report());
483         ruleset.apply(makeCompilationUnits(), context);
484 
485         assertEquals("Invalid number of Violations Reported", size, context.getReport().size());
486 
487         Iterator<RuleViolation> violations = context.getReport().iterator();
488         while (violations.hasNext()) {
489             RuleViolation violation = violations.next();
490 
491             reportedValues.add(violation);
492             assertTrue("Unexpected Violation Returned: " + violation, values.contains(violation));
493         }
494 
495         Iterator<RuleViolation> expected = values.iterator();
496         while (expected.hasNext()) {
497             RuleViolation violation = expected.next();
498             assertTrue("Expected Violation not Returned: " + violation, reportedValues.contains(violation));
499         }
500     }
