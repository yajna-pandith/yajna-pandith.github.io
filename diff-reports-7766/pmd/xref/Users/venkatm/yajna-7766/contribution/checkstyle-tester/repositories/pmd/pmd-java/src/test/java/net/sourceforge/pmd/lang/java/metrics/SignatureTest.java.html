38      // common to operation and field signatures
39      @Test
40      public void visibilityTest() {
41          final String TEST = "class Bzaz{ "
42              + "public int bar;"
43              + "String k;"
44              + "protected double d;"
45              + "private int i;"
46              + "protected int x;"
47              + "public Bzaz(){} "
48              + "void bar(){} "
49              + "protected void foo(int x){}"
50              + "private Bzaz(int y){}"
51              + "}";
52  
53  
54          List<ASTMethodOrConstructorDeclaration> operationDeclarations = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST);
55          List<ASTFieldDeclaration> fieldDeclarations = getOrderedNodes(ASTFieldDeclaration.class, TEST);
56          List<JavaSignature> sigs = new ArrayList<>();
57  
58          for (ASTMethodOrConstructorDeclaration node : operationDeclarations) {
59              sigs.add(JavaOperationSignature.buildFor(node));
60          }
61  
62          // operations
63          assertEquals(Visibility.PUBLIC, sigs.get(0).visibility);
64          assertEquals(Visibility.PACKAGE, sigs.get(1).visibility);
65          assertEquals(Visibility.PROTECTED, sigs.get(2).visibility);
66          assertEquals(Visibility.PRIVATE, sigs.get(3).visibility);
67  
68          sigs.clear();
69          for (ASTFieldDeclaration node : fieldDeclarations) {
70              sigs.add(JavaFieldSignature.buildFor(node));
71          }
72  
73          // fields
74          assertEquals(Visibility.PUBLIC, sigs.get(0).visibility);
75          assertEquals(Visibility.PACKAGE, sigs.get(1).visibility);
76          assertEquals(Visibility.PROTECTED, sigs.get(2).visibility);
77          assertEquals(Visibility.PRIVATE, sigs.get(3).visibility);
78      }

80      @Test
81      public void operationRoleTest() {
82          final String TEST = "class Bzaz{ int x; "
83              + "public static void foo(){} "
84              + "Bzaz(){} "
85              + "int getX(){return x;}"
86              + " void setX(int a){x=a;}"
87              + " public void doSomething(){}}";
88  
89  
90          List<ASTMethodOrConstructorDeclaration> nodes = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST);
91          List<JavaOperationSignature> sigs = new ArrayList<>();
92  
93          for (ASTMethodOrConstructorDeclaration node : nodes) {
94              sigs.add(JavaOperationSignature.buildFor(node));
95          }
96  
97          assertEquals(Role.STATIC, sigs.get(0).role);
98          assertEquals(Role.CONSTRUCTOR, sigs.get(1).role);
99          assertEquals(Role.GETTER_OR_SETTER, sigs.get(2).role);
100         assertEquals(Role.GETTER_OR_SETTER, sigs.get(3).role);
101         assertEquals(Role.METHOD, sigs.get(4).role);
102     }

204     // Ensure only one instance of a signature is created.
205     @Test
206     public void operationPoolTest() {
207         final String TEST = "class Bzaz{ "
208             + "public static void foo(){} "
209             + "public static void az(){} "
210             + "public static int getX(){return x;}}";
211 
212         final String TEST2 = "class Bzaz{ "
213             + "void foo(){} "
214             + "void az(){} "
215             + "int rand(){return x;}}";
216 
217 
218         List<ASTMethodOrConstructorDeclaration> nodes = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST);
219         List<ASTMethodOrConstructorDeclaration> nodes2 = getOrderedNodes(ASTMethodOrConstructorDeclaration.class, TEST2);
220 
221         List<JavaOperationSignature> sigs = new ArrayList<>();
222         List<JavaOperationSignature> sigs2 = new ArrayList<>();
223 
224         for (int i = 0; i < sigs.size(); i++) {
225             sigs.add(JavaOperationSignature.buildFor(nodes.get(i)));
226             sigs2.add(JavaOperationSignature.buildFor(nodes2.get(i)));
227 
228         }
229 
230         for (int i = 0; i < sigs.size() - 1; i++) {
231             assertTrue(sigs.get(i) == sigs.get(i + 1));
232             assertTrue(sigs2.get(i) == sigs2.get(i + 1));
233         }
234     }
