5   package net.sourceforge.pmd.lang.xml.ast;
6   
7   import java.io.IOException;
8   import java.io.Reader;
9   import java.io.StringReader;
10  import java.util.HashMap;
11  import java.util.Map;
12  import javax.xml.parsers.DocumentBuilder;
13  import javax.xml.parsers.DocumentBuilderFactory;
14  import javax.xml.parsers.ParserConfigurationException;
15  
16  import org.apache.commons.io.IOUtils;
17  import org.w3c.dom.Document;
18  import org.w3c.dom.Node;
19  import org.xml.sax.InputSource;
20  import org.xml.sax.SAXException;
21  
22  import net.sourceforge.pmd.lang.ast.ParseException;
23  import net.sourceforge.pmd.lang.ast.RootNode;
24  import net.sourceforge.pmd.lang.xml.XmlParserOptions;
25  
26  
27  public class XmlParser {
28      protected final XmlParserOptions parserOptions;
29      protected Map<org.w3c.dom.Node, XmlNode> nodeCache = new HashMap<>();
30  
31  
32      public XmlParser(XmlParserOptions parserOptions) {
33          this.parserOptions = parserOptions;
34      }
35  
36  
37      protected Document parseDocument(Reader reader) throws ParseException {
38          nodeCache.clear();
39          try {
40              String xmlData = IOUtils.toString(reader);
41  
42              DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
43              dbf.setNamespaceAware(parserOptions.isNamespaceAware());
44              dbf.setValidating(parserOptions.isValidating());
45              dbf.setIgnoringComments(parserOptions.isIgnoringComments());
46              dbf.setIgnoringElementContentWhitespace(parserOptions.isIgnoringElementContentWhitespace());
47              dbf.setExpandEntityReferences(parserOptions.isExpandEntityReferences());
48              dbf.setCoalescing(parserOptions.isCoalescing());
49              dbf.setXIncludeAware(parserOptions.isXincludeAware());
50              dbf.setFeature("http://xml.org/sax/features/external-general-entities", false);
51              dbf.setFeature("http://xml.org/sax/features/external-parameter-entities", false);
52              DocumentBuilder documentBuilder = dbf.newDocumentBuilder();
53              documentBuilder.setEntityResolver(parserOptions.getEntityResolver());
54              Document document = documentBuilder.parse(new InputSource(new StringReader(xmlData)));
55              DOMLineNumbers lineNumbers = new DOMLineNumbers(document, xmlData);
56              lineNumbers.determine();
57              return document;
58          } catch (ParserConfigurationException | SAXException | IOException e) {
59              throw new ParseException(e);
60          }
61      }
62  
63  
64      public XmlNode parse(Reader reader) {
65          Document document = parseDocument(reader);
66          XmlNode root = new RootXmlNode(this, document);
67          nodeCache.put(document, root);
68          return root;
69      }
70  
71  
72      /**
73       * Gets the wrapper for a DOM node, implementing PMD interfaces.
74       *
75       * @param domNode The node to wrap
76       *
77       * @return The wrapper
78       */
79      XmlNode wrapDomNode(Node domNode) {
80          XmlNode wrapper = nodeCache.get(domNode);
81          if (wrapper == null) {
82              wrapper = new XmlNodeWrapper(this, domNode);
83              nodeCache.put(domNode, wrapper);
84          }
85          return wrapper;
86      }
87  
88  
89      /**
90       * The root should implement {@link RootNode}.
91       */
92      public static class RootXmlNode extends XmlNodeWrapper implements RootNode {
93          RootXmlNode(XmlParser parser, Node domNode) {
94              super(parser, domNode);
95          }
96      }
97  
98  }
