454     @Test
455     public void testFilterByEntry() {
456         Map<Integer, Integer> map = new HashMap<>();
457         for (int index = 1; index <= Byte.SIZE; index++) {
458             map.put(index, index & 0x01);
459         }
460 
461         Map<Integer, Integer> expected = new HashMap<>();
462         Predicate<Map.Entry<Integer, Integer>> selector = e -> (e.getKey() > 3) && (e.getKey() <= 7) && ((e.getValue() & 0x01) == 0);
463         map.forEach((key, value) -> {
464             if (selector.test(Pair.of(key, value))) {
465                 expected.put(key, value);
466             }
467         });
468 
469         Map<Integer, Integer> actual = MapUtils.filterByEntry(map, selector, TreeMap::new);
470         assertMapEquals(getCurrentTestName(), expected, actual);
471     }
