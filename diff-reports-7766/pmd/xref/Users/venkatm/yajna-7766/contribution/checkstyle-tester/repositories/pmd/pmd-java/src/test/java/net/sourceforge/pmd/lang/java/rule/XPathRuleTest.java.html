122     /**
123      * Test for problem reported in bug #1219 PrimarySuffix/@Image does not work
124      * in some cases in xpath 2.0
125      * 
126      * @throws Exception
127      *             any error
128      */
129     @Test
130     public void testImageOfPrimarySuffix() throws Exception {
131         final String SUFFIX = "import java.io.File;\n" + "\n" + "public class TestSuffix {\n"
132                 + "    public static void main(String args[]) {\n" + "        new File(\"subdirectory\").list();\n"
133                 + "    }\n" + "}";
134         LanguageVersion language = LanguageRegistry.getLanguage(JavaLanguageModule.NAME).getDefaultVersion();
135         ParserOptions parserOptions = language.getLanguageVersionHandler().getDefaultParserOptions();
136         Parser parser = language.getLanguageVersionHandler().getParser(parserOptions);
137         ASTCompilationUnit cu = (ASTCompilationUnit) parser.parse("test", new StringReader(SUFFIX));
138         RuleContext ruleContext = new RuleContext();
139         ruleContext.setLanguageVersion(language);
140 
141         String xpath = "//PrimarySuffix[@Image='list']";
142 
143         // XPATH version 1.0
144         XPathRuleQuery xpathRuleQuery = new JaxenXPathRuleQuery();
145         xpathRuleQuery.setXPath(xpath);
146         xpathRuleQuery.setProperties(new HashMap<PropertyDescriptor<?>, Object>());
147         xpathRuleQuery.setVersion(XPathRuleQuery.XPATH_1_0);
148         List<Node> nodes = xpathRuleQuery.evaluate(cu, ruleContext);
149         assertEquals(1, nodes.size());
150 
151         // XPATH version 1.0 Compatibility
152         xpathRuleQuery = new SaxonXPathRuleQuery();
153         xpathRuleQuery.setXPath(xpath);
154         xpathRuleQuery.setProperties(new HashMap<PropertyDescriptor<?>, Object>());
155         xpathRuleQuery.setVersion(XPathRuleQuery.XPATH_1_0_COMPATIBILITY);
156         nodes = xpathRuleQuery.evaluate(cu, ruleContext);
157         assertEquals(1, nodes.size());
158 
159         // XPATH version 2.0
160         xpathRuleQuery = new SaxonXPathRuleQuery();
161         xpathRuleQuery.setXPath(xpath);
162         xpathRuleQuery.setProperties(new HashMap<PropertyDescriptor<?>, Object>());
163         xpathRuleQuery.setVersion(XPathRuleQuery.XPATH_2_0);
164         nodes = xpathRuleQuery.evaluate(cu, ruleContext);
165         assertEquals(1, nodes.size());
166     }

168     /**
169      * Following sibling check: See https://sourceforge.net/p/pmd/bugs/1209/
170      * 
171      * @throws Exception
172      *             any error
173      */
174     @Test
175     public void testFollowingSibling() throws Exception {
176         final String SOURCE = "public class dummy {\n" + "  public String toString() {\n"
177                 + "    String test = \"bad example\";\n" + "    test = \"a\";\n" + "    return test;\n" + "  }\n" + "}";
178         LanguageVersion language = LanguageRegistry.getLanguage(JavaLanguageModule.NAME).getDefaultVersion();
179         ParserOptions parserOptions = language.getLanguageVersionHandler().getDefaultParserOptions();
180         Parser parser = language.getLanguageVersionHandler().getParser(parserOptions);
181         ASTCompilationUnit cu = (ASTCompilationUnit) parser.parse("test", new StringReader(SOURCE));
182         RuleContext ruleContext = new RuleContext();
183         ruleContext.setLanguageVersion(language);
184 
185         String xpath = "//Block/BlockStatement/following-sibling::BlockStatement";
186 
187         // XPATH version 1.0
188         XPathRuleQuery xpathRuleQuery = new JaxenXPathRuleQuery();
189         xpathRuleQuery.setXPath(xpath);
190         xpathRuleQuery.setProperties(new HashMap<PropertyDescriptor<?>, Object>());
191         xpathRuleQuery.setVersion(XPathRuleQuery.XPATH_1_0);
192         List<Node> nodes = xpathRuleQuery.evaluate(cu, ruleContext);
193         assertEquals(2, nodes.size());
194         assertEquals(4, nodes.get(0).getBeginLine());
195         assertEquals(5, nodes.get(1).getBeginLine());
196 
197         // XPATH version 2.0
198         xpathRuleQuery = new SaxonXPathRuleQuery();
199         xpathRuleQuery.setXPath(xpath);
200         xpathRuleQuery.setProperties(new HashMap<PropertyDescriptor<?>, Object>());
201         xpathRuleQuery.setVersion(XPathRuleQuery.XPATH_2_0);
202         nodes = xpathRuleQuery.evaluate(cu, ruleContext);
203         assertEquals(2, nodes.size());
204         assertEquals(4, nodes.get(0).getBeginLine());
205         assertEquals(5, nodes.get(1).getBeginLine());
206     }

208     private static Report getReportForTestString(Rule r, String test) throws PMDException {
209         PMD p = new PMD();
210         RuleContext ctx = new RuleContext();
211         Report report = new Report();
212         ctx.setReport(report);
213         ctx.setSourceCodeFile(new File("n/a"));
214         RuleSet rules = RulesetsFactoryUtils.defaultFactory().createSingleRuleRuleSet(r);
215         p.getSourceCodeProcessor().processSourceCode(new StringReader(test), new RuleSets(rules), ctx);
216         return report;
217     }
