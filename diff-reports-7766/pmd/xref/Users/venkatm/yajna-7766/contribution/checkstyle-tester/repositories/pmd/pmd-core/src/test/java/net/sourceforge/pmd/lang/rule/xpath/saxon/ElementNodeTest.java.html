5   package net.sourceforge.pmd.lang.rule.xpath.saxon;
6   
7   import org.junit.Assert;
8   import org.junit.Test;
9   
10  import net.sourceforge.pmd.lang.ast.DummyNode;
11  import net.sourceforge.pmd.lang.ast.xpath.saxon.DocumentNode;
12  import net.sourceforge.pmd.lang.ast.xpath.saxon.ElementNode;
13  
14  public class ElementNodeTest {
15  
16      @Test
17      public void testCompareOrder() {
18          DummyNode node = new DummyNode(1, false, "dummy");
19          DummyNode foo1 = new DummyNode(2, false, "foo");
20          foo1.testingOnlySetBeginLine(1);
21          DummyNode foo2 = new DummyNode(2, false, "foo");
22          foo2.testingOnlySetBeginLine(2);
23          node.jjtAddChild(foo1, 0);
24          node.jjtAddChild(foo2, 1);
25  
26          DocumentNode document = new DocumentNode(node);
27          ElementNode elementFoo1 = document.nodeToElementNode.get(foo1);
28          ElementNode elementFoo2 = document.nodeToElementNode.get(foo2);
29  
30          Assert.assertFalse(elementFoo1.isSameNodeInfo(elementFoo2));
31          Assert.assertFalse(elementFoo2.isSameNodeInfo(elementFoo1));
32          Assert.assertTrue(elementFoo1.compareOrder(elementFoo2) < 0);
33          Assert.assertTrue(elementFoo2.compareOrder(elementFoo1) > 0);
34  
35          Assert.assertEquals(0, elementFoo1.compareOrder(elementFoo1));
36      }
37  
38      @Test
39      public void testCompareOrderSamePosition() {
40          DummyNode node = new DummyNode(1, false, "dummy");
41          DummyNode foo1 = new DummyNode(2, false, "foo");
42          foo1.testingOnlySetBeginLine(1);
43          foo1.testingOnlySetBeginColumn(1);
44          DummyNode foo2 = new DummyNode(2, false, "foo");
45          foo2.testingOnlySetBeginLine(1);
46          foo2.testingOnlySetBeginColumn(1);
47          node.jjtAddChild(foo1, 0);
48          node.jjtAddChild(foo2, 1);
49  
50          DocumentNode document = new DocumentNode(node);
51          ElementNode elementFoo1 = document.nodeToElementNode.get(foo1);
52          ElementNode elementFoo2 = document.nodeToElementNode.get(foo2);
53  
54          Assert.assertFalse(elementFoo1.isSameNodeInfo(elementFoo2));
55          Assert.assertFalse(elementFoo2.isSameNodeInfo(elementFoo1));
56          Assert.assertTrue(elementFoo1.compareOrder(elementFoo2) < 0);
57          Assert.assertTrue(elementFoo2.compareOrder(elementFoo1) > 0);
58      }
59  }
