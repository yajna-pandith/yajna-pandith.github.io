251   /**
252    * Magic we put ahead of a serialized protobuf message.
253    * For example, all znode content is protobuf messages with the below magic
254    * for preamble.
255    */
256   public static final byte [] PB_MAGIC = new byte [] {'P', 'B', 'U', 'F'};
257   private static final String PB_MAGIC_STR = Bytes.toString(PB_MAGIC);
258 
259   /**
260    * Prepend the passed bytes with four bytes of magic, {@link #PB_MAGIC}, to flag what
261    * follows as a protobuf in hbase.  Prepend these bytes to all content written to znodes, etc.
262    * @param bytes Bytes to decorate
263    * @return The passed <code>bytes</codes> with magic prepended (Creates a new
264    * byte array that is <code>bytes.length</code> plus {@link #PB_MAGIC}.length.
265    */
266   public static byte [] prependPBMagic(final byte [] bytes) {
267     return Bytes.add(PB_MAGIC, bytes);
268   }
269 
270   /**
271    * @param bytes Bytes to check.
272    * @return True if passed <code>bytes</code> has {@link #PB_MAGIC} for a prefix.
273    */
274   public static boolean isPBMagicPrefix(final byte [] bytes) {
275     if (bytes == null) return false;
276     return isPBMagicPrefix(bytes, 0, bytes.length);
277   }
278 
279   /**
280    * @param bytes Bytes to check.
281    * @return True if passed <code>bytes</code> has {@link #PB_MAGIC} for a prefix.
282    */
283   public static boolean isPBMagicPrefix(final byte [] bytes, int offset, int len) {
284     if (bytes == null || len < PB_MAGIC.length) return false;
285     return Bytes.compareTo(PB_MAGIC, 0, PB_MAGIC.length, bytes, offset, PB_MAGIC.length) == 0;
286   }

305   /**
306    * Return the IOException thrown by the remote server wrapped in
307    * ServiceException as cause.
308    *
309    * @param se ServiceException that wraps IO exception thrown by the server
310    * @return Exception wrapped in ServiceException or
311    *   a new IOException that wraps the unexpected ServiceException.
312    */
313   public static IOException getRemoteException(ServiceException se) {
314     Throwable e = se.getCause();
315     if (e == null) {
316       return new IOException(se);
317     }
318     if (ExceptionUtil.isInterrupt(e)) {
319       return ExceptionUtil.asInterrupt(e);
320     }
321     if (e instanceof RemoteException) {
322       e = ((RemoteException) e).unwrapRemoteException();
323     }
324     return e instanceof IOException ? (IOException) e : new IOException(se);
325   }

441   /**
442    * Convert a protocol buffer Get to a client Get
443    *
444    * @param proto the protocol buffer Get to convert
445    * @return the converted client Get
446    * @throws IOException
447    */
448   public static Get toGet(
449       final ClientProtos.Get proto) throws IOException {
450     if (proto == null) return null;
451     byte[] row = proto.getRow().toByteArray();
452     Get get = new Get(row);
453     if (proto.hasCacheBlocks()) {
454       get.setCacheBlocks(proto.getCacheBlocks());
455     }
456     if (proto.hasMaxVersions()) {
457       get.setMaxVersions(proto.getMaxVersions());
458     }
459     if (proto.hasStoreLimit()) {
460       get.setMaxResultsPerColumnFamily(proto.getStoreLimit());
461     }
462     if (proto.hasStoreOffset()) {
463       get.setRowOffsetPerColumnFamily(proto.getStoreOffset());
464     }
465     if (proto.hasTimeRange()) {
466       HBaseProtos.TimeRange timeRange = proto.getTimeRange();
467       long minStamp = 0;
468       long maxStamp = Long.MAX_VALUE;
469       if (timeRange.hasFrom()) {
470         minStamp = timeRange.getFrom();
471       }
472       if (timeRange.hasTo()) {
473         maxStamp = timeRange.getTo();
474       }
475       get.setTimeRange(minStamp, maxStamp);
476     }
477     if (proto.hasFilter()) {
478       FilterProtos.Filter filter = proto.getFilter();
479       get.setFilter(ProtobufUtil.toFilter(filter));
480     }
481     for (NameBytesPair attribute: proto.getAttributeList()) {
482       get.setAttribute(attribute.getName(), attribute.getValue().toByteArray());
483     }
484     if (proto.getColumnCount() > 0) {
485       for (Column column: proto.getColumnList()) {
486         byte[] family = column.getFamily().toByteArray();
487         if (column.getQualifierCount() > 0) {
488           for (ByteString qualifier: column.getQualifierList()) {
489             get.addColumn(family, qualifier.toByteArray());
490           }
491         } else {
492           get.addFamily(family);
493         }
494       }
495     }
496     if (proto.hasExistenceOnly() && proto.getExistenceOnly()){
497       get.setCheckExistenceOnly(true);
498     }
499     if (proto.hasClosestRowBefore() && proto.getClosestRowBefore()){
500       get.setClosestRowBefore(true);
501     }
502     if (proto.hasConsistency()) {
503       get.setConsistency(toConsistency(proto.getConsistency()));
504     }
505     return get;
506   }

536   /**
537    * Convert a protocol buffer Mutate to a Put.
538    *
539    * @param proto The protocol buffer MutationProto to convert
540    * @param cellScanner If non-null, the Cell data that goes with this proto.
541    * @return A client Put.
542    * @throws IOException
543    */
544   public static Put toPut(final MutationProto proto, final CellScanner cellScanner)
545   throws IOException {
546     // TODO: Server-side at least why do we convert back to the Client types?  Why not just pb it?
547     MutationType type = proto.getMutateType();
548     assert type == MutationType.PUT: type.name();
549     long timestamp = proto.hasTimestamp()? proto.getTimestamp(): HConstants.LATEST_TIMESTAMP;
550     Put put = null;
551     int cellCount = proto.hasAssociatedCellCount()? proto.getAssociatedCellCount(): 0;
552     if (cellCount > 0) {
553       // The proto has metadata only and the data is separate to be found in the cellScanner.
554       if (cellScanner == null) {
555         throw new DoNotRetryIOException("Cell count of " + cellCount + " but no cellScanner: " +
556             toShortString(proto));
557       }
558       for (int i = 0; i < cellCount; i++) {
559         if (!cellScanner.advance()) {

564         if (put == null) {
565           put = new Put(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(), timestamp);
566         }
567         put.add(cell);
568       }
569     } else {
570       if (proto.hasRow()) {
571         put = new Put(proto.getRow().asReadOnlyByteBuffer(), timestamp);
572       } else {

574       }
575       // The proto has the metadata and the data itself
576       for (ColumnValue column: proto.getColumnValueList()) {

580             throw new DoNotRetryIOException(
581                 "Missing required field: qualifier value");
582           }
583           ByteBuffer qualifier =
584               qv.hasQualifier() ? qv.getQualifier().asReadOnlyByteBuffer() : null;
585           ByteBuffer value =
586               qv.hasValue() ? qv.getValue().asReadOnlyByteBuffer() : null;
587           long ts = timestamp;
588           if (qv.hasTimestamp()) {
589             ts = qv.getTimestamp();
590           }
591           byte[] tags;
592           if (qv.hasTags()) {
593             tags = qv.getTags().toByteArray();
594             Object[] array = Tag.asList(tags, 0, (short)tags.length).toArray();
595             Tag[] tagArray = new Tag[array.length];
596             for(int i = 0; i< array.length; i++) {
597               tagArray[i] = (Tag)array[i];
598             }
599             if(qv.hasDeleteType()) {
600               byte[] qual = qv.hasQualifier() ? qv.getQualifier().toByteArray() : null;
601               put.add(new KeyValue(proto.getRow().toByteArray(), family, qual, ts,
602                   fromDeleteType(qv.getDeleteType()), null, tags));
603             } else {
604               put.addImmutable(family, qualifier, ts, value, tagArray);
605             }
606           } else {
607             if(qv.hasDeleteType()) {
608               byte[] qual = qv.hasQualifier() ? qv.getQualifier().toByteArray() : null;
609               put.add(new KeyValue(proto.getRow().toByteArray(), family, qual, ts,
610                   fromDeleteType(qv.getDeleteType())));
611             } else{
612               put.addImmutable(family, qualifier, ts, value);
613             }
614           }

1939   /**
1940    * Converts a Permission proto to a client TablePermission object.
1941    *
1942    * @param proto the protobuf Permission
1943    * @return the converted TablePermission
1944    */
1945   public static TablePermission toTablePermission(AccessControlProtos.Permission proto) {
1946     if(proto.getType() == AccessControlProtos.Permission.Type.Global) {
1947       AccessControlProtos.GlobalPermission perm = proto.getGlobalPermission();
1948       List<Permission.Action> actions = toPermissionActions(perm.getActionList());
1949 
1950       return new TablePermission(null, null, null,
1951           actions.toArray(new Permission.Action[actions.size()]));
1952     }
1953     if(proto.getType() == AccessControlProtos.Permission.Type.Namespace) {
1954       AccessControlProtos.NamespacePermission perm = proto.getNamespacePermission();
1955       List<Permission.Action> actions = toPermissionActions(perm.getActionList());
1956 
1957       if(!proto.hasNamespacePermission()) {
1958         throw new IllegalStateException("Namespace must not be empty in NamespacePermission");
1959       }
1960       String namespace = perm.getNamespaceName().toStringUtf8();
1961       return new TablePermission(namespace, actions.toArray(new Permission.Action[actions.size()]));
1962     }
1963     if(proto.getType() == AccessControlProtos.Permission.Type.Table) {
1964       AccessControlProtos.TablePermission perm = proto.getTablePermission();
1965       List<Permission.Action> actions = toPermissionActions(perm.getActionList());
1966 
1967       byte[] qualifier = null;
1968       byte[] family = null;
1969       TableName table = null;
1970 
1971       if (!perm.hasTableName()) {
1972         throw new IllegalStateException("TableName cannot be empty");
1973       }
1974       table = ProtobufUtil.toTableName(perm.getTableName());
1975 
1976       if (perm.hasFamily()) family = perm.getFamily().toByteArray();
1977       if (perm.hasQualifier()) qualifier = perm.getQualifier().toByteArray();
1978 
1979       return new TablePermission(table, family, qualifier,
1980           actions.toArray(new Permission.Action[actions.size()]));
1981     }
1982     throw new IllegalStateException("Unrecognize Perm Type: "+proto.getType());
1983   }

2686       return "region= " + getStringForByteString(r.getRegion().getValue()) +
2687           ", row=" + getStringForByteString(r.getGet().getRow());

2689       ClientProtos.MultiRequest r = (ClientProtos.MultiRequest) m;

2692       String row = actions.getActionCount() <= 0? "":
2693         getStringForByteString(actions.getAction(0).hasGet()?
2694           actions.getAction(0).getGet().getRow():
2695           actions.getAction(0).getMutation().getRow());
2696       return "region= " + getStringForByteString(actions.getRegion().getValue()) +
2697           ", for " + r.getRegionActionCount() +

2699     } else if (m instanceof ClientProtos.MutateRequest) {
2700       ClientProtos.MutateRequest r = (ClientProtos.MutateRequest) m;
2701       return "region= " + getStringForByteString(r.getRegion().getValue()) +
2702           ", row=" + getStringForByteString(r.getMutation().getRow());
2703     }
2704     return "TODO: " + m.getClass().toString();
