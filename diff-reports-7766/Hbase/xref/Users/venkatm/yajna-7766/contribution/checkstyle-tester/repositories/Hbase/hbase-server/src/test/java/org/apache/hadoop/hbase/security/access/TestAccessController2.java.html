161   @Test
162   public void testCreateWithCorrectOwner() throws Exception {
163     // Create a test user
164     final User testUser = User.createUserForTesting(TEST_UTIL.getConfiguration(), "TestUser",
165       new String[0]);
166     // Grant the test user the ability to create tables
167     SecureTestUtil.grantGlobal(TEST_UTIL, testUser.getShortName(), Action.CREATE);
168     verifyAllowed(new AccessTestAction() {
169       @Override
170       public Object run() throws Exception {
171         HTableDescriptor desc = new HTableDescriptor(TEST_TABLE.getTableName());
172         desc.addFamily(new HColumnDescriptor(TEST_FAMILY));
173         try (Connection connection =
174             ConnectionFactory.createConnection(TEST_UTIL.getConfiguration(), testUser)) {
175           try (Admin admin = connection.getAdmin()) {
176             createTable(TEST_UTIL, admin, desc);
177           }
178         }
179         return null;
180       }
181     }, testUser);
182     TEST_UTIL.waitTableAvailable(TEST_TABLE.getTableName());
183     // Verify that owner permissions have been granted to the test user on the
184     // table just created
185     List<TablePermission> perms =
186       AccessControlLists.getTablePermissions(conf, TEST_TABLE.getTableName())
187        .get(testUser.getShortName());
188     assertNotNull(perms);
189     assertFalse(perms.isEmpty());
190     // Should be RWXCA
191     assertTrue(perms.get(0).implies(Permission.Action.READ));
192     assertTrue(perms.get(0).implies(Permission.Action.WRITE));
193     assertTrue(perms.get(0).implies(Permission.Action.EXEC));
194     assertTrue(perms.get(0).implies(Permission.Action.CREATE));
195     assertTrue(perms.get(0).implies(Permission.Action.ADMIN));
196   }

198   @Test
199   public void testACLTableAccess() throws Exception {
200     final Configuration conf = TEST_UTIL.getConfiguration();
201 
202     // Superuser
203     User superUser = User.createUserForTesting(conf, "admin", new String[] { "supergroup" });
204 
205     // Global users
206     User globalRead = User.createUserForTesting(conf, "globalRead", new String[0]);
207     User globalWrite = User.createUserForTesting(conf, "globalWrite", new String[0]);
208     User globalCreate = User.createUserForTesting(conf, "globalCreate", new String[0]);
209     User globalAdmin = User.createUserForTesting(conf, "globalAdmin", new String[0]);
210     SecureTestUtil.grantGlobal(TEST_UTIL, globalRead.getShortName(), Action.READ);
211     SecureTestUtil.grantGlobal(TEST_UTIL, globalWrite.getShortName(), Action.WRITE);
212     SecureTestUtil.grantGlobal(TEST_UTIL, globalCreate.getShortName(), Action.CREATE);
213     SecureTestUtil.grantGlobal(TEST_UTIL, globalAdmin.getShortName(), Action.ADMIN);
214 
215     // Namespace users
216     User nsRead = User.createUserForTesting(conf, "nsRead", new String[0]);
217     User nsWrite = User.createUserForTesting(conf, "nsWrite", new String[0]);
218     User nsCreate = User.createUserForTesting(conf, "nsCreate", new String[0]);
219     User nsAdmin = User.createUserForTesting(conf, "nsAdmin", new String[0]);
220     SecureTestUtil.grantOnNamespace(TEST_UTIL, nsRead.getShortName(),
221       TEST_TABLE.getTableName().getNamespaceAsString(), Action.READ);
222     SecureTestUtil.grantOnNamespace(TEST_UTIL, nsWrite.getShortName(),
223       TEST_TABLE.getTableName().getNamespaceAsString(), Action.WRITE);
224     SecureTestUtil.grantOnNamespace(TEST_UTIL, nsCreate.getShortName(),
225       TEST_TABLE.getTableName().getNamespaceAsString(), Action.CREATE);
226     SecureTestUtil.grantOnNamespace(TEST_UTIL, nsAdmin.getShortName(),
227       TEST_TABLE.getTableName().getNamespaceAsString(), Action.ADMIN);
228 
229     // Table users
230     User tableRead = User.createUserForTesting(conf, "tableRead", new String[0]);
231     User tableWrite = User.createUserForTesting(conf, "tableWrite", new String[0]);
232     User tableCreate = User.createUserForTesting(conf, "tableCreate", new String[0]);
233     User tableAdmin = User.createUserForTesting(conf, "tableAdmin", new String[0]);
234     SecureTestUtil.grantOnTable(TEST_UTIL, tableRead.getShortName(),
235       TEST_TABLE.getTableName(), null, null, Action.READ);
236     SecureTestUtil.grantOnTable(TEST_UTIL, tableWrite.getShortName(),
237       TEST_TABLE.getTableName(), null, null, Action.WRITE);
238     SecureTestUtil.grantOnTable(TEST_UTIL, tableCreate.getShortName(),
239       TEST_TABLE.getTableName(), null, null, Action.CREATE);
240     SecureTestUtil.grantOnTable(TEST_UTIL, tableAdmin.getShortName(),
241       TEST_TABLE.getTableName(), null, null, Action.ADMIN);
242 
243     // Write tests
244 
245     AccessTestAction writeAction = new AccessTestAction() {
246       @Override
247       public Object run() throws Exception {
248         try(Connection conn = ConnectionFactory.createConnection(conf);
249             Table t = conn.getTable(AccessControlLists.ACL_TABLE_NAME)) {
250           t.put(new Put(TEST_ROW).add(AccessControlLists.ACL_LIST_FAMILY, TEST_QUALIFIER,
251             TEST_VALUE));
252           return null;
253         }
254       }
255     };
256 
257     // All writes to ACL table denied except for GLOBAL WRITE permission and superuser
258 
259     verifyDenied(writeAction, globalAdmin, globalCreate, globalRead);
260     verifyDenied(writeAction, nsAdmin, nsCreate, nsRead, nsWrite);
261     verifyDenied(writeAction, tableAdmin, tableCreate, tableRead, tableWrite);
262     verifyAllowed(writeAction, superUser, globalWrite);
263 
264     // Read tests
265 
266     AccessTestAction scanAction = new AccessTestAction() {
267       @Override
268       public Object run() throws Exception {
269         try(Connection conn = ConnectionFactory.createConnection(conf);
270             Table t = conn.getTable(AccessControlLists.ACL_TABLE_NAME)) {
271           ResultScanner s = t.getScanner(new Scan());
272           try {
273             for (Result r = s.next(); r != null; r = s.next()) {
274               // do nothing
275             }
276           } finally {
277             s.close();
278           }
279           return null;
280         }
281       }
282     };
283 
284     // All reads from ACL table denied except for GLOBAL READ and superuser
285 
286     verifyDenied(scanAction, globalAdmin, globalCreate, globalWrite);
287     verifyDenied(scanAction, nsCreate, nsAdmin, nsRead, nsWrite);
288     verifyDenied(scanAction, tableCreate, tableAdmin, tableRead, tableWrite);
289     verifyAllowed(scanAction, superUser, globalRead);
290   }

406     // Verify user from a group whose column family level access has been revoked can't read any

427   @Test
428   public void testACLZNodeDeletion() throws Exception {
429     String baseAclZNode = "/hbase/acl/";
430     String ns = "testACLZNodeDeletionNamespace";
431     NamespaceDescriptor desc = NamespaceDescriptor.create(ns).build();
432     createNamespace(TEST_UTIL, desc);
433 
434     final TableName table = TableName.valueOf(ns, "testACLZNodeDeletionTable");
435     final byte[] family = Bytes.toBytes("f1");
436     HTableDescriptor htd = new HTableDescriptor(table);
437     htd.addFamily(new HColumnDescriptor(family));
438     createTable(TEST_UTIL, htd);
439 
440     // Namespace needs this, as they follow the lazy creation of ACL znode.
441     grantOnNamespace(TEST_UTIL, TESTGROUP1_USER1.getShortName(), ns, Action.ADMIN);
442     ZooKeeperWatcher zkw = TEST_UTIL.getMiniHBaseCluster().getMaster().getZooKeeper();
443     assertTrue("The acl znode for table should exist",  ZKUtil.checkExists(zkw, baseAclZNode +
444         table.getNameAsString()) != -1);
445     assertTrue("The acl znode for namespace should exist", ZKUtil.checkExists(zkw, baseAclZNode +
446         convertToNamespace(ns)) != -1);
447 
448     revokeFromNamespace(TEST_UTIL, TESTGROUP1_USER1.getShortName(), ns, Action.ADMIN);
449     deleteTable(TEST_UTIL, table);
450     deleteNamespace(TEST_UTIL, ns);
451 
452     assertTrue("The acl znode for table should have been deleted",
453         ZKUtil.checkExists(zkw, baseAclZNode + table.getNameAsString()) == -1);
454     assertTrue( "The acl znode for namespace should have been deleted",
455         ZKUtil.checkExists(zkw, baseAclZNode + convertToNamespace(ns)) == -1);
456   }
