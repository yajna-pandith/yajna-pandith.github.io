188   @Test
189   public void testLogRoll() throws Exception {
190     long seq = 0;
191     long baseline = 1000;
192     long time = baseline;
193     KeyValue kv = new KeyValue(r1, f1, r1);
194     WALEdit edit = new WALEdit();
195     edit.add(kv);
196 
197     List<WALActionsListener> listeners = new ArrayList<WALActionsListener>();
198     listeners.add(replication);
199     final WALFactory wals = new WALFactory(utility.getConfiguration(), listeners,
200         URLEncoder.encode("regionserver:60020", "UTF8"));
201     final WAL wal = wals.getWAL(hri.getEncodedNameAsBytes());
202     final AtomicLong sequenceId = new AtomicLong(1);
203     manager.init();
204     HTableDescriptor htd = new HTableDescriptor();
205     htd.addFamily(new HColumnDescriptor(f1));
206     // Testing normal log rolling every 20
207     for(long i = 1; i < 101; i++) {
208       if(i > 1 && i % 20 == 0) {
209         wal.rollWriter();
210       }
211       LOG.info(i);
212       final long txid = wal.append(htd, hri, new WALKey(hri.getEncodedNameAsBytes(), test,
213           System.currentTimeMillis()), edit, sequenceId, true ,null);
214       wal.sync(txid);
215     }
216 
217     // Simulate a rapid insert that's followed
218     // by a report that's still not totally complete (missing last one)
219     LOG.info(baseline + " and " + time);
220     baseline += 101;
221     time = baseline;
222     LOG.info(baseline + " and " + time);
223 
224     for (int i = 0; i < 3; i++) {
225       wal.append(htd, hri, new WALKey(hri.getEncodedNameAsBytes(), test,
226           System.currentTimeMillis()), edit, sequenceId, true, null);
227     }
228     wal.sync();
229 
230     assertEquals(6, manager.getWALs().get(slaveId).size());
231 
232     wal.rollWriter();
233 
234     manager.logPositionAndCleanOldLogs(manager.getSources().get(0).getCurrentPath(),
235         "1", 0, false, false);
236 
237     wal.append(htd, hri, new WALKey(hri.getEncodedNameAsBytes(), test,
238         System.currentTimeMillis()), edit, sequenceId, true, null);
239     wal.sync();
240 
241     assertEquals(1, manager.getWALs().size());
242 
243 
244     // TODO Need a case with only 2 WALs and we only want to delete the first one
245   }

247   @Test
248   public void testClaimQueues() throws Exception {
249     LOG.debug("testNodeFailoverWorkerCopyQueuesFromRSUsingMulti");
250     conf.setBoolean(HConstants.ZOOKEEPER_USEMULTI, true);
251     final Server server = new DummyServer("hostname0.example.org");
252     ReplicationQueues rq =
253         ReplicationFactory.getReplicationQueues(server.getZooKeeper(), server.getConfiguration(),
254           server);
255     rq.init(server.getServerName().toString());
256     // populate some znodes in the peer znode
257     files.add("log1");
258     files.add("log2");
259     for (String file : files) {
260       rq.addLog("1", file);
261     }
262     // create 3 DummyServers
263     Server s1 = new DummyServer("dummyserver1.example.org");
264     Server s2 = new DummyServer("dummyserver2.example.org");
265     Server s3 = new DummyServer("dummyserver3.example.org");
266 
267     // create 3 DummyNodeFailoverWorkers
268     DummyNodeFailoverWorker w1 = new DummyNodeFailoverWorker(
269         server.getServerName().getServerName(), s1);
270     DummyNodeFailoverWorker w2 = new DummyNodeFailoverWorker(
271         server.getServerName().getServerName(), s2);
272     DummyNodeFailoverWorker w3 = new DummyNodeFailoverWorker(
273         server.getServerName().getServerName(), s3);
274 
275     latch = new CountDownLatch(3);
276     // start the threads
277     w1.start();
278     w2.start();
279     w3.start();
280     // make sure only one is successful
281     int populatedMap = 0;
282     // wait for result now... till all the workers are done.
283     latch.await();
284     populatedMap += w1.isLogZnodesMapPopulated() + w2.isLogZnodesMapPopulated()
285         + w3.isLogZnodesMapPopulated();
286     assertEquals(1, populatedMap);
287     server.abort("", null);
288   }

290   @Test
291   public void testCleanupFailoverQueues() throws Exception {
292     final Server server = new DummyServer("hostname1.example.org");
293     ReplicationQueues rq =
294         ReplicationFactory.getReplicationQueues(server.getZooKeeper(), server.getConfiguration(),
295           server);
296     rq.init(server.getServerName().toString());
297     // populate some znodes in the peer znode
298     SortedSet<String> files = new TreeSet<String>();
299     files.add("log1");
300     files.add("log2");
301     for (String file : files) {
302       rq.addLog("1", file);
303     }
304     Server s1 = new DummyServer("dummyserver1.example.org");
305     ReplicationQueues rq1 =
306         ReplicationFactory.getReplicationQueues(s1.getZooKeeper(), s1.getConfiguration(), s1);
307     rq1.init(s1.getServerName().toString());
308     ReplicationPeers rp1 =
309         ReplicationFactory.getReplicationPeers(s1.getZooKeeper(), s1.getConfiguration(), s1);
310     rp1.init();
311     NodeFailoverWorker w1 =
312         manager.new NodeFailoverWorker(server.getServerName().getServerName(), rq1, rp1, new UUID(
313             new Long(1), new Long(2)));
314     w1.start();
315     w1.join(5000);
316     assertEquals(1, manager.getWalsByIdRecoveredQueues().size());
317     String id = "1-" + server.getServerName().getServerName();
318     assertEquals(files, manager.getWalsByIdRecoveredQueues().get(id));
319     manager.cleanOldLogs("log2", id, true);
320     // log1 should be deleted
321     assertEquals(Sets.newHashSet("log2"), manager.getWalsByIdRecoveredQueues().get(id));
322   }

324   @Test
325   public void testNodeFailoverDeadServerParsing() throws Exception {
326     LOG.debug("testNodeFailoverDeadServerParsing");
327     conf.setBoolean(HConstants.ZOOKEEPER_USEMULTI, true);
328     final Server server = new DummyServer("ec2-54-234-230-108.compute-1.amazonaws.com");
329     ReplicationQueues repQueues =
330         ReplicationFactory.getReplicationQueues(server.getZooKeeper(), conf, server);
331     repQueues.init(server.getServerName().toString());
332     // populate some znodes in the peer znode
333     files.add("log1");
334     files.add("log2");
335     for (String file : files) {
336       repQueues.addLog("1", file);
337     }
338 
339     // create 3 DummyServers
340     Server s1 = new DummyServer("ip-10-8-101-114.ec2.internal");
341     Server s2 = new DummyServer("ec2-107-20-52-47.compute-1.amazonaws.com");
342     Server s3 = new DummyServer("ec2-23-20-187-167.compute-1.amazonaws.com");
343 
344     // simulate three servers fail sequentially
345     ReplicationQueues rq1 =
346         ReplicationFactory.getReplicationQueues(s1.getZooKeeper(), s1.getConfiguration(), s1);
347     rq1.init(s1.getServerName().toString());
348     SortedMap<String, SortedSet<String>> testMap =
349         rq1.claimQueues(server.getServerName().getServerName());
350     ReplicationQueues rq2 =
351         ReplicationFactory.getReplicationQueues(s2.getZooKeeper(), s2.getConfiguration(), s2);
352     rq2.init(s2.getServerName().toString());
353     testMap = rq2.claimQueues(s1.getServerName().getServerName());
354     ReplicationQueues rq3 =
355         ReplicationFactory.getReplicationQueues(s3.getZooKeeper(), s3.getConfiguration(), s3);
356     rq3.init(s3.getServerName().toString());
357     testMap = rq3.claimQueues(s2.getServerName().getServerName());
358 
359     ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(testMap.firstKey());
360     List<String> result = replicationQueueInfo.getDeadRegionServers();
361 
362     // verify
363     assertTrue(result.contains(server.getServerName().getServerName()));
364     assertTrue(result.contains(s1.getServerName().getServerName()));
365     assertTrue(result.contains(s2.getServerName().getServerName()));
366 
367     server.abort("", null);
368   }

371   static class DummyNodeFailoverWorker extends Thread {
372     private SortedMap<String, SortedSet<String>> logZnodesMap;
373     Server server;
374     private String deadRsZnode;
375     ReplicationQueues rq;
376 
377     public DummyNodeFailoverWorker(String znode, Server s) throws Exception {
378       this.deadRsZnode = znode;
379       this.server = s;
380       this.rq =
381           ReplicationFactory.getReplicationQueues(server.getZooKeeper(), server.getConfiguration(),
382             server);
383       this.rq.init(this.server.getServerName().toString());
384     }
385 
386     @Override
387     public void run() {
388       try {
389         logZnodesMap = rq.claimQueues(deadRsZnode);
390         server.abort("Done with testing", null);
391       } catch (Exception e) {
392         LOG.error("Got exception while running NodeFailoverWorker", e);
393       } finally {
394         latch.countDown();
395       }
396     }
397 
398     /**
399      * @return 1 when the map is not empty.
400      */
401     private int isLogZnodesMapPopulated() {
402       Collection<SortedSet<String>> sets = logZnodesMap.values();
403       if (sets.size() > 1) {
404         throw new RuntimeException("unexpected size of logZnodesMap: " + sets.size());
405       }
406       if (sets.size() == 1) {
407         SortedSet<String> s = sets.iterator().next();
408         for (String file : files) {
409           // at least one file was missing
410           if (!s.contains(file)) {
411             return 0;
412           }
413         }
414         return 1; // we found all the files
415       }
416       return 0;
417     }
418   }
