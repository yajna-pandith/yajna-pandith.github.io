172   private void startAndWriteData() throws IOException, InterruptedException {
173     // When the hbase:meta table can be opened, the region servers are running
174     new HTable(TEST_UTIL.getConfiguration(), TableName.META_TABLE_NAME);
175     this.server = cluster.getRegionServerThreads().get(0).getRegionServer();
176 
177     Table table = createTestTable(this.tableName);
178 
179     server = TEST_UTIL.getRSForFirstRegionInTable(table.getName());
180     for (int i = 1; i <= 256; i++) {    // 256 writes should cause 8 log rolls
181       doPut(table, i);
182       if (i % 32 == 0) {
183         // After every 32 writes sleep to let the log roller run
184         try {
185           Thread.sleep(2000);
186         } catch (InterruptedException e) {
187           // continue
188         }
189       }
190     }
191   }

252   void validateData(Table table, int rownum) throws IOException {
253     String row = "row" + String.format("%1$04d", rownum);
254     Get get = new Get(Bytes.toBytes(row));
255     get.addFamily(HConstants.CATALOG_FAMILY);
256     Result result = table.get(get);
257     assertTrue(result.size() == 1);
258     assertTrue(Bytes.equals(value,
259                 result.getValue(HConstants.CATALOG_FAMILY, null)));
260     LOG.info("Validated row " + row);
261   }

263   void batchWriteAndWait(Table table, final FSHLog log, int start, boolean expect, int timeout)
264       throws IOException {
265     for (int i = 0; i < 10; i++) {
266       Put put = new Put(Bytes.toBytes("row"
267           + String.format("%1$04d", (start + i))));
268       put.add(HConstants.CATALOG_FAMILY, null, value);
269       table.put(put);
270     }
271     Put tmpPut = new Put(Bytes.toBytes("tmprow"));
272     tmpPut.add(HConstants.CATALOG_FAMILY, null, value);
273     long startTime = System.currentTimeMillis();
274     long remaining = timeout;
275     while (remaining > 0) {
276       if (log.isLowReplicationRollEnabled() == expect) {
277         break;
278       } else {
279         // Trigger calling FSHlog#checkLowReplication()
280         table.put(tmpPut);
281         try {
282           Thread.sleep(200);
283         } catch (InterruptedException e) {
284           // continue
285         }
286         remaining = timeout - (System.currentTimeMillis() - startTime);
287       }
288     }
289   }

291   /**
292    * Tests that logs are rolled upon detecting datanode death
293    * Requires an HDFS jar with HDFS-826 & syncFs() support (HDFS-200)
294    */
295   @Test
296   public void testLogRollOnDatanodeDeath() throws Exception {
297     TEST_UTIL.ensureSomeRegionServersAvailable(2);
298     assertTrue("This test requires WAL file replication set to 2.",
299       fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()) == 2);
300     LOG.info("Replication=" +
301       fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()));
302 
303     this.server = cluster.getRegionServer(0);
304 
305     // Create the test table and open it
306     HTableDescriptor desc = new HTableDescriptor(TableName.valueOf(getName()));
307     desc.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
308 
309     admin.createTable(desc);
310     Table table = TEST_UTIL.getConnection().getTable(desc.getTableName());
311     assertTrue(((HTable) table).isAutoFlush());
312 
313     server = TEST_UTIL.getRSForFirstRegionInTable(desc.getTableName());
314     final FSHLog log = (FSHLog) server.getWAL(null);
315     final AtomicBoolean lowReplicationHookCalled = new AtomicBoolean(false);
316 
317     log.registerWALActionsListener(new WALActionsListener.Base() {
318       @Override
319       public void logRollRequested(boolean lowReplication) {
320         if (lowReplication) {
321           lowReplicationHookCalled.lazySet(true);
322         }
323       }
324     });
325 
326     // don't run this test without append support (HDFS-200 & HDFS-142)
327     assertTrue("Need append support for this test", FSUtils
328         .isAppendSupported(TEST_UTIL.getConfiguration()));
329 
330     // add up the datanode count, to ensure proper replication when we kill 1
331     // This function is synchronous; when it returns, the dfs cluster is active
332     // We start 3 servers and then stop 2 to avoid a directory naming conflict
333     //  when we stop/start a namenode later, as mentioned in HBASE-5163
334     List<DataNode> existingNodes = dfsCluster.getDataNodes();
335     int numDataNodes = 3;
336     dfsCluster.startDataNodes(TEST_UTIL.getConfiguration(), numDataNodes, true,
337         null, null);
338     List<DataNode> allNodes = dfsCluster.getDataNodes();
339     for (int i = allNodes.size()-1; i >= 0; i--) {
340       if (existingNodes.contains(allNodes.get(i))) {
341         dfsCluster.stopDataNode( i );
342       }
343     }
344 
345     assertTrue("DataNodes " + dfsCluster.getDataNodes().size() +
346         " default replication " +
347         fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()),
348     dfsCluster.getDataNodes().size() >=
349       fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()) + 1);
350 
351     writeData(table, 2);
352 
353     long curTime = System.currentTimeMillis();
354     LOG.info("log.getCurrentFileName(): " + log.getCurrentFileName());
355     long oldFilenum = DefaultWALProvider.extractFileNumFromWAL(log);
356     assertTrue("Log should have a timestamp older than now",
357         curTime > oldFilenum && oldFilenum != -1);
358 
359     assertTrue("The log shouldn't have rolled yet",
360         oldFilenum == DefaultWALProvider.extractFileNumFromWAL(log));
361     final DatanodeInfo[] pipeline = log.getPipeLine();
362     assertTrue(pipeline.length ==
363         fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()));
364 
365     // kill a datanode in the pipeline to force a log roll on the next sync()
366     // This function is synchronous, when it returns the node is killed.
367     assertTrue(dfsCluster.stopDataNode(pipeline[0].getName()) != null);
368 
369     // this write should succeed, but trigger a log roll
370     writeData(table, 2);
371     long newFilenum = DefaultWALProvider.extractFileNumFromWAL(log);
372 
373     assertTrue("Missing datanode should've triggered a log roll",
374         newFilenum > oldFilenum && newFilenum > curTime);
375 
376     assertTrue("The log rolling hook should have been called with the low replication flag",
377         lowReplicationHookCalled.get());
378 
379     // write some more log data (this should use a new hdfs_out)
380     writeData(table, 3);
381     assertTrue("The log should not roll again.",
382         DefaultWALProvider.extractFileNumFromWAL(log) == newFilenum);
383     // kill another datanode in the pipeline, so the replicas will be lower than
384     // the configured value 2.
385     assertTrue(dfsCluster.stopDataNode(pipeline[1].getName()) != null);
386 
387     batchWriteAndWait(table, log, 3, false, 14000);
388     int replication = log.getLogReplication();
389     assertTrue("LowReplication Roller should've been disabled, current replication="
390             + replication, !log.isLowReplicationRollEnabled());
391 
392     dfsCluster
393         .startDataNodes(TEST_UTIL.getConfiguration(), 1, true, null, null);
394 
395     // Force roll writer. The new log file will have the default replications,
396     // and the LowReplication Roller will be enabled.
397     log.rollWriter(true);
398     batchWriteAndWait(table, log, 13, true, 10000);
399     replication = log.getLogReplication();
400     assertTrue("New log file should have the default replication instead of " +
401       replication,
402       replication == fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()));
403     assertTrue("LowReplication Roller should've been enabled", log.isLowReplicationRollEnabled());
404   }

406   /**
407    * Test that WAL is rolled when all data nodes in the pipeline have been
408    * restarted.
409    * @throws Exception
410    */
411   @Test
412   public void testLogRollOnPipelineRestart() throws Exception {
413     LOG.info("Starting testLogRollOnPipelineRestart");
414     assertTrue("This test requires WAL file replication.",
415       fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()) > 1);
416     LOG.info("Replication=" +
417       fs.getDefaultReplication(TEST_UTIL.getDataTestDirOnTestFS()));
418     // When the hbase:meta table can be opened, the region servers are running
419     Table t = new HTable(TEST_UTIL.getConfiguration(), TableName.META_TABLE_NAME);
420     try {
421       this.server = cluster.getRegionServer(0);
422 
423       // Create the test table and open it
424       HTableDescriptor desc = new HTableDescriptor(TableName.valueOf(getName()));
425       desc.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
426 
427       admin.createTable(desc);
428       Table table = new HTable(TEST_UTIL.getConfiguration(), desc.getTableName());
429 
430       server = TEST_UTIL.getRSForFirstRegionInTable(desc.getTableName());
431       final WAL log = server.getWAL(null);
432       final List<Path> paths = new ArrayList<Path>();
433       final List<Integer> preLogRolledCalled = new ArrayList<Integer>();
434 
435       paths.add(DefaultWALProvider.getCurrentFileName(log));
436       log.registerWALActionsListener(new WALActionsListener.Base() {
437 
438         @Override
439         public void preLogRoll(Path oldFile, Path newFile)  {
440           LOG.debug("preLogRoll: oldFile="+oldFile+" newFile="+newFile);
441           preLogRolledCalled.add(new Integer(1));
442         }
443         @Override
444         public void postLogRoll(Path oldFile, Path newFile) {
445           paths.add(newFile);
446         }
447       });
448 
449       // don't run this test without append support (HDFS-200 & HDFS-142)
450       assertTrue("Need append support for this test", FSUtils
451           .isAppendSupported(TEST_UTIL.getConfiguration()));
452 
453       writeData(table, 1002);
454 
455       long curTime = System.currentTimeMillis();
456       LOG.info("log.getCurrentFileName()): " + DefaultWALProvider.getCurrentFileName(log));
457       long oldFilenum = DefaultWALProvider.extractFileNumFromWAL(log);
458       assertTrue("Log should have a timestamp older than now",
459           curTime > oldFilenum && oldFilenum != -1);
460 
461       assertTrue("The log shouldn't have rolled yet", oldFilenum ==
462           DefaultWALProvider.extractFileNumFromWAL(log));
463 
464       // roll all datanodes in the pipeline
465       dfsCluster.restartDataNodes();
466       Thread.sleep(1000);
467       dfsCluster.waitActive();
468       LOG.info("Data Nodes restarted");
469       validateData(table, 1002);
470 
471       // this write should succeed, but trigger a log roll
472       writeData(table, 1003);
473       long newFilenum = DefaultWALProvider.extractFileNumFromWAL(log);
474 
475       assertTrue("Missing datanode should've triggered a log roll",
476           newFilenum > oldFilenum && newFilenum > curTime);
477       validateData(table, 1003);
478 
479       writeData(table, 1004);
480 
481       // roll all datanode again
482       dfsCluster.restartDataNodes();
483       Thread.sleep(1000);
484       dfsCluster.waitActive();
485       LOG.info("Data Nodes restarted");
486       validateData(table, 1004);
487 
488       // this write should succeed, but trigger a log roll
489       writeData(table, 1005);
490 
491       // force a log roll to read back and verify previously written logs
492       log.rollWriter(true);
493       assertTrue("preLogRolledCalled has size of " + preLogRolledCalled.size(),
494           preLogRolledCalled.size() >= 1);
495 
496       // read back the data written
497       Set<String> loggedRows = new HashSet<String>();
498       FSUtils fsUtils = FSUtils.getInstance(fs, TEST_UTIL.getConfiguration());
499       for (Path p : paths) {
500         LOG.debug("recovering lease for " + p);
501         fsUtils.recoverFileLease(((HFileSystem)fs).getBackingFs(), p,
502           TEST_UTIL.getConfiguration(), null);
503 
504         LOG.debug("Reading WAL "+FSUtils.getPath(p));
505         WAL.Reader reader = null;
506         try {
507           reader = WALFactory.createReader(fs, p, TEST_UTIL.getConfiguration());
508           WAL.Entry entry;
509           while ((entry = reader.next()) != null) {
510             LOG.debug("#"+entry.getKey().getLogSeqNum()+": "+entry.getEdit().getCells());
511             for (Cell cell : entry.getEdit().getCells()) {
512               loggedRows.add(Bytes.toStringBinary(cell.getRow()));
513             }
514           }
515         } catch (EOFException e) {
516           LOG.debug("EOF reading file "+FSUtils.getPath(p));
517         } finally {
518           if (reader != null) reader.close();
519         }
520       }
521 
522       // verify the written rows are there
523       assertTrue(loggedRows.contains("row1002"));
524       assertTrue(loggedRows.contains("row1003"));
525       assertTrue(loggedRows.contains("row1004"));
526       assertTrue(loggedRows.contains("row1005"));
527 
528       // flush all regions
529       for (Region r: server.getOnlineRegionsLocalContext()) {
530         r.flush(true);
531       }
532 
533       ResultScanner scanner = table.getScanner(new Scan());
534       try {
535         for (int i=2; i<=5; i++) {
536           Result r = scanner.next();
537           assertNotNull(r);
538           assertFalse(r.isEmpty());
539           assertEquals("row100"+i, Bytes.toString(r.getRow()));
540         }
541       } finally {
542         scanner.close();
543       }
544 
545       // verify that no region servers aborted
546       for (JVMClusterUtil.RegionServerThread rsThread:
547         TEST_UTIL.getHBaseCluster().getRegionServerThreads()) {
548         assertFalse(rsThread.getRegionServer().isAborted());
549       }
550     } finally {
551       if (t != null) t.close();
552     }
553   }

555   /**
556    * Tests that logs are deleted when some region has a compaction
557    * record in WAL and no other records. See HBASE-8597.
558    */
559   @Test
560   public void testCompactionRecordDoesntBlockRolling() throws Exception {
561     Table table = null;
562     Table table2 = null;
563 
564     // When the hbase:meta table can be opened, the region servers are running
565     Table t = new HTable(TEST_UTIL.getConfiguration(), TableName.META_TABLE_NAME);
566     try {
567       table = createTestTable(getName());
568       table2 = createTestTable(getName() + "1");
569 
570       server = TEST_UTIL.getRSForFirstRegionInTable(table.getName());
571       final WAL log = server.getWAL(null);
572       Region region = server.getOnlineRegions(table2.getName()).get(0);
573       Store s = region.getStore(HConstants.CATALOG_FAMILY);
574 
575       //have to flush namespace to ensure it doesn't affect wall tests
576       admin.flush(TableName.NAMESPACE_TABLE_NAME);
577 
578       // Put some stuff into table2, to make sure we have some files to compact.
579       for (int i = 1; i <= 2; ++i) {
580         doPut(table2, i);
581         admin.flush(table2.getName());
582       }
583       doPut(table2, 3); // don't flush yet, or compaction might trigger before we roll WAL
584       assertEquals("Should have no WAL after initial writes", 0,
585           DefaultWALProvider.getNumRolledLogFiles(log));
586       assertEquals(2, s.getStorefilesCount());
587 
588       // Roll the log and compact table2, to have compaction record in the 2nd WAL.
589       log.rollWriter();
590       assertEquals("Should have WAL; one table is not flushed", 1,
591           DefaultWALProvider.getNumRolledLogFiles(log));
592       admin.flush(table2.getName());
593       region.compact(false);
594       // Wait for compaction in case if flush triggered it before us.
595       Assert.assertNotNull(s);
596       for (int waitTime = 3000; s.getStorefilesCount() > 1 && waitTime > 0; waitTime -= 200) {
597         Threads.sleepWithoutInterrupt(200);
598       }
599       assertEquals("Compaction didn't happen", 1, s.getStorefilesCount());
600 
601       // Write some value to the table so the WAL cannot be deleted until table is flushed.
602       doPut(table, 0); // Now 2nd WAL will have compaction record for table2 and put for table.
603       log.rollWriter(); // 1st WAL deleted, 2nd not deleted yet.
604       assertEquals("Should have WAL; one table is not flushed", 1,
605           DefaultWALProvider.getNumRolledLogFiles(log));
606 
607       // Flush table to make latest WAL obsolete; write another record, and roll again.
608       admin.flush(table.getName());
609       doPut(table, 1);
610       log.rollWriter(); // Now 2nd WAL is deleted and 3rd is added.
611       assertEquals("Should have 1 WALs at the end", 1,
612           DefaultWALProvider.getNumRolledLogFiles(log));
613     } finally {
614       if (t != null) t.close();
615       if (table != null) table.close();
616       if (table2 != null) table2.close();
617     }
618   }

621     Put put = new Put(Bytes.toBytes("row" + String.format("%1$04d", i)));
622     put.add(HConstants.CATALOG_FAMILY, null, value);
623     table.put(put);
624   }
