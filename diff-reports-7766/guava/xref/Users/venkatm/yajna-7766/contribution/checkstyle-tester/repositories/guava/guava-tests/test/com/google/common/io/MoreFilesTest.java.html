122           });

199   public void testEqual() throws IOException {
200     try (FileSystem fs = Jimfs.newFileSystem(Configuration.unix())) {
201       Path fooPath = fs.getPath("foo");
202       Path barPath = fs.getPath("bar");
203       MoreFiles.asCharSink(fooPath, UTF_8).write("foo");
204       MoreFiles.asCharSink(barPath, UTF_8).write("barbar");
205 
206       assertThat(MoreFiles.equal(fooPath, barPath)).isFalse();
207       assertThat(MoreFiles.equal(fooPath, fooPath)).isTrue();
208       assertThat(MoreFiles.asByteSource(fooPath).contentEquals(MoreFiles.asByteSource(fooPath)))
209           .isTrue();
210 
211       Path fooCopy = Files.copy(fooPath, fs.getPath("fooCopy"));
212       assertThat(Files.isSameFile(fooPath, fooCopy)).isFalse();
213       assertThat(MoreFiles.equal(fooPath, fooCopy)).isTrue();
214 
215       MoreFiles.asCharSink(fooCopy, UTF_8).write("boo");
216       assertThat(MoreFiles.asByteSource(fooPath).size())
217           .isEqualTo(MoreFiles.asByteSource(fooCopy).size());
218       assertThat(MoreFiles.equal(fooPath, fooCopy)).isFalse();
219 
220       // should also assert that a Path that erroneously reports a size 0 can still be compared,
221       // not sure how to do that with the Path API
222     }
223   }

396   /**
397    * Creates a new file system for testing that supports the given features in addition to
398    * supporting symbolic links. The file system is created initially having the following file
399    * structure:
400    *
401    * <pre>
402    *   /
403    *      work/
404    *         dir/
405    *            a
406    *            b/
407    *               g
408    *               h -> ../a
409    *               i/
410    *                  j/
411    *                     k
412    *                     l/
413    *            c
414    *            d -> b/i
415    *            e/
416    *            f -> /dontdelete
417    *      dontdelete/
418    *         a
419    *         b/
420    *         c
421    *      symlinktodir -> work/dir
422    * </pre>
423    */
424   static FileSystem newTestFileSystem(Feature... supportedFeatures) throws IOException {
425     FileSystem fs =
426         Jimfs.newFileSystem(
427             Configuration.unix()
428                 .toBuilder()
429                 .setSupportedFeatures(ObjectArrays.concat(SYMBOLIC_LINKS, supportedFeatures))
430                 .build());
431     Files.createDirectories(fs.getPath("dir/b/i/j/l"));
432     Files.createFile(fs.getPath("dir/a"));
433     Files.createFile(fs.getPath("dir/c"));
434     Files.createSymbolicLink(fs.getPath("dir/d"), fs.getPath("b/i"));
435     Files.createDirectory(fs.getPath("dir/e"));
436     Files.createSymbolicLink(fs.getPath("dir/f"), fs.getPath("/dontdelete"));
437     Files.createFile(fs.getPath("dir/b/g"));
438     Files.createSymbolicLink(fs.getPath("dir/b/h"), fs.getPath("../a"));
439     Files.createFile(fs.getPath("dir/b/i/j/k"));
440     Files.createDirectory(fs.getPath("/dontdelete"));
441     Files.createFile(fs.getPath("/dontdelete/a"));
442     Files.createDirectory(fs.getPath("/dontdelete/b"));
443     Files.createFile(fs.getPath("/dontdelete/c"));
444     Files.createSymbolicLink(fs.getPath("/symlinktodir"), fs.getPath("work/dir"));
445     return fs;
446   }

572   /**
573    * This test attempts to create a situation in which one thread is constantly changing a file from
574    * being a real directory to being a symlink to another directory. It then calls
575    * deleteDirectoryContents thousands of times on a directory whose subtree contains the file
576    * that's switching between directory and symlink to try to ensure that under no circumstance does
577    * deleteDirectoryContents follow the symlink to the other directory and delete that directory's
578    * contents.
579    *
580    * <p>We can only test this with a file system that supports SecureDirectoryStream, because it's
581    * not possible to protect against this if the file system doesn't.
582    */
583   public void testDirectoryDeletion_directorySymlinkRace() throws IOException {
584     for (DirectoryDeleteMethod method : EnumSet.allOf(DirectoryDeleteMethod.class)) {
585       try (FileSystem fs = newTestFileSystem(SECURE_DIRECTORY_STREAM)) {
586         Path dirToDelete = fs.getPath("dir/b/i");
587         Path changingFile = dirToDelete.resolve("j/l");
588         Path symlinkTarget = fs.getPath("/dontdelete");
589 
590         ExecutorService executor = Executors.newSingleThreadExecutor();
591         startDirectorySymlinkSwitching(changingFile, symlinkTarget, executor);
592 
593         try {
594           for (int i = 0; i < 5000; i++) {
595             try {
596               Files.createDirectories(changingFile);
597               Files.createFile(dirToDelete.resolve("j/k"));
598             } catch (FileAlreadyExistsException expected) {
599               // if a file already exists, that's fine... just continue
600             }
601 
602             try {
603               method.delete(dirToDelete);
604             } catch (FileSystemException expected) {
605               // the delete method may or may not throw an exception, but if it does that's fine
606               // and expected
607             }
608 
609             // this test is mainly checking that the contents of /dontdelete aren't deleted under
610             // any circumstances
611             assertEquals(3, MoreFiles.listFiles(symlinkTarget).size());
612 
613             Thread.yield();
614           }
615         } finally {
616           executor.shutdownNow();
617         }
618       }
619     }
620   }

640       assertTrue(Files.isDirectory(realSymlinkTarget, NOFOLLOW_LINKS));
641     }
642   }

644   /**
645    * Starts a new task on the given executor that switches (deletes and replaces) a file between
646    * being a directory and being a symlink. The given {@code file} is the file that should switch
647    * between being a directory and being a symlink, while the given {@code target} is the target the
648    * symlink should have.
649    */
650   private static void startDirectorySymlinkSwitching(
651       final Path file, final Path target, ExecutorService executor) {
652     @SuppressWarnings("unused") // go/futurereturn-lsc
653     Future<?> possiblyIgnoredError =
654         executor.submit(
655             new Runnable() {
656               @Override
657               public void run() {
658                 boolean createSymlink = false;
659                 while (!Thread.interrupted()) {
660                   try {
661                     // trying to switch between a real directory and a symlink (dir -> /a)
662                     if (Files.deleteIfExists(file)) {
663                       if (createSymlink) {
664                         Files.createSymbolicLink(file, target);
665                       } else {
666                         Files.createDirectory(file);
667                       }
668                       createSymlink = !createSymlink;
669                     }
670                   } catch (IOException tolerated) {
671                     // it's expected that some of these will fail
672                   }
673 
674                   Thread.yield();
675                 }
676               }
677             });
678   }
