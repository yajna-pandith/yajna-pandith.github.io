76  /**
77   * Combined workspace/project properties page for setting FindBugs properties.
78   * <p>
79   * There are two different preference files: FindBugs core preferences, saved in
80   * the "*.fbprefs" file, and Eclipse plugin preferences, saved in the
81   * "com.github.spotbugs.plugin.eclipse.prefs" file. The difference is, that FB
82   * core prefs are saved/read by the FB core API and there should go all FB
83   * engine settings and none of Eclipse related stuff.
84   * <p>
85   * To retrieve FB core preferences, one should use
86   * {@link #getOriginalUserPreferences()} and
87   * {@link #getCurrentUserPreferences()}.
88   * <p>
89   * To retrieve Eclipse plugin preferences, one should use
90   * {@link #getPreferenceStore()}.
91   * <p>
92   * In both cases project settings are only available in the context of the
93   * project properties page if the project settings are enabled, global settings
94   * are used otherwise.
95   *
96   * @author Andrei Loskutov
97   * @author Peter Friese
98   * @author David Hovemeyer
99   * @author Phil Crosby
100  * @version 2.0
101  * @since 17.06.2004
102  */

182     private void initPreferencesStore(IProject currProject) {
183         workspaceStore = new ScopedPreferenceStore(new InstanceScope(), FindbugsPlugin.PLUGIN_ID);
184         if (currProject != null) {
185             projectStore = new ScopedPreferenceStore(new ProjectScope(currProject), FindbugsPlugin.PLUGIN_ID);
186             projectPropsInitiallyEnabled = FindbugsPlugin.isProjectSettingsEnabled(currProject);
187             if (!projectPropsInitiallyEnabled) {
188                 // use workspace properties instead
189                 currProject = null;
190             }
191             setPreferenceStore(projectStore);
192         } else {
193             setPreferenceStore(workspaceStore);
194         }
195         loadPreferences(currProject);
196     }

198     /**
199      * @param currProject
200      *            if null, workspace properties are used
201      */
202     private UserPreferences loadPreferences(IProject currProject) {
203         // Get current user preferences for project
204         if (currProject == null) {
205             origUserPreferences = FindbugsPlugin.getCorePreferences(null, true);
206         } else {
207             origUserPreferences = FindbugsPlugin.getProjectPreferences(currProject, true);
208         }
209         currentUserPreferences = origUserPreferences.clone();
210         return currentUserPreferences;
211     }

375                 String id = "FindBugsPreferencePage";
376                 int result = PreferencesUtil.createPreferenceDialogOn(getShell(), id, new String[] { id }, null).open();

445     /**
446      * Will be called when the user presses the OK button.
447      *
448      * @see IPreferencePage#performOk()
449      */
450     @Override
451     public boolean performOk() {
452         reportConfigurationTab.performOk();
453         boolean analysisSettingsChanged = false;
454         boolean reporterSettingsChanged = false;
455         boolean needRedisplayMarkers = false;
456         if (workspaceSettingsTab != null) {
457             workspaceSettingsTab.performOK();
458         }
459 
460         boolean pluginsChanged = false;
461         // Have user preferences for project changed?
462         // If so, write them to the user preferences file & re-run builder
463         if (!currentUserPreferences.equals(origUserPreferences)) {
464             pluginsChanged = !currentUserPreferences.getCustomPlugins().equals(origUserPreferences.getCustomPlugins());
465             // save only if we in the workspace page OR in the project page with
466             // enabled
467             // project settings
468             if (getProject() == null || enableProjectCheck.getSelection()) {
469                 try {
470                     FindbugsPlugin.saveUserPreferences(getProject(), currentUserPreferences);
471                 } catch (CoreException e) {
472                     FindbugsPlugin.getDefault().logException(e, "Could not store SpotBugs preferences for project");
473                 }
474             }
475             if(pluginsChanged) {
476                 FindbugsPlugin.applyCustomDetectors(true);
477             }
478         }
479 
480         analysisSettingsChanged = pluginsChanged || areAnalysisPrefsChanged(currentUserPreferences, origUserPreferences);
481 
482         reporterSettingsChanged = !currentUserPreferences.getFilterSettings().equals(origUserPreferences.getFilterSettings());
483 
484         boolean markerSeveritiesChanged = reportConfigurationTab.isMarkerSeveritiesChanged();
485 
486         needRedisplayMarkers = pluginsChanged || markerSeveritiesChanged || reporterSettingsChanged;
487 
488         boolean builderEnabled = false;
489 
490         if (getProject() != null) {
491             builderEnabled = chkEnableFindBugs.getSelection();
492             // Update whether or not FindBugs is run automatically.
493             if (!natureEnabled && builderEnabled) {
494                 addNature();
495             } else if (natureEnabled && !builderEnabled) {
496                 removeNature();
497             }
498 
499             // update the flag to match the incremental/not property
500             builderEnabled &= chkRunAtFullBuild.getSelection();
501             boolean newSelection = enableProjectCheck.getSelection();
502             if (projectPropsInitiallyEnabled != newSelection) {
503                 analysisSettingsChanged = true;
504                 FindbugsPlugin.setProjectSettingsEnabled(project, projectStore, newSelection);
505             }
506         }
507 
508         if (analysisSettingsChanged) {
509             // trigger a Findbugs rebuild here
510             if (builderEnabled) {
511                 runFindbugsBuilder();
512                 needRedisplayMarkers = false;
513             } else {
514                 if (!getPreferenceStore().getBoolean(FindBugsConstants.DONT_REMIND_ABOUT_FULL_BUILD)) {
515                     remindAboutFullBuild();
516                 }
517             }
518         }
519 
520         if (needRedisplayMarkers) {
521             redisplayMarkers();
522         }
523 
524         return true;
525     }

527     protected void remindAboutFullBuild() {
528         MessageDialogWithToggle dialog = MessageDialogWithToggle.openInformation(getShell(), "Full SpotBugs build required",
529                 "SpotBugs analysis settings have changed. Run SpotBugs analysis again to see updated results.",
530                 "Do not show this warning again", false, null, null);
531 
532         getPreferenceStore().setValue(FindBugsConstants.DONT_REMIND_ABOUT_FULL_BUILD, dialog.getToggleState());
533     }

535     private void redisplayMarkers() {
536         // if filter settings changed, and builder is not enabled, manually
537         // trigger update
538         if (project != null) {
539             MarkerUtil.redisplayMarkers(JavaCore.create(project));
540         } else {
541             // trigger redisplay for workbench change too
542             List<IProject> projects = ProjectUtilities.getFindBugsProjects();
543             for (IProject aProject : projects) {
544                 MarkerUtil.redisplayMarkers(JavaCore.create(aProject));
545             }
546         }
547     }

549     private boolean areAnalysisPrefsChanged(UserPreferences pref1, UserPreferences pref2) {
550         String effort1 = "" + pref1.getEffort();
551         String effort2 = pref2.getEffort();
552         return !effort1.equals(effort2) || isDetectorConfigurationChanged(pref1, pref2)
553                 || !pref1.getExcludeBugsFiles().equals(pref2.getExcludeBugsFiles())
554                 || !pref1.getExcludeFilterFiles().equals(pref2.getExcludeFilterFiles())
555                 || !pref1.getIncludeFilterFiles().equals(pref2.getIncludeFilterFiles());
556     }
557 
558     boolean isDetectorConfigurationChanged(UserPreferences pref1, UserPreferences pref2) {
559 
560         Iterator<DetectorFactory> iterator = DetectorFactoryCollection.instance().factoryIterator();
561         while (iterator.hasNext()) {
562             DetectorFactory factory = iterator.next();
563             // Only compare non-hidden factories
564             if (factory.isHidden() && !detectorTab.isHiddenVisible()) {
565                 continue;
566             }
567             if (pref1.isDetectorEnabled(factory) ^ pref2.isDetectorEnabled(factory)) {
568                 return true;
569             }
570         }
571         return false;
572     }

578     private void runFindbugsBuilder() {
579         IProject myProject = getProject();
580         if (myProject != null) {
581             runBuild(myProject);
582         } else {
583             // workspace settings change: trigger workspace build
584             List<IProject> projects = ProjectUtilities.getFindBugsProjects();
585             for (IProject iProject : projects) {
586                 runBuild(iProject);
587             }
588         }
589     }

619     private final class NatureWorker extends FindBugsJob {
620         private boolean add = true;
621 
622         public NatureWorker(boolean add) {
623             super((add ? "Adding SpotBugs nature to " : "Removing SpotBugs nature from ") + getProject(), getProject());
624             this.add = add;
625             // adding/removing nature uses workspace scope
626             setRule(ResourcesPlugin.getWorkspace().getRoot());
627         }
628 
629         @Override
630         protected void runWithProgress(IProgressMonitor monitor) throws CoreException {
631             if (add) {
632                 ProjectUtilities.addFindBugsNature(getProject(), monitor);
633             } else {
634                 ProjectUtilities.removeFindBugsNature(getProject(), monitor);
635             }
636         }
637     }
