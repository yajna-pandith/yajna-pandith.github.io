85      private static final class BugPatternTableSorter extends ViewerSorter implements Comparator<DetectorFactory> {

94          BugPatternTableSorter(DetectorConfigurationTab tab) {
95              this.tab = tab;
96          }

98          @Override
99          public int compare(Viewer viewer, Object e1, Object e2) {
100             return compare((DetectorFactory) e1, (DetectorFactory) e2);
101         }
102 
103         @Override
104         public int compare(DetectorFactory factory1, DetectorFactory factory2) {
105             int result = 0;
106             String s1, s2;
107             switch (getSortColumnId()) {
108             case BUG_CODES:
109                 s1 = tab.getBugsAbbreviation(factory1);
110                 s2 = tab.getBugsAbbreviation(factory2);
111                 break;
112             case DETECTOR_SPEED:
113                 s1 = factory1.getSpeed();
114                 s2 = factory2.getSpeed();
115                 break;
116             case PLUGIN:
117                 s1 = factory1.getPlugin().getPluginId();
118                 s2 = factory2.getPlugin().getPluginId();
119                 break;
120             case BUG_CATEGORIES:
121                 s1 = tab.getBugsCategories(factory1);
122                 s2 = tab.getBugsCategories(factory2);
123                 break;
124             case DETECTOR_NAME:
125             default:
126                 s1 = "" + factory1.getShortName(); //$NON-NLS-1$
127                 s2 = factory2.getShortName();
128                 break;
129             }
130             if (s1 == null) {
131                 s1 = "";
132             }
133             if (s2 == null) {
134                 s2 = "";
135             }
136             result = s1.compareTo(s2);
137 
138             // second sort if elements are equals - on only 2 criteria
139             if (result == 0) {
140                 switch (getSortColumnId()) {
141                 case DETECTOR_NAME:
142                     s1 = tab.getBugsAbbreviation(factory1);
143                     s2 = tab.getBugsAbbreviation(factory2);
144                     break;
145                 case BUG_CODES:
146                 default:
147                     s1 = "" + factory1.getShortName(); //$NON-NLS-1$
148                     s2 = factory2.getShortName();
149                     break;
150                 }
151                 result = s1.compareTo(s2);
152             } else if (revertOrder) {
153                 // same column selected twice - revert first order
154                 result = -Integer.signum(result);
155             }
156             return result;
157         }
158 
159         @Override
160         public boolean isSorterProperty(Object element, String property) {
161             return property.equals(COLUMN.DETECTOR_NAME.name()) || property.equals(COLUMN.BUG_CODES.name())
162                     || property.equals(COLUMN.DETECTOR_SPEED.name()) || property.equals(COLUMN.PLUGIN.name());
163         }
164 
165         /**
166          * @param columnId
167          *            The sortColumnId to set.
168          */
169         public void setSortColumnIndex(COLUMN columnId) {
170             this.lastSortColumnId = this.sortColumnId;
171             this.sortColumnId = columnId;
172             revertOrder = !revertOrder && lastSortColumnId == columnId;
173         }
174 
175         /**
176          * @return Returns the sortColumnId.
177          */
178         public COLUMN getSortColumnId() {
179             return sortColumnId;
180         }

280         /**
281          * Return whether or not given DetectorFactory reports bug patterns in
282          * one of the currently-enabled set of bug categories.
283          *
284          * @param factory
285          *            the DetectorFactory
286          * @return true if the factory reports bug patterns in one of the
287          *         currently-enabled bug categories, false if not
288          */
289         private boolean isFactoryVisible(DetectorFactory factory) {
290             Map<DetectorFactory, Boolean> enabledDetectors = tab.propertyPage.getVisibleDetectors();
291             Boolean enabled = enabledDetectors.get(factory);
292 
293             if (enabled != null) {
294                 return enabled.booleanValue();
295             }
296             ProjectFilterSettings filterSettings = tab.getCurrentProps().getFilterSettings();
297             for (BugPattern pattern : factory.getReportedBugPatterns()) {
298                 if (filterSettings.containsCategory(pattern.getCategory())) {
299                     enabledDetectors.put(factory, Boolean.TRUE);
300                     return true;
301                 }
302             }
303             enabledDetectors.put(factory, Boolean.FALSE);
304             return false;
305         }

309             return null;

389     private static String getDetailedText(DetectorFactory factory) {
390         if (factory == null) {
391             return "";
392         }
393         StringBuffer sb = new StringBuffer(factory.getFullName());
394         sb.append("\n");
395         sb.append(getDescriptionWithoutHtml(factory));
396         sb.append("\n\nReported patterns:\n");
397         Collection<BugPattern> patterns = factory.getReportedBugPatterns();
398         for (Iterator<BugPattern> iter = patterns.iterator(); iter.hasNext();) {
399             BugPattern pattern = iter.next();
400             sb.append(pattern.getType()).append(" ").append(" (").append(pattern.getAbbrev()).append(", ")
401                     .append(pattern.getCategory()).append("):").append("  ");
402             sb.append(pattern.getShortDescription());
403             if (iter.hasNext()) {
404                 sb.append("\n");
405             }
406         }
407         if (patterns.isEmpty()) {
408             sb.append("none");
409         }
410         sb.append(getPluginDescription(factory.getPlugin()));
411         return sb.toString();
412     }

414     private static String getPluginDescription(Plugin plugin) {
415         StringBuilder sb = new StringBuilder();
416         sb.append("\n\nPlugin: ").append(plugin.getPluginId());
417         String version = plugin.getVersion();
418         if (version.length() > 0) {
419             sb.append("\nVersion: ").append(version);
420         }
421 
422         sb.append("\nProvider: ").append(plugin.getProvider());
423         String website = plugin.getWebsite();
424         if (website != null && website.length() > 0) {
425             sb.append(" (").append(website).append(")");
426         }
427         return sb.toString();
428     }

430     /**
431      * Tries to trim all the html out of the
432      * {@link DetectorFactory#getDetailHTML()} return value. See also private
433      * {@link PluginLoader} .init() method.
434      */
435     private static String getDescriptionWithoutHtml(DetectorFactory factory) {
436         String detailHTML = factory.getDetailHTML();
437         // cut beginning and the end of the html document
438         detailHTML = trimHtml(detailHTML, "<BODY>", "</BODY>");
439         // replace any amount of white space with newline between through one
440         // space
441         detailHTML = detailHTML.replaceAll("\\s*[\\n]+\\s*", " ");
442         // remove all valid html tags
443         detailHTML = detailHTML.replaceAll("<[a-zA-Z]+>", "");
444         detailHTML = detailHTML.replaceAll("</[a-zA-Z]+>", "");
445         // convert some of the entities which are used in current FB
446         // messages.xml
447         detailHTML = detailHTML.replaceAll("&nbsp;", "");
448         detailHTML = detailHTML.replaceAll("&lt;", "<");
449         detailHTML = detailHTML.replaceAll("&gt;", ">");
450         detailHTML = detailHTML.replaceAll("&amp;", "&");
451         return detailHTML.trim();
452     }
453 
454     private static String trimHtml(String detailHTML, String startTag, String endTag) {
455         if (detailHTML.indexOf(startTag) > 0) {
456             detailHTML = detailHTML.substring(detailHTML.indexOf(startTag) + startTag.length());
457         }
458         if (detailHTML.indexOf(endTag) > 0) {
459             detailHTML = detailHTML.substring(0, detailHTML.lastIndexOf(endTag));
460         }
461         return detailHTML;
462     }

464     /**
465      * @param factory
466      * @return
467      */
468     private String getBugsCategories(DetectorFactory factory) {
469         Collection<BugPattern> patterns = factory.getReportedBugPatterns();
470         String category = null;
471         Set<String> categories = new TreeSet<>();
472         for (BugPattern bugPattern : patterns) {
473             String category2 = bugPattern.getCategory();
474             if (category == null) {
475                 category = category2;
476             } else if (!category.equals(category2)) {
477                 categories.add(category);
478                 categories.add(category2);
479             }
480         }
481         if (!categories.isEmpty()) {
482             StringBuilder sb = new StringBuilder();
483             for (String string : categories) {
484                 sb.append(I18N.instance().getBugCategoryDescription(string)).append("|");
485             }
486             category = sb.toString();
487         } else {
488             category = I18N.instance().getBugCategoryDescription(category);
489         }
490         return category;
491     }

493     void refreshUI(UserPreferences preferences) {
494         // Enable only those detectors that are enabled by preferences
495         TableItem[] itemList = availableFactoriesTableViewer.getTable().getItems();

502         syncUserPreferencesWithTable();
503     }

514         TableItem[] itemList = availableFactoriesTableViewer.getTable().getItems();

575         currentColumnIdx++;
576         TableColumn speedColumn = createColumn(currentColumnIdx, factoriesTable, getMessage("property.speed"), 70,
577                 COLUMN.DETECTOR_SPEED);
578         addColumnSelectionListener(sorter, speedColumn, COLUMN.DETECTOR_SPEED);
579 
580         currentColumnIdx++;
581         TableColumn pluginColumn = createColumn(currentColumnIdx, factoriesTable, getMessage("property.provider"), 100,
582                 COLUMN.PLUGIN);
583         addColumnSelectionListener(sorter, pluginColumn, COLUMN.PLUGIN);

585         currentColumnIdx++;
586         TableColumn categoryColumn = createColumn(currentColumnIdx, factoriesTable, getMessage("property.category"), 75,
587                 COLUMN.BUG_CATEGORIES);
588         addColumnSelectionListener(sorter, categoryColumn, COLUMN.BUG_CATEGORIES);
589 
590         factoriesTable.setLinesVisible(true);
591         factoriesTable.setHeaderVisible(true);
592         // initial sort indicator
593         factoriesTable.setSortDirection(sorter.revertOrder ? SWT.UP : SWT.DOWN);
594         factoriesTable.setSortColumn(factoryNameColumn);
595         sorter.setSortColumnIndex(COLUMN.DETECTOR_NAME);
596 
597         availableFactoriesTableViewer.setContentProvider(new DetectorFactoriesContentProvider());
598         availableFactoriesTableViewer.setLabelProvider(new DetectorFactoryLabelProvider(this));
599 
600         availableFactoriesTableViewer.setSorter(sorter);
601 
602         populateAvailableRulesTable(project);
603         factoriesTable.setEnabled(true);
604 
605         return factoriesTable;
606     }

640     /**
641      * Populate the rule table
642      */
643     private void populateAvailableRulesTable(IProject project) {
644         List<DetectorFactory> allAvailableList = new ArrayList<>();
645         factoriesToBugAbbrev = new HashMap<>();
646         Iterator<DetectorFactory> iterator = DetectorFactoryCollection.instance().factoryIterator();
647         while (iterator.hasNext()) {
648             DetectorFactory factory = iterator.next();
649 
650             // Only configure non-hidden factories
651             if (factory.isHidden() && !isHiddenVisible()) {
652                 continue;
653             }
654 
655             allAvailableList.add(factory);
656             addBugsAbbreviation(factory);
657         }
658 
659         availableFactoriesTableViewer.setInput(allAvailableList);
660         TableItem[] itemList = availableFactoriesTableViewer.getTable().getItems();
661         UserPreferences userPreferences = getCurrentProps();
662         for (int i = 0; i < itemList.length; i++) {
663             DetectorFactory rule = (DetectorFactory) itemList[i].getData();
664             // set enabled if defined in configuration
665             if (userPreferences.isDetectorEnabled(rule)) {
666                 itemList[i].setChecked(true);
667             }
668         }
669     }

682     protected String getBugsAbbreviation(DetectorFactory factory) {
683         String abbr = factoriesToBugAbbrev.get(factory);
684         if (abbr == null) {
685             abbr = createBugsAbbreviation(factory);
686         }
687         return abbr;
688     }

690     @Nonnull
691     private String createBugsAbbreviation(DetectorFactory factory) {
692         StringBuffer sb = new StringBuffer();
693         Collection<BugPattern> patterns = factory.getReportedBugPatterns();
694         LinkedHashSet<String> abbrs = new LinkedHashSet<>();
695         for (Iterator<BugPattern> iter = patterns.iterator(); iter.hasNext();) {
696             BugPattern pattern = iter.next();
697             String abbr = pattern.getAbbrev();
698             abbrs.add(abbr);
699         }
700         for (Iterator<String> iter = abbrs.iterator(); iter.hasNext();) {
701             String element = iter.next();
702             sb.append(element);
703             if (iter.hasNext()) {
704                 sb.append("|"); //$NON-NLS-1$
705             }
706         }
707         return sb.toString();
708     }
