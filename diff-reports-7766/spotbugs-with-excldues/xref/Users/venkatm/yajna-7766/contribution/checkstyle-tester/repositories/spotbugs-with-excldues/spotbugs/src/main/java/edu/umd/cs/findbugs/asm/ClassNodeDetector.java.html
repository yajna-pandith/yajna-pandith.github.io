20  package edu.umd.cs.findbugs.asm;
21  
22  import java.util.ArrayList;
23  
24  import org.objectweb.asm.tree.ClassNode;
25  
26  import edu.umd.cs.findbugs.BugReporter;
27  import edu.umd.cs.findbugs.Detector2;
28  import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;
29  import edu.umd.cs.findbugs.classfile.ClassDescriptor;
30  import edu.umd.cs.findbugs.classfile.Global;
31  import edu.umd.cs.findbugs.classfile.engine.asm.FindBugsASM;
32  
33  /**
34   * Abstract base class to to reduce boilerplate needed for writing ASM-based
35   * Detectors implemented as ClassNode visitors
36   *
37   * @author pugh
38   */
39  abstract public class ClassNodeDetector extends ClassNode implements Detector2 {
40  
41      protected final BugReporter bugReporter;
42  
43      /**
44       * Construct a ClassNodeDetector. The bugReporter is passed to the
45       * constructor and stored in a protected final field.
46       *
47       * @param bugReporter
48       *            the BugReporter that bug should be reporter to.
49       */
50      public ClassNodeDetector(BugReporter bugReporter) {
51          super(FindBugsASM.ASM_VERSION);
52          this.bugReporter = bugReporter;
53      }
54  
55      @Override
56      public String getDetectorClassName() {
57          return this.getClass().getName();
58      }
59  
60      @Override
61      public void visitClass(ClassDescriptor classDescriptor) throws CheckedAnalysisException {
62  
63          FBClassReader cr = Global.getAnalysisCache().getClassAnalysis(FBClassReader.class, classDescriptor);
64          this.interfaces = new ArrayList<>();
65          this.innerClasses = new ArrayList<>();
66          this.fields = new ArrayList<>();
67          this.methods = new ArrayList<>();
68          cr.accept(this, 0);
69      }
70  
71      @Override
72      public void finishPass() {
73          // do nothing
74      }
75  
76  }
