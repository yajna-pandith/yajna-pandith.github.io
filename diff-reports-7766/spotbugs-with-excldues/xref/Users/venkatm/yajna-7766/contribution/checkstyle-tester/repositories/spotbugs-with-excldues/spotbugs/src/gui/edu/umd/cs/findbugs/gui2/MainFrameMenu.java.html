66      JMenuItem createRecentItem(final File f, final SaveType localSaveType) {
67          if (MainFrame.GUI2_DEBUG) {
68              System.out.println("createRecentItem(" + f + ", " + localSaveType + ")");
69          }
70          String name = f.getName();
71  
72          final JMenuItem item = new JMenuItem(name);
73          item.addActionListener(e -> {
74              try {
75                  mainFrame.setCursor(new Cursor(Cursor.WAIT_CURSOR));
76  
77                  if (!f.exists()) {
78                      JOptionPane.showMessageDialog(null,
79                              L10N.getLocalString("msg.proj_not_found", "This project can no longer be found"));
80                      GUISaveState.getInstance().fileNotFound(f);
81                      return;
82                  }
83                  GUISaveState.getInstance().fileReused(f);// Move to front in
84                  // GUISaveState, so
85                  // it will be last
86                  // thing to be
87                  // removed from the
88                  // list
89  
90                  recentMenuCache.addRecentFile(f);
91  
92                  if (!f.exists()) {
93                      throw new IllegalStateException("User used a recent projects menu item that didn't exist.");
94                  }
95  
96                  // Moved this outside of the thread, and above the line
97                  // saveFile=f.getParentFile()
98                  // Since if this save goes on in the thread below, there is
99                  // no way to stop the save from
100                 // overwriting the files we are about to load.
101                 if (mainFrame.getCurProject() != null && mainFrame.isProjectChanged()) {
102                     int response = JOptionPane.showConfirmDialog(mainFrame, L10N.getLocalString("dlg.save_current_changes",
103                             "The current project has been changed, Save current changes?"), L10N.getLocalString(
104                                     "dlg.save_changes", "Save Changes?"), JOptionPane.YES_NO_CANCEL_OPTION,
105                             JOptionPane.WARNING_MESSAGE);
106 
107                     if (response == JOptionPane.YES_OPTION) {
108                         if (mainFrame.getSaveFile() != null) {
109                             mainFrame.getMainFrameLoadSaveHelper().save();
110                         } else {
111                             mainFrame.getMainFrameLoadSaveHelper().saveAs();
112                         }
113                     } else if (response == JOptionPane.CANCEL_OPTION)
114                     {
115                         return;
116                         // IF no, do nothing.
117                     }
118                 }
119 
120                 SaveType st = SaveType.forFile(f);
121                 boolean result = true;
122                 switch (st) {
123                 case XML_ANALYSIS:
124                     result = mainFrame.openAnalysis(f, st);
125                     break;
126                 case FBP_FILE:
127                     result = mainFrame.getMainFrameLoadSaveHelper().openFBPFile(f);
128                     break;
129                 case FBA_FILE:
130                     result = mainFrame.getMainFrameLoadSaveHelper().openFBAFile(f);
131                     break;
132                 default:
133                     mainFrame.error("Wrong file type in recent menu item.");
134                 }
135 
136                 if (!result) {
137                     JOptionPane.showMessageDialog(MainFrame.getInstance(), "There was an error in opening the file",
138                             "Recent Menu Opening Error", JOptionPane.WARNING_MESSAGE);
139                 }
140             } finally {
141                 mainFrame.setCursor(new Cursor(Cursor.DEFAULT_CURSOR));
142                 mainFrame.setSaveType(localSaveType);
143             }
144         });
145         item.setFont(item.getFont().deriveFont(Driver.getFontSize()));
146         return item;
147     }

329     void setViewMenu() {
330 
331         viewMenu.removeAll();
332         viewMenu.add(groupByMenuItem);
333         if (mainFrame.getProjectPackagePrefixes().size() > 0 && mainFrame.getBugCollection() != null) {
334             JMenuItem selectPackagePrefixMenu = new JMenuItem("Select class search strings by project...");
335             selectPackagePrefixMenu.addActionListener(e -> mainFrame.selectPackagePrefixByProject());
336             viewMenu.add(selectPackagePrefixMenu);
337 
338         }
339         if (viewMenu.getItemCount() > 0) {
340             viewMenu.addSeparator();
341         }
342 
343         ButtonGroup rankButtonGroup = new ButtonGroup();
344         for (final ViewFilter.RankFilter r : ViewFilter.RankFilter.values()) {
345             JRadioButtonMenuItem rbMenuItem = new JRadioButtonMenuItem(r.toString());
346             AnnotatedString.localiseButton(rbMenuItem, "menu.rankFilter_"+r.name().toLowerCase(Locale.ENGLISH), r.toString(), true);
347             rankButtonGroup.add(rbMenuItem);
348             if (r == ViewFilter.RankFilter.ALL) {
349                 rbMenuItem.setSelected(true);
350             }
351             rbMenuItem.addActionListener(e -> {
352                 mainFrame.getViewFilter().setRank(r);
353                 mainFrame.resetViewCache();
354             });
355             viewMenu.add(rbMenuItem);
356         }
357 
358         viewMenu.addSeparator();
359 
360         ButtonGroup priorityButtonGroup = new ButtonGroup();
361         for (final ViewFilter.PriorityFilter r : ViewFilter.PriorityFilter.values()) {
362             JRadioButtonMenuItem rbMenuItem = new JRadioButtonMenuItem(r.toString());
363             AnnotatedString.localiseButton(rbMenuItem, "menu.priorityFilter_"+r.name().toLowerCase(Locale.ENGLISH), r.toString(), true);
364             priorityButtonGroup.add(rbMenuItem);
365             if (r == ViewFilter.PriorityFilter.ALL_BUGS) {
366                 rbMenuItem.setSelected(true);
367             }
368             rbMenuItem.addActionListener(e -> {
369                 mainFrame.getViewFilter().setPriority(r);
370                 mainFrame.resetViewCache();
371             });
372             viewMenu.add(rbMenuItem);
373         }
374 
375         viewMenu.addSeparator();
376         final Filter suppressionFilter = MainFrame.getInstance().getProject().getSuppressionFilter();
377         Collection<Matcher> filters = suppressionFilter.getChildren();
378         JMenuItem filterMenu = new JMenuItem(filters.isEmpty() ? "Add Filters..." : "Filters...");
379 
380         filterMenu.addActionListener(e -> {
381             PreferencesFrame preferenceFrame = PreferencesFrame.getInstance();
382             preferenceFrame.showFilterPane();
383             preferenceFrame.setLocationRelativeTo(mainFrame);
384             preferenceFrame.setVisible(true);
385 
386 
387         });
388         viewMenu.add(filterMenu);
389         for(final Matcher m : filters) {
390             JCheckBoxMenuItem f = new JCheckBoxMenuItem(m.toString(), suppressionFilter.isEnabled(m));
391             viewMenu.add(f);
392             f.addItemListener(e -> {
393                 boolean enabled = e.getStateChange() == ItemEvent.SELECTED;
394                 suppressionFilter.setEnabled(m, enabled);
395                 FilterActivity.notifyListeners(enabled ? Action.FILTERING : Action.UNFILTERING, null);
396 
397             });
398 
399 
400         }

405     /**
406      * This checks if the xmlFile is in the GUISaveState. If not adds it. Then
407      * adds the file to the recentMenuCache.
408      *
409      * @param xmlFile
410      */
411     /*
412      * If the file already existed, its already in the preferences, as well as
413      * the recent projects menu items, only add it if they change the name,
414      * otherwise everything we're storing is still accurate since all we're
415      * storing is the location of the file.
416      */
417     public void addFileToRecent(File xmlFile) {
418         ArrayList<File> xmlFiles = GUISaveState.getInstance().getRecentFiles();
419         if (!xmlFiles.contains(xmlFile)) {
420             GUISaveState.getInstance().addRecentFile(xmlFile);
421         }
422         this.recentMenuCache.addRecentFile(xmlFile);
423     }

464     public void initOSX() throws ClassNotFoundException, NoSuchMethodException, InvocationTargetException, IllegalAccessException {
465         osxAdapter = Class.forName("edu.umd.cs.findbugs.gui2.OSXAdapter");
466         Method registerMethod = osxAdapter.getDeclaredMethod("registerMacOSXApplication", MainFrame.class);
467         if (registerMethod != null) {
468             registerMethod.invoke(osxAdapter, mainFrame);
469         }
470         osxPrefsEnableMethod = osxAdapter.getDeclaredMethod("enablePrefs", boolean.class);
471     }

482         saveMenuItem.setEnabled(haveBugs);
483         saveAsMenuItem.setEnabled(haveBugs);
484         reconfigMenuItem.setEnabled(haveBugs);
485         groupByMenuItem.setEnabled(haveBugs);
486     }
