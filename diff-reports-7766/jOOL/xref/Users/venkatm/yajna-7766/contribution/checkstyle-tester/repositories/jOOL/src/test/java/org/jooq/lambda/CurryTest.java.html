16  package org.jooq.lambda;
17  
18  import static org.jooq.lambda.tuple.Tuple.tuple;
19  import static org.junit.Assert.assertEquals;
20  
21  import org.jooq.lambda.function.Function5;
22  import org.jooq.lambda.tuple.Tuple2;
23  import org.jooq.lambda.tuple.Tuple3;
24  
25  import org.junit.Test;
26  
27  public class CurryTest {
28  
29      @Test
30      public void testFunction5to3() {
31          Tuple2<Integer, Integer> t1 = tuple(4, 4);
32          Tuple3<Integer, Integer, Integer> t2 = tuple(5, 3, 2);
33  
34          // Concat the two and three tuples and apply them together.
35          int normal1 = lift(this::fiveArgMethod).apply(t1.concat(t2));
36  
37          // Curry with the first two values, then apply with the remaining three
38          int curriedExplicitExplicit = lift(this::fiveArgMethod).curry(t1.v1, t1.v2).apply(t2.v1, t2.v2, t2.v3);
39          int curriedExplicitTuple = lift(this::fiveArgMethod).curry(t1.v1, t1.v2).apply(t2);
40          int curriedTupleExplicit = lift(this::fiveArgMethod).curry(t1).apply(t2.v1, t2.v2, t2.v3);
41          int curriedTupleTuple = lift(this::fiveArgMethod).curry(t1).apply(t2);
42  
43          assertEquals(normal1, curriedExplicitExplicit);
44          assertEquals(normal1, curriedExplicitTuple);
45          assertEquals(normal1, curriedTupleExplicit);
46          assertEquals(normal1, curriedTupleTuple);
47      }
48  
49      private <A, B, C, D, E, F> Function5<A, B, C, D, E, F> lift(Function5<A, B, C, D, E, F> func) {
50          return func;
51      }
52  
53      private int fiveArgMethod(int a, int b, int c, int d, int e) {
54          return a + b * c / d - e;
55      }
56  }
