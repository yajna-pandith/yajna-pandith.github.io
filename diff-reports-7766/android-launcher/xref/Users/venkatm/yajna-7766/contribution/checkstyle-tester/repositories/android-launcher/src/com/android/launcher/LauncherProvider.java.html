88      @Override
89      public String getType(Uri uri) {
90          SqlArguments args = new SqlArguments(uri, null, null);
91          if (TextUtils.isEmpty(args.where)) {
92              return "vnd.android.cursor.dir/" + args.table;
93          } else {
94              return "vnd.android.cursor.item/" + args.table;
95          }
96      }

98      @Override
99      public Cursor query(Uri uri, String[] projection, String selection,
100             String[] selectionArgs, String sortOrder) {
101 
102         SqlArguments args = new SqlArguments(uri, selection, selectionArgs);
103         SQLiteQueryBuilder qb = new SQLiteQueryBuilder();
104         qb.setTables(args.table);
105 
106         SQLiteDatabase db = mOpenHelper.getWritableDatabase();
107         Cursor result = qb.query(db, projection, args.where, args.args, null, null, sortOrder);
108         result.setNotificationUri(getContext().getContentResolver(), uri);
109 
110         return result;
111     }
112 
113     @Override
114     public Uri insert(Uri uri, ContentValues initialValues) {
115         SqlArguments args = new SqlArguments(uri);
116 
117         SQLiteDatabase db = mOpenHelper.getWritableDatabase();
118         final long rowId = db.insert(args.table, null, initialValues);
119         if (rowId <= 0) return null;
120 
121         uri = ContentUris.withAppendedId(uri, rowId);
122         sendNotify(uri);
123 
124         return uri;
125     }
126 
127     @Override
128     public int bulkInsert(Uri uri, ContentValues[] values) {
129         SqlArguments args = new SqlArguments(uri);
130 
131         SQLiteDatabase db = mOpenHelper.getWritableDatabase();
132         db.beginTransaction();
133         try {
134             int numValues = values.length;
135             for (int i = 0; i < numValues; i++) {
136                 if (db.insert(args.table, null, values[i]) < 0) return 0;
137             }
138             db.setTransactionSuccessful();
139         } finally {
140             db.endTransaction();
141         }
142 
143         sendNotify(uri);
144         return values.length;
145     }
146 
147     @Override
148     public int delete(Uri uri, String selection, String[] selectionArgs) {
149         SqlArguments args = new SqlArguments(uri, selection, selectionArgs);
150 
151         SQLiteDatabase db = mOpenHelper.getWritableDatabase();
152         int count = db.delete(args.table, args.where, args.args);
153         if (count > 0) sendNotify(uri);
154 
155         return count;
156     }
157 
158     @Override
159     public int update(Uri uri, ContentValues values, String selection, String[] selectionArgs) {
160         SqlArguments args = new SqlArguments(uri, selection, selectionArgs);
161 
162         SQLiteDatabase db = mOpenHelper.getWritableDatabase();
163         int count = db.update(args.table, values, args.where, args.args);
164         if (count > 0) sendNotify(uri);
165 
166         return count;
167     }
168 
169     private void sendNotify(Uri uri) {
170         String notify = uri.getQueryParameter(PARAMETER_NOTIFY);
171         if (notify == null || "true".equals(notify)) {
172             getContext().getContentResolver().notifyChange(uri, null);
173         }
174     }

251         private boolean convertDatabase(SQLiteDatabase db) {
252             if (LOGD) Log.d(LOG_TAG, "converting database from an older format, but not onUpgrade");
253             boolean converted = false;
254 
255             final Uri uri = Uri.parse("content://" + Settings.AUTHORITY +
256                     "/old_favorites?notify=true");
257             final ContentResolver resolver = mContext.getContentResolver();
258             Cursor cursor = null;
259 
260             try {
261                 cursor = resolver.query(uri, null, null, null, null);
262             } catch (Exception e) {
263 	            // Ignore
264             }
265 
266             // We already have a favorites database in the old provider
267             if (cursor != null && cursor.getCount() > 0) {
268                 try {
269                     converted = copyFromCursor(db, cursor) > 0;
270                 } finally {
271                     cursor.close();
272                 }
273 
274                 if (converted) {
275                     resolver.delete(uri, null, null);
276                 }
277             }
278             
279             if (converted) {
280                 // Convert widgets from this import into widgets
281                 if (LOGD) Log.d(LOG_TAG, "converted and now triggering widget upgrade");
282                 convertWidgets(db);
283             }
284 
285             return converted;
286         }

288         private int copyFromCursor(SQLiteDatabase db, Cursor c) {
289             final int idIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites._ID);
290             final int intentIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.INTENT);
291             final int titleIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.TITLE);
292             final int iconTypeIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.ICON_TYPE);
293             final int iconIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.ICON);
294             final int iconPackageIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.ICON_PACKAGE);
295             final int iconResourceIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.ICON_RESOURCE);
296             final int containerIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.CONTAINER);
297             final int itemTypeIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.ITEM_TYPE);
298             final int screenIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.SCREEN);
299             final int cellXIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.CELLX);
300             final int cellYIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.CELLY);
301             final int uriIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.URI);
302             final int displayModeIndex = c.getColumnIndexOrThrow(LauncherSettings.Favorites.DISPLAY_MODE);
303 
304             ContentValues[] rows = new ContentValues[c.getCount()];
305             int i = 0;
306             while (c.moveToNext()) {
307                 ContentValues values = new ContentValues(c.getColumnCount());
308                 values.put(LauncherSettings.Favorites._ID, c.getLong(idIndex));
309                 values.put(LauncherSettings.Favorites.INTENT, c.getString(intentIndex));
310                 values.put(LauncherSettings.Favorites.TITLE, c.getString(titleIndex));
311                 values.put(LauncherSettings.Favorites.ICON_TYPE, c.getInt(iconTypeIndex));
312                 values.put(LauncherSettings.Favorites.ICON, c.getBlob(iconIndex));
313                 values.put(LauncherSettings.Favorites.ICON_PACKAGE, c.getString(iconPackageIndex));
314                 values.put(LauncherSettings.Favorites.ICON_RESOURCE, c.getString(iconResourceIndex));
315                 values.put(LauncherSettings.Favorites.CONTAINER, c.getInt(containerIndex));
316                 values.put(LauncherSettings.Favorites.ITEM_TYPE, c.getInt(itemTypeIndex));
317                 values.put(LauncherSettings.Favorites.APPWIDGET_ID, -1);
318                 values.put(LauncherSettings.Favorites.SCREEN, c.getInt(screenIndex));
319                 values.put(LauncherSettings.Favorites.CELLX, c.getInt(cellXIndex));
320                 values.put(LauncherSettings.Favorites.CELLY, c.getInt(cellYIndex));
321                 values.put(LauncherSettings.Favorites.URI, c.getString(uriIndex));
322                 values.put(LauncherSettings.Favorites.DISPLAY_MODE, c.getInt(displayModeIndex));
323                 rows[i++] = values;
324             }
325 
326             db.beginTransaction();
327             int total = 0;
328             try {
329                 int numValues = rows.length;
330                 for (i = 0; i < numValues; i++) {
331                     if (db.insert(TABLE_FAVORITES, null, rows[i]) < 0) {
332                         return 0;
333                     } else {
334                         total++;
335                     }
336                 }
337                 db.setTransactionSuccessful();
338             } finally {
339                 db.endTransaction();
340             }
341 
342             return total;
343         }

345         @Override
346         public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
347             if (LOGD) Log.d(LOG_TAG, "onUpgrade triggered");
348             
349             int version = oldVersion;
350             if (version < 3) {
351                 // upgrade 1,2 -> 3 added appWidgetId column
352                 db.beginTransaction();
353                 try {
354                     // Insert new column for holding appWidgetIds
355                     db.execSQL("ALTER TABLE favorites " +
356                         "ADD COLUMN appWidgetId INTEGER NOT NULL DEFAULT -1;");
357                     db.setTransactionSuccessful();
358                     version = 3;
359                 } catch (SQLException ex) {
360                     // Old version remains, which means we wipe old data
361                     Log.e(LOG_TAG, ex.getMessage(), ex);
362                 } finally {
363                     db.endTransaction();
364                 }
365                 
366                 // Convert existing widgets only if table upgrade was successful
367                 if (version == 3) {
368                     convertWidgets(db);
369                 }
370             }
371 
372             if (version < 4) {
373                 version = 4;
374             }
375 
376             if (version < 5) {
377                 if (updateContactsShortcuts(db)) {
378                     version = 5;
379                 }
380             }
381             
382             if (version != DATABASE_VERSION) {
383                 Log.w(LOG_TAG, "Destroying all old data.");
384                 db.execSQL("DROP TABLE IF EXISTS " + TABLE_FAVORITES);
385                 onCreate(db);
386             }
387         }

389         private boolean updateContactsShortcuts(SQLiteDatabase db) {
390             Cursor c = null;
391             final String selectWhere = buildOrWhereString(LauncherSettings.Favorites.ITEM_TYPE,
392                     new int[] { LauncherSettings.Favorites.ITEM_TYPE_SHORTCUT });
393 
394             db.beginTransaction();
395             try {
396                 // Select and iterate through each matching widget
397                 c = db.query(TABLE_FAVORITES, new String[] { LauncherSettings.Favorites._ID,
398                         LauncherSettings.Favorites.INTENT }, selectWhere, null, null, null, null);
399                 
400                 if (LOGD) Log.d(LOG_TAG, "found upgrade cursor count=" + c.getCount());
401                 
402                 final ContentValues values = new ContentValues();
403                 final int idIndex = c.getColumnIndex(LauncherSettings.Favorites._ID);
404                 final int intentIndex = c.getColumnIndex(LauncherSettings.Favorites.INTENT);
405                 
406                 while (c != null && c.moveToNext()) {
407                     long favoriteId = c.getLong(idIndex);
408                     final String intentUri = c.getString(intentIndex);
409                     if (intentUri != null) {
410                         try {
411                             Intent intent = Intent.parseUri(intentUri, 0);
412                             android.util.Log.d("Home", intent.toString());
413                             final Uri uri = intent.getData();
414                             final String data = uri.toString();
415                             if (Intent.ACTION_VIEW.equals(intent.getAction()) &&
416                                     (data.startsWith("content://contacts/people/") ||
417                                     data.startsWith("content://com.android.contacts/contacts/lookup/"))) {
418 
419                                 intent = new Intent("com.android.contacts.action.QUICK_CONTACT");
420                                 intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK |
421                                         Intent.FLAG_ACTIVITY_CLEAR_TOP |
422                                         Intent.FLAG_ACTIVITY_RESET_TASK_IF_NEEDED);
423 
424                                 intent.setData(uri);
425                                 intent.putExtra("mode", 3);
426                                 intent.putExtra("exclude_mimes", (String[]) null);
427 
428                                 values.clear();
429                                 values.put(LauncherSettings.Favorites.INTENT, intent.toUri(0));
430     
431                                 String updateWhere = LauncherSettings.Favorites._ID + "=" + favoriteId;
432                                 db.update(TABLE_FAVORITES, values, updateWhere, null);                                
433                             }
434                         } catch (RuntimeException ex) {
435                             Log.e(LOG_TAG, "Problem upgrading shortcut", ex);
436                         } catch (URISyntaxException e) {
437                             Log.e(LOG_TAG, "Problem upgrading shortcut", e);                            
438                         }
439                     }
440                 }
441                 
442                 db.setTransactionSuccessful();
443             } catch (SQLException ex) {
444                 Log.w(LOG_TAG, "Problem while upgrading contacts", ex);
445                 return false;
446             } finally {
447                 db.endTransaction();
448                 if (c != null) {
449                     c.close();
450                 }
451             }
452 
453             return true;
454         }

457         /**
458          * Upgrade existing clock and photo frame widgets into their new widget
459          * equivalents. This method allocates appWidgetIds, and then hands off to
460          * LauncherAppWidgetBinder to finish the actual binding.
461          */
462         private void convertWidgets(SQLiteDatabase db) {
463             final int[] bindSources = new int[] {
464                     Favorites.ITEM_TYPE_WIDGET_CLOCK,
465                     Favorites.ITEM_TYPE_WIDGET_PHOTO_FRAME,
466             };
467             
468             final ArrayList<ComponentName> bindTargets = new ArrayList<ComponentName>();
469             bindTargets.add(new ComponentName("com.android.alarmclock",
470                     "com.android.alarmclock.AnalogAppWidgetProvider"));
471             bindTargets.add(new ComponentName("com.android.camera",
472                     "com.android.camera.PhotoAppWidgetProvider"));
473             
474             final String selectWhere = buildOrWhereString(Favorites.ITEM_TYPE, bindSources);
475             
476             Cursor c = null;
477             boolean allocatedAppWidgets = false;
478             
479             db.beginTransaction();
480             try {
481                 // Select and iterate through each matching widget
482                 c = db.query(TABLE_FAVORITES, new String[] { Favorites._ID },
483                         selectWhere, null, null, null, null);
484                 
485                 if (LOGD) Log.d(LOG_TAG, "found upgrade cursor count="+c.getCount());
486                 
487                 final ContentValues values = new ContentValues();
488                 while (c != null && c.moveToNext()) {
489                     long favoriteId = c.getLong(0);
490                     
491                     // Allocate and update database with new appWidgetId
492                     try {
493                         int appWidgetId = mAppWidgetHost.allocateAppWidgetId();
494                         
495                         if (LOGD) Log.d(LOG_TAG, "allocated appWidgetId="+appWidgetId+" for favoriteId="+favoriteId);
496                         
497                         values.clear();
498                         values.put(LauncherSettings.Favorites.APPWIDGET_ID, appWidgetId);
499                         
500                         // Original widgets might not have valid spans when upgrading
501                         values.put(LauncherSettings.Favorites.SPANX, 2);
502                         values.put(LauncherSettings.Favorites.SPANY, 2);
503 
504                         String updateWhere = Favorites._ID + "=" + favoriteId;
505                         db.update(TABLE_FAVORITES, values, updateWhere, null);
506                         
507                         allocatedAppWidgets = true;
508                     } catch (RuntimeException ex) {
509                         Log.e(LOG_TAG, "Problem allocating appWidgetId", ex);
510                     }
511                 }
512                 
513                 db.setTransactionSuccessful();
514             } catch (SQLException ex) {
515                 Log.w(LOG_TAG, "Problem while allocating appWidgetIds for existing widgets", ex);
516             } finally {
517                 db.endTransaction();
518                 if (c != null) {
519                     c.close();
520                 }
521             }
522             
523             // If any appWidgetIds allocated, then launch over to binder
524             if (allocatedAppWidgets) {
525                 launchAppWidgetBinder(bindSources, bindTargets);
526             }
527         }
528 
529         /**
530          * Launch the widget binder that walks through the Launcher database,
531          * binding any matching widgets to the corresponding targets. We can't
532          * bind ourselves because our parent process can't obtain the
533          * BIND_APPWIDGET permission.
534          */
535         private void launchAppWidgetBinder(int[] bindSources, ArrayList<ComponentName> bindTargets) {
536             final Intent intent = new Intent();
537             intent.setComponent(new ComponentName("com.android.settings",
538                     "com.android.settings.LauncherAppWidgetBinder"));
539             intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
540             
541             final Bundle extras = new Bundle();
542             extras.putIntArray(EXTRA_BIND_SOURCES, bindSources);
543             extras.putParcelableArrayList(EXTRA_BIND_TARGETS, bindTargets);
544             intent.putExtras(extras);
545             
546             mContext.startActivity(intent);
547         }

549         /**
550          * Loads the default set of favorite packages from an xml file.
551          *
552          * @param db The database to write the values into
553          */
554         private int loadFavorites(SQLiteDatabase db) {
555             Intent intent = new Intent(Intent.ACTION_MAIN, null);
556             intent.addCategory(Intent.CATEGORY_LAUNCHER);
557             ContentValues values = new ContentValues();
558 
559             PackageManager packageManager = mContext.getPackageManager();
560             int i = 0;
561             try {
562                 XmlResourceParser parser = mContext.getResources().getXml(R.xml.default_workspace);
563                 AttributeSet attrs = Xml.asAttributeSet(parser);
564                 XmlUtils.beginDocument(parser, TAG_FAVORITES);
565 
566                 final int depth = parser.getDepth();
567 
568                 int type;
569                 while (((type = parser.next()) != XmlPullParser.END_TAG ||
570                         parser.getDepth() > depth) && type != XmlPullParser.END_DOCUMENT) {
571 
572                     if (type != XmlPullParser.START_TAG) {
573                         continue;
574                     }
575 
576                     boolean added = false;
577                     final String name = parser.getName();
578 
579                     TypedArray a = mContext.obtainStyledAttributes(attrs, R.styleable.Favorite);
580 
581                     values.clear();                    
582                     values.put(LauncherSettings.Favorites.CONTAINER,
583                             LauncherSettings.Favorites.CONTAINER_DESKTOP);
584                     values.put(LauncherSettings.Favorites.SCREEN,
585                             a.getString(R.styleable.Favorite_screen));
586                     values.put(LauncherSettings.Favorites.CELLX,
587                             a.getString(R.styleable.Favorite_x));
588                     values.put(LauncherSettings.Favorites.CELLY,
589                             a.getString(R.styleable.Favorite_y));
590 
591                     if (TAG_FAVORITE.equals(name)) {
592                         added = addAppShortcut(db, values, a, packageManager, intent);
593                     } else if (TAG_SEARCH.equals(name)) {
594                         added = addSearchWidget(db, values);
595                     } else if (TAG_CLOCK.equals(name)) {
596                         added = addClockWidget(db, values);
597                     } else if (TAG_SHORTCUT.equals(name)) {
598                         added = addShortcut(db, values, a);
599                     }
600 
601                     if (added) i++;
602 
603                     a.recycle();
604                 }
605             } catch (XmlPullParserException e) {
606                 Log.w(LOG_TAG, "Got exception parsing favorites.", e);
607             } catch (IOException e) {
608                 Log.w(LOG_TAG, "Got exception parsing favorites.", e);
609             }
610 
611             return i;
612         }

614         private boolean addAppShortcut(SQLiteDatabase db, ContentValues values, TypedArray a,
615                 PackageManager packageManager, Intent intent) {
616 
617             ActivityInfo info;
618             String packageName = a.getString(R.styleable.Favorite_packageName);
619             String className = a.getString(R.styleable.Favorite_className);
620             try {
621                 ComponentName cn = new ComponentName(packageName, className);
622                 info = packageManager.getActivityInfo(cn, 0);
623                 intent.setComponent(cn);
624                 intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK
625                         | Intent.FLAG_ACTIVITY_RESET_TASK_IF_NEEDED);
626                 values.put(Favorites.INTENT, intent.toUri(0));
627                 values.put(Favorites.TITLE, info.loadLabel(packageManager).toString());
628                 values.put(Favorites.ITEM_TYPE, Favorites.ITEM_TYPE_APPLICATION);
629                 values.put(Favorites.SPANX, 1);
630                 values.put(Favorites.SPANY, 1);
631                 db.insert(TABLE_FAVORITES, null, values);
632             } catch (PackageManager.NameNotFoundException e) {
633                 Log.w(LOG_TAG, "Unable to add favorite: " + packageName +
634                         "/" + className, e);
635                 return false;
636             }
637             return true;
638         }

640         private boolean addShortcut(SQLiteDatabase db, ContentValues values, TypedArray a) {
641             Resources r = mContext.getResources();
642             
643             final int iconResId = a.getResourceId(R.styleable.Favorite_icon, 0);
644             final int titleResId = a.getResourceId(R.styleable.Favorite_title, 0);
645             
646             Intent intent;
647             String uri = null;
648             try {
649                 uri = a.getString(R.styleable.Favorite_uri);
650                 intent = Intent.parseUri(uri, 0);
651             } catch (URISyntaxException e) {
652                 w(LauncherModel.LOG_TAG, "Shortcut has malformed uri: " + uri);
653                 return false; // Oh well
654             }
655             
656             if (iconResId == 0 || titleResId == 0) {
657                 w(LauncherModel.LOG_TAG, "Shortcut is missing title or icon resource ID");
658                 return false;
659             }
660             
661             intent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
662             values.put(Favorites.INTENT, intent.toUri(0));
663             values.put(Favorites.TITLE, r.getString(titleResId));
664             values.put(Favorites.ITEM_TYPE, Favorites.ITEM_TYPE_SHORTCUT);
665             values.put(Favorites.SPANX, 1);
666             values.put(Favorites.SPANY, 1);
667             values.put(Favorites.ICON_TYPE, Favorites.ICON_TYPE_RESOURCE);
668             values.put(Favorites.ICON_PACKAGE, mContext.getPackageName());
669             values.put(Favorites.ICON_RESOURCE, mContext.getResources().getResourceName(iconResId));
670 
671             db.insert(TABLE_FAVORITES, null, values);
672 
673             return true;
674         }

715             }
716 
717             return allocatedAppWidgets;

721     /**
722      * Build a query string that will match any row where the column matches
723      * anything in the values list.
724      */
725     static String buildOrWhereString(String column, int[] values) {
726         StringBuilder selectWhere = new StringBuilder();
727         for (int i = values.length - 1; i >= 0; i--) {
728             selectWhere.append(column).append("=").append(values[i]);
729             if (i > 0) {
730                 selectWhere.append(" OR ");
731             }
732         }
733         return selectWhere.toString();
734     }

736     static class SqlArguments {
737         public final String table;
738         public final String where;
739         public final String[] args;
740 
741         SqlArguments(Uri url, String where, String[] args) {
742             if (url.getPathSegments().size() == 1) {
743                 this.table = url.getPathSegments().get(0);
744                 this.where = where;
745                 this.args = args;
746             } else if (url.getPathSegments().size() != 2) {
747                 throw new IllegalArgumentException("Invalid URI: " + url);
748             } else if (!TextUtils.isEmpty(where)) {
749                 throw new UnsupportedOperationException("WHERE clause not supported: " + url);
750             } else {
751                 this.table = url.getPathSegments().get(0);
752                 this.where = "_id=" + ContentUris.parseId(url);                
753                 this.args = null;
754             }
755         }
756 
757         SqlArguments(Uri url) {
758             if (url.getPathSegments().size() == 1) {
759                 table = url.getPathSegments().get(0);
760                 where = null;
761                 args = null;
762             } else {
763                 throw new IllegalArgumentException("Invalid URI: " + url);
764             }
765         }
766     }
