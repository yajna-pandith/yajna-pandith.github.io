204     /**
205      * Can receive timeout if subscribe never receives an onError/onCompleted ... which reveals a race condition.
206      */
207     @Test(timeout = 10000)
208     public void testSubscribeCompletionRaceCondition() {
209         /*
210          * With non-threadsafe code this fails most of the time on my dev laptop and is non-deterministic enough
211          * to act as a unit test to the race conditions.
212          * 
213          * With the synchronization code in place I can not get this to fail on my laptop.
214          */
215         for (int i = 0; i < 50; i++) {
216             final AsyncSubject<String> subject = AsyncSubject.create();
217             final AtomicReference<String> value1 = new AtomicReference<String>();
218 
219             subject.subscribe(new Action1<String>() {
220 
221                 @Override
222                 public void call(String t1) {
223                     try {
224                         // simulate a slow observer
225                         Thread.sleep(50);
226                     } catch (InterruptedException e) {
227                         e.printStackTrace();
228                     }
229                     value1.set(t1);
230                 }
231 
232             });
233 
234             Thread t1 = new Thread(new Runnable() {
235 
236                 @Override
237                 public void run() {
238                     subject.onNext("value");
239                     subject.onCompleted();
240                 }
241             });
242 
243             SubjectObserverThread t2 = new SubjectObserverThread(subject);
244             SubjectObserverThread t3 = new SubjectObserverThread(subject);
245             SubjectObserverThread t4 = new SubjectObserverThread(subject);
246             SubjectObserverThread t5 = new SubjectObserverThread(subject);
247 
248             t2.start();
249             t3.start();
250             t1.start();
251             t4.start();
252             t5.start();
253             try {
254                 t1.join();
255                 t2.join();
256                 t3.join();
257                 t4.join();
258                 t5.join();
259             } catch (InterruptedException e) {
260                 throw new RuntimeException(e);
261             }
262 
263             assertEquals("value", value1.get());
264             assertEquals("value", t2.value.get());
265             assertEquals("value", t3.value.get());
266             assertEquals("value", t4.value.get());
267             assertEquals("value", t5.value.get());
268         }
269 
270     }

272     private static class SubjectObserverThread extends Thread {
273 
274         private final AsyncSubject<String> subject;
275         private final AtomicReference<String> value = new AtomicReference<String>();
276 
277         public SubjectObserverThread(AsyncSubject<String> subject) {
278             this.subject = subject;
279         }
280 
281         @Override
282         public void run() {
283             try {
284                 // a timeout exception will happen if we don't get a terminal state 
285                 String v = subject.timeout(2000, TimeUnit.MILLISECONDS).toBlocking().single();
286                 value.set(v);
287             } catch (Exception e) {
288                 e.printStackTrace();
289             }
290         }
291     }

293     @Test
294     public void testOnErrorThrowsDoesntPreventDelivery() {
295         AsyncSubject<String> ps = AsyncSubject.create();
296 
297         ps.subscribe();
298         TestSubscriber<String> ts = new TestSubscriber<String>();
299         ps.subscribe(ts);
300 
301         try {
302             ps.onError(new RuntimeException("an exception"));
303             fail("expect OnErrorNotImplementedException");
304         } catch (OnErrorNotImplementedException e) {
305             // ignore
306         }
307         // even though the onError above throws we should still receive it on the other subscriber 
308         assertEquals(1, ts.getOnErrorEvents().size());
309     }

311     /**
312      * This one has multiple failures so should get a CompositeException
313      */
314     @Test
315     public void testOnErrorThrowsDoesntPreventDelivery2() {
316         AsyncSubject<String> ps = AsyncSubject.create();
317 
318         ps.subscribe();
319         ps.subscribe();
320         TestSubscriber<String> ts = new TestSubscriber<String>();
321         ps.subscribe(ts);
322         ps.subscribe();
323         ps.subscribe();
324         ps.subscribe();
325 
326         try {
327             ps.onError(new RuntimeException("an exception"));
328             fail("expect OnErrorNotImplementedException");
329         } catch (CompositeException e) {
330             // we should have 5 of them
331             assertEquals(5, e.getExceptions().size());
332         }
333         // even though the onError above throws we should still receive it on the other subscriber 
334         assertEquals(1, ts.getOnErrorEvents().size());
335     }

337     @Test
338     public void testCurrentStateMethodsNormal() {
339         AsyncSubject<Object> as = AsyncSubject.create();
340         
341         assertFalse(as.hasValue());
342         assertFalse(as.hasThrowable());
343         assertFalse(as.hasCompleted());
