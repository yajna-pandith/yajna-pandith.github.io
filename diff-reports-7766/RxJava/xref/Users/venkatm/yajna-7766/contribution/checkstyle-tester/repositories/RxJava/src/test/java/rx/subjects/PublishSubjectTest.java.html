223     @Test
224     public void testNestedSubscribe() {
225         final PublishSubject<Integer> s = PublishSubject.create();
226 
227         final AtomicInteger countParent = new AtomicInteger();
228         final AtomicInteger countChildren = new AtomicInteger();
229         final AtomicInteger countTotal = new AtomicInteger();
230 
231         final ArrayList<String> list = new ArrayList<String>();
232 
233         s.flatMap(new Func1<Integer, Observable<String>>() {
234 
235             @Override
236             public Observable<String> call(final Integer v) {
237                 countParent.incrementAndGet();
238 
239                 // then subscribe to subject again (it will not receive the previous value)
240                 return s.map(new Func1<Integer, String>() {
241 
242                     @Override
243                     public String call(Integer v2) {
244                         countChildren.incrementAndGet();
245                         return "Parent: " + v + " Child: " + v2;
246                     }
247 
248                 });
249             }
250 
251         }).subscribe(new Action1<String>() {
252 
253             @Override
254             public void call(String v) {
255                 countTotal.incrementAndGet();
256                 list.add(v);
257             }
258 
259         });
260 
261         for (int i = 0; i < 10; i++) {
262             s.onNext(i);
263         }
264         s.onCompleted();
265 
266         //            System.out.println("countParent: " + countParent.get());
267         //            System.out.println("countChildren: " + countChildren.get());
268         //            System.out.println("countTotal: " + countTotal.get());
269 
270         // 9+8+7+6+5+4+3+2+1+0 == 45
271         assertEquals(45, list.size());
272     }

274     /**
275      * Should be able to unsubscribe all Observers, have it stop emitting, then subscribe new ones and it start emitting again.
276      */
277     @Test
278     public void testReSubscribe() {
279         final PublishSubject<Integer> ps = PublishSubject.create();
280 
281         @SuppressWarnings("unchecked")
282         Observer<Integer> o1 = mock(Observer.class);
283         Subscription s1 = ps.subscribe(o1);
284 
285         // emit
286         ps.onNext(1);
287 
288         // validate we got it
289         InOrder inOrder1 = inOrder(o1);
290         inOrder1.verify(o1, times(1)).onNext(1);
291         inOrder1.verifyNoMoreInteractions();
292 
293         // unsubscribe
294         s1.unsubscribe();
295 
296         // emit again but nothing will be there to receive it
297         ps.onNext(2);
298 
299         @SuppressWarnings("unchecked")
300         Observer<Integer> o2 = mock(Observer.class);
301         Subscription s2 = ps.subscribe(o2);
302 
303         // emit
304         ps.onNext(3);
305 
306         // validate we got it
307         InOrder inOrder2 = inOrder(o2);
308         inOrder2.verify(o2, times(1)).onNext(3);
309         inOrder2.verifyNoMoreInteractions();
310 
311         s2.unsubscribe();
312     }

316     @Test(timeout = 1000)
317     public void testUnsubscriptionCase() {
318         PublishSubject<String> src = PublishSubject.create();
319 
320         for (int i = 0; i < 10; i++) {
321             @SuppressWarnings("unchecked")
322             final Observer<Object> o = mock(Observer.class);
323             InOrder inOrder = inOrder(o);
324             String v = "" + i;
325             System.out.printf("Turn: %d%n", i);
326             src.first()
327                 .flatMap(new Func1<String, Observable<String>>() {
328 
329                     @Override
330                     public Observable<String> call(String t1) {
331                         return Observable.just(t1 + ", " + t1);
332                     }
333                 })
334                 .subscribe(new Observer<String>() {
335                     @Override
336                     public void onNext(String t) {
337                         o.onNext(t);
338                     }
339 
340                     @Override
341                     public void onError(Throwable e) {
342                         o.onError(e);
343                     }
344 
345                     @Override
346                     public void onCompleted() {
347                         o.onCompleted();
348                     }
349                 });
350             src.onNext(v);
351             
352             inOrder.verify(o).onNext(v + ", " + v);
353             inOrder.verify(o).onCompleted();
354             verify(o, never()).onError(any(Throwable.class));
355         }

359     @Test
360     public void testOnErrorThrowsDoesntPreventDelivery() {
361         PublishSubject<String> ps = PublishSubject.create();
362 
363         ps.subscribe();
364         TestSubscriber<String> ts = new TestSubscriber<String>();
365         ps.subscribe(ts);
366 
367         try {
368             ps.onError(new RuntimeException("an exception"));
369             fail("expect OnErrorNotImplementedException");
370         } catch (OnErrorNotImplementedException e) {
371             // ignore
372         }
373         // even though the onError above throws we should still receive it on the other subscriber 
374         assertEquals(1, ts.getOnErrorEvents().size());
375     }
