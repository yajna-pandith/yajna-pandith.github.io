46      @Test
47      public void testRefCountAsync() {
48          final AtomicInteger subscribeCount = new AtomicInteger();
49          final AtomicInteger nextCount = new AtomicInteger();
50          Observable<Long> r = Observable.timer(0, 5, TimeUnit.MILLISECONDS)
51                  .doOnSubscribe(new Action0() {
52  
53                      @Override
54                      public void call() {
55                          subscribeCount.incrementAndGet();
56                      }
57  
58                  })
59                  .doOnNext(new Action1<Long>() {
60  
61                      @Override
62                      public void call(Long l) {
63                          nextCount.incrementAndGet();
64                      }
65  
66                  })
67                  .publish().refCount();
68  
69          final AtomicInteger receivedCount = new AtomicInteger();
70          Subscription s1 = r.subscribe(new Action1<Long>() {
71  
72              @Override
73              public void call(Long l) {
74                  receivedCount.incrementAndGet();
75              }
76  
77          });
78          Subscription s2 = r.subscribe();
79  
80          // give time to emit
81          try {
82              Thread.sleep(52);
83          } catch (InterruptedException e) {
84          }
85  
86          // now unsubscribe
87          s2.unsubscribe(); // unsubscribe s2 first as we're counting in 1 and there can be a race between unsubscribe and one subscriber getting a value but not the other
88          s1.unsubscribe();
89  
90          System.out.println("onNext: " + nextCount.get());
91  
92          // should emit once for both subscribers
93          assertEquals(nextCount.get(), receivedCount.get());
94          // only 1 subscribe
95          assertEquals(1, subscribeCount.get());
96      }

182     @Test
183     public void testRepeat() {
184         final AtomicInteger subscribeCount = new AtomicInteger();
185         final AtomicInteger unsubscribeCount = new AtomicInteger();
186         Observable<Long> r = Observable.timer(0, 1, TimeUnit.MILLISECONDS)
187                 .doOnSubscribe(new Action0() {
188 
189                     @Override
190                     public void call() {
191                         System.out.println("******************************* Subscribe received");
192                         // when we are subscribed
193                         subscribeCount.incrementAndGet();
194                     }
195 
196                 })
197                 .doOnUnsubscribe(new Action0() {
198 
199                     @Override
200                     public void call() {
201                         System.out.println("******************************* Unsubscribe received");
202                         // when we are unsubscribed
203                         unsubscribeCount.incrementAndGet();
204                     }
205 
206                 })
207                 .publish().refCount();
208 
209         for (int i = 0; i < 10; i++) {
210             TestSubscriber<Long> ts1 = new TestSubscriber<Long>();
211             TestSubscriber<Long> ts2 = new TestSubscriber<Long>();
212             r.subscribe(ts1);
213             r.subscribe(ts2);
214             try {
215                 Thread.sleep(50);
216             } catch (InterruptedException e) {
217             }
218             ts1.unsubscribe();
219             ts2.unsubscribe();
220             ts1.assertNoErrors();
221             ts2.assertNoErrors();
222             assertTrue(ts1.getOnNextEvents().size() > 0);
223             assertTrue(ts2.getOnNextEvents().size() > 0);
224         }
225 
226         assertEquals(10, subscribeCount.get());
227         assertEquals(10, unsubscribeCount.get());
228     }

230     @Test
231     public void testConnectUnsubscribe() throws InterruptedException {
232         final CountDownLatch unsubscribeLatch = new CountDownLatch(1);
233         final CountDownLatch subscribeLatch = new CountDownLatch(1);
234         Observable<Long> o = synchronousInterval()
235                 .doOnSubscribe(new Action0() {
236 
237                     @Override
238                     public void call() {
239                         System.out.println("******************************* Subscribe received");
240                         // when we are subscribed
241                         subscribeLatch.countDown();
242                     }
243 
244                 })
245                 .doOnUnsubscribe(new Action0() {
246 
247                     @Override
248                     public void call() {
249                         System.out.println("******************************* Unsubscribe received");
250                         // when we are unsubscribed
251                         unsubscribeLatch.countDown();
252                     }
253 
254                 });
255         TestSubscriber<Long> s = new TestSubscriber<Long>();
256         o.publish().refCount().subscribeOn(Schedulers.newThread()).subscribe(s);
257         System.out.println("send unsubscribe");
258         // wait until connected
259         subscribeLatch.await();
260         // now unsubscribe
261         s.unsubscribe();
262         System.out.println("DONE sending unsubscribe ... now waiting");
263         if (!unsubscribeLatch.await(3000, TimeUnit.MILLISECONDS)) {
264             System.out.println("Errors: " + s.getOnErrorEvents());
265             if (s.getOnErrorEvents().size() > 0) {
266                 s.getOnErrorEvents().get(0).printStackTrace();
267             }
268             fail("timed out waiting for unsubscribe");
269         }
270         s.assertNoErrors();
271     }

280     @Test
281     public void testConnectUnsubscribeRaceCondition() throws InterruptedException {
282         final AtomicInteger subUnsubCount = new AtomicInteger();
283         Observable<Long> o = synchronousInterval()
284                 .doOnUnsubscribe(new Action0() {
285 
286                     @Override
287                     public void call() {
288                         System.out.println("******************************* Unsubscribe received");
289                         // when we are unsubscribed
290                         subUnsubCount.decrementAndGet();
291                     }
292 
293                 })
294                 .doOnSubscribe(new Action0() {
295 
296                     @Override
297                     public void call() {
298                         System.out.println("******************************* SUBSCRIBE received");
299                         subUnsubCount.incrementAndGet();
300                     }
301 
302                 });
303 
304         TestSubscriber<Long> s = new TestSubscriber<Long>();
305         
306         o.publish().refCount().subscribeOn(Schedulers.computation()).subscribe(s);
307         System.out.println("send unsubscribe");
308         // now immediately unsubscribe while subscribeOn is racing to subscribe
309         s.unsubscribe();
310         // this generally will mean it won't even subscribe as it is already unsubscribed by the time connect() gets scheduled
311         // give time to the counter to update
312         Thread.sleep(1);
313         // either we subscribed and then unsubscribed, or we didn't ever even subscribe
314         assertEquals(0, subUnsubCount.get());
315 
316         System.out.println("DONE sending unsubscribe ... now waiting");
317         System.out.println("Errors: " + s.getOnErrorEvents());
318         if (s.getOnErrorEvents().size() > 0) {
319             s.getOnErrorEvents().get(0).printStackTrace();
320         }
321         s.assertNoErrors();
322     }

324     private Observable<Long> synchronousInterval() {
325         return Observable.create(new OnSubscribe<Long>() {
326 
327             @Override
328             public void call(Subscriber<? super Long> subscriber) {
329                 while (!subscriber.isUnsubscribed()) {
330                     try {
331                         Thread.sleep(100);
332                     } catch (InterruptedException e) {
333                     }
334                     subscriber.onNext(1L);
335                 }
336             }
337         });
338     }

341     public void onlyFirstShouldSubscribeAndLastUnsubscribe() {
342         final AtomicInteger subscriptionCount = new AtomicInteger();
343         final AtomicInteger unsubscriptionCount = new AtomicInteger();
344         Observable<Integer> observable = Observable.create(new OnSubscribe<Integer>() {
345             @Override
346             public void call(Subscriber<? super Integer> observer) {
347                 subscriptionCount.incrementAndGet();
348                 observer.add(Subscriptions.create(new Action0() {
349                     @Override
350                     public void call() {
351                         unsubscriptionCount.incrementAndGet();
352                     }
353                 }));
354             }
355         });
356         Observable<Integer> refCounted = observable.publish().refCount();
357         @SuppressWarnings("unchecked")
358         Observer<Integer> observer = mock(Observer.class);
359         Subscription first = refCounted.subscribe(observer);
360         assertEquals(1, subscriptionCount.get());
361         Subscription second = refCounted.subscribe(observer);
362         assertEquals(1, subscriptionCount.get());
363         first.unsubscribe();
364         assertEquals(0, unsubscriptionCount.get());
365         second.unsubscribe();
366         assertEquals(1, unsubscriptionCount.get());

369     @Test
370     public void testRefCount() {
371         TestScheduler s = new TestScheduler();
372         Observable<Long> interval = Observable.interval(100, TimeUnit.MILLISECONDS, s).publish().refCount();
373 
374         // subscribe list1
375         final List<Long> list1 = new ArrayList<Long>();
376         Subscription s1 = interval.subscribe(new Action1<Long>() {
377 
378             @Override
379             public void call(Long t1) {
380                 list1.add(t1);
381             }
382 
383         });
384         s.advanceTimeBy(200, TimeUnit.MILLISECONDS);
385 
386         assertEquals(2, list1.size());
387         assertEquals(0L, list1.get(0).longValue());
388         assertEquals(1L, list1.get(1).longValue());
389 
390         // subscribe list2
391         final List<Long> list2 = new ArrayList<Long>();
392         Subscription s2 = interval.subscribe(new Action1<Long>() {
393 
394             @Override
395             public void call(Long t1) {
396                 list2.add(t1);
397             }
398 
399         });
400         s.advanceTimeBy(300, TimeUnit.MILLISECONDS);
401 
402         // list 1 should have 5 items
403         assertEquals(5, list1.size());
404         assertEquals(2L, list1.get(2).longValue());
405         assertEquals(3L, list1.get(3).longValue());
406         assertEquals(4L, list1.get(4).longValue());
407 
408         // list 2 should only have 3 items
409         assertEquals(3, list2.size());
410         assertEquals(2L, list2.get(0).longValue());
411         assertEquals(3L, list2.get(1).longValue());
412         assertEquals(4L, list2.get(2).longValue());
413 
414         // unsubscribe list1
415         s1.unsubscribe();
416 
417         // advance further
418         s.advanceTimeBy(300, TimeUnit.MILLISECONDS);
419 
420         // list 1 should still have 5 items
421         assertEquals(5, list1.size());
422 
423         // list 2 should have 6 items
424         assertEquals(6, list2.size());
425         assertEquals(5L, list2.get(3).longValue());
426         assertEquals(6L, list2.get(4).longValue());
427         assertEquals(7L, list2.get(5).longValue());
428 
429         // unsubscribe list2
430         s2.unsubscribe();
431 
432         // advance further
433         s.advanceTimeBy(1000, TimeUnit.MILLISECONDS);
434 
435         // subscribing a new one should start over because the source should have been unsubscribed
436         // subscribe list3
437         final List<Long> list3 = new ArrayList<Long>();
438         interval.subscribe(new Action1<Long>() {
439 
440             @Override
441             public void call(Long t1) {
442                 list3.add(t1);
443             }
444 
445         });
446         s.advanceTimeBy(200, TimeUnit.MILLISECONDS);
447 
448         assertEquals(2, list3.size());
449         assertEquals(0L, list3.get(0).longValue());
450         assertEquals(1L, list3.get(1).longValue());
451 
452     }

529     @Test(timeout = 10000)
530     public void testUpstreamErrorAllowsRetry() throws InterruptedException {
531         final AtomicInteger intervalSubscribed = new AtomicInteger();
532         Observable<String> interval =
533                 Observable.interval(200,TimeUnit.MILLISECONDS)
534                         .doOnSubscribe(
535                                 new Action0() {
536                                     @Override
537                                     public void call() {
538                                         System.out.println("Subscribing to interval " + intervalSubscribed.incrementAndGet());
539                                     }
540                                 }
541                          )
542                         .flatMap(new Func1<Long, Observable<String>>() {
543                             @Override
544                             public Observable<String> call(Long t1) {
545                                 return Observable.defer(new Func0<Observable<String>>() {
546                                     @Override
547                                     public Observable<String> call() {
548                                         return Observable.<String>error(new Exception("Some exception"));
549                                     }
550                                 });
551                             }
552                         })
553                         .onErrorResumeNext(new Func1<Throwable, Observable<String>>() {
554                             @Override
555                             public Observable<String> call(Throwable t1) {
556                                 return Observable.error(t1);
557                             }
558                         })
559                         .publish()
560                         .refCount();
561 
562         interval
563                 .doOnError(new Action1<Throwable>() {
564                     @Override
565                     public void call(Throwable t1) {
566                         System.out.println("Subscriber 1 onError: " + t1);
567                     }
568                 })
569                 .retry(5)
570                 .subscribe(new Action1<String>() {
571                     @Override
572                     public void call(String t1) {
573                         System.out.println("Subscriber 1: " + t1);
574                     }
575                 });
576         Thread.sleep(100);
577         interval
578         .doOnError(new Action1<Throwable>() {
579             @Override
580             public void call(Throwable t1) {
581                 System.out.println("Subscriber 2 onError: " + t1);
582             }
583         })
584         .retry(5)
585                 .subscribe(new Action1<String>() {
586                     @Override
587                     public void call(String t1) {
588                         System.out.println("Subscriber 2: " + t1);
589                     }
590                 });
591         
592         Thread.sleep(1300);
593         
594         System.out.println(intervalSubscribed.get());
595         assertEquals(6, intervalSubscribed.get());
596     }
