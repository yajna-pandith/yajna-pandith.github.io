82      @Test
83      public void testCompletedStopsEmittingData() {
84          ReplaySubject<Integer> channel = ReplaySubject.create();
85          @SuppressWarnings("unchecked")
86          Observer<Object> observerA = mock(Observer.class);
87          @SuppressWarnings("unchecked")
88          Observer<Object> observerB = mock(Observer.class);
89          @SuppressWarnings("unchecked")
90          Observer<Object> observerC = mock(Observer.class);
91          @SuppressWarnings("unchecked")
92          Observer<Object> observerD = mock(Observer.class);
93  
94          Subscription a = channel.subscribe(observerA);
95          channel.subscribe(observerB);
96  
97          InOrder inOrderA = inOrder(observerA);
98          InOrder inOrderB = inOrder(observerB);
99          InOrder inOrderC = inOrder(observerC);
100         InOrder inOrderD = inOrder(observerD);
101 
102         channel.onNext(42);
103 
104         // both A and B should have received 42 from before subscription
105         inOrderA.verify(observerA).onNext(42);
106         inOrderB.verify(observerB).onNext(42);
107 
108         a.unsubscribe();
109 
110         // a should receive no more
111         inOrderA.verifyNoMoreInteractions();
112 
113         channel.onNext(4711);
114 
115         // only be should receive 4711 at this point
116         inOrderB.verify(observerB).onNext(4711);
117 
118         channel.onCompleted();
119 
120         // B is subscribed so should receive onCompleted
121         inOrderB.verify(observerB).onCompleted();
122 
123         channel.subscribe(observerC);
124 
125         // when C subscribes it should receive 42, 4711, onCompleted
126         inOrderC.verify(observerC).onNext(42);
127         inOrderC.verify(observerC).onNext(4711);
128         inOrderC.verify(observerC).onCompleted();
129 
130         // if further events are propagated they should be ignored
131         channel.onNext(13);
132         channel.onNext(14);
133         channel.onNext(15);
134         channel.onError(new RuntimeException());
135 
136         // a new subscription should only receive what was emitted prior to terminal state onCompleted
137         channel.subscribe(observerD);
138 
139         inOrderD.verify(observerD).onNext(42);
140         inOrderD.verify(observerD).onNext(4711);
141         inOrderD.verify(observerD).onCompleted();
142 
143         Mockito.verifyNoMoreInteractions(observerA);
144         Mockito.verifyNoMoreInteractions(observerB);
145         Mockito.verifyNoMoreInteractions(observerC);
146         Mockito.verifyNoMoreInteractions(observerD);
147 
148     }

269     @Test(timeout = 2000)
270     public void testNewSubscriberDoesntBlockExisting() throws InterruptedException {
271 
272         final AtomicReference<String> lastValueForObserver1 = new AtomicReference<String>();
273         Subscriber<String> observer1 = new Subscriber<String>() {
274 
275             @Override
276             public void onCompleted() {
277 
278             }
279 
280             @Override
281             public void onError(Throwable e) {
282 
283             }
284 
285             @Override
286             public void onNext(String v) {
287                 System.out.println("observer1: " + v);
288                 lastValueForObserver1.set(v);
289             }
290 
291         };
292 
293         final AtomicReference<String> lastValueForObserver2 = new AtomicReference<String>();
294         final CountDownLatch oneReceived = new CountDownLatch(1);
295         final CountDownLatch makeSlow = new CountDownLatch(1);
296         final CountDownLatch completed = new CountDownLatch(1);
297         Subscriber<String> observer2 = new Subscriber<String>() {
298 
299             @Override
300             public void onCompleted() {
301                 completed.countDown();
302             }
303 
304             @Override
305             public void onError(Throwable e) {
306 
307             }
308 
309             @Override
310             public void onNext(String v) {
311                 System.out.println("observer2: " + v);
312                 if (v.equals("one")) {
313                     oneReceived.countDown();
314                 } else {
315                     try {
316                         makeSlow.await();
317                     } catch (InterruptedException e) {
318                         e.printStackTrace();
319                     }
320                     lastValueForObserver2.set(v);
321                 }
322             }
323 
324         };
325 
326         ReplaySubject<String> subject = ReplaySubject.create();
327         subject.subscribe(observer1);
328         subject.onNext("one");
329         assertEquals("one", lastValueForObserver1.get());
330         subject.onNext("two");
331         assertEquals("two", lastValueForObserver1.get());
332 
333         // use subscribeOn to make this async otherwise we deadlock as we are using CountDownLatches
334         subject.subscribeOn(Schedulers.newThread()).subscribe(observer2);
335 
336         System.out.println("before waiting for one");
337 
338         // wait until observer2 starts having replay occur
339         oneReceived.await();
340 
341         System.out.println("after waiting for one");
342 
343         subject.onNext("three");
344         
345         System.out.println("sent three");
346         
347         // if subscription blocked existing subscribers then 'makeSlow' would cause this to not be there yet 
348         assertEquals("three", lastValueForObserver1.get());
349         
350         System.out.println("about to send onCompleted");
351         
352         subject.onCompleted();
353 
354         System.out.println("completed subject");
355         
356         // release 
357         makeSlow.countDown();
358         
359         System.out.println("makeSlow released");
360         
361         completed.await();
362         // all of them should be emitted with the last being "three"
363         assertEquals("three", lastValueForObserver2.get());
364 
365     }
