91    public void testDirectExecutorServiceServiceInThreadExecution() throws Exception {
92      final ListeningExecutorService executor = newDirectExecutorService();
93      final ThreadLocal<Integer> threadLocalCount =
94          new ThreadLocal<Integer>() {
95            @Override
96            protected Integer initialValue() {
97              return 0;
98            }
99          };
100     final AtomicReference<Throwable> throwableFromOtherThread = new AtomicReference<>(null);
101     final Runnable incrementTask =
102         new Runnable() {
103           @Override
104           public void run() {
105             threadLocalCount.set(threadLocalCount.get() + 1);
106           }
107         };
108 
109     Thread otherThread =
110         new Thread(
111             new Runnable() {
112               @Override
113               public void run() {
114                 try {
115                   Future<?> future = executor.submit(incrementTask);
116                   assertTrue(future.isDone());
117                   assertEquals(1, threadLocalCount.get().intValue());
118                 } catch (Throwable t) {
119                   throwableFromOtherThread.set(t);
120                 }
121               }
122             });
123 
124     otherThread.start();
125 
126     ListenableFuture<?> future = executor.submit(incrementTask);
127     assertTrue(future.isDone());
128     assertListenerRunImmediately(future);
129     assertEquals(1, threadLocalCount.get().intValue());
130     otherThread.join(1000);
131     assertEquals(Thread.State.TERMINATED, otherThread.getState());
132     Throwable throwable = throwableFromOtherThread.get();
133     assertNull(
134         "Throwable from other thread: "
135             + (throwable == null ? null : Throwables.getStackTraceAsString(throwable)),
136         throwableFromOtherThread.get());
137   }

170   public void testDirectExecutorServiceServiceTermination() throws Exception {
171     final ExecutorService executor = newDirectExecutorService();
172     final CyclicBarrier barrier = new CyclicBarrier(2);
173     final AtomicReference<Throwable> throwableFromOtherThread = new AtomicReference<>(null);
174     final Runnable doNothingRunnable =
175         new Runnable() {
176           @Override
177           public void run() {}
178         };
179 
180     Thread otherThread =
181         new Thread(
182             new Runnable() {
183               @Override
184               public void run() {
185                 try {
186                   Future<?> future =
187                       executor.submit(
188                           new Callable<Void>() {
189                             @Override
190                             public Void call() throws Exception {
191                               // WAIT #1
192                               barrier.await(1, TimeUnit.SECONDS);
193 
194                               // WAIT #2
195                               barrier.await(1, TimeUnit.SECONDS);
196                               assertTrue(executor.isShutdown());
197                               assertFalse(executor.isTerminated());
198 
199                               // WAIT #3
200                               barrier.await(1, TimeUnit.SECONDS);
201                               return null;
202                             }
203                           });
204                   assertTrue(future.isDone());
205                   assertTrue(executor.isShutdown());
206                   assertTrue(executor.isTerminated());
207                 } catch (Throwable t) {
208                   throwableFromOtherThread.set(t);
209                 }
210               }
211             });
212 
213     otherThread.start();
214 
215     // WAIT #1
216     barrier.await(1, TimeUnit.SECONDS);
217     assertFalse(executor.isShutdown());
218     assertFalse(executor.isTerminated());
219 
220     executor.shutdown();
221     assertTrue(executor.isShutdown());
222     try {
223       executor.submit(doNothingRunnable);
224       fail("Should have encountered RejectedExecutionException");
225     } catch (RejectedExecutionException ex) {
226       // good to go
227     }
228     assertFalse(executor.isTerminated());
229 
230     // WAIT #2
231     barrier.await(1, TimeUnit.SECONDS);
232     assertFalse(executor.awaitTermination(20, TimeUnit.MILLISECONDS));
233 
234     // WAIT #3
235     barrier.await(1, TimeUnit.SECONDS);
236     assertTrue(executor.awaitTermination(1, TimeUnit.SECONDS));
237     assertTrue(executor.awaitTermination(0, TimeUnit.SECONDS));
238     assertTrue(executor.isShutdown());
239     try {
240       executor.submit(doNothingRunnable);
241       fail("Should have encountered RejectedExecutionException");
242     } catch (RejectedExecutionException ex) {
243       // good to go
244     }
245     assertTrue(executor.isTerminated());
246 
247     otherThread.join(1000);
248     assertEquals(Thread.State.TERMINATED, otherThread.getState());
249     Throwable throwable = throwableFromOtherThread.get();
250     assertNull(
251         "Throwable from other thread: "
252             + (throwable == null ? null : Throwables.getStackTraceAsString(throwable)),
253         throwableFromOtherThread.get());
254   }

284   /** Wait for the given thread to reach the {@link State#TIMED_WAITING} thread state. */
285   void awaitTimedWaiting(Thread thread) {
286     while (true) {
287       switch (thread.getState()) {
288         case BLOCKED:
289         case NEW:
290         case RUNNABLE:
291         case WAITING:
292           Thread.yield();
293           break;
294         case TIMED_WAITING:
295           return;
296         case TERMINATED:
297         default:
298           throw new AssertionError();
299       }
300     }
301   }

355   public void testListeningDecorator_scheduleSuccess() throws Exception {
356     final CountDownLatch completed = new CountDownLatch(1);
357     ScheduledThreadPoolExecutor delegate =
358         new ScheduledThreadPoolExecutor(1) {
359           @Override
360           protected void afterExecute(Runnable r, Throwable t) {
361             completed.countDown();
362           }
363         };
364     ListeningScheduledExecutorService service = listeningDecorator(delegate);
365     ListenableFuture<Integer> future =
366         service.schedule(Callables.returning(42), 1, TimeUnit.MILLISECONDS);
367 
368     /*
369      * Wait not just until the Future's value is set (as in future.get()) but
370      * also until ListeningScheduledExecutorService's wrapper task is done
371      * executing listeners, as detected by yielding control to afterExecute.
372      */
373     completed.await();
374     assertTrue(future.isDone());
375     assertThat(future.get()).isEqualTo(42);
376     assertListenerRunImmediately(future);
377     assertEquals(0, delegate.getQueue().size());
378   }

446   private static final class ThrowingRunnable implements Runnable {
447     final int throwAfterCount;
448     final RuntimeException thrown;
449     int count;
450 
451     ThrowingRunnable(int throwAfterCount, RuntimeException thrown) {
452       this.throwAfterCount = throwAfterCount;
453       this.thrown = thrown;
454     }
455 
456     @Override
457     public void run() {
458       if (++count >= throwAfterCount) {
459         throw thrown;
460       }
461     }

725   public void testShutdownAndAwaitTermination_interruptedInternal() throws Exception {
726     final ExecutorService service = mock(ExecutorService.class);
727     when(service.awaitTermination(HALF_SECOND_NANOS, NANOSECONDS))
728         .thenThrow(new InterruptedException());
729 
730     final AtomicBoolean terminated = new AtomicBoolean();
731     // we need to keep this in a flag because t.isInterrupted() returns false after t.join()
732     final AtomicBoolean interrupted = new AtomicBoolean();
733     // we need to use another thread because it will be interrupted and thus using
734     // the current one, owned by JUnit, would make the test fail
735     Thread thread =
736         new Thread(
737             new Runnable() {
738               @Override
739               public void run() {
740                 terminated.set(shutdownAndAwaitTermination(service, 1L, SECONDS));
741                 interrupted.set(Thread.currentThread().isInterrupted());
742               }
743             });
744     thread.start();
745     thread.join();
746     verify(service).shutdown();
747     verify(service).awaitTermination(HALF_SECOND_NANOS, NANOSECONDS);
748     verify(service).shutdownNow();
749     assertTrue(interrupted.get());
750     assertFalse(terminated.get());
751   }
