213 		if (type == null) {
214 			return Collections.emptyList();
215 		}
216 		Factory shadowFactory = createFactory();
217 		CtTypeReference<?> shadowTypeRef = shadowFactory.Type().createReference(type.getActualClass());
218 		CtType<?> shadowType = shadowTypeRef.getTypeDeclaration();
219 
220 		assertFalse(type.isShadow());
221 		assertTrue(shadowType.isShadow());
222 
223 		ShadowEqualsVisitor sev = new ShadowEqualsVisitor(new HashSet<>(Arrays.asList(
224 				//shadow classes has no body
225 				CtRole.STATEMENT,

227 				// shadow classes have no default expression

229 
230 				// shadow classes have no comments
231 				CtRole.COMMENT)));
232 
233 		return sev.checkDiffs(type, shadowType);

246 	private static class ShadowEqualsChecker extends EqualsChecker {
247 		Diff currentDiff;
248 		List<Diff> differences = new ArrayList<>();
249 
250 		@Override
251 		protected void setNotEqual(CtRole role) {
252 			if (role == CtRole.MODIFIER) {
253 				if (currentDiff.element instanceof CtTypeMember) {
254 					CtTypeMember tm = (CtTypeMember) currentDiff.element;
255 					CtType<?> type = tm.getDeclaringType();
256 					if (type != null) {
257 						Set<ModifierKind> elementModifiers = ((CtModifiable) currentDiff.element).getModifiers();
258 						Set<ModifierKind> otherModifiers = ((CtModifiable) currentDiff.other).getModifiers();
259 						if (type.isInterface()) {
260 							if (removeModifiers(elementModifiers, ModifierKind.PUBLIC, ModifierKind.ABSTRACT, ModifierKind.FINAL)
261 									.equals(removeModifiers(otherModifiers, ModifierKind.PUBLIC, ModifierKind.ABSTRACT, ModifierKind.FINAL))) {
262 								//it is OK, that type memebers of interface differs in public abstract modifiers
263 								return;
264 							}
265 						} else if (type.isEnum()) {
266 							CtType<?> type2 = type.getDeclaringType();
267 							if (type2 != null) {
268 								if (type2.isInterface()) {
269 									if (removeModifiers(elementModifiers, ModifierKind.PUBLIC, ModifierKind.FINAL/*, ModifierKind.STATIC*/)
270 											.equals(removeModifiers(otherModifiers, ModifierKind.PUBLIC, ModifierKind.FINAL/*, ModifierKind.STATIC*/))) {
271 										//it is OK, that type memebers of interface differs in public abstract modifiers
272 										return;
273 									}
274 								}
275 							}
276 						}
277 					}
278 				}
279 			}
280 			currentDiff.roles.add(role);
281 		}
282 
283 		private Set<ModifierKind> removeModifiers(Set<ModifierKind> elementModifiers, ModifierKind... modifiers) {
284 			Set<ModifierKind> copy = new HashSet<>(elementModifiers);
285 			for (ModifierKind modifierKind : modifiers) {
286 				copy.remove(modifierKind);
287 			}
288 			return copy;
289 		}
290 
291 		@Override
292 		public void scan(CtElement element) {
293 			currentDiff = new Diff(element, other);
294 			super.scan(element);
295 			if (!currentDiff.roles.isEmpty()) {
296 				differences.add(currentDiff);
297 			}
298 		}
299 	}

301 	private static class ShadowEqualsVisitor extends EqualsVisitor {
302 		CtElement rootOfOther;
303 		CtElementPathBuilder pathBuilder = new CtElementPathBuilder();
304 		List<String> differences;
305 		Set<CtRole> ignoredRoles;
306 
307 		ShadowEqualsVisitor(Set<CtRole> ignoredRoles) {
308 			super(new ShadowEqualsChecker());
309 			this.ignoredRoles = ignoredRoles;
310 		}
311 		List<Diff> getDiffs() {
312 			return ((ShadowEqualsChecker) checker).differences;
313 		}
314 		@Override
315 		protected boolean fail(CtRole role, Object element, Object other) {
316 			if (role == null) {
317 				this.isNotEqual = false;
318 				return false;
319 			}
320 			if (ignoredRoles.contains(role)) {
321 				this.isNotEqual = false;
322 				return false;
323 			}
324 			if (element instanceof CtEnumValue && role == CtRole.VALUE) {
325 				//CtStatementImpl.InsertType.BEFORE contains a value with nested type. Java reflection doesn't support that
326 				this.isNotEqual = false;
327 				return false;
328 			}
329 
330 			CtElement parentOfOther = stack.peek();
331 			try {
332 				differences.add("Difference on path: " + pathBuilder.fromElement(parentOfOther, rootOfOther).toString() + "#" + role.getCamelCaseName()
333 				+ "\nShadow: " + String.valueOf(other)
334 				+ "\nNormal: " + String.valueOf(element) + "\n");
335 			} catch (CtPathException e) {
336 				throw new SpoonException(e);
337 			}
338 			return false;
339 		}
340 		@Override
341 		public void biScan(CtRole role, CtElement element, CtElement other) {
342 			if (element instanceof CtParameter) {
343 				CtParameter param = (CtParameter) element;
344 				CtParameter otherParam = (CtParameter) other;
345 				if (otherParam.getSimpleName().startsWith("arg")) {
346 					otherParam.setSimpleName(param.getSimpleName());
347 				}
348 				if (param.isFinal()) {
349 					//modifier final of parameters isn't accessible in runtime
350 					otherParam.addModifier(ModifierKind.FINAL);
351 				}
352 			}
353 			if (element instanceof CtAnnotation) {
354 				CtAnnotation myAnnotation = (CtAnnotation) element;
355 				if (myAnnotation.getAnnotationType().getQualifiedName().equals(Override.class.getName())) {
356 					return;
357 				}
358 				if (myAnnotation.getAnnotationType().getQualifiedName().equals(Root.class.getName())) {
359 					return;
360 				}
361 			}
362 			if (role == CtRole.SUPER_TYPE && other == null && element != null && ((CtTypeReference<?>) element).getQualifiedName().equals(Object.class.getName())) {
363 				//class X<T extends Object> cannot be distinguished in runtime from X<T>
364 				return;
365 			}
366 			super.biScan(role, element, other);
367 		}
368 		@Override
369 		protected void biScan(CtRole role, Collection<? extends CtElement> elements, Collection<? extends CtElement> others) {
370 			if (role == CtRole.TYPE_MEMBER) {
371 				//sort type members so they match together
372 				Map<String, CtTypeMember> elementsByName = groupTypeMembersBySignature((Collection) elements);
373 				Map<String, CtTypeMember> othersByName = groupTypeMembersBySignature((Collection) others);
374 				for (Map.Entry<String, CtTypeMember> e : elementsByName.entrySet()) {
375 					String name = e.getKey();
376 					CtTypeMember other = othersByName.remove(name);
377 					if (other == null) {
378 						if (e.getValue().isImplicit()) {
379 							//it is OK, that implicit elements are not available in runtime
380 							continue;
381 						}
382 						differences.add("Missing shadow typeMember: " + name);
383 					}
384 					biScan(role, e.getValue(), other);
385 				}
386 				for (Map.Entry<String, CtTypeMember> e : othersByName.entrySet()) {
387 					differences.add("Unexpected shadow typeMember: " + e.getKey());
388 				}
389 				return;
390 			}
391 			if (role == CtRole.ANNOTATION) {
392 				//remove all RetentionPolicy#SOURCE level annotations from elements
393 				List<CtAnnotation<?>> fileteredElements = ((List<CtAnnotation<?>>) elements).stream().filter(a -> {
394 					CtTypeReference<?> at = a.getAnnotationType();
395 					Class ac = at.getActualClass();
396 					return ac != Override.class && ac != SuppressWarnings.class && ac != Root.class;
397 				}).collect(Collectors.toList());
398 				super.biScan(role, fileteredElements, others);
399 				return;
400 			}
401 			super.biScan(role, elements, others);
402 		}
403 		public List<String> checkDiffs(CtType<?> type, CtType<?> shadowType) {
404 			differences = new ArrayList<>();
405 			rootOfOther = shadowType;
406 			biScan(null, type, shadowType);
407 			for (Diff diff : getDiffs()) {
408 				try {
409 					CtElement parentOf;
410 					CtElement rootOf;
411 					if (diff.other != null) {
412 						parentOf = diff.other.getParent();
413 						rootOf = rootOfOther;
414 					} else {
415 						parentOf = diff.element.getParent();
416 						rootOf = type;
417 					}
418 					differences.add("Diff on path: " + pathBuilder.fromElement(parentOf, rootOf).toString() + "#"
419 					+ diff.roles.stream().map(CtRole::getCamelCaseName).collect(Collectors.joining(", ", "[", "]"))
420 					+ "\nShadow: " + String.valueOf(diff.other)
421 					+ "\nNormal: " + String.valueOf(diff.element) + "\n");
422 				} catch (CtPathException e) {
423 					throw new SpoonException(e);
424 				}
425 
426 			}
427 			return differences;
428 		}
429 	}

431 	private static Map<String, CtTypeMember> groupTypeMembersBySignature(Collection<CtTypeMember> typeMembers) {
432 		Map<String, CtTypeMember> typeMembersByName = new HashMap<>();
433 		for (CtTypeMember tm : typeMembers) {
434 			String name;
435 			if (tm instanceof CtExecutable) {
436 				CtExecutable<?> exec = ((CtExecutable) tm);
437 				name = exec.getSignature();
438 			} else {
439 				name = tm.getSimpleName();
440 			}
441 			CtTypeMember conflictTM = typeMembersByName.put(name, tm);
442 			if (conflictTM != null) {
443 				throw new SpoonException("There are two type members with name: " + name + " in " + tm.getParent(CtType.class).getQualifiedName());
444 			}
445 		}
446 		return typeMembersByName;
447 	}
