44  	/**
45  	 * Inserts all the methods, fields, constructors, initialization blocks (if
46  	 * target is a class), inner types, and super interfaces (except
47  	 * {@link Template}) from a given template by substituting all the template
48  	 * parameters by their values. Members annotated with
49  	 * {@link spoon.template.Local} or {@link Parameter} are not inserted.
50  	 *
51  	 * @param targetType
52  	 * 		the target type
53  	 * @param template
54  	 * 		the source template
55  	 */
56  	public static <T extends Template<?>> void insertAll(CtType<?> targetType, T template) {
57  
58  		CtClass<T> templateClass = getTemplateCtClass(targetType, template);
59  		// insert all the interfaces
60  		insertAllSuperInterfaces(targetType, template);
61  		// insert all the methods
62  		insertAllMethods(targetType, template);
63  		// insert all the constructors and all the initialization blocks (only for classes)
64  		insertAllConstructors(targetType, template);
65  		for (CtTypeMember typeMember : templateClass.getTypeMembers()) {
66  			if (typeMember instanceof CtField) {
67  				// insert all the fields
68  				insertGeneratedField(targetType, template, (CtField<?>) typeMember);
69  			} else if (typeMember instanceof CtType) {
70  				// insert all the inner types
71  				insertGeneratedNestedType(targetType, template, (CtType) typeMember);
72  			}
73  		}
74  	}

76  	/**
77  	 * Generates a type (class, interface, enum, ...) from the template model `templateOfType`
78  	 * by by substituting all the template parameters by their values.
79  	 *
80  	 * Inserts all the methods, fields, constructors, initialization blocks (if
81  	 * target is a class), inner types, super class and super interfaces.
82  	 *
83  	 * Note!
84  	 * This algorithm does NOT handle interfaces or annotations
85  	 * {@link Template}, {@link spoon.template.Local}, {@link TemplateParameter} or {@link Parameter}
86  	 * in a special way, it means they all will be added to the generated type too.
87  	 * If you do not want to add them then clone your templateOfType and remove these nodes from that model before.
88  	 *
89  	 * @param qualifiedTypeName
90  	 * 		the qualified name of the new type
91  	 * @param templateOfType
92  	 * 		the model used as source of generation.
93  	 * @param templateParameters
94  	 * 		the substitution parameters
95  	 */
96  	public static <T extends CtType<?>> T createTypeFromTemplate(String qualifiedTypeName, CtType<?> templateOfType, Map<String, Object> templateParameters) {
97  		return PatternBuilder
98  				.create(templateOfType)
99  				.configurePatternParameters(pc -> {
100 					pc.byTemplateParameter(templateParameters);
101 					pc.byParameterValues(templateParameters);
102 				})
103 				.build()
104 				.generator()
105 				.generateType(qualifiedTypeName, templateParameters);
106 	}

109 	 * Inserts all the super interfaces (except {@link Template}) from a given
110 	 * template by substituting all the template parameters by their values.
111 	 *
112 	 * @param targetType
113 	 * 		the target type
114 	 * @param template
115 	 * 		the source template
116 	 */
117 	public static void insertAllSuperInterfaces(CtType<?> targetType, Template<?> template) {
118 
119 		CtClass<? extends Template<?>> sourceClass = getTemplateCtClass(targetType, template);
120 		insertAllSuperInterfaces(targetType, template, sourceClass);
121 	}
122 	/**
123 	 * Inserts all the super interfaces (except {@link Template}) from a given
124 	 * template by substituting all the template parameters by their values.
125 	 *
126 	 * @param targetType
127 	 * 		the target type
128 	 * @param template
129 	 * 		the source template
130 	 * @param sourceClass
131 	 * 		the model of source template
132 	 */
133 	static void insertAllSuperInterfaces(CtType<?> targetType, Template<?> template, CtClass<? extends Template<?>> sourceClass) {
134 
135 		// insert all the interfaces
136 		for (CtTypeReference<?> t : sourceClass.getSuperInterfaces()) {
137 			if (!t.equals(targetType.getFactory().Type().createReference(Template.class))) {
138 				CtTypeReference<?> t1 = t;
139 				// substitute ref if needed
140 				if (Parameters.getNames(sourceClass).contains(t.getSimpleName())) {
141 					Object o = Parameters.getValue(template, t.getSimpleName(), null);
142 					if (o instanceof CtTypeReference) {
143 						t1 = (CtTypeReference<?>) o;
144 					} else if (o instanceof Class) {
145 						t1 = targetType.getFactory().Type().createReference((Class<?>) o);
146 					} else if (o instanceof String) {
147 						t1 = targetType.getFactory().Type().createReference((String) o);
148 					}
149 				}
150 				if (!t1.equals(targetType.getReference())) {
151 					Class<?> c = null;
152 					try {
153 						c = t1.getActualClass();
154 					} catch (Exception e) {
155 						// swallow it
156 					}
157 					if (c != null && c.isInterface()) {
158 						targetType.addSuperInterface(t1);
159 					}
160 					if (c == null) {
161 						targetType.addSuperInterface(t1);
162 					}
163 				}
164 			}
165 		}
166 	}

169 	 * Inserts all the methods from a given template by substituting all the
170 	 * template parameters by their values. Members annotated with
171 	 * {@link spoon.template.Local} or {@link Parameter} are not inserted.
172 	 *
173 	 * @param targetType
174 	 * 		the target type
175 	 * @param template
176 	 * 		the source template
177 	 */
178 	public static void insertAllMethods(CtType<?> targetType, Template<?> template) {
179 
180 		CtClass<?> sourceClass = getTemplateCtClass(targetType, template);
181 		insertAllMethods(targetType, template, sourceClass);
182 	}
183 	/**
184 	 * Inserts all the methods from a given template by substituting all the
185 	 * template parameters by their values. Members annotated with
186 	 * {@link spoon.template.Local} or {@link Parameter} are not inserted.
187 	 *
188 	 * @param targetType
189 	 * 		the target type
190 	 * @param template
191 	 * 		the source template
192 	 * @param sourceClass
193 	 * 		the model of source template
194 	 */
195 	static void insertAllMethods(CtType<?> targetType, Template<?> template, CtClass<?> sourceClass) {
196 		Set<CtMethod<?>> methodsOfTemplate = sourceClass.getFactory().Type().get(Template.class).getMethods();
197 		// insert all the methods
198 		for (CtMethod<?> m : sourceClass.getMethods()) {
199 			if (m.getAnnotation(Local.class) != null) {
200 				continue;
201 			}
202 			if (m.getAnnotation(Parameter.class) != null) {
203 				continue;
204 			}
205 
206 			boolean isOverridingTemplateItf = false;
207 			for (CtMethod m2 : methodsOfTemplate) {
208 				if (m.isOverriding(m2)) {
209 					isOverridingTemplateItf = true;
210 				}
211 			}
212 
213 			if (isOverridingTemplateItf) {
214 				continue;
215 			}
216 
217 			insertMethod(targetType, template, m);
218 		}
219 	}

243 	 * Inserts the field by substituting all the
244 	 * template parameters by their values. Field annotated with
245 	 * {@link spoon.template.Local} or {@link Parameter} is not inserted.
246 
247 	 * @param targetType
248 	 * @param template
249 	 * @param field
250 	 */
251 	static void insertGeneratedField(CtType<?> targetType, Template<?> template, CtField<?> field) {
252 
253 		if (field.getAnnotation(Local.class) != null) {
254 			return;
255 		}
256 		if (Parameters.isParameterSource(field.getReference())) {
257 			return;
258 		}
259 
260 		insertField(targetType, template, field);

320 	/**
321 	 * Inserts all constructors and initialization blocks from a given template
322 	 * by substituting all the template parameters by their values. Members
323 	 * annotated with {@link spoon.template.Local} or {@link Parameter} are not
324 	 * inserted.
325 	 *
326 	 * @param targetType
327 	 * 		the target type
328 	 * @param template
329 	 * 		the source template
330 	 * @param sourceClass
331 	 * 		the model of source template
332 	 */
333 	static void insertAllConstructors(CtType<?> targetType, Template<?> template, CtClass<?> sourceClass) {
334 
335 		// insert all the constructors
336 		if (targetType instanceof CtClass) {
337 			for (CtConstructor<?> c : sourceClass.getConstructors()) {
338 				if (c.isImplicit()) {
339 					continue;
340 				}
341 				if (c.getAnnotation(Local.class) != null) {
342 					continue;
343 				}
344 				insertConstructor((CtClass<?>) targetType, template, c);
345 			}
346 		}
347 		// insert all the initialization blocks (only for classes)
348 		if (targetType instanceof CtClass) {
349 			for (CtAnonymousExecutable e : sourceClass.getAnonymousExecutables()) {
350 				((CtClass<?>) targetType).addAnonymousExecutable(substitute(targetType, template, e));
351 			}
352 		}
353 	}

358 	 *
359 	 * @param targetClass

361 	 * @param template

375 		targetClass.addConstructor(newConstructor);
376 		return newConstructor;

402 	/**
403 	 * Generates a constructor from a template constructor by substituting all
404 	 * the template parameters by their values.
405 	 *
406 	 * @param targetClass
407 	 * 		the target class where to insert the generated constructor
408 	 * @param template
409 	 * 		the template instance that holds the source template
410 	 * 		constructor and that defines the parameter values
411 	 * @param sourceConstructor
412 	 * 		the source template constructor
413 	 * @return the generated constructor
414 	 */
415 	@SuppressWarnings("unchecked")
416 	public static <T> CtConstructor<T> insertConstructor(CtClass<T> targetClass, Template<?> template, CtConstructor<?> sourceConstructor) {
417 
418 		CtConstructor<T> newConstrutor = substitute(targetClass, template, (CtConstructor<T>) sourceConstructor);
419 		// remove the implicit constructor if clashing
420 		if (newConstrutor.getParameters().isEmpty()) {
421 			CtConstructor<?> c = targetClass.getConstructor();
422 			if (c != null && c.isImplicit()) {
423 				targetClass.removeConstructor((CtConstructor<T>) c);
424 			}
425 		}
426 		targetClass.addConstructor(newConstrutor);
427 		return newConstrutor;
428 	}

499 	 * Substitutes all the template parameters in a random piece of code.
500 	 *
501 	 * @param targetType
502 	 * 		the target type
503 	 * @param template
504 	 * 		the template instance
505 	 * @param code
506 	 * 		the code
507 	 * @return the code where all the template parameters has been substituted
508 	 * by their values
509 	 */
510 	@SuppressWarnings("unchecked")
511 	public static <E extends CtElement> E substitute(CtType<?> targetType, Template<?> template, E code) {
512 		if (code == null) {
513 			return null;
514 		}
515 		if (targetType == null) {
516 			throw new RuntimeException("target is null in substitution");
517 		}
518 		TemplateBuilder tb = TemplateBuilder.createPattern(code, template);
519 		if (template instanceof AbstractTemplate) {
520 			tb.setAddGeneratedBy(((AbstractTemplate) template).isAddGeneratedBy());
521 		}
522 		return (E) tb.substituteSingle(targetType, CtElement.class);
523 	}

566 	 * A helper method that recursively redirects all the type references from a
567 	 * source type to a target type in the given element.
568 	 */
569 	public static void redirectTypeReferences(CtElement element, CtTypeReference<?> source, CtTypeReference<?> target) {
570 
571 		List<CtTypeReference<?>> refs = Query.getReferences(element, new ReferenceTypeFilter<>(CtTypeReference.class));
572 
573 		String srcName = source.getQualifiedName();
574 		String targetName = target.getSimpleName();
575 		CtPackageReference targetPackage = target.getPackage();
576 
577 		for (CtTypeReference<?> ref : refs) {
578 			if (ref.getQualifiedName().equals(srcName)) {
579 				ref.setSimpleName(targetName);
580 				ref.setPackage(targetPackage);
581 			}
582 		}
583 	}

586 	 * @param targetType - the element which is going to receive the model produced by the template.
587 	 * It is needed here just to provide the spoon factory, which contains the model of the template
588 	 *
589 	 * @param template - java instance of the template
590 	 *
591 	 * @return - CtClass from the already built spoon model, which represents the template
592 	 */
593 	static <T> CtClass<T> getTemplateCtClass(CtType<?> targetType, Template<?> template) {
594 		Factory factory;
595 		// we first need a factory
596 		if (targetType != null) {
597 			// if it's template with reference replacement
598 			factory = targetType.getFactory();
599 		} else {
600 			// else we have at least one template parameter with a factory
601 			factory = getFactory(template);
602 		}
603 		return getTemplateCtClass(factory, template);
604 	}

606 	/**
607 	 * @param factory - the factory, which contains the model of the template
608 	 *
609 	 * @param template - java instance of the template
610 	 *
611 	 * @return - CtClass from the already built spoon model, which represents the template
612 	 */
613 	public static <T> CtClass<T> getTemplateCtClass(Factory factory, Template<?> template) {
614 		CtClass<T> c = factory.Class().get(template.getClass());
615 		if (c.isShadow()) {
616 			throw new SpoonException("The template " + template.getClass().getName() + " is not part of model. Add template sources to spoon template path.");
617 		}
618 		checkTemplateContracts(c);
619 		return c;
620 	}

622 	private static <T> void checkTemplateContracts(CtClass<T> c) {
623 		for (CtField f : c.getFields()) {
624 			Parameter templateParamAnnotation = f.getAnnotation(Parameter.class);
625 			if (templateParamAnnotation != null && !templateParamAnnotation.value().isEmpty()) {
626 				String proxyName = templateParamAnnotation.value();
627 				// contract: if value, then the field type must be String or CtTypeReference
628 				String fieldTypeQName = f.getType().getQualifiedName();
629 				if (fieldTypeQName.equals(String.class.getName())) {
630 					// contract: the name of the template parameter must correspond to the name of the field
631 					// as found, by Pavel, this is not good contract because it prevents easy refactoring of templates
632 					// we remove it but keep the commented code in case somebody would come up with this bad idea
633 //					if (!f.getSimpleName().equals("_" + f.getAnnotation(Parameter.class).value())) {
634 //						throw new TemplateException("the field name of a proxy template parameter must be called _" + f.getSimpleName());
635 //					}
636 
637 					// contract: if a proxy parameter is declared and named "x" (@Parameter("x")), then a type member named "x" must exist.
638 					boolean found = false;
639 					for (CtTypeMember member: c.getTypeMembers()) {
640 						if (member.getSimpleName().equals(proxyName)) {
641 							found = true;
642 						}
643 					}
644 					if (!found) {
645 						throw new TemplateException("if a proxy parameter is declared and named \"" + proxyName + "\", then a type member named \"\" + proxyName + \"\" must exist.");
646 					}
647 				} else if (fieldTypeQName.equals(CtTypeReference.class.getName())) {
648 					//OK it is CtTypeReference
649 				} else {
650 					throw new TemplateException("proxy template parameter must be typed as String or CtTypeReference, but it is " + fieldTypeQName);
651 				}
652 			}
653 		}
654 	}

656 	/**
657 	 * returns a Spoon factory object from the first template parameter that contains one
658 	 */
659 	static Factory getFactory(Template<?> template) {
660 		try {
661 			for (Field f : Parameters.getAllTemplateParameterFields(template.getClass())) {
662 				if (f.get(template) != null && f.get(template) instanceof FactoryAccessor) {
663 					return ((FactoryAccessor) f.get(template)).getFactory();
664 				}
665 			}
666 		} catch (Exception e) {
667 			throw new SpoonException(e);
668 		}
669 		throw new TemplateException("no factory found in template " + template.getClass().getName());
670 	}
