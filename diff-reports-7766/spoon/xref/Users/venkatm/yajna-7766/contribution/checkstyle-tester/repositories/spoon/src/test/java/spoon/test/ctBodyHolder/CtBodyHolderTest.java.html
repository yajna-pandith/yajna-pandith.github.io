116 	private void checkCtBody(CtBodyHolder p_bodyHolder, String p_constant, int off) {
117 		CtStatement body = p_bodyHolder.getBody();
118 		assertTrue(body instanceof CtBlock<?>);
119 
120 		CtBlock<?> block = (CtBlock) body;
121 		assertEquals(1 + off, block.getStatements().size());
122 
123 		assertTrue(block.getStatement(off) instanceof CtAssignment);
124 
125 		CtAssignment assignment = block.getStatement(off);
126 		assertEquals(p_constant, ((CtLiteral<String>) assignment.getAssignment().partiallyEvaluate()).getValue());
127 
128 		Factory f = body.getFactory();
129 
130 		CtStatement newStat = new CWBStatementTemplate("xx").apply(body.getParent(CtType.class));
131 		try {
132 			newStat.getParent();
133 			fail();
134 		} catch (ParentNotInitializedException e) {
135 			//expected exception
136 		}
137 		//try to set statement and get CtBlock
138 		p_bodyHolder.setBody(newStat);
139 		CtBlock newBlock = (CtBlock) p_bodyHolder.getBody();
140 		assertSame(p_bodyHolder, newBlock.getParent());
141 		assertSame(newBlock, newStat.getParent());
142 
143 		//try to set CtBlock and get the same CtBlock
144 		CtStatement newStat2 = newStat.clone();
145 		try {
146 			newStat2.getParent();
147 			fail();
148 		} catch (ParentNotInitializedException e) {
149 			//expected exception
150 		}
151 		CtBlock newBlock2 = f.Code().createCtBlock(newStat2);
152 		assertSame(newBlock2, newStat2.getParent());
153 		try {
154 			newBlock2.getParent();
155 			fail();
156 		} catch (ParentNotInitializedException e) {
157 			//expected exception
158 		}
159 
160 		p_bodyHolder.setBody(newBlock2);
161 		assertSame(newBlock2, p_bodyHolder.getBody());
162 		assertSame(p_bodyHolder, newBlock2.getParent());
163 		assertSame(newBlock2, newStat2.getParent());
164 	}
