103 	@Override
104 	public void run(String[] args) {
105 		this.setArgs(args);
106 		if (args.length != 0) {
107 			this.run();
108 			// display GUI
109 			if (this.jsapActualArgs.getBoolean("gui")) {
110 				new SpoonModelTree(getFactory());
111 			}
112 		} else {
113 			this.printUsage();
114 		}
115 	}
116 
117 	public void setArgs(String[] args2) {
118 		this.commandLineArgs = args2;
119 		if (processed) {
120 			throw new SpoonException("You cannot process twice the same launcher instance.");
121 		}
122 		processed = true;
123 
124 		processArguments();
125 	}
126 
127 	public void printUsage() {
128 		this.commandLineArgs = new String[] { "--help" };
129 		processArguments();
130 	}

442 	protected void processArguments() {
443 		jsapActualArgs = getArguments();
444 
445 		Environment environment = factory.getEnvironment();
446 		// environment initialization
447 		environment.setComplianceLevel(jsapActualArgs.getInt("compliance"));
448 		environment.setLevel(jsapActualArgs.getString("level"));
449 		if (jsapActualArgs.getBoolean("imports")) {
450 			environment.setPrettyPrintingMode(Environment.PRETTY_PRINTING_MODE.AUTOIMPORT);
451 		} else {
452 			environment.setPrettyPrintingMode(Environment.PRETTY_PRINTING_MODE.FULLYQUALIFIED);
453 		}
454 
455 		if (jsapActualArgs.getBoolean("noclasspath")) {
456 			Launcher.LOGGER.warn("The usage of --noclasspath argument is now deprecated: noclasspath is now the default behaviour.");
457 		} else {
458 			Launcher.LOGGER.warn("Spoon is now using the 'no classpath mode' by default. If you want to ensure using Spoon in full classpath mode, please use the new flag: --cpmode fullclasspath.");
459 		}
460 
461 		String cpmode = jsapActualArgs.getString("cpmode").toUpperCase();
462 		CLASSPATH_MODE classpath_mode = CLASSPATH_MODE.valueOf(cpmode);
463 		switch (classpath_mode) {
464 			case NOCLASSPATH:
465 				environment.setNoClasspath(true);
466 				break;
467 
468 			case FULLCLASSPATH:
469 				environment.setNoClasspath(false);
470 				break;
471 		}
472 
473 		environment.setPreserveLineNumbers(jsapActualArgs.getBoolean("lines"));
474 		environment.setTabulationSize(jsapActualArgs.getInt("tabsize"));
475 		environment.useTabulations(jsapActualArgs.getBoolean("tabs"));
476 		environment.setCopyResources(!jsapActualArgs.getBoolean("no-copy-resources"));
477 
478 		if (jsapActualArgs.getBoolean("enable-comments")) {
479 			Launcher.LOGGER.warn("The option --enable-comments (-c) is deprecated as it is now the default behaviour in Spoon.");
480 		} else {
481 			Launcher.LOGGER.warn("Spoon now parses by default the comments. Consider using the option --disable-comments if you want the old behaviour.");
482 		}
483 
484 		if (jsapActualArgs.getBoolean("disable-comments")) {
485 			environment.setCommentEnabled(false);
486 		} else {
487 			environment.setCommentEnabled(true);
488 		}
489 
490 		environment.setShouldCompile(jsapActualArgs.getBoolean("compile"));
491 		if (jsapActualArgs.getBoolean("disable-model-self-checks")) {
492 			environment.disableConsistencyChecks();
493 		}
494 
495 		String outputString = jsapActualArgs.getString("output-type");
496 		OutputType outputType = OutputType.fromString(outputString);
497 		if (outputType == null) {
498 			throw  new SpoonException("Unknown output type: " + outputString);
499 		} else {
500 			environment.setOutputType(outputType);
501 		}
502 
503 		try {
504 			Charset charset = Charset.forName(jsapActualArgs.getString("encoding"));
505 			environment.setEncoding(charset);
506 		} catch (Exception e) {
507 			throw new SpoonException(e);
508 		}
509 
510 		if (getArguments().getString("generate-files") != null) {
511 			setOutputFilter(getArguments().getString("generate-files").split(":"));
512 		}
513 
514 		// now we are ready to create a spoon compiler
515 		modelBuilder = createCompiler();
516 
517 		if (getArguments().getString("input") != null) {
518 			for (String s : getArguments().getString("input").split("[" + File.pathSeparatorChar + "]")) {
519 				try {
520 					modelBuilder.addInputSource(SpoonResourceHelper.createResource(new File(s)));
521 				} catch (FileNotFoundException e) {
522 					throw new SpoonException(e);
523 				}
524 			}
525 		}
526 
527 		if (jsapActualArgs.getBoolean("precompile")) {
528 			modelBuilder.compile(InputType.FILES);
529 			getEnvironment().setSourceClasspath(new String[]{getEnvironment().getBinaryOutputDirectory()});
530 		}
531 
532 		if (getArguments().getFile("output") != null) {
533 			setSourceOutputDirectory(getArguments().getFile("output"));
534 		}
535 
536 		// Adding template from command-line
537 		if (getArguments().getString("template") != null) {
538 			for (String s : getArguments().getString("template").split("[" + File.pathSeparatorChar + "]")) {
539 				try {
540 					modelBuilder.addTemplateSource(SpoonResourceHelper.createResource(new File(s)));
541 				} catch (FileNotFoundException e) {
542 					environment.report(null, Level.ERROR, "Unable to add template file: " + e.getMessage());
543 					LOGGER.error(e.getMessage(), e);
544 				}
545 			}
546 		}
547 
548 		if (getArguments().getString("processors") != null) {
549 			for (String processorName : getArguments().getString("processors").split(File.pathSeparator)) {
550 				addProcessor(processorName);
551 			}
552 		}
553 
554 	}

572 	/**
573 	 * Parses the arguments given by the command line.
574 	 *
575 	 * @return the JSAP-presented arguments
576 	 */
577 	protected JSAPResult parseArgs() {
578 		if (jsapSpec == null) {
579 			throw new IllegalStateException("no args, please call setArgs before");
580 		}
581 		JSAPResult arguments = jsapSpec.parse(commandLineArgs);
582 		if (!arguments.success()) {
583 			// print out specific error messages describing the problems
584 			for (java.util.Iterator<?> errs = arguments.getErrorMessageIterator(); errs.hasNext();) {
585 				System.err.println("Error: " + errs.next());
586 			}
587 		}
588 		if (!arguments.success() || arguments.getBoolean("help")) {
589 			System.err.println(getVersionMessage());
590 			System.err.println("Usage: java <launcher name> [option(s)]");
591 			System.err.println();
592 			System.err.println("Options : ");
593 			System.err.println();
594 			System.err.println(jsapSpec.getHelp());
595 			System.exit(-1);
596 		}
597 
598 		return arguments;
599 	}

613 	public SpoonModelBuilder createCompiler(Factory factory) {
614 		SpoonModelBuilder comp = new JDTBasedSpoonCompiler(factory);
615 		Environment env = getEnvironment();
616 		// building
617 		comp.setBinaryOutputDirectory(jsapActualArgs.getFile("destination"));
618 
619 		// backward compatibility
620 		// we don't have to set the source classpath
621 		if (jsapActualArgs.contains("source-classpath")) {
622 			comp.setSourceClasspath(jsapActualArgs.getString("source-classpath").split(System.getProperty("path.separator")));
623 		}
624 
625 		env.debugMessage("destination: " + comp.getBinaryOutputDirectory());
626 		env.debugMessage("source classpath: " + Arrays.toString(comp.getSourceClasspath()));
627 		env.debugMessage("template classpath: " + Arrays.toString(comp.getTemplateClasspath()));
628 
629 		return comp;
630 	}

694 	/**
695 	 * Runs Spoon using the given compiler, with the given run options. A Spoon
696 	 * run will perform the following tasks:
697 	 *
698 	 * <ol>
699 	 * <li>Source model building in the given compiler:
700 	 * {@link SpoonModelBuilder#build()}.</li>
701 	 * <li>Template model building in the given factory (if any template source
702 	 * is given): {@link SpoonModelBuilder#build()}.</li>
703 	 * <li>Model processing with the list of given processors if any:
704 	 * {@link SpoonModelBuilder#instantiateAndProcess(List)}.</li>
705 	 * <li>Processed Source code printing and generation (can be disabled with
706 	 * {@link OutputType#NO_OUTPUT}):
707 	 * {@link SpoonModelBuilder#generateProcessedSourceFiles(OutputType)}.</li>
708 	 * <li>Processed source code compilation (optional):
709 	 * </ol>
710 	 */
711 	@Override
712 	public void run() {
713 		Environment env = modelBuilder.getFactory().getEnvironment();
714 		env.reportProgressMessage(getVersionMessage());
715 		env.reportProgressMessage("running Spoon...");
716 
717 		env.reportProgressMessage("start processing...");
718 
719 		long tstart = System.currentTimeMillis();
720 
721 		buildModel();
722 
723 		process();
724 
725 		prettyprint();
726 
727 		if (env.shouldCompile()) {
728 			// we compile the types from the factory, they may have been modified by some processors
729 			modelBuilder.compile(InputType.CTTYPES);
730 		}
731 
732 		long t = System.currentTimeMillis();
733 
734 		env.debugMessage("program spooning done in " + (t - tstart) + " ms");
735 		env.reportEnd();
736 
737 	}

783 	@Override
784 	public void prettyprint() {
785 		long tstart = System.currentTimeMillis();
786 		try {
787 			modelBuilder.generateProcessedSourceFiles(getEnvironment().getOutputType(), typeFilter);
788 		} catch (Exception e) {
789 			throw new SpoonException(e);
790 		}
791 
792 		if (!getEnvironment().getOutputType().equals(OutputType.NO_OUTPUT) && getEnvironment().isCopyResources()) {
793 			for (File dirInputSource : modelBuilder.getInputSources()) {
794 				if (dirInputSource.isDirectory()) {
795 					final Collection<?> resources = FileUtils.listFiles(dirInputSource, RESOURCES_FILE_FILTER, ALL_DIR_FILTER);
796 					for (Object resource : resources) {
797 						final String resourceParentPath = ((File) resource).getParent();
798 						final String packageDir = resourceParentPath.substring(dirInputSource.getPath().length());
799 						final String targetDirectory = getEnvironment().getDefaultFileGenerator().getOutputDirectory() + packageDir;
800 						try {
801 							FileUtils.copyFileToDirectory((File) resource, new File(targetDirectory));
802 						} catch (IOException e) {
803 							throw new SpoonException(e);
804 						}
805 					}
806 				}
807 			}
808 		}
809 
810 		getEnvironment().debugMessage("pretty-printed in " + (System.currentTimeMillis() - tstart) + " ms");
811 	}

863 	/** returns the AST of an inline class */
864 	public static CtClass<?> parseClass(String code) {
865 		Launcher launcher = new Launcher();
866 		launcher.addInputResource(new VirtualFile(code));
867 		launcher.getEnvironment().setNoClasspath(true);
868 		launcher.getEnvironment().setAutoImports(true);
869 		Collection<CtType<?>> allTypes = launcher.buildModel().getAllTypes();
870 		if (allTypes.size() != 1) {
871 			throw new SpoonException("parseClass only considers one class. Please consider using a Launcher object for more advanced usage.");
872 		}
873 		try {
874 			return (CtClass<?>) allTypes.stream().findFirst().get();
875 		} catch (ClassCastException e) {
876 			throw new SpoonException("parseClass only considers classes (and not interfaces and enums). Please consider using a Launcher object for more advanced usage.");
877 		}
878 	}
