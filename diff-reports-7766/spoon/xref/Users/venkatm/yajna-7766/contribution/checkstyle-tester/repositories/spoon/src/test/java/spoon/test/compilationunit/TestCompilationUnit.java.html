201 	@Test
202 	public void testCompilationUnitModelContracts() {
203 		final Launcher launcher = new Launcher();
204 		launcher.getEnvironment().setAutoImports(true);
205 		launcher.addInputResource("./src/test/java/spoon/test/api/testclasses/Bar.java");
206 		launcher.buildModel();
207 
208 		CtType type = launcher.getFactory().Type().get(Bar.class);
209 		//contract: type built from model has BodyHolderSourcePosition
210 		assertTrue(type.getPosition() instanceof BodyHolderSourcePosition);
211 		//contract: type has compilation unit in position
212 		CompilationUnit compilationUnit = type.getPosition().getCompilationUnit();
213 
214 		//contract: parent of compilationUnit is always null
215 		//compilation unit is not part of factory.getModel()
216 		assertNull(compilationUnit.getParent());
217 
218 		//contract: parent of CtImport is CompilationUnit
219 		CtImport anImport = compilationUnit.getImports().iterator().next();
220 		assertSame(compilationUnit, anImport.getParent());
221 		
222 		//contract: parent of type declared in Compilation unit is a package (never CompilationUnit)
223 		assertTrue(compilationUnit.getMainType().getParent() instanceof CtPackage);
224 		
225 		//contract: compilation unit which contains types has null declared module
226 		assertNull(compilationUnit.getDeclaredModule());
227 		//contract: compilation unit knows declared package
228 		assertSame(type.getPackage(), compilationUnit.getDeclaredPackage());
229 		
230 		//the package declaration exists and points to correct package
231 		assertEquals(type.getPackage().getReference(), compilationUnit.getPackageDeclaration().getReference());
232 		
233 		assertSame(compilationUnit, compilationUnit.getPackageDeclaration().getParent());
234 		
235 		//contract: types and imports are scanned exactly once when scanning starts from compilation unit
236 		//note: therefore compilationUnit.getDeclaredPackage() must return null
237 		List<CtType<?>> types = new ArrayList<>();
238 		List<CtTypeReference<?>> typeRefs = new ArrayList<>();
239 		List<CtImport> imports = new ArrayList<>();
240 		new CtScanner() {
241 			public <T> void visitCtTypeReference(CtTypeReference<T> reference) {
242 				typeRefs.add(reference);
243 			}
244 			public <T> void visitCtClass(CtClass<T> ctClass) {
245 				types.add(ctClass);
246 			}
247 			public <T> void visitCtInterface(CtInterface<T> intrface) {
248 				types.add(intrface);
249 			}
250 			public void visitCtImport(CtImport ctImport) {
251 				imports.add(ctImport);
252 				assertSame(compilationUnit, ctImport.getParent());
253 			}
254 		}.scan(compilationUnit);
255 		
256 		assertEquals(0, types.size());
257 		assertEquals(compilationUnit.getDeclaredTypeReferences(), typeRefs);
258 		assertEquals(compilationUnit.getImports(), imports);
259 		
260 		//contract: compilation unit is not visited by scanner when scanning started from model
261 		new CtScanner() {
262 			@Override
263 			public void visitCtCompilationUnit(CtCompilationUnit compilationUnit) {
264 				fail("CtCompilation unit must not be scanned when started from model unnamed module");
265 			}
266 		}.scan(type.getFactory().getModel().getUnnamedModule());
267 
268 		new CtScanner() {
269 			@Override
270 			public void visitCtCompilationUnit(CtCompilationUnit compilationUnit) {
271 				fail("CtCompilation unit must not be scanned when started from model root package");
272 			}
273 		}.scan(type.getFactory().getModel().getRootPackage());
274 	}

276 	private Charset detectEncodingDummy(SpoonFile unused, byte[] fileBytes) {
277 		if (fileBytes.length == 76) {
278 			return Charset.forName("Cp1251");
279 		} else if (fileBytes.length == 86) {
280 			return Charset.forName("UTF-8");
281 		}
282 		throw new SpoonException("unexpected length");
283 	}

285 	@Test
286 	public void testDifferentEncodings() throws Exception {
287 		//contract: both utf-8 and cp1251 files in the same project should be handled properly
288 		final Launcher launcher = new Launcher();
289 		launcher.addInputResource("./src/test/resources/encodings/Cp1251.java");
290 		launcher.addInputResource("./src/test/resources/encodings/Utf8.java");
291 		launcher.getEnvironment().setEncodingProvider(this::detectEncodingDummy);
292 		CtModel model = launcher.buildModel();
293 
294 		CtType<?> utf8Type = model.getAllTypes()
295 				.stream()
296 				.filter(t -> "Utf8".equals(t.getSimpleName()))
297 				.findFirst()
298 				.get();
299 
300 		CtType<?> cp1251Type = model.getAllTypes()
301 				.stream()
302 				.filter(t -> "Cp1251".equals(t.getSimpleName()))
303 				.findFirst()
304 				.get();
305 
306 		assertEquals("\"Привет мир\"", utf8Type.getField("s1").getAssignment().toString());
307 		assertEquals("\"Привет мир\"", cp1251Type.getField("s1").getAssignment().toString());
308 		assertEquals(utf8Type.getField("s1"), cp1251Type.getField("s1"));
309 		assertNotEquals(utf8Type.getField("s2"), cp1251Type.getField("s2"));
310 	}
