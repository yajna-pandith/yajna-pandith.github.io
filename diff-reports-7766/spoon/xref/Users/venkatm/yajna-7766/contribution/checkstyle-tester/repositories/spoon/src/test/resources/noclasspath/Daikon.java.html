542   /**

1556   // The function filters out the reflexive invs in binary slices,
1557   // reflexive and partially reflexive invs in ternary slices
1558   // and also filters out the invariants that have not seen enough
1559   // samples in ternary slices.
1560   static List<Invariant> filter_invs(List<Invariant> invs) {
1561     List<Invariant> new_list = new ArrayList<Invariant>();
1562 
1563     for (Invariant inv : invs) {
1564       VarInfo[] vars = inv.ppt.var_infos;
1565 
1566       // This check is the most non-intrusive way to filter out the invs
1567       // Filter out reflexive invariants in the binary invs
1568       if (!((inv.ppt instanceof PptSlice2) && vars[0] == vars[1])) {
1569 
1570         // Filter out the reflexive and partially reflexive invs in the
1571         // ternary slices
1572         if (!((inv.ppt instanceof PptSlice3) && (vars[0] == vars[1]
1573             || vars[1] == vars[2] || vars[0] == vars[2]))) {
1574           if (inv.ppt.num_values() != 0) {
1575 
1576             // filters out "warning: too few samples for
1577             // daikon.inv.ternary.threeScalar.LinearTernary invariant"
1578             if (inv.isActive()) {
1579               new_list.add(inv);
1580             }
1581           }
1582         }
1583       }
1584     }
1585

1588   /**
1589    * Add orig() variables to the given EXIT/EXITnn point, Does nothing if
1590    * exit_ppt is not an EXIT/EXITnn.
1591    */
1592   private static void create_orig_vars(PptTopLevel exit_ppt, PptMap ppts) {
1593     if (! exit_ppt.ppt_name.isExitPoint()) {
1594       return;
1595     }
1596 
1597     if (debugInit.isLoggable(Level.FINE)) {
1598       debugInit.fine ("Doing create and relate orig vars for: "
1599                        + exit_ppt.name());
1600     }
1601 
1602     PptTopLevel entry_ppt = ppts.get(exit_ppt.ppt_name.makeEnter());
1603     Assert.assertTrue(entry_ppt != null, exit_ppt.name());
1604 
1605     // Add "orig(...)" (prestate) variables to the program point.
1606     // Don't bother to include the constants.  Walk through
1607     // entry_ppt's vars.  For each non-constant, put it on the
1608     // new_vis worklist after fixing its comparability information.
1609     exit_ppt.num_orig_vars = entry_ppt.num_tracevars;
1610     VarInfo[] new_vis = new VarInfo[exit_ppt.num_orig_vars];
1611     {
1612       VarInfo[] entry_ppt_vis = entry_ppt.var_infos;
1613       int new_vis_index = 0;
1614       for (int k = 0; k < entry_ppt.num_declvars; k++) {
1615         VarInfo vi = entry_ppt_vis[k];
1616         Assert.assertTrue(!vi.isDerived(), "Derived when making orig(): "
1617                           + vi.name());
1618         if (vi.isStaticConstant())
1619           continue;
1620         VarInfo origvar = VarInfo.origVarInfo(vi);
1621         // Fix comparability
1622         VarInfo postvar = exit_ppt.find_var_by_name (vi.name());
1623         if (postvar == null) {
1624           System.out.printf ("Cant find var %s in exit of ppt %s%n", vi,
1625                              exit_ppt.name());
1626           for (VarInfo cvi : entry_ppt.var_infos)
1627             System.out.printf ("  entry var = %s%n", cvi);
1628           for (VarInfo cvi : exit_ppt.var_infos)
1629             System.out.printf ("  exit var = %s%n", cvi);
1630           assert false;
1631           throw new RuntimeException("this can't happen: postvar is null");
1632         }
1633         origvar.postState = postvar;
1634         origvar.comparability = postvar.comparability.makeAlias();
1635 
1636         // Add to new_vis
1637         new_vis[new_vis_index] = origvar;
1638         new_vis_index++;
1639         //System.out.printf ("adding origvar %s to ppt %s%n", origvar.name(),
1640         //                   exit_ppt.name());
1641       }
1642       Assert.assertTrue(new_vis_index == exit_ppt.num_orig_vars);
1643     }
1644     exit_ppt.addVarInfos(new_vis);
1645   }
