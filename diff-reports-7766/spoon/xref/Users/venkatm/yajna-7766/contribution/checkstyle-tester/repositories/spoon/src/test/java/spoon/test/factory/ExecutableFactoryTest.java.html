17  package spoon.test.factory;
18  
19  import org.junit.Test;
20  import spoon.reflect.factory.ExecutableFactory;
21  import spoon.reflect.factory.Factory;
22  import spoon.reflect.reference.CtExecutableReference;
23  import spoon.reflect.reference.CtTypeReference;
24  
25  import java.util.List;
26  
27  import static org.junit.Assert.assertEquals;
28  import static spoon.testing.utils.ModelUtils.createFactory;
29  
30  public class ExecutableFactoryTest {
31  
32  	@Test
33  	public void testCreateReference() {
34  		Factory f = createFactory();
35  		ExecutableFactory ef = f.Executable();
36  		String signature = "boolean Object#equals(Object)";
37  		CtExecutableReference<Object> eref = ef.createReference(signature);
38  
39  		String type = eref.getType().getQualifiedName();
40  		String decltype = eref.getDeclaringType().getQualifiedName();
41  		String name = eref.getSimpleName();
42  		List<CtTypeReference<?>> params = eref.getParameters();
43  		List<CtTypeReference<?>> atas = eref.getActualTypeArguments();
44  
45  		assertEquals("boolean", type);
46  		assertEquals("Object", decltype);
47  		assertEquals("equals", name);
48  		assertEquals(1, params.size());
49  		assertEquals(0, atas.size());
50  	}
51  }
