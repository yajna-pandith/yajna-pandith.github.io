439 	private List<CtMethod<?>> orderByName(Collection<CtMethod<?>> meths) {
440 		List<CtMethod<?>> ordered = new ArrayList<>(meths);
441 		ordered.sort(new Comparator<CtMethod<?>>() {
442 			@Override
443 			public int compare(CtMethod<?> o1, CtMethod<?> o2) {
444 				return o1.getParent(CtType.class).getQualifiedName().compareTo(o2.getParent(CtType.class).getQualifiedName());
445 			}
446 		});
447 		return ordered;
448 	}
