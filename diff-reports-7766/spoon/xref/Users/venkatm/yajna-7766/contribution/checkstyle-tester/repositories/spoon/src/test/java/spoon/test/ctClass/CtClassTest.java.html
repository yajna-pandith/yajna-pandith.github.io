66  	@Test
67  	public void getConstructor() throws Exception {
68  		final Factory build = build(Foo.class);
69  		final CtClass<?> foo = (CtClass<?>) build.Type().get(Foo.class);
70  
71  		assertEquals(3, foo.getConstructors().size());
72  
73  		CtTypeReference<Object> typeString = build.Code().createCtTypeReference(String.class);
74  		CtConstructor<?> constructor = foo.getConstructor(typeString);
75  		assertEquals(typeString, constructor.getParameters().get(0).getType());
76  
77  		CtArrayTypeReference<Object> typeStringArray = build.Core().createArrayTypeReference();
78  		typeStringArray.setComponentType(typeString);
79  		constructor = foo.getConstructor(typeStringArray);
80  		assertEquals(typeStringArray, constructor.getParameters().get(0).getType());
81  
82  		CtArrayTypeReference<Object> typeStringArrayArray = build.Core().createArrayTypeReference();
83  		typeStringArrayArray.setComponentType(typeStringArray);
84  		constructor = foo.getConstructor(typeStringArrayArray);
85  		assertEquals(typeStringArrayArray, constructor.getParameters().get(0).getType());
86  
87  		// contract: one could add a type member that already exists (equals but not same) and modify it afterwards
88  		// this adds some flexibility for client code
89  		// see https://github.com/INRIA/spoon/issues/1862
90  		CtConstructor cons = foo.getConstructors().toArray(new CtConstructor[0])[0].clone();
91  		foo.addConstructor(cons);
92  		// as long as we have not changed the signature, getConstructors, which is based on signatures,
93  		// thinks there is one single constructor (and that's OK)
94  		assertEquals(3, foo.getConstructors().size());
95  		cons.addParameter(cons.getFactory().createParameter().setType(cons.getFactory().Type().OBJECT));
96  		// now that we have changed the signature we can call getConstructors safely
97  		assertEquals(4, foo.getConstructors().size());
98  		// we cloned the first constructor, so it has the same position, and comes before the 2nd and 3rd constructor
99  		assertSame(cons, foo.getTypeMembers().get(1));
100 		// the parent is set (the core problem described in the issue has been fixed)
101 		assertSame(foo, cons.getParent());
102 
103 		// now we clone and reset the position
104 		CtConstructor cons2 = foo.getConstructors().toArray(new CtConstructor[0])[0].clone();
105 		cons2.setPosition(null);
106 		// adding the constructor, this time, without a position
107 		foo.addConstructor(cons2);
108 		// without position, it has been addded at the end
109 		assertSame(cons2, foo.getTypeMembers().get(4));
110 	}

114 		// contract: When we have a static class which extends a superclass in the classpath,
115 		// the enclosing class don't have a superclass. This is probably a bug in JDT but good

123 		launcher.getEnvironment().setNoClasspath(true);
124 		launcher.run();
125

127 
128 		final CtType<?> staticClass = aClass.getNestedType("InvariantCheckProcessor");

134 
135 		canBeBuilt("./target/class", 8, true);

156 	@Test
157 	public void testAllTypeReferencesToALocalTypeShouldNotStartWithNumber() throws Exception {
158 		// contract: When we have a local type, we should never start with its number. But this
159 		// number isn't rewrite only in the class declaration but in all type references and in
160 		// the constructor.
161 		final CtType<Pozole> aPozole = buildClass(Pozole.class);
162 		final CtClass<?> cook = aPozole.getNestedType("1Cook");
163 
164 		assertEquals("1Cook", cook.getSimpleName());
165 		assertEquals("spoon.test.ctClass.testclasses.Pozole$1Cook", cook.getQualifiedName());
166 		final Set<? extends CtConstructor<?>> constructors = cook.getConstructors();
167 		final String expectedConstructor = "public Cook() {" + System.lineSeparator() + "}";
168 		assertEquals(expectedConstructor, constructors.toArray(new CtConstructor[constructors.size()])[0].toString());
169 		CtLocalVariable m = cook.getMethod("m").getBody().getStatement(0);
170 		assertEquals("final java.lang.Class<Cook> cookClass = Cook.class", m.toString());
171 		CtFieldAccess ac = (CtFieldAccess) m.getAssignment();
172 		assertEquals("class", ac.getVariable().getSimpleName());
173 		assertEquals(true, ac.getTarget() instanceof CtTypeAccess);
174 
175 		// contract: one can call getModifiers on ".class" of "Aclass.class", it does not crash and it returns 0
176 		assertEquals(0, ac.getVariable().getModifiers().size());
177 
178 		Factory factory = aPozole.getFactory();
179 
180 		aPozole.removeModifier(ModifierKind.PUBLIC);
181 		factory.Code().createCodeSnippetStatement(aPozole.toString()).compile();
182 
183 		CtClass internalClass = factory.Core().createClass();
184 		internalClass.setSimpleName("Foo");
185 		cook.getParent(CtBlock.class).addStatement(internalClass);
186 		assertEquals("Foo", internalClass.getSimpleName());
187 		assertEquals("spoon.test.ctClass.testclasses.Pozole$Foo", internalClass.getQualifiedName());
188 
189 		internalClass.addConstructor(factory.Core().createConstructor());
190 		CtConstructor cons = (CtConstructor) internalClass.getConstructors().toArray(new CtConstructor[0])[0];
191 		cons.setBody(factory.Core().createBlock());
192 
193 		CtConstructorCall call = cook.getFactory().Core().createConstructorCall();
194 		call.setExecutable(cons.getReference());
195 		assertEquals(internalClass, internalClass.getReference().getDeclaration());
196 		assertEquals("new Foo()", call.toString());
197 		internalClass.insertAfter(call);
198 
199 		factory.getEnvironment().setAutoImports(true);
200 		factory.Code().createCodeSnippetStatement(aPozole.toString()).compile();
201 
202 		factory.getEnvironment().setAutoImports(false);
203 		factory.Code().createCodeSnippetStatement(aPozole.toString()).compile();
204 	}

224 	@Test
225 	public void toStringWithImports() {
226 		String newLine = System.getProperty("line.separator");
227 
228 		final Launcher launcher2 = new Launcher();
229 		launcher2.addInputResource("./src/test/java/spoon/test/ctClass/");
230 		launcher2.getEnvironment().setNoClasspath(true);
231 		launcher2.buildModel();
232 		final CtClass<Object> aClass2 = launcher2.getFactory().Class().get(AnonymousClass.class);
233 		DefaultJavaPrettyPrinter djpp = new DefaultJavaPrettyPrinter(launcher2.getEnvironment());
234 		aClass2.accept(djpp);
235 
236 		// contract: a class can be printed with full context
237 		assertEquals("package spoon.test.ctClass.testclasses;" + newLine +
238 				"/**" + newLine +
239 				" * Created by urli on 11/10/2017." + newLine +
240 				" */" + newLine +
241 				"public class AnonymousClass {" + newLine +
242 				"    final int machin = new java.util.Comparator<java.lang.Integer>() {" + newLine +
243 				"        @java.lang.Override" + newLine +
244 				"        public int compare(java.lang.Integer o1, java.lang.Integer o2) {" + newLine +
245 				"            return 0;" + newLine +
246 				"        }" + newLine +
247 				"    }.compare(1, 2);" + newLine +
248 				"}", aClass2.toStringWithImports());
249 
250 		// contract: a class can be printed with full context in autoimports
251 		aClass2.getFactory().getEnvironment().setAutoImports(true);
252 		assertEquals("package spoon.test.ctClass.testclasses;" + newLine +
253 				"import java.util.Comparator;" + newLine +
254 				"/**" + newLine +
255 				" * Created by urli on 11/10/2017." + newLine +
256 				" */" + newLine +
257 				"public class AnonymousClass {" + newLine +
258 				"    final int machin = new Comparator<Integer>() {" + newLine +
259 				"        @Override" + newLine +
260 				"        public int compare(Integer o1, Integer o2) {" + newLine +
261 				"            return 0;" + newLine +
262 				"        }" + newLine +
263 				"    }.compare(1, 2);" + newLine +
264 				"}", aClass2.toStringWithImports());
265 
266 		// contract: toStringWithImports works with a new class with no position
267 		assertEquals("package foo;" + newLine +
268 				"import java.io.File;" + newLine +
269 				"class Bar extends File {}", launcher2.getFactory().createClass("foo.Bar").setSuperclass(launcher2.getFactory().Type().get(File.class).getReference()).toStringWithImports());
270 	}

292 	public void testCloneAnonymousClassInvocation() {
293 		// contract: after cloning an anonymous class invocation, we still should be able to print it, when not using autoimport
294 
295 		final Launcher launcher = new Launcher();
296 		launcher.addInputResource("./src/test/java/spoon/test/ctClass/testclasses/AnonymousClass.java");
297 		launcher.getEnvironment().setAutoImports(false);
298 		launcher.buildModel();
299 
300 		CtModel model = launcher.getModel();
301 		CtNewClass newClassInvocation = launcher.getModel().getElements(new TypeFilter<>(CtNewClass.class)).get(0);
302 		CtNewClass newClassInvocationCloned = newClassInvocation.clone();
303 
304 		CtClass anonymousClass = newClassInvocation.getAnonymousClass();
305 		CtClass anonymousClassCloned = newClassInvocationCloned.getAnonymousClass();
306 
307 		// The test stops failing if we set the parent below
308 		//newClassInvocationCloned.setParent(launcher.getFactory().Class().get(AnonymousClass.class));
309 
310 		assertEquals(0, anonymousClass.getAllFields().size());
311 		assertEquals(0, anonymousClassCloned.getAllFields().size());
312 
313 		assertFalse(newClassInvocation.toString().isEmpty());
314 		assertFalse(newClassInvocationCloned.toString().isEmpty());
315 
316 		assertEquals(newClassInvocation.toString(), newClassInvocationCloned.toString());
317 	}
