85  	 *
86  	 * @param anonymousQualifiedName
87  	 * 		Qualified name which contains the anonymous name.
88  	 * @return Anonymous simple name.
89  	 */
90  	static String computeAnonymousName(char[] anonymousQualifiedName) {
91  		final String poolName = CharOperation.charToString(anonymousQualifiedName);

96  	 * Creates a qualified type name from a two-dimensional array.
97  	 *
98  	 * @param typeName
99  	 * 		two-dimensional array which represents the qualified name expected.
100 	 * @return Qualified name.
101 	 */
102 	static String createQualifiedTypeName(char[][] typeName) {
103 		StringBuilder s = new StringBuilder();
104 		for (int i = 0; i < typeName.length - 1; i++) {
105 			s.append(CharOperation.charToString(typeName[i])).append(".");
106 		}
107 		s.append(CharOperation.charToString(typeName[typeName.length - 1]));
108 		return s.toString();
109 	}

112 	 * Creates a catch variable from a type reference.
113 	 *
114 	 * @param typeReference
115 	 * 		Corresponds to the exception type declared in the catch.
116 	 * @return a catch variable.
117 	 */
118 	CtCatchVariable<Throwable> createCatchVariable(TypeReference typeReference) {
119 		final Argument jdtCatch = (Argument) jdtTreeBuilder.getContextBuilder().stack.peekFirst().node;
120 		final Set<CtExtendedModifier> modifiers = getModifiers(jdtCatch.modifiers, false, false);
121 
122 		CtCatchVariable<Throwable> result = jdtTreeBuilder.getFactory().Core().createCatchVariable();
123 		result.<CtCatchVariable>setSimpleName(CharOperation.charToString(jdtCatch.name)).setExtendedModifiers(modifiers);
124 		if (typeReference instanceof UnionTypeReference) {
125 			//do not set type of variable yet. It will be initialized later by visit of multiple types. Each call then ADDs one type
126 			return result;
127 		} else {
128 			CtTypeReference ctTypeReference = jdtTreeBuilder.getReferencesBuilder().<Throwable>getTypeReference(typeReference.resolvedType);
129 			return result.<CtCatchVariable>setType(ctTypeReference);
130 		}
131 	}

165 	/**
166 	 * Analyzes if {@code singleNameReference} points to a {@link CtVariable} visible in current
167 	 * scope and, if existent, returns its corresponding {@link CtVariableAccess}. Returns
168 	 * {@code null} if {@code singleNameReference} could not be resolved as variable access. Since
169 	 * we are in noclasspath mode this function may also return {@code null} if
170 	 * {@code singleNameReference} points to a variable declared by an unknown class.
171 	 *
172 	 * @param singleNameReference
173 	 * 		The potential variable access.
174 	 * @return A {@link CtVariableAccess} if {@code singleNameReference} points to a variable
175 	 * 		visible in current scope, {@code null} otherwise.
176 	 */
177 	<T> CtVariableAccess<T> createVariableAccessNoClasspath(SingleNameReference singleNameReference) {
178 		final CoreFactory coreFactory = jdtTreeBuilder.getFactory().Core();
179 		final ContextBuilder contextBuilder = jdtTreeBuilder.getContextBuilder();
180 		final ReferenceBuilder referenceBuilder = jdtTreeBuilder.getReferencesBuilder();
181 		final PositionBuilder positionBuilder = jdtTreeBuilder.getPositionBuilder();
182 
183 		final String name = CharOperation.charToString(singleNameReference.token);
184 		final CtVariable<T> variable = contextBuilder.getVariableDeclaration(name);
185 		if (variable == null) {
186 			return null;
187 		}
188 
189 		final CtVariableReference<T> variableReference;
190 		final CtVariableAccess<T> variableAccess;
191 		if (variable instanceof CtParameter) {
192 			// create variable of concrete type to avoid type casting while calling methods
193 			final CtParameterReference<T> parameterReference = coreFactory.createParameterReference();
194 			if (variable.getParent() instanceof CtLambda) {
195 				// nothing
196 
197 			} else {
198 				// Unfortunately, we can not use `variable.getReference()` here as some parent
199 				// references (in terms of Java objects) have not been set up yet. Thus, we need to
200 				// create the required parameter reference by our own.
201 
202 				// Since the given parameter has not been declared in a lambda expression it must
203 				// have been declared by a method/constructor.
204 				final CtExecutable executable = (CtExecutable) variable.getParent();
205 
206 				// find executable's corresponding jdt element
207 				AbstractMethodDeclaration executableJDT = null;
208 				for (final ASTPair astPair : contextBuilder.stack) {
209 					if (astPair.element == executable) {
210 						executableJDT = (AbstractMethodDeclaration) astPair.node;
211 					}
212 				}
213 				assert executableJDT != null;
214 
215 				// create a reference to executable's declaring class
216 				final CtTypeReference declaringReferenceOfExecutable =
217 						// `binding` may be null for anonymous classes which means we have to
218 						// create an 'empty' type reference since we have no further information
219 						// available
220 						executableJDT.binding == null ? coreFactory.createTypeReference()
221 								: referenceBuilder.getTypeReference(executableJDT.binding.declaringClass);
222 			}
223 			variableReference = parameterReference;
224 			variableAccess = isLhsAssignment(contextBuilder, singleNameReference)
225 					? coreFactory.<T>createVariableWrite() : coreFactory.<T>createVariableRead();
226 		} else if (variable instanceof CtField) {
227 			variableReference = variable.getReference();
228 			variableAccess = isLhsAssignment(contextBuilder, singleNameReference)
229 					? coreFactory.<T>createFieldWrite() : coreFactory.<T>createFieldRead();
230 		} else { // CtLocalVariable, CtCatchVariable, ...
231 			variableReference = variable.getReference();
232 			variableAccess = isLhsAssignment(contextBuilder, singleNameReference)
233 					? coreFactory.<T>createVariableWrite() : coreFactory.<T>createVariableRead();
234 		}
235 		variableReference.setSimpleName(name);
236 		variableReference.setPosition(positionBuilder.buildPosition(
237 				singleNameReference.sourceStart(), singleNameReference.sourceEnd()));
238 		variableAccess.setVariable(variableReference);
239 		return variableAccess;
240 	}

242 	/**
243 	 * Creates a variable or a field access from its qualified name.
244 	 *
245 	 * @param qualifiedNameReference
246 	 * 		Used to build the variable access. See all sub methods of this class to understand its usage.
247 	 * @return a variable access.
248 	 */
249 	<T> CtVariableAccess<T> createVariableAccess(QualifiedNameReference qualifiedNameReference) {
250 		long[] positions = qualifiedNameReference.sourcePositions;
251 		int sourceStart = qualifiedNameReference.sourceStart();
252 		int sourceEnd = qualifiedNameReference.sourceEnd();
253 		if (qualifiedNameReference.indexOfFirstFieldBinding < positions.length) {
254 			sourceEnd = (int) (positions[qualifiedNameReference.indexOfFirstFieldBinding] >>> 32) - 2;
255 		}
256 		CtVariableAccess<T> va;
257 		CtVariableReference<T> ref;
258 		boolean fromAssignment = isLhsAssignment(jdtTreeBuilder.getContextBuilder(), qualifiedNameReference);
259 		boolean isOtherBinding = qualifiedNameReference.otherBindings == null || qualifiedNameReference.otherBindings.length == 0;
260 		if (qualifiedNameReference.binding instanceof FieldBinding) {
261 			ref = jdtTreeBuilder.getReferencesBuilder().getVariableReference(qualifiedNameReference.fieldBinding());
262 			ref.setPosition(jdtTreeBuilder.getPositionBuilder().buildPosition(getVariableReferenceSourceStart(sourceStart, sourceEnd), sourceEnd));
263 
264 			va = createFieldAccess(ref, createTargetFieldAccess(qualifiedNameReference, (CtFieldReference<Object>) ref), isOtherBinding && fromAssignment);
265 		} else {
266 			ref = jdtTreeBuilder.getReferencesBuilder().getVariableReference((VariableBinding) qualifiedNameReference.binding);
267 			ref.setPosition(jdtTreeBuilder.getPositionBuilder().buildPosition(getVariableReferenceSourceStart(sourceStart, sourceEnd), sourceEnd));
268 
269 			va = createVariableAccess(ref, isOtherBinding && fromAssignment);
270 		}
271 
272 		if (qualifiedNameReference.otherBindings != null) {
273 			int i = 0; //positions index;
274 			va.setPosition(jdtTreeBuilder.getPositionBuilder().buildPosition(sourceStart, sourceEnd));
275 			sourceStart = (int) (positions[qualifiedNameReference.indexOfFirstFieldBinding - 1] >>> 32);
276 			for (FieldBinding b : qualifiedNameReference.otherBindings) {
277 				isOtherBinding = qualifiedNameReference.otherBindings.length == i + 1;
278 				CtFieldAccess<T> other = createFieldAccess(
279 						jdtTreeBuilder.getReferencesBuilder().<T>getVariableReference(b, qualifiedNameReference.tokens[i + 1]), va, isOtherBinding && fromAssignment);
280 				//set source position of fa
281 				if (i + qualifiedNameReference.indexOfFirstFieldBinding >= qualifiedNameReference.otherBindings.length) {
282 					sourceEnd = qualifiedNameReference.sourceEnd();
283 				} else {
284 					sourceEnd = (int) (positions[qualifiedNameReference.indexOfFirstFieldBinding + i + 1] >>> 32) - 2;
285 				}
286 				other.setPosition(jdtTreeBuilder.getPositionBuilder().buildPosition(sourceStart, sourceEnd));
287 				va = other;
288 				i++;
289 			}
290 		} else if (!(qualifiedNameReference.binding instanceof FieldBinding) && qualifiedNameReference.tokens.length > 1) {
291 			sourceStart = (int) (positions[0] >>> 32);
292 			for (int i = 1; i < qualifiedNameReference.tokens.length; i++) {
293 				isOtherBinding = qualifiedNameReference.tokens.length == i + 1;
294 				CtFieldAccess<T> other = createFieldAccess(//
295 						jdtTreeBuilder.getReferencesBuilder().<T>getVariableReference(null, qualifiedNameReference.tokens[i]), va, isOtherBinding && fromAssignment);
296 				//set source position of va;
297 				sourceEnd = (int) (positions[i]);
298 				va.setPosition(jdtTreeBuilder.getPositionBuilder().buildPosition(sourceStart, sourceEnd));
299 				va = other;
300 			}
301 		}
302 		va.setPosition(jdtTreeBuilder.getPositionBuilder().buildPosition(qualifiedNameReference.sourceStart(), qualifiedNameReference.sourceEnd()));
303 		return va;
304 	}

306 	private int getVariableReferenceSourceStart(int sourceStart, int sourceEnd) {
307 		char[] contents = jdtTreeBuilder.getContextBuilder().getCompilationUnitContents();
308 		//search for last dot, which is not contained in comment
309 		int lastIdentifierStart = sourceStart;
310 		while (true) {
311 			int dotIdx = PositionBuilder.findNextChar(contents, sourceEnd, lastIdentifierStart, '.');
312 			if (dotIdx < 0) {
313 				return lastIdentifierStart;
314 			}
315 			lastIdentifierStart = dotIdx + 1;
316 		}
317 	}

348 		}

355 				va.setTarget(jdtTreeBuilder.getFactory().Code().createThisAccess(jdtTreeBuilder.getReferencesBuilder().getTypeReference(singleNameReference.actualReceiverType), true));
356 			}
357 		}
358 		return va;
359 	}

388 	 * In no classpath mode, when we build a field access, we have a binding typed by ProblemBinding.
389 	 * We try to get all information we can get from this binding.
390 	 *
391 	 * @param qualifiedNameReference
392 	 * 		Used to get the problem binding of the field access and the name of the declaring type.
393 	 * @return a field access.
394 	 */
395 	<T> CtFieldAccess<T> createFieldAccessNoClasspath(QualifiedNameReference qualifiedNameReference) {
396 		boolean fromAssignment = isLhsAssignment(jdtTreeBuilder.getContextBuilder(), qualifiedNameReference);
397 		CtFieldAccess<T> fieldAccess = createFieldAccess(jdtTreeBuilder.getReferencesBuilder().<T>getVariableReference((ProblemBinding) qualifiedNameReference.binding), null, fromAssignment);
398 		// In no classpath mode and with qualified name, the type given by JDT is wrong...
399 		final char[][] declaringClass = CharOperation.subarray(qualifiedNameReference.tokens, 0, qualifiedNameReference.tokens.length - 1);
400 		final MissingTypeBinding declaringType = jdtTreeBuilder.getContextBuilder().compilationunitdeclaration.scope.environment.createMissingType(null, declaringClass);
401 		final CtTypeReference<T> declaringRef = jdtTreeBuilder.getReferencesBuilder().getTypeReference(declaringType);
402 		fieldAccess.getVariable().setDeclaringType(declaringRef);
403 		fieldAccess.getVariable().setStatic(true);
404 		fieldAccess.setTarget(jdtTreeBuilder.getFactory().Code().createTypeAccess(declaringRef));
405 		// In no classpath mode and with qualified name, the binding doesn't have a good name.
406 		fieldAccess.getVariable()
407 				.setSimpleName(createQualifiedTypeName(CharOperation.subarray(qualifiedNameReference.tokens, qualifiedNameReference.tokens.length - 1, qualifiedNameReference.tokens.length)));
408 		return fieldAccess;
409 	}

430 	/**
431 	 * Creates a type access from its qualified name and with a field reference.
432 	 *
433 	 * @param qualifiedNameReference
434 	 * 		Used to update the field reference if necessary.
435 	 * @param fieldReference
436 	 * 		Used to get its declaring class and to put it in the type access.
437 	 * @return a type access.
438 	 */
439 	CtTypeAccess<?> createTypeAccess(QualifiedNameReference qualifiedNameReference, CtFieldReference<?> fieldReference) {
440 		final TypeBinding receiverType = qualifiedNameReference.actualReceiverType;
441 		if (receiverType != null) {
442 			final CtTypeReference<Object> qualifiedRef = jdtTreeBuilder.getReferencesBuilder().getQualifiedTypeReference(
443 					qualifiedNameReference.tokens, receiverType, qualifiedNameReference.fieldBinding().declaringClass.enclosingType(), new JDTTreeBuilder.OnAccessListener() {
444 						@Override
445 						public boolean onAccess(char[][] tokens, int index) {
446 							return !CharOperation.equals(tokens[index + 1], tokens[tokens.length - 1]);
447 						}
448 					});
449 			if (qualifiedRef != null) {
450 				fieldReference.setDeclaringType(qualifiedRef);
451 			} else {
452 				fieldReference.setDeclaringType(jdtTreeBuilder.getReferencesBuilder().getTypeReference(receiverType));
453 			}
454 		}
455 		CtTypeAccess<?> typeAccess = jdtTreeBuilder.getFactory().Code().createTypeAccess(fieldReference.getDeclaringType());
456 		if (qualifiedNameReference.indexOfFirstFieldBinding > 1) {
457 			// the array sourcePositions contains the position of each element of the qualifiedNameReference
458 			// the last element contains the position of the field
459 			long[] positions = qualifiedNameReference.sourcePositions;
460 			typeAccess.setPosition(
461 					jdtTreeBuilder.getPositionBuilder().buildPosition(qualifiedNameReference.sourceStart(), (int) (positions[qualifiedNameReference.indexOfFirstFieldBinding - 1] >>> 32) - 2));
462 			handleImplicit(
463 					qualifiedNameReference.actualReceiverType.getPackage(),
464 					qualifiedNameReference, fieldReference.getSimpleName(), typeAccess.getAccessedType());
465 		} else {
466 			typeAccess.setImplicit(qualifiedNameReference.isImplicitThis());
467 		}
468 		return typeAccess;
469 	}

479 	static void handleImplicit(QualifiedTypeReference qualifiedTypeReference, CtTypeReference<?> typeRef) {
480 		if (qualifiedTypeReference.resolvedType instanceof ProblemReferenceBinding || qualifiedTypeReference.resolvedType == null) {
481 			//cannot detect implicitness of parts of qualifiedTypeReference, when its type is not known
482 			return;
483 		}
484 		handleImplicit(
485 				qualifiedTypeReference.resolvedType.getPackage(),
486 				qualifiedTypeReference.tokens,
487 				0,
488 				qualifiedTypeReference, null, typeRef);
489 	}

491 	private static void handleImplicit(PackageBinding packageBinding, char[][] tokens, int countOfOtherBindings, Object qualifiedNameReference, String simpleName, CtTypeReference<?> typeRef) {
492 		char[][] packageNames = null;
493 		if (packageBinding != null) {
494 			packageNames = packageBinding.compoundName;
495 		}
496 		CtTypeReference<?> originTypeRef = typeRef;
497 		//get component type of arrays
498 		while (typeRef instanceof CtArrayTypeReference<?>) {
499 			typeRef = ((CtArrayTypeReference<?>) typeRef).getComponentType();
500 		}
501 		int off = tokens.length - 1;
502 		off = off - countOfOtherBindings;
503 		if (off > 0) {
504 			if (simpleName != null) {
505 				if (!simpleName.equals(new String(tokens[off]))) {
506 					throw new SpoonException("Unexpected field reference simple name: \"" + new String(tokens[off]) + "\" expected: \"" + simpleName + "\"");
507 				}
508 				off--;
509 			}
510 			while (off >= 0) {
511 				String token = new String(tokens[off]);
512 				if (!typeRef.getSimpleName().equals(token)) {
513 					/*
514 					 * Actually it may happen when type reference full qualified name
515 					 * (e.g. spoon.test.imports.testclasses.internal.SuperClass.InnerClassProtected) doesn't match with access path
516 					 * (e.g. spoon.test.imports.testclasses.internal.ChildClass.InnerClassProtected)
517 					 * In such rare case we cannot detect and set implicitness
518 					 */
519 					typeRef.getFactory().getEnvironment().debugMessage("Compiler's type path: " + qualifiedNameReference + " doesn't matches with Spoon qualified type name: " + originTypeRef);
520 					return;
521 					//throw new SpoonException("Unexpected type reference simple name: \"" + token + "\" expected: \"" + typeRef.getSimpleName() + "\"");
522 				}
523 				CtTypeReference<?> declTypeRef = typeRef.getDeclaringType();
524 				if (declTypeRef != null) {
525 					typeRef = declTypeRef;
526 					off--;
527 					continue;
528 				}
529 				CtPackageReference packageRef = typeRef.getPackage();
530 				if (packageRef != null) {
531 					if (packageNames != null && packageNames.length == off) {
532 						//there is full package name
533 						//keep it explicit
534 						return;
535 					}
536 					if (off == 0) {
537 						//the package name is implicit
538 						packageRef.setImplicit(true);
539 						return;
540 					}
541 					if (packageBinding == null || (packageBinding.tagBits & TagBits.HasMissingType) != 0) {
542 						/*
543 						 * the packageBinding points to unresolved type like: import static daikon.PptRelation.PptRelationType;
544 						 * where packageBinding is 'daikon.PptRelation'
545 						 * but it is not clear whether `PptRelation` is type or package.
546 						 * JDT compiler expects it is package, while Spoon model expects it is type
547 						 */
548 						//keep it explicit
549 						return;
550 					}
551 					throw new SpoonException("Unexpected QualifiedNameReference tokens " + qualifiedNameReference + " for typeRef: " + originTypeRef);
552 				}
553 			}
554 			typeRef.setImplicit(true);
555 		}
556 	}

712 	 * Creates a class, an enum, an interface or a annotation type.
713 	 *
714 	 * @return a type.
715 	 */
716 	CtType<?> createType(TypeDeclaration typeDeclaration) {
717 		CtType<?> type;
718 		if ((typeDeclaration.modifiers & ClassFileConstants.AccAnnotation) != 0) {
719 			type = jdtTreeBuilder.getFactory().Core().<java.lang.annotation.Annotation>createAnnotationType();
720 		} else if ((typeDeclaration.modifiers & ClassFileConstants.AccEnum) != 0) {
721 			type = jdtTreeBuilder.getFactory().Core().createEnum();
722 		} else if ((typeDeclaration.modifiers & ClassFileConstants.AccInterface) != 0) {
723 			type = jdtTreeBuilder.getFactory().Core().createInterface();
724 		} else {
725 			type = jdtTreeBuilder.getFactory().Core().createClass();
726 		}
727 
728 		// Setting modifiers
729 		type.setExtendedModifiers(getModifiers(typeDeclaration.modifiers, false, false));
730 
731 		jdtTreeBuilder.getContextBuilder().enter(type, typeDeclaration);
732 
733 		if (typeDeclaration.superInterfaces != null) {
734 			for (TypeReference ref : typeDeclaration.superInterfaces) {
735 				final CtTypeReference superInterface = jdtTreeBuilder.references.buildTypeReference(ref, null);
736 				type.addSuperInterface(superInterface);
737 			}
738 		}
739 
740 		if (type instanceof CtClass) {
741 			if (typeDeclaration.superclass != null) {
742 				((CtClass) type).setSuperclass(jdtTreeBuilder.references.buildTypeReference(typeDeclaration.superclass, typeDeclaration.scope));
743 			}
744 			if (typeDeclaration.binding != null && (typeDeclaration.binding.isAnonymousType() || (typeDeclaration.binding instanceof LocalTypeBinding && typeDeclaration.binding.enclosingMethod() != null))) {
745 				type.setSimpleName(computeAnonymousName(typeDeclaration.binding.constantPoolName()));
746 			} else {
747 				type.setSimpleName(new String(typeDeclaration.name));
748 			}
749 		} else {
750 			type.setSimpleName(new String(typeDeclaration.name));
751 		}
752 
753 		return type;
754 	}

757 	 * Creates an entire object CtModule from a module declaration.
758 	 * @return a CtModule
759 	 */
760 	CtModule createModule(ModuleDeclaration moduleDeclaration) {
761 		CtModule module = jdtTreeBuilder.getFactory().Module().getOrCreate(new String(moduleDeclaration.moduleName));
762 		module.setIsOpenModule(moduleDeclaration.isOpen());
763 		jdtTreeBuilder.getContextBuilder().enter(module, moduleDeclaration);
764 
765 		if (moduleDeclaration.requires != null && moduleDeclaration.requires.length > 0) {
766 			List<CtModuleRequirement> moduleRequirements = new ArrayList<>();
767 			for (RequiresStatement requiresStatement : moduleDeclaration.requires) {
768 				moduleRequirements.add(this.createModuleRequirement(requiresStatement));
769 			}
770 
771 			module.setRequiredModules(moduleRequirements);
772 		}
773 
774 		if (moduleDeclaration.exports != null && moduleDeclaration.exports.length > 0) {
775 			List<CtPackageExport> moduleExports = new ArrayList<>();
776 			for (ExportsStatement exportsStatement : moduleDeclaration.exports) {
777 				moduleExports.add(this.createModuleExport(exportsStatement));
778 			}
779 
780 			module.setExportedPackages(moduleExports);
781 		}
782 
783 		if (moduleDeclaration.opens != null && moduleDeclaration.opens.length > 0) {
784 			List<CtPackageExport> moduleOpens = new ArrayList<>();
785 			for (OpensStatement opensStatement : moduleDeclaration.opens) {
786 				moduleOpens.add(this.createModuleExport(opensStatement));
787 			}
788 
789 			module.setOpenedPackages(moduleOpens);
790 		}
791 
792 		if (moduleDeclaration.uses != null && moduleDeclaration.uses.length > 0) {
793 			List<CtUsedService> consumedServices = new ArrayList<>();
794 			for (UsesStatement consumedService : moduleDeclaration.uses) {
795 				consumedServices.add(this.createUsedService(consumedService));
796 			}
797 
798 			module.setUsedServices(consumedServices);
799 		}
800 
801 		if (moduleDeclaration.services != null && moduleDeclaration.services.length > 0) {
802 			List<CtProvidedService> moduleProvidedServices = new ArrayList<>();
803 			for (ProvidesStatement providesStatement : moduleDeclaration.services) {
804 				moduleProvidedServices.add(this.createModuleProvidedService(providesStatement));
805 			}
806 
807 			module.setProvidedServices(moduleProvidedServices);
808 		}
809 		module.setPosition(this.jdtTreeBuilder.getPositionBuilder().buildPosition(moduleDeclaration.declarationSourceStart, moduleDeclaration.declarationSourceEnd));
810 		return module;
811 	}
