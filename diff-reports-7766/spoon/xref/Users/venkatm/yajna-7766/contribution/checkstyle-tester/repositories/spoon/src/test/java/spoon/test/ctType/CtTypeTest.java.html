153 	@Test
154 	public void testIsSubTypeOfonTypeReferences() {
155 		final Launcher launcher = new Launcher();
156 		launcher.setArgs(new String[]{"-c"});
157 		launcher.addInputResource("./src/test/java/spoon/test/ctType/testclasses/SubtypeModel.java");
158 		launcher.buildModel();
159 		Factory factory = launcher.getFactory();
160 
161 		CtType<?> oCtType = factory.Class().get("spoon.test.ctType.testclasses.SubtypeModel");
162 		CtMethod<?> O_FooMethod = oCtType.filterChildren(new NamedElementFilter<>(CtMethod.class, "foo")).first();
163 
164 		Map<String, CtTypeReference<?>> nameToTypeRef = new HashMap<>();
165 		O_FooMethod.filterChildren(new TypeFilter<>(CtLocalVariable.class)).forEach((CtLocalVariable var) -> {
166 			nameToTypeRef.put(var.getSimpleName(), var.getType());
167 		});
168 
169 		int[] count = new int[1];
170 
171 		O_FooMethod.filterChildren(new TypeFilter<>(CtAssignment.class)).forEach((CtAssignment ass) -> {
172 			for (CtComment comment : ass.getComments()) {
173 				checkIsNotSubtype(comment, nameToTypeRef);
174 				count[0]++;
175 			}
176 			count[0]++;
177 			checkIsSubtype(((CtVariableAccess) ass.getAssigned()).getVariable().getType(), ((CtVariableAccess) ass.getAssignment()).getVariable().getType(), nameToTypeRef);
178 		});
179 
180 		assertTrue(count[0] > (9 * 8));
181 
182 		// contract: isSubTypeOf does not throw any exception
183 		// #2288 cannot be reproduced, probably fixed by #2406
184 		CtTypeReference<Object> typeReferenceWithNoDeclaration = launcher.getFactory().createTypeReference();
185 		typeReferenceWithNoDeclaration.setSimpleName("DoesNotExist");
186 		assertFalse(typeReferenceWithNoDeclaration.isSubtypeOf(oCtType.getReference()));
187 		assertFalse(oCtType.isSubtypeOf(typeReferenceWithNoDeclaration));
188 	}

190 	private void checkIsSubtype(CtTypeReference superType, CtTypeReference subType, Map<String, CtTypeReference<?>> nameToTypeRef) {
191 		String msg = getTypeName(subType) + " isSubTypeOf " + getTypeName(superType);
192 		assertTrue(msg, subType.isSubtypeOf(superType));
193 	}
194 
195 	private static final Pattern assignment = Pattern.compile("\\s*(\\w+)\\s*=\\s*(\\w+);");
196 
197 	private void checkIsNotSubtype(CtComment comment, Map<String, CtTypeReference<?>> nameToTypeRef) {
198 		Matcher m = assignment.matcher(comment.getContent());
199 		assertTrue(m.matches());
200 		CtTypeReference<?> superType = nameToTypeRef.get(m.group(1));
201 		CtTypeReference<?> subType = nameToTypeRef.get(m.group(2));
202 		String msg = getTypeName(subType) + " is NOT SubTypeOf " + getTypeName(superType);
203 		assertFalse(msg, subType.isSubtypeOf(superType));
204 	}
205 
206 	private String getTypeName(CtTypeReference<?> ref) {
207 		String name;
208 		CtReference r = ref.getParent(CtReference.class);
209 		if (r != null) {
210 			name = r.getSimpleName();
211 		} else {
212 			name = ref.getParent(CtNamedElement.class).getSimpleName();
213 		}
214 		return ref.toString() + " " + name;
215 	}
