182 	private void testJavaDocComment(CtClass type, String EOL) {
183 		CtJavaDoc classJavaDoc = (CtJavaDoc) type.getComments().get(0);
184 		//contract: test that java doc is printed correctly
185 		String str = classJavaDoc.toString();
186 		StringTokenizer st = new StringTokenizer(str, System.getProperty("line.separator"));
187 		boolean first = true;
188 		while (st.hasMoreTokens()) {
189 			String line = st.nextToken();
190 			if (first) {
191 				//first
192 				first = false;
193 				assertEquals(3, line.length());
194 				assertEquals("/**", line);
195 			} else {
196 				if (st.hasMoreTokens()) {
197 					//in the middle
198 					assertTrue(line.length() >= 2);
199 					assertEquals(" *", line.substring(0, 2));
200 				} else {
201 					//last
202 					assertTrue(line.length() == 3);
203 					assertEquals(" */", line.substring(0, 3));
204 				}
205 			}
206 		}
207 		assertEquals("JavaDoc test class." + EOL + EOL
208 				+ "Long description", classJavaDoc.getContent());
209 
210 		List<CtJavaDocTag> elements = type.getElements(new TypeFilter<>(CtJavaDocTag.class));
211 		assertEquals(8, elements.size());
212 
213 
214 		List<CtJavaDocTag> authorTags = getTagByType(elements, CtJavaDocTag.TagType.AUTHOR);
215 		assertEquals(1, authorTags.size());
216 		assertEquals("Thomas Durieux", authorTags.get(0).getContent());
217 
218 		List<CtJavaDocTag> deprecatedTags = getTagByType(elements, CtJavaDocTag.TagType.DEPRECATED);
219 		assertEquals(1, deprecatedTags.size());
220 		assertEquals("", deprecatedTags.get(0).getContent());
221 
222 
223 		List<CtJavaDocTag> sinceTags = getTagByType(elements, CtJavaDocTag.TagType.SINCE);
224 		assertEquals(2, sinceTags.size());
225 		assertEquals("1.3", sinceTags.get(0).getContent());
226 		assertEquals("1.3", sinceTags.get(1).getContent());
227 
228 		List<CtJavaDocTag> paramTags = getTagByType(elements, CtJavaDocTag.TagType.PARAM);
229 		assertEquals(1, paramTags.size());
230 		assertEquals("the parameters", paramTags.get(0).getContent());
231 		assertEquals("i", paramTags.get(0).getParam());
232 
233 		CtJavaDocTag tagClone = paramTags.get(0).clone();
234 		assertEquals("the parameters", tagClone.getContent());
235 		assertEquals("i", tagClone.getParam());
236 
237 		List<CtJavaDocTag> throwsTags = getTagByType(elements, CtJavaDocTag.TagType.THROWS);
238 		assertEquals(1, throwsTags.size());
239 		assertEquals("an exception", throwsTags.get(0).getContent());
240 		assertEquals("RuntimeException", throwsTags.get(0).getParam());
241 
242 
243 		assertEquals("JavaDoc test class.", classJavaDoc.getShortDescription());
244 		assertEquals("Long description", classJavaDoc.getLongDescription());
245 
246 		CtJavaDocTag deprecatedTag = classJavaDoc.getTags().get(0);
247 		assertTrue(classJavaDoc.toString().contains("@deprecated"));
248 		classJavaDoc.removeTag(0);
249 		assertEquals(-1, classJavaDoc.toString().indexOf("@deprecated"));
250 		classJavaDoc.addTag(deprecatedTag);
251 		assertTrue(classJavaDoc.toString().contains("@deprecated"));
252 	}

816

818 			assertTrue(x.getSimpleName() + " has no documentation", !x.getDocComment().isEmpty());
819

821 			if (launcher.getModel().getElements(new AbstractFilter<CtElement>() {
822 				@Override
823 				public boolean matches(CtElement element) {
824 					return (element instanceof CtNamedElement) && ((CtNamedElement) element).getSimpleName().equals(x.getSimpleName() + "Impl") && (element instanceof CtClass) && !((CtClass) element).hasModifier(ModifierKind.ABSTRACT);

826 			}).isEmpty()) {
827 				return;
828 			}
829 
830 			// we don't consider references
831 			if (x.getSimpleName().endsWith("Reference")) {
832 				return;
833 			}
834 
835 			if (x.isSubtypeOf(launcher.getFactory().Type().get(CtStatement.class).getReference())
836 					|| x.isSubtypeOf(launcher.getFactory().Type().get(CtExpression.class).getReference())
837 					) {
838 
839 				// no meaningful snippet
840 				if ("CtCodeSnippetStatement".equals(x.getSimpleName())) {
841 					return;
842 				}
843 
844 				// no meaningful snippet
845 				if ("CtCodeSnippetExpression".equals(x.getSimpleName())) {
846 					return;
847 				}
848 
849 				// no comment in snippet mode
850 				if ("CtComment".equals(x.getSimpleName())) {
851 					return;
852 				}
853 
854 				// a statement in really rare cases
855 				if ("CtEnum".equals(x.getSimpleName())) {
856 					return;
857 				}
858

860 				if ("CtAnnotationFieldAccess".equals(x.getSimpleName())) {
861 					return;
862 				}

865 				codeElementsDocumentationPage.append("[(javadoc)](http://spoon.gforge.inria.fr/mvnsites/spoon-core/apidocs/" + x.getQualifiedName().replace('.', '/') + ".html)\n\n");
866 				codeElementsDocumentationPage.append("```java" + "\n");

890 
891 		}
