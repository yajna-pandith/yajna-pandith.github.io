42  	@Test
43  	public void testRemoveStatement() {
44  		Launcher spoon = new Launcher();
45  		spoon.addInputResource("./src/test/java/spoon/test/ctBlock/testclasses/Toto.java");
46  		spoon.buildModel();
47  
48  		List<CtMethod> methods = spoon.getModel().getElements(new NamedElementFilter<>(CtMethod.class, "foo"));
49  
50  		assertEquals(1, methods.size());
51  
52  		CtMethod foo = methods.get(0);
53  
54  		CtBlock block = foo.getBody();
55  		CtStatement lastStatement = block.getLastStatement();
56  
57  		assertEquals("i++", lastStatement.toString());
58  
59  		block.removeStatement(lastStatement);
60  
61  		CtStatement newLastStatement = block.getLastStatement();
62  
63  		assertNotSame(newLastStatement, lastStatement);
64  		assertTrue(newLastStatement instanceof CtIf);
65  	}

67  	@Test
68  	public void testAddStatementInBlock() {
69  		// contract: we can add a statement at a specific index
70  		// the statements with a higher index are pushed after
71  
72  		Launcher spoon = new Launcher();
73  		spoon.addInputResource("./src/test/java/spoon/test/ctBlock/testclasses/Toto.java");
74  		spoon.buildModel();
75  
76  		CtClass<?> toto = spoon.getFactory().Class().get(Toto.class);
77  		CtMethod foo = toto.getMethodsByName("foo").get(0);
78  		CtBlock block = foo.getBody();
79  		CtBlock originalBlock = block.clone();
80  
81  		assertEquals(4, block.getStatements().size());
82  		block.addStatement(1, (CtStatement) spoon.getFactory().createInvocation().setExecutable(foo.getReference()));
83  
84  		assertEquals(5, block.getStatements().size());
85  
86  		assertEquals(originalBlock.getStatement(0), block.getStatement(0));
87  
88  		for (int i = 1; i < 4; i++) {
89  			assertEquals(originalBlock.getStatement(i), block.getStatement(i + 1));
90  		}
91  	}

93  	@Test
94  	public void testAddStatementInCase() {
95  		// contract: we can add a statement at a specific index
96  		// the statements with a higher index are pushed after
97  
98  		Launcher spoon = new Launcher();
99  		spoon.addInputResource("./src/test/java/spoon/test/ctBlock/testclasses/Toto.java");
100 		spoon.buildModel();
101 
102 		CtClass<?> toto = spoon.getFactory().Class().get(Toto.class);
103 		CtMethod bar = toto.getMethodsByName("bar").get(0);
104 		CtSwitch<?> ctSwitch = (CtSwitch) bar.getBody().getStatement(0);
105 
106 		CtCase firstCase = ctSwitch.getCases().get(0);
107 		CtCase originalFirstCase = firstCase.clone();
108 
109 		CtCase secondCase = ctSwitch.getCases().get(1);
110 		CtCase originalSecondCase = secondCase.clone();
111 
112 
113 		assertEquals(3, firstCase.getStatements().size());
114 		firstCase.addStatement(3, spoon.getFactory().createBreak());
115 		assertEquals(4, firstCase.getStatements().size());
116 
117 		for (int i = 0; i < 3; i++) {
118 			assertEquals(originalFirstCase.getStatements().get(i), firstCase.getStatements().get(i));
119 		}
120 
121 		assertEquals(2, secondCase.getStatements().size());
122 		secondCase.addStatement(1, (CtStatement) spoon.getFactory().createInvocation().setExecutable(bar.getReference()));
123 		assertEquals(3, secondCase.getStatements().size());
124 
125 		assertEquals(originalSecondCase.getStatements().get(0), secondCase.getStatements().get(0));
126 		assertEquals(originalSecondCase.getStatements().get(1), secondCase.getStatements().get(2));
127 	}
