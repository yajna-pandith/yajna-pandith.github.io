106 		String className = "ReferencingClass";
107 		String qualifiedName = packageName + "." + className;
108 		String referencedQualifiedName = packageName + "." + "ReferencedClass";
109 
110 		// we only create the model for ReferecingClass
111 		List<SpoonResource> fileToBeSpooned = SpoonResourceHelper.resources("./src/test/resources/reference-test/input/" + qualifiedName.replace('.', '/') + ".java");
112 		comp.addInputSources(fileToBeSpooned);
113 		assertEquals(1, fileToBeSpooned.size()); // for ReferecingClass
114 
115 		// Spoon requires the binary version of ReferencedClass
116 		List<SpoonResource> classpath = SpoonResourceHelper.resources("./src/test/resources/reference-test/ReferenceTest.jar");
117 		String[] dependencyClasspath = { classpath.get(0).getPath() };
118 
119 		factory.getEnvironment().setSourceClasspath(dependencyClasspath);
120 		assertEquals(1, classpath.size());
121 
122 		// now we can build the model
123 		comp.build();
124 
125 		// we can get the model of ReferecingClass
126 		CtType<?> theClass = factory.Type().get(qualifiedName);
127 
128 		// now we retrieve the reference to ReferencedClass
129 		CtTypeReference referencedType = null;
130 		ReferenceTypeFilter<CtTypeReference> referenceTypeFilter = new ReferenceTypeFilter<>(CtTypeReference.class);
131 		List<CtTypeReference> elements = Query.getElements(theClass, referenceTypeFilter);
132 		for (CtTypeReference reference : elements) {
133 			if (reference.getQualifiedName().equals(referencedQualifiedName)) {

138 		assertNotNull(referencedType);
139 
140 		// we can get the actual class from the reference, because it is loaded from the class path
141 		Class referencedClass = referencedType.getActualClass();
142 		assertEquals(referencedQualifiedName, referencedClass.getName());
143 	}

146 	@Test
147 	public void doNotCloseLoader() throws Exception {
148 
149 		/* Given the following scenario:
150 		 * 	- ClassA has a field of ClassB.
151 		 *	- ClassB has a field of ClassC.
152 		 * 	- Spoon only models ClassA.
153 		 *
154 		 * We want to get the field of ClassB, which should be accessible because
155 		 * the definitions of ClassB and ClassC were provided in the class path.
156 		 */
157 
158 		SpoonModelBuilder comp = new Launcher().createCompiler();
159 		Factory factory = comp.getFactory();
160 
161 		String qualifiedName = "spoontest.a.ClassA";
162 		String referenceQualifiedName = "spoontest.b.ClassB";
163 
164 		// we only create the model for ClassA
165 		List<SpoonResource> fileToBeSpooned = SpoonResourceHelper.resources("./src/test/resources/reference-test-2/" + qualifiedName.replace('.', '/') + ".java");
166 		comp.addInputSources(fileToBeSpooned);
167 		assertEquals(1, fileToBeSpooned.size()); // for ClassA
168 
169 		// Spoon requires the binary version of dependencies
170 		List<SpoonResource> classpath = SpoonResourceHelper.resources("./src/test/resources/reference-test-2/ReferenceTest2.jar");
171 		String[] dependencyClasspath = { classpath.get(0).getPath() };
172 
173 		factory.getEnvironment().setSourceClasspath(dependencyClasspath);
174 		assertEquals(1, classpath.size());
175 
176 		// now we can build the model
177 		comp.build();
178 
179 		// we can get the model of ClassA
180 		CtType<?> theClass = factory.Type().get(qualifiedName);
181 
182 		// we get ClassA's field of type ClassB
183 		List<CtField<?>> fields = theClass.getFields();
184 		assertEquals(1, fields.size());
185 
186 		CtField<?> bField = fields.get(0);
187 		CtTypeReference referencedType = bField.getType();
188 		assertEquals(referenceQualifiedName, referencedType.getQualifiedName());
189 
190 		// we get ClassB's field of type ClassC
191 		Collection<CtFieldReference<?>> fieldsOfB = referencedType.getAllFields();
192 		if (fieldsOfB.size() == 2) {
193 			// Jacoco instruments all dependencies with an agent.
194 			// So, when we use reflection on ClassB, we don't have one field but two fields.
195 			// First, it is the field of ClassB. Second, it is the field of Jacoco.
196 			final CtFieldReference<?> potentialJacoco = (CtFieldReference<?>) fieldsOfB.toArray()[1];
197 			if ("$jacocoData".equals(potentialJacoco.getSimpleName())) {
198 				fieldsOfB.remove(potentialJacoco);
199 			}
200 		}
201 		assertEquals(1, fieldsOfB.size());
202 
203 		CtFieldReference<?> cField = fieldsOfB.iterator().next();
204 		assertEquals("spoontest.c.ClassC", cField.getType().getQualifiedName());
205 	}
