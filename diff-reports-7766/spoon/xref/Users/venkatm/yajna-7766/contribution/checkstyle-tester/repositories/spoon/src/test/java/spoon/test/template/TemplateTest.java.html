272 	/*
273 	 * Matches comments like:
274 	 * Generated by spoon.test.template.testclasses.inheritance.SuperTemplate#toBeOverriden(SuperTemplate.java:7)
275 	 * Generated by spoon.test.template.testclasses.inheritance.SubTemplate#methodWithTemplatedParameters(SubTemplate.java:20)
276 	 * Generated by spoon.test.template.testclasses.inheritance.SubTemplate#toBeOverriden(SubTemplate.java:16)
277 	 * Generated by spoon.test.template.testclasses.inheritance.SubTemplate#var(SubTemplate.java:51)
278 	 * Generated by spoon.test.template.testclasses.inheritance.SubTemplate$InnerClass(SubTemplate.java:76)
279 	 */
280 	private static final Pattern generatedByRE = Pattern.compile(".*Generated by ([^$#\\(]+)([^\\(]*)\\(([^\\.]+)\\.java:(\\d+)\\).*", Pattern.MULTILINE | Pattern.DOTALL);
281 	private static final Pattern typeMemberRE = Pattern.compile("[^\\.$#]+$");
282 	
283 	private void assertCommentHasGeneratedBy(CtElement e, String templateQName, Map<CtElement, String> elementToGeneratedByMember) {
284 		String docComment = e.getDocComment();
285 		String generatedByMember = elementToGeneratedByMember.get(e);
286 		if (generatedByMember == null) {
287 			assertEquals("", docComment);
288 //			assertTrue(e.getPosition().isValidPosition() == false);
289 		} else {
290 			assertNotNull("Javadoc comment is missing for "+e.toString(), docComment);
291 			int idx = docComment.indexOf("Generated by");
292 			assertTrue("Javadoc comment doesn't contain Generated by. There is:\n"+docComment, idx>=0);
293 			Matcher m = generatedByRE.matcher(docComment);
294 			assertTrue("Unexpected Generated by:\n"+docComment, m.matches());
295 			assertEquals(templateQName, m.group(1));
296 			assertEquals(generatedByMember, m.group(2));
297 			assertTrue(templateQName.endsWith("."+m.group(3)));
298 			int lineNr = Integer.parseInt(m.group(4))-1;
299 			assertTrue(lineNr>=0);
300 			String[] sourceLines = e.getPosition().getCompilationUnit().getOriginalSourceCode().split("\\r\\n|\\r|\\n");
301 			assertTrue(sourceLines.length>lineNr);
302 			String sourceLine = sourceLines[lineNr];
303 			Matcher m2 = typeMemberRE.matcher(generatedByMember);
304 			assertTrue(m2.find());
305 			String memberSimpleName = m2.group();
306 			assertTrue(memberSimpleName.length()>2);
307 			assertTrue("Source file "+e.getPosition().getFile().getAbsolutePath()+":"+lineNr+" doesn't contain member name "+memberSimpleName, sourceLine.contains(memberSimpleName));
308 		}
309 	}

639 			assertEquals("The arguments of keyValues must be in pairs", 0, keyValues.length % 2);
640 			Map<String, Object> allParams = new HashMap<>(match.getParameters().asMap());
641 			int count = keyValues.length / 2;
642 			for (int i = 0; i < count; i++) {
643 				String key = keyValues[i * 2];
644 				String expectedValue = keyValues[i * 2 + 1];
645 				Object realValue = allParams.remove(key);

647 			}

649 			return true;
