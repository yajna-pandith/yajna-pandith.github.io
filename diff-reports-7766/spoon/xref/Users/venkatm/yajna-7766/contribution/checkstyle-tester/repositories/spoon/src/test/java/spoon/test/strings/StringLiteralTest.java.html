17  package spoon.test.strings;
18  
19  import org.junit.Test;
20  import spoon.reflect.declaration.CtClass;
21  import spoon.reflect.declaration.CtField;
22  import spoon.reflect.factory.Factory;
23  
24  import static org.junit.Assert.assertEquals;
25  import static spoon.testing.utils.ModelUtils.createFactory;
26  
27  public class StringLiteralTest {
28  	@SuppressWarnings("unused")
29  	@Test
30  	public void testSnippetFullClass() {
31  		Factory factory = createFactory();
32  		CtClass<?> clazz = factory.Code().createCodeSnippetStatement(
33  				"class StringValueUTF {\n" +
34  				"	String f0 = \"toto\";\n" +
35  				"	String f1 = \"\\n\";\n" +
36  				"	char c1 = '\\n';\n" +
37  				"	String f2 = \"\\u20ac\";\n" +
38  				"	char c2 = '\\u20ac';\n" +
39  				"	String f3 = \"€\";\n" +
40  				"	char c3 = '€';\n" +
41  				"	String f4 = \"\\t\";\n" +
42  				"	char c4 = '\\t';\n" +
43  				"	String f5 = \"	\";\n" +
44  				"	char c5 = '	';\n" +
45  				"	String f6 = \"€\\u20ac\";\n" +
46  				"}"
47  		).compile();
48  		CtField<?> f0 = (CtField<?>) clazz.getFields().toArray()[0];
49  		CtField<?> f1 = (CtField<?>) clazz.getFields().toArray()[1];
50  		CtField<?> c1 = (CtField<?>) clazz.getFields().toArray()[2];
51  		CtField<?> f2 = (CtField<?>) clazz.getFields().toArray()[3];
52  		CtField<?> c2 = (CtField<?>) clazz.getFields().toArray()[4];
53  		CtField<?> f3 = (CtField<?>) clazz.getFields().toArray()[5];
54  		CtField<?> c3 = (CtField<?>) clazz.getFields().toArray()[6];
55  		CtField<?> f4 = (CtField<?>) clazz.getFields().toArray()[7];
56  		CtField<?> c4 = (CtField<?>) clazz.getFields().toArray()[8];
57  		CtField<?> f5 = (CtField<?>) clazz.getFields().toArray()[9];
58  		CtField<?> c5 = (CtField<?>) clazz.getFields().toArray()[10];
59  		CtField<?> f6 = (CtField<?>) clazz.getFields().toArray()[11];
60  
61  		assertEquals("java.lang.String f0 = \"toto\";", f0.toString());
62  		assertEquals("java.lang.String f1 = \"\\n\";", f1.toString());
63  		assertEquals("char c1 = '\\n';", c1.toString());
64  		assertEquals("java.lang.String f2 = \"\\u20ac\";", f2.toString());
65  		assertEquals("char c2 = '\\u20ac';", c2.toString());
66  		assertEquals("java.lang.String f3 = \"€\";", f3.toString());
67  		assertEquals("char c3 = '€';", c3.toString());
68  		assertEquals("java.lang.String f4 = \"\\t\";", f4.toString());
69  		assertEquals("char c4 = '\\t';", c4.toString());
70  		assertEquals("java.lang.String f5 = \"	\";", f5.toString());
71  		assertEquals("char c5 = '	';", c5.toString());
72  
73  		// spoon cannot handle unicode and unicode in the same literal
74  		// assertEquals("java.lang.String f6 = \"€\\u20ac\";", f6.toString());
75  	}
76  
77  }
