57  	private void checkJavaVersion() {
58  		String property = System.getProperty("java.version");
59  		if (property != null && !property.isEmpty()) {
60  
61  			// java 8 and less are versionning "1.X" where 9 and more are directly versioned "X"
62  			Assume.assumeFalse(property.startsWith("1."));
63  		}
64  	}

98  	@Test
99  	public void testCompleteModuleInfoContentNoClasspath() {
100 		// contract: all information of the module-info should be available through the model
101 		final Launcher launcher = new Launcher();
102 		launcher.addInputResource("./src/test/resources/spoon/test/module/simple_module/module-info.java");
103 		launcher.getEnvironment().setNoClasspath(true);
104 		launcher.getEnvironment().setComplianceLevel(9);
105 		launcher.buildModel();
106 
107 		assertEquals(2, launcher.getModel().getAllModules().size());
108 
109 		CtModule unnamedModule = launcher.getFactory().Module().getOrCreate(CtModule.TOP_LEVEL_MODULE_NAME);
110 		assertSame(unnamedModule, launcher.getModel().getUnnamedModule());
111 		CtModule moduleGreetings = launcher.getFactory().Module().getOrCreate("simple_module");
112 
113 		assertEquals("simple_module", moduleGreetings.getSimpleName());
114 
115 		assertEquals(7, moduleGreetings.getModuleDirectives().size());
116 
117 		List<CtModuleRequirement> requiredModules = moduleGreetings.getRequiredModules();
118 		assertEquals(1, requiredModules.size());
119 
120 		CtModuleRequirement moduleRequirement = requiredModules.get(0);
121 		assertEquals("java.logging", moduleRequirement.getModuleReference().getSimpleName());
122 		assertTrue(moduleRequirement.getRequiresModifiers().contains(CtModuleRequirement.RequiresModifier.TRANSITIVE));
123 
124 		List<CtPackageExport> moduleExports = moduleGreetings.getExportedPackages();
125 		assertEquals(1, moduleExports.size());
126 
127 		assertEquals("com.greetings.pkg", moduleExports.get(0).getPackageReference().getQualifiedName());
128 
129 		assertEquals(2, moduleExports.get(0).getTargetExport().size());
130 
131 		for (CtModuleReference target : moduleExports.get(0).getTargetExport()) {
132 			if (!"com.other.module".equals(target.getSimpleName()) && !"com.second.module".equals(target.getSimpleName())) {
133 				fail();
134 			}
135 		}
136 
137 		List<CtPackageExport> moduleOpened = moduleGreetings.getOpenedPackages();
138 		assertEquals(2, moduleOpened.size());
139 
140 		CtPackageExport openedFirst = moduleOpened.get(0);
141 		CtPackageExport openedSecond = moduleOpened.get(1);
142 
143 		assertEquals("com.greetings.otherpkg", openedFirst.getPackageReference().getSimpleName());
144 		assertTrue(openedFirst.getTargetExport().isEmpty());
145 
146 		assertEquals("com.greetings.openpkg", openedSecond.getPackageReference().getSimpleName());
147 		assertEquals(1, openedSecond.getTargetExport().size());
148 		assertEquals("com.third.module", openedSecond.getTargetExport().iterator().next().getSimpleName());
149 
150 		List<CtUsedService> consumedService = moduleGreetings.getUsedServices();
151 		assertEquals(1, consumedService.size());
152 		assertEquals("com.greetings.pkg.ConsumedService", consumedService.get(0).getServiceType().getQualifiedName());
153 
154 		List<CtProvidedService> providedServices = moduleGreetings.getProvidedServices();
155 		assertEquals(2, providedServices.size());
156 
157 		CtProvidedService providedService1 = providedServices.get(0);
158 		CtProvidedService providedService2 = providedServices.get(1);
159 
160 		assertEquals("com.greetings.pkg.ConsumedService", providedService1.getServiceType().getQualifiedName());
161 		assertEquals(2, providedService1.getImplementationTypes().size());
162 		assertEquals("com.greetings.pkg.ProvidedClass1", providedService1.getImplementationTypes().get(0).getQualifiedName());
163 		assertEquals("com.greetings.otherpkg.ProvidedClass2", providedService1.getImplementationTypes().get(1).getQualifiedName());
164 
165 		assertEquals("java.logging.Service", providedService2.getServiceType().getQualifiedName());
166 		assertEquals(1, providedService2.getImplementationTypes().size());
167 		assertEquals("com.greetings.logging.Logger", providedService2.getImplementationTypes().get(0).getQualifiedName());
168 	}

170 	@Test
171 	public void testModuleInfoShouldBeCorrectlyPrettyPrinted() throws IOException {
172 		// contract: module-info with complete information should be correctly pretty printed
173 
174 		File input = new File("./src/test/resources/spoon/test/module/simple_module/module-info.java");
175 		File output = new File("./target/spoon-module");
176 		final Launcher launcher = new Launcher();
177 		launcher.getEnvironment().setNoClasspath(true);
178 		launcher.getEnvironment().setComplianceLevel(9);
179 		launcher.setSourceOutputDirectory(output.getPath());
180 		launcher.addInputResource(input.getPath());
181 		launcher.run();
182 
183 		assertEquals(2, launcher.getModel().getAllModules().size());
184 
185 		try (Stream<Path> files = Files.list(output.toPath())) {
186 			assertEquals(2, files.count()); // should be 1 but for now we also have the module-info-tpl.
187 		}
188 		File fileOuput = new File(output, "simple_module/module-info.java");
189 		List<String> originalLines = Files.readAllLines(input.toPath());
190 		List<String> createdLines = Files.readAllLines(fileOuput.toPath());
191 
192 		assertEquals(originalLines.size(), createdLines.size());
193 
194 		for (int i = 0; i < originalLines.size(); i++) {
195 			assertEquals(originalLines.get(i), createdLines.get(i));
196 		}
197 	}

199 	@Test
200 	public void testModuleInfoWithComments() {
201 		// contract: documentation on module-info elements should be managed
202 
203 		final Launcher launcher = new Launcher();
204 		launcher.getEnvironment().setNoClasspath(true);
205 		launcher.getEnvironment().setComplianceLevel(9);
206 		launcher.getEnvironment().setCommentEnabled(true);
207 		launcher.addInputResource(MODULE_RESOURCES_PATH+"/module_with_comments/module-info.java");
208 		launcher.buildModel();
209 
210 		assertEquals(2, launcher.getModel().getAllModules().size());
211 		CtModule module = launcher.getFactory().Module().getModule("module_with_comments");
212 		assertNotNull(module);
213 
214 		assertTrue(module.isOpenModule());
215 
216 		List<CtComment> comments = module.getComments();
217 		assertEquals(1, comments.size());
218 
219 		CtComment comment = comments.get(0);
220 		assertEquals("This is the main module of the application", comment.getContent());
221 		assertEquals(CtComment.CommentType.JAVADOC, comment.getCommentType());
222 
223 		assertEquals(3, module.getModuleDirectives().size());
224 
225 		CtModuleRequirement moduleRequirement = module.getRequiredModules().get(0);
226 		comments = moduleRequirement.getComments();
227 		assertEquals(1, comments.size());
228 
229 		comment = comments.get(0);
230 		assertEquals("this is needed for logging stuff", comment.getContent());
231 		assertEquals(CtComment.CommentType.INLINE, comment.getCommentType());
232 
233 		CtProvidedService providedService = module.getProvidedServices().get(0);
234 		comments = providedService.getComments();
235 		assertEquals(1, comments.size());
236 
237 		comment = comments.get(0);
238 		assertEquals("A specific implementation", comment.getContent());
239 		assertEquals(CtComment.CommentType.JAVADOC, comment.getCommentType());
240 
241 		CtUsedService usedService = module.getUsedServices().get(0);
242 		comments = usedService.getComments();
243 		assertEquals(1, comments.size());
244 
245 		comment = comments.get(0);
246 		assertEquals("A simple implementation", comment.getContent());
247 		assertEquals(CtComment.CommentType.BLOCK, comment.getCommentType());
248 	}

295 	@Test
296 	public void testSimpleModuleCanBeBuilt() {
297 		checkJavaVersion();
298 		// contract: Spoon is able to build a simple model with a module in full classpath
299 		final Launcher launcher = new Launcher();
300 		launcher.getEnvironment().setComplianceLevel(9);
301 		launcher.getEnvironment().setNoClasspath(false);
302 		launcher.addInputResource("./src/test/resources/spoon/test/module/simple_module_with_code");
303 		launcher.run();
304 
305 		CtModel model = launcher.getModel();
306 
307 		// unnamed module and module 'simple_module_with_code'
308 		assertEquals(2, model.getAllModules().size());
309 		assertEquals(1, model.getAllTypes().size());
310 
311 		CtClass simpleClass = model.getElements(new TypeFilter<>(CtClass.class)).get(0);
312 		assertEquals("SimpleClass", simpleClass.getSimpleName());
313 
314 		CtModule simpleModule = model.getElements(new NamedElementFilter<>(CtModule.class, "simple_module_with_code")).get(0);
315 		assertNotNull(simpleModule);
316 		assertEquals("simple_module_with_code", simpleModule.getSimpleName());
317 
318 		CtModule module = simpleClass.getParent(CtModule.class);
319 		assertNotNull(module);
320 		assertSame(simpleModule, module);
321 	}
