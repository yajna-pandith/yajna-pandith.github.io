134 	public void testGetterSetterForRole() {
135 		// contract: all roles in spoon metamodel must at least have a setter and a getter
136 		SpoonAPI interfaces = new Launcher();
137 		interfaces.addInputResource("src/main/java/spoon/reflect/declaration");
138 		interfaces.addInputResource("src/main/java/spoon/reflect/code");
139 		interfaces.addInputResource("src/main/java/spoon/reflect/reference");
140 		interfaces.buildModel();
141 		Factory factory = interfaces.getFactory();
142 		CtTypeReference propertyGetter = factory.Type().get(PropertyGetter.class).getReference();
143 		CtTypeReference propertySetter = factory.Type().get(PropertySetter.class).getReference();
144 
145 		Set<String> expectedRoles = Arrays.stream(CtRole.values()).map(r -> r.name()).collect(Collectors.toSet());
146 
147 		List<CtMethod<?>> getters = interfaces.getModel().getElements(new AnnotationFilter<>(PropertyGetter.class));
148 		Set<String> getterRoles = getters.stream().map(g -> ((CtFieldRead) g.getAnnotation(propertyGetter).getValue("role")).getVariable().getSimpleName()).collect(Collectors.toSet());
149 		Set<CtMethod<?>> isNotGetter = getters.stream().filter(m -> !(m.getSimpleName().startsWith("get") || m.getSimpleName().startsWith("is"))).collect(Collectors.toSet());
150 
151 		List<CtMethod<?>> setters = interfaces.getModel().getElements(new AnnotationFilter<>(PropertySetter.class));
152 		Set<String> setterRoles = setters.stream().map(g -> ((CtFieldRead) g.getAnnotation(propertySetter).getValue("role")).getVariable().getSimpleName()).collect(Collectors.toSet());
153 		Set<CtMethod<?>> isNotSetter = setters.stream().filter(m -> !(m.getSimpleName().startsWith("set") || m.getSimpleName().startsWith("add") || m.getSimpleName().startsWith("insert") || m.getSimpleName().startsWith("remove"))).collect(Collectors.toSet());
154 
155 		assertEquals(expectedRoles, getterRoles);
156 		//derived roles with no setter:
157 		expectedRoles.remove(CtRole.DECLARED_MODULE.name());
158 		expectedRoles.remove(CtRole.DECLARED_TYPE.name());
159 		expectedRoles.remove(CtRole.EMODIFIER.name());
160 
161 		assertSetEquals("", expectedRoles, setterRoles);
162 
163 		assertEquals(Collections.EMPTY_SET, isNotGetter);
164 		assertEquals(Collections.EMPTY_SET, isNotSetter);
165 	}

179 	@Test
180 	public void testRoleOnField() {
181 		//  contract: all non-final / transient / static fields must be annotated with {@link spoon.reflect.annotations.MetamodelPropertyField}
182 		SpoonAPI implementations = new Launcher();
183 		implementations.addInputResource("src/main/java/spoon/support/reflect");
184 		implementations.buildModel();
185 
186 		String nl = System.getProperty("line.separator");
187 		Factory factory = implementations.getFactory();
188 
189 		CtTypeReference metamodelPropertyField = factory.Type().get(MetamodelPropertyField.class).getReference();
190 
191 		final List<String> result = new ArrayList();
192 		List<CtField> fieldWithoutAnnotation = (List<CtField>) implementations.getModel().getElements(new TypeFilter<CtField>(CtField.class) {
193 			@Override
194 			public boolean matches(CtField candidate) {
195 				if (candidate.hasModifier(ModifierKind.FINAL) || candidate.hasModifier(ModifierKind.STATIC) || candidate.hasModifier(ModifierKind.TRANSIENT)) {
196 					return false;
197 				}
198 				if (IGNORED_FIELD_NAMES.contains(candidate.getSimpleName())) {
199 					// not a role
200 					return false;
201 				}
202 				CtClass parent = candidate.getParent(CtClass.class);
203 				return parent != null
204 						// code and reference element fields must be annotated
205 						&&
206 						(parent.isSubtypeOf(candidate.getFactory().createCtTypeReference(CtReference.class))
207 						|| parent.isSubtypeOf(candidate.getFactory().createCtTypeReference(CtElement.class))
208 						);
209 			}
210 		}).stream().map(x -> {
211 			result.add(x.toString()); return x;
212 		}).filter(f -> f.getAnnotation(metamodelPropertyField) == null).collect(Collectors.toList());
213 
214 		assertTrue(result.contains("@spoon.reflect.annotations.MetamodelPropertyField(role = spoon.reflect.path.CtRole.IS_SHADOW)" + nl + "boolean isShadow;"));
215 		assertTrue(result.contains("@spoon.reflect.annotations.MetamodelPropertyField(role = spoon.reflect.path.CtRole.TYPE)" + nl + "spoon.reflect.reference.CtTypeReference<T> type;"));
216 		assertTrue(result.size() > 100);
217 		assertEquals(Collections.emptyList(), fieldWithoutAnnotation);
218 
219 		final CtTypeReference propertySetter = factory.Type().get(PropertySetter.class).getReference();
220 		final CtTypeReference propertyGetter = factory.Type().get(PropertyGetter.class).getReference();
221 
222 		List<CtField> fields = factory.getModel().getElements(new AnnotationFilter<>(MetamodelPropertyField.class));
223 		for (CtField field : fields) {
224 			CtClass parent = field.getParent(CtClass.class);
225 			CtExpression roleExpression = field.getAnnotation(metamodelPropertyField).getValue("role");
226 			List<String> roles = new ArrayList<>();
227 			if (roleExpression instanceof CtFieldRead) {
228 				roles.add(((CtFieldRead) roleExpression).getVariable().getSimpleName());
229 			} else  if (roleExpression instanceof CtNewArray) {
230 				List<CtFieldRead> elements = ((CtNewArray) roleExpression).getElements();
231 				for (CtFieldRead ctFieldRead : elements) {
232 					roles.add(ctFieldRead.getVariable().getSimpleName());
233 				}
234 			}
235 
236 			CtQuery superQuery = parent.map(new SuperInheritanceHierarchyFunction());
237 
238 			List<CtMethod> methods = superQuery.map((CtType type) -> type.getMethodsAnnotatedWith(propertyGetter, propertySetter)).list();
239 
240 			boolean setterFound = false;
241 			boolean getterFound = false;
242 			for (CtMethod method : methods) {
243 				CtAnnotation getterAnnotation = method.getAnnotation(propertyGetter);
244 				CtAnnotation setterAnnotation = method.getAnnotation(propertySetter);
245 				if (getterAnnotation != null) {
246 					getterFound |= roles.contains(((CtFieldRead) getterAnnotation.getValue("role")).getVariable().getSimpleName());
247 				}
248 				if (setterAnnotation != null) {
249 					setterFound |= roles.contains(((CtFieldRead) setterAnnotation.getValue("role")).getVariable().getSimpleName());
250 				}
251 			}
252 
253 			assertTrue(roles + " must have a getter in " + parent.getQualifiedName(), getterFound);
254 			assertTrue(roles + " must have a setter in " + parent.getQualifiedName(), setterFound);
255 		}
256 	}

277 	private void assertConceptsEqual(MetamodelConcept expectedConcept, MetamodelConcept runtimeConcept) {
278 		assertEquals(expectedConcept.getName(), runtimeConcept.getName());
279 		if (expectedConcept.getImplementationClass() == null) {
280 			assertNull(runtimeConcept.getImplementationClass());
281 		} else {
282 			assertNotNull(runtimeConcept.getImplementationClass());
283 			assertSame(expectedConcept.getImplementationClass().getActualClass(), runtimeConcept.getImplementationClass().getActualClass());
284 		}
285 		assertSame(expectedConcept.getMetamodelInterface().getActualClass(), runtimeConcept.getMetamodelInterface().getActualClass());
286 		assertEquals(expectedConcept.getKind(), runtimeConcept.getKind());
287 		assertEquals(expectedConcept.getSuperConcepts().size(), runtimeConcept.getSuperConcepts().size());
288 		for (int i = 0; i < expectedConcept.getSuperConcepts().size(); i++) {
289 			assertConceptsEqual(expectedConcept.getSuperConcepts().get(i), runtimeConcept.getSuperConcepts().get(i));
290 		}
291 		Map<CtRole, MetamodelProperty> expectedRoleToProperty = new HashMap(expectedConcept.getRoleToProperty());
292 		for (Map.Entry<CtRole, MetamodelProperty> e : runtimeConcept.getRoleToProperty().entrySet()) {
293 			MetamodelProperty runtimeProperty = e.getValue();
294 			MetamodelProperty expectedProperty = expectedRoleToProperty.remove(e.getKey());
295 			assertPropertiesEqual(expectedProperty, runtimeProperty);
296 		}
297 		assertEquals(0, expectedRoleToProperty.size());
298 	}

375 		Class ifaceClass =  iface.getActualClass();
376 		//contract: check that for each Spoon model interface we have correct list of Role handlers
377 		List<RoleHandler> roleHandlersOfIFace = new ArrayList<>(RoleHandlerHelper.getRoleHandlers(ifaceClass));

379 		RoleHandlerHelper.forEachRoleHandler(rh -> allRoleHandlers.add(rh));

385 			}
386 		}
387 		assertTrue("There are unexpected RoleHandlers " + roleHandlersOfIFace + " for " + ifaceClass, roleHandlersOfIFace.isEmpty());

520 		//contract: single value roles supports multivalue interface too
521 		Launcher launcher = new Launcher();
522 		Factory factory = launcher.getFactory();
523 		CtTypeReference<?> typeRef = factory.Type().createReference("some.test.package.TestType");
524 		RoleHandler rh = RoleHandlerHelper.getRoleHandler(typeRef.getClass(), CtRole.PACKAGE_REF);
525 
526 		//contract: single value role provides a List
527 		List<CtPackageReference> packages = rh.asList(typeRef);
528 		assertListContracts(packages, typeRef, 1, "some.test.package");
529 
530 		//contract: adding of existing value fails and changes nothing
531 		try {
532 			packages.add(typeRef.getPackage());
533 			fail();
534 		} catch (Exception e) {
535 			//OK
536 		}
537 		assertListContracts(packages, typeRef, 1, "some.test.package");
538 
539 		//contract: adding of null fails and changes nothing
540 		try {
541 			assertFalse(packages.add(null));
542 			fail();
543 		} catch (Exception e) {
544 			//OK
545 		}
546 		assertListContracts(packages, typeRef, 1, "some.test.package");
547 
548 		//contract: adding of different value fails, and changes nothing
549 		try {
550 			packages.add(factory.Package().createReference("some.test.another_package"));
551 			fail();
552 		} catch (SpoonException e) {
553 			//OK
554 		}
555 		assertListContracts(packages, typeRef, 1, "some.test.package");
556 
557 		//contract remove of different value changes nothing
558 		assertFalse(packages.remove(factory.Package().createReference("some.test.another_package")));
559 		assertListContracts(packages, typeRef, 1, "some.test.package");
560 
561 		//contract remove of null value changes nothing
562 		assertFalse(packages.remove(null));
563 		assertListContracts(packages, typeRef, 1, "some.test.package");
564 
565 		//contract remove of existing value sets value to null and size to 0
566 		assertTrue(packages.remove(factory.Package().createReference("some.test.package")));
567 		assertListContracts(packages, typeRef, 0, null);
568 
569 		//contract add of null into empty collection changes size to 1, but value is still null
570 		assertTrue(packages.add(null));
571 		assertListContracts(packages, typeRef, 1, null);
572 
573 		//contract: adding of new value into collection with single null value fails and changes nothing
574 		try {
575 			packages.add(factory.Package().createReference("some.test.another_package"));
576 			fail();
577 		} catch (SpoonException e) {
578 			//OK
579 		}
580 		assertListContracts(packages, typeRef, 1, null);
581 
582 		//contract: set of new value replaces existing value
583 		assertNull(packages.set(0, factory.Package().createReference("some.test.package")));
584 		assertListContracts(packages, typeRef, 1, "some.test.package");
585 
586 		//contract: set of null value keeps size==1 even if value is replaced by null
587 		assertEquals("some.test.package", packages.set(0, null).getQualifiedName());
588 		assertListContracts(packages, typeRef, 1, null);
589 
590 		//contract: remove of null value by index sets size==0 the value is still null
591 		assertNull(packages.remove(0));
592 		assertListContracts(packages, typeRef, 0, null);
593 
594 		//contract: add of null value sets size==1 the value is still null
595 		assertTrue(packages.add(null));
596 		assertListContracts(packages, typeRef, 1, null);
597 
598 		//contract: remove of null value by value sets size==0 the value is still null
599 		assertTrue(packages.remove(null));
600 		assertListContracts(packages, typeRef, 0, null);
601 
602 		//contract: set of new value on empty collection fails with IndexOutOfBounds and changes nothing
603 		try {
604 			packages.set(0, factory.Package().createReference("some.test.another_package"));
605 			fail();
606 		} catch (IndexOutOfBoundsException e) {
607 			//OK
608 		}
609 		assertListContracts(packages, typeRef, 0, null);
610 
611 		//contract: adding of value into empty collection adds value
612 		assertTrue(packages.add(factory.Package().createReference("some.test.another_package")));
613 		assertListContracts(packages, typeRef, 1, "some.test.another_package");
614 
615 		//contract: remove of value by index from collection removes that value
616 		assertEquals("some.test.another_package", packages.remove(0).getQualifiedName());
617 		assertListContracts(packages, typeRef, 0, null);
618 	}

620 	private void assertListContracts(List<CtPackageReference> packages, CtTypeReference<?> typeRef, int expectedSize, String expectedValue) {
621 		if (expectedSize == 0) {
622 			assertEquals(0, packages.size());
623 			assertNull(typeRef.getPackage());
624 			//contract: get(x) fails for each x when called on empty collection
625 			for (int i = -1; i < 3; i++) {
626 				try {
627 					packages.get(i);
628 					fail();
629 				} catch (IndexOutOfBoundsException e) {
630 					//OK
631 				}
632 			}
633 		} else if (expectedSize == 1) {
634 			assertEquals(1, packages.size());
635 			assertPackageName(expectedValue, typeRef.getPackage());
636 			//contract: get(x) fails for each x when called on collection with one item, excluding for index == 0
637 			for (int i = -1; i < 3; i++) {
638 				if (i == 0) {
639 					assertPackageName(expectedValue, packages.get(0));
640 				} else {
641 					try {
642 						packages.get(i);
643 						fail();
644 					} catch (IndexOutOfBoundsException e) {
645 						//OK
646 					}
647 				}
648 			}
649 		} else {
650 			fail();
651 		}
652 	}

656 			assertNull(packageRef);
657 		} else {
658 			assertEquals(expectedPackageName, packageRef.getQualifiedName());

663 	public void listValueRoleSetOn() {
664 		//contract: multi-value role supports set(int, Object)
665 		Launcher launcher = new Launcher();
666 		Factory factory = launcher.getFactory();
667 		CtClass<?> ctClass = factory.Class().create("some.test.TestClass");
668 		RoleHandler rh = RoleHandlerHelper.getRoleHandler(ctClass.getClass(), CtRole.TYPE_MEMBER);
669 		List<CtTypeMember> typeMembers = rh.asList(ctClass);
670 		assertEquals(0, typeMembers.size());
671 		CtField<?> field1 = createField(factory, "field1");
672 		CtField<?> field2 = createField(factory, "field2");
673 		CtField<?> field3 = createField(factory, "field3");
674 		//check that field was not added in type yet
675 		assertEquals(0, typeMembers.size());
676 		//contract: call of add on RoleHandler collection adds the item into real collection too
677 		typeMembers.add(field1);
678 		assertEquals(1, typeMembers.size());
679 		assertEquals(1, ctClass.getTypeMembers().size());
680 		assertSame(ctClass, field1.getDeclaringType());
681 		assertThat(Arrays.asList("field1"), is(ctClass.filterChildren(new TypeFilter(CtField.class)).map((CtField e) -> e.getSimpleName()).list()));
682 		//contract: call of add on RoleHandler collection adds the item into real collection too
683 		typeMembers.add(field2);
684 		assertSame(ctClass, field2.getDeclaringType());
685 		assertThat(Arrays.asList("field1", "field2"), is(ctClass.filterChildren(new TypeFilter(CtField.class)).map((CtField e) -> e.getSimpleName()).list()));
686 		//contract: call of set on RoleHandler collection replaces the item in real collection
687 		typeMembers.set(0, field3);
688 		assertSame(ctClass, field3.getDeclaringType());
689 		assertThat(Arrays.asList("field3", "field2"), is(ctClass.filterChildren(new TypeFilter(CtField.class)).map((CtField e) -> e.getSimpleName()).list()));
690 		typeMembers.set(1, field1);
691 		assertThat(Arrays.asList("field3", "field1"), is(ctClass.filterChildren(new TypeFilter(CtField.class)).map((CtField e) -> e.getSimpleName()).list()));
692 		//contract: call of remove(int) on RoleHandler collection removes the item in real collection
693 		assertSame(field3, typeMembers.remove(0));
694 		assertThat(Arrays.asList("field1"), is(ctClass.filterChildren(new TypeFilter(CtField.class)).map((CtField e) -> e.getSimpleName()).list()));
695 		//contract: call of remove(Object) which does not exist does nothing
696 		assertFalse(typeMembers.remove(field2));
697 		assertThat(Arrays.asList("field1"), is(ctClass.filterChildren(new TypeFilter(CtField.class)).map((CtField e) -> e.getSimpleName()).list()));
698 		//contract: call of remove(Object) on RoleHandler collection removes the item in real collection
699 		assertTrue(typeMembers.remove(field1));
700 		assertThat(Arrays.asList(), is(ctClass.filterChildren(new TypeFilter(CtField.class)).map((CtField e) -> e.getSimpleName()).list()));
701 	}
