42  /**
43   * {@link PrettyPrinter} implementation, which copies as much as possible from the origin sources
44   * and tries to only print the changed elements.
45   */
46  @Experimental
47  public class SniperJavaPrettyPrinter extends DefaultJavaPrettyPrinter {
48  
49  	private final MutableTokenWriter mutableTokenWriter;
50  	private ChangeResolver changeResolver;
51  	private final Deque<SourceFragmentContext> sourceFragmentContextStack = new ArrayDeque<>();
52  
53  	/**
54  	 * Creates a new {@link PrettyPrinter} which copies origin sources and prints only changes.
55  	 */
56  	public SniperJavaPrettyPrinter(Environment env) {
57  		super(env);
58  		inlineElseIf = false;
59  		// required for sniper mode
60  		env.useTabulations(true);
61  		env.setCommentEnabled(true);
62  		env.setOutputType(OutputType.COMPILATION_UNITS);
63  		//create a TokenWriter which can be configured to ignore tokens coming from DJPP
64  		mutableTokenWriter = new MutableTokenWriter(env);
65  		//wrap that TokenWriter to listen on all incoming events and set wrapped version to DJPP
66  		setPrinterTokenWriter(createTokenWriterListener(mutableTokenWriter));
67  	}
68  
69  	/**
70  	 * @return {@link ChangeCollector} registered in spoon environment
71  	 */
72  	private ChangeCollector getChangeCollector() {
73  		ChangeCollector changeCollector = ChangeCollector.getChangeCollector(env);
74  		if (changeCollector == null) {
75  			throw new SpoonException(ChangeCollector.class.getSimpleName() + " was not attached to the Environment");
76  		}
77  		return changeCollector;
78  	}
79  
80  	/**
81  	 * @return new or cached instance of root ChangeResolver
82  	 */
83  	private ChangeResolver getChangeResolver() {
84  		if (changeResolver == null) {
85  			changeResolver = new ChangeResolver(getChangeCollector(), null);
86  		}
87  		return changeResolver;
88  	}

90  	/**
91  	 * wrap a `tokenWriter` by a proxy which intercepts all {@link TokenWriter} writeXxx(String) calls
92  	 * and calls {@link #onTokenWriterWrite(TokenType, String, CtComment, Runnable)}
93  	 * where {@link Runnable} can be used to invoke same event on the wrapped {@link TokenWriter}
94  	 * @param tokenWriter to be wrapped {@link TokenWriter}
95  	 * @return a proxy of {@link TokenWriter}
96  	 */
97  	private TokenWriter createTokenWriterListener(TokenWriter tokenWriter) {
98  		return new TokenWriterProxy(this::onTokenWriterWrite, tokenWriter);
99  	}

101 	@Override
102 	protected void scanCompilationUnit(CtCompilationUnit compilationUnit) {
103 		//use line separator of origin source file
104 		setLineSeparator(detectLineSeparator(compilationUnit.getOriginalSourceCode()));
105 		runInContext(new SourceFragmentContextList(mutableTokenWriter,
106 				compilationUnit,
107 				Collections.singletonList(compilationUnit.getOriginalSourceFragment()),
108 				new ChangeResolver(getChangeCollector(), compilationUnit)),
109 		() -> {
110 			super.scanCompilationUnit(compilationUnit);
111 		});
112 	}

114 	private static final String CR = "\r";
115 	private static final String CRLF = "\r\n";
116 	private static final String LF = "\n";
117 
118 	/**
119 	 * Detect line separator used in origin code
120 	 * @return character sequence used as line separator in `text`
121 	 */
122 	private String detectLineSeparator(String text) {
123 		if (text != null) {
124 			int len = text.length();
125 			for (int i = 0; i < len; i++) {
126 				char c = text.charAt(i);
127 				if (c == '\n') {
128 					return LF;
129 				} else if (c == '\r') {
130 					i++;
131 					if (i < len && text.charAt(i) == '\n') {
132 						return CRLF;
133 					}
134 					return CR;
135 				}
136 			}
137 		}
138 		return System.getProperty("line.separator");
139 	}

141 	/**
142 	 * Called for each printed token
143 	 * @param tokenType the type of {@link TokenWriter} method
144 	 * @param token the actual token value. It may be null for some `tokenWriterMethodName`
145 	 * @param comment the comment when `tokenType` == `COMMENT`
146 	 * @param printAction the executor of the action, we are listening for. Call it send token to output
147 	 */
148 	void onTokenWriterWrite(TokenType tokenType, String token, CtComment comment, Runnable printAction) {
149 		onPrintEvent(new TokenPrinterEvent(tokenType, token, comment) {
150 			@Override
151 			public void print(Boolean muted) {
152 				runInMutedState(muted, printAction);
153 			}
154 			@Override
155 			public void printSourceFragment(SourceFragment fragment, Boolean isModified) {
156 				boolean isCollectionStarted = false;
157 				if (fragment instanceof CollectionSourceFragment) {
158 					CollectionSourceFragment csf = (CollectionSourceFragment) fragment;
159 					//we started scanning of collection of elements
160 					SourceFragmentContext listContext = csf.isOrdered()
161 							? new SourceFragmentContextList(mutableTokenWriter, null, csf.getItems(), getChangeResolver())
162 							: new SourceFragmentContextSet(mutableTokenWriter, null, csf.getItems(), getChangeResolver());
163 					//push the context of this collection
164 					sourceFragmentContextStack.push(listContext);
165 					isCollectionStarted = true;
166 				}
167 				if (isModified == null || isModified) {
168 					//print origin token
169 					printAction.run();
170 					return;
171 				}
172 				mutableTokenWriter.getPrinterHelper().directPrint(fragment.getSourceCode());
173 				if (isCollectionStarted) {
174 					mutableTokenWriter.setMuted(true);
175 				}
176 			}
177 		});
178 	}

182 	private static boolean hasImplicitAncestor(CtElement el) {
183 		if (el == null) {
184 			return false;
185 		}
186 		if (el == el.getFactory().getModel().getRootPackage()) {
187 			return false;
188 		} else if (el.isImplicit()) {
189 			return true;
190 		} else {
191 			return hasImplicitAncestor(el.getParent());
192 		}
193 	}

195 	/**
196 	 * SniperPrettyPrinter does not apply preprocessor to a CtElement when calling toString()
197 	 * @param element
198 	 * @return
199 	 */
200 	@Override
201 	public String printElement(CtElement element) {
202 		if (element != null && !hasImplicitAncestor(element)) {
203 			CompilationUnit compilationUnit = element.getPosition().getCompilationUnit();
204 			if (compilationUnit != null
205 					&& !(compilationUnit instanceof NoSourcePosition.NullCompilationUnit)) {
206 
207 				//use line separator of origin source file
208 				setLineSeparator(detectLineSeparator(compilationUnit.getOriginalSourceCode()));
209 
210 				CtRole role = getRoleInCompilationUnit(element);
211 				ElementSourceFragment esf = element.getOriginalSourceFragment();
212 
213 				runInContext(
214 					new SourceFragmentContextList(mutableTokenWriter,
215 						element,
216 						Collections.singletonList(esf),
217 						new ChangeResolver(getChangeCollector(), element)),
218 					() -> onPrintEvent(new ElementPrinterEvent(role, element) {
219 						@Override
220 						public void print(Boolean muted) {
221 							superScanInContext(element, SourceFragmentContextPrettyPrint.INSTANCE, muted);
222 						}
223 
224 						@Override
225 						public void printSourceFragment(SourceFragment fragment, Boolean isModified) {
226 							scanInternal(role, element, fragment, isModified);
227 						}
228 					})
229 				);
230 			}
231 		}
232 
233 		return toString().replaceFirst("^\\s+", "");
234 	}

237 	/**
238 	 * Called whenever {@link DefaultJavaPrettyPrinter} scans/prints an element
239 	 * Warning: DO not call on a cloned element. Use scanClone instead.
240 	 */
241 	@Override
242 	public SniperJavaPrettyPrinter scan(CtElement element) {
243 		if (element != null) {
244 			CtRole role = getRoleInCompilationUnit(element);
245 			onPrintEvent(new ElementPrinterEvent(role, element) {
246 				@Override
247 				public void print(Boolean muted) {
248 					superScanInContext(element, SourceFragmentContextPrettyPrint.INSTANCE, muted);
249 				}
250 				@Override
251 				public void printSourceFragment(SourceFragment fragment, Boolean isModified) {
252 					scanInternal(role, element, fragment, isModified);
253 				}
254 			});
255 		}
256 		return this;
257 	}
258 
259 	private CtRole getRoleInCompilationUnit(CtElement element) {
260 		CtRole role = element.getRoleInParent();
261 		if (role == CtRole.CONTAINED_TYPE) {
262 			role = CtRole.DECLARED_TYPE;
263 		}
264 		return role;
265 	}
266 
267 	/**
268 	 * Called whenever {@link DefaultJavaPrettyPrinter} scans/prints an element or writes a token
269 	 */
270 	private void onPrintEvent(PrinterEvent event) {
271 		SourceFragmentContext sfc = detectCurrentContext(event);
272 		if (sfc == null) {
273 			throw new SpoonException("Missing SourceFragmentContext");
274 		}
275 		//there is an context let it handle scanning
276 		if (mutableTokenWriter.isMuted()) {
277 			//it is already muted by an parent. Simply scan and ignore all tokens,
278 			event.print(null);
279 			return;
280 		}
281 		//let context handle the event
282 		sfc.onPrintEvent(event);
283 	}
284 
285 	/**
286 	 * Checks if `event` can be still handled by the current context.
287 	 * If not then finish current context and enter parent context
288 	 * @param event a to be processed {@link PrinterEvent}
289 	 * @return context which can be used to process `event`
290 	 */
291 	private SourceFragmentContext detectCurrentContext(PrinterEvent event) {
292 		SourceFragmentContext sfc;
293 		while ((sfc = sourceFragmentContextStack.peek()) != null && sfc.matchesPrinterEvent(event) == false) {
294 			//this context handles only subset of roles, which just finished
295 			//leave it and return back to parent context
296 			sourceFragmentContextStack.pop();
297 			sfc.onFinished();
298 		}
299 		return sfc;
300 	}

302 	/**
303 	 * scans the `element` which exist on `role` in it's parent
304 	 * @param role {@link CtRole} of `element` in scope of it's parent
305 	 * @param element a scanned element
306 	 * @param fragment origin source fragment of element
307 	 * @param isFragmentModified true if any part of `fragment` is modified, false if whole fragment is not modified, null if caller doesn't know
308 	 */
309 	private void scanInternal(CtRole role, CtElement element, SourceFragment fragment, Boolean isFragmentModified) {
310 		if (mutableTokenWriter.isMuted()) {
311 			throw new SpoonException("Unexpected state of sniper pretty printer. TokenWriter is muted.");
312 		}
313 		//it is not muted yet, so this element or any sibling is modified
314 		if (fragment == null) {
315 			throw new SpoonException("Missing source fragment. Call PrintEvent#print instead.");
316 		}
317 		//we have sources of fragment
318 		if (fragment instanceof CollectionSourceFragment) {
319 			CollectionSourceFragment csf = (CollectionSourceFragment) fragment;
320 			//we started scanning of collection of elements
321 			SourceFragmentContext listContext = csf.isOrdered()
322 					? new SourceFragmentContextList(mutableTokenWriter, element, csf.getItems(), getChangeResolver())
323 					: new SourceFragmentContextSet(mutableTokenWriter, element, csf.getItems(), getChangeResolver());
324 			//push the context of this collection
325 			sourceFragmentContextStack.push(listContext);
326 			//and scan first element of that collection again in new context of that collection
327 			if (Boolean.FALSE.equals(isFragmentModified)) {
328 				mutableTokenWriter.getPrinterHelper().directPrint(fragment.getSourceCode());
329 				//and mute the token writer and let DJPP scan it and ignore everything
330 				mutableTokenWriter.setMuted(true);
331 				//TODO check if DJPP needs this call somewhere (because of some state)... may be we can skip this scan completely??
332 				scan(element);
333 				//and keep it muted until SourceFragmentContextList is finished
334 			} else {
335 				scan(element);
336 			}
337 		} else if (fragment instanceof ElementSourceFragment) {
338 			ElementSourceFragment sourceFragment = (ElementSourceFragment) fragment;
339 			//it is fragment with single value
340 			ChangeResolver changeResolver = null;
341 			if (isFragmentModified == null) {
342 				changeResolver = new ChangeResolver(getChangeCollector(), element);
343 				isFragmentModified = changeResolver.hasChangedRole();
344 			}
345 			if (isFragmentModified == false) {
346 				//nothing is changed, we can print origin sources of this element
347 				mutableTokenWriter.getPrinterHelper().directPrint(fragment.getSourceCode());
348 				//and mute the token writer and let DJPP scan it and ignore everything
349 				//TODO check if DJPP needs this call somewhere (because of some state)... may be we can skip this scan completely??
350 				superScanInContext(element, SourceFragmentContextPrettyPrint.INSTANCE, true);
351 				return;
352 			}
353 			//check what roles of this element are changed
354 			if (changeResolver == null) {
355 				changeResolver = new ChangeResolver(getChangeCollector(), element);
356 			}
357 			//changeResolver.hasChangedRole() is false when element is added
358 			//something is changed in this element
359 			superScanInContext(element, new SourceFragmentContextNormal(mutableTokenWriter, sourceFragment, changeResolver), false);
360 		} else {
361 			throw new SpoonException("Unsupported fragment type: " + fragment.getClass());
362 		}
363 	}

365 	/**
366 	 * Call normal java printing in defined `context`
367 	 * @param element to be printed element
368 	 * @param context to be used context
369 	 * @param muted whether it has to be processed using
370 	 * 	true - muted,
371 	 * 	false - not muted
372 	 * 	null - same like before
373 	 */
374 	private void superScanInContext(CtElement element, SourceFragmentContext context, Boolean muted) {
375 		runInContext(context,
376 				() -> runInMutedState(muted,
377 						() -> super.scan(element)));
378 	}

380 	/**
381 	 * Run `code` in scope of `context`
382 	 * @param context a to be entered `context`
383 	 * @param code a to be processed {@link Runnable}
384 	 */
385 	private void runInContext(SourceFragmentContext context, Runnable code) {
386 		sourceFragmentContextStack.push(context);
387 		try {
388 			code.run();
389 		} finally {
390 			//remove `context` and all it's child contexts
391 			while (true) {
392 				if (sourceFragmentContextStack.isEmpty()) {
393 					throw new SpoonException("Inconsistent sourceFragmentContextStack");
394 				}
395 				SourceFragmentContext c = sourceFragmentContextStack.pop();
396 				c.onFinished();
397 				if (c == context) {
398 					break;
399 				}
400 			}
401 		}
402 	}
403 	/**
404 	 * Run code using {@link MutableTokenWriter} in defined state.
405 	 * After this function leaves, the muted status is restored.
406 	 * @param muted required muted status
407 	 * @param code to be processed {@link Runnable}
408 	 */
409 	private void runInMutedState(Boolean muted, Runnable code) {
410 		boolean originMuted = mutableTokenWriter.isMuted();
411 		if (muted == null) {
412 			muted = originMuted;
413 		}
414 		try {
415 			mutableTokenWriter.setMuted(muted);
416 			code.run();
417 		} finally {
418 			//assure that muted status did not changed in between
419 			if (mutableTokenWriter.isMuted() != muted) {
420 				if (mutableTokenWriter.isMuted()) {
421 					throw new SpoonException("Unexpected state: Token writer is muted after scanning");
422 				} else {
423 					throw new SpoonException("Unexpected state: Token writer is not muted after scanning");
424 				}
425 			}
426 			mutableTokenWriter.setMuted(originMuted);
427 		}
