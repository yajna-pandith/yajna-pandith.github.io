72      public MovePath calculateMoveTurn() {
73          long enter = System.currentTimeMillis();
74          int initiative = 0;
75          MoveOption min = null;
76  
77          System.out.println("beginning movement calculations...");
78  
79          //first check and that someone else has moved so we don't replan
80          Object[] enemy_array = this.getEnemyEntities().toArray();
81          for (int j = 0; j < enemy_array.length; j++) {
82              if (!((Entity) enemy_array[j]).isSelectableThisTurn()) {
83                  initiative++;
84              }
85          }
86          // if nobody's moved and we have a valid move waiting, use that
87          if (initiative == enemies_moved && old_moves != null) {
88              min = this.old_moves.getResult();
89              if (min == null
90                  || !min.isMoveLegal()
91                  || (min.isPhysical && centities.get(min.getPhysicalTargetId()).isPhysicalTarget)) {
92                  this.old_moves = null;
93                  System.out.println("recalculating moves since the old move was invalid");
94                  return calculateMoveTurn();
95              }
96          } else {
97              enemies_moved = initiative;
98              Vector possible = new Vector();
99  
100             Enumeration e = game.getEntities();
101 
102             while (e.hasMoreElements()) {
103                 Entity entity = (Entity) e.nextElement();
104                 
105                 // ignore loaded units
106                 if (entity.getPosition().isValidPosition() == false) {
107                     continue;
108                 }
109                 
110                 CEntity cen = centities.get(entity);
111                 cen.refresh();
112                 firstPass(cen);
113             }
114 
115             Iterator i = this.getEntitiesOwned().iterator();
116             boolean short_circuit = false;
117 
118             while (i.hasNext() && !short_circuit) {
119                 Entity entity = (Entity) i.next();
120 
121                 // ignore loaded units 
122                 // (not really necessary unless bot manages to load units)
123                 if (entity.getPosition().isValidPosition() == false) {
124                     continue;
125                 }
126 
127                 // if we can't move this entity right now, ignore it
128                 if (!game.getTurn().isValidEntity(entity, game)) {
129                     continue;
130                 }
131 
132                 CEntity cen = centities.get(entity);
133 
134                 System.out.println("Contemplating movement of " + entity.getShortName() + " " + entity.getId());
135 
136                 MoveOption[] result = calculateMove(entity);
137 
138                 if (game.getOptions().booleanOption("skip_ineligable_movement") && cen.getEntity().isImmobile()) {
139                     cen.moved = true;
140                 } else if (!cen.moved) {
141                     if (result.length < 6) {
142                         min = result.length > 0 ? (MoveOption) result[0] : null;
143                         short_circuit = true;
144                     }
145                     possible.add(result);
146                 }
147             }
148 
149             //should ignore mechs that are not engaged
150             //and only do the below when there are 2 or mechs left to move
151             if (!short_circuit) {
152                 if (this.getEntitiesOwned().size() > 1) {
153                     GALance lance = new GALance(this, possible, 50, 80);
154                     lance.evolve();
155                     min = lance.getResult();
156                     this.old_moves = lance;
157                 } else if (
158                     ((MoveOption[]) possible.elementAt(0)) != null
159                         && ((MoveOption[]) possible.elementAt(0)).length > 0) {
160                     min = ((MoveOption[]) possible.elementAt(0))[0];
161                 }
162             }
163         }
164         if (min == null) {
165             min = new MoveOption(game, centities.get(getFirstEntityNum()));
166         }
167         for (int d = 0; d < enemy_array.length; d++) {
168             Entity en = (Entity) enemy_array[d];
169             
170             // ignore loaded units
171             if (en.getPosition().isValidPosition() == false) {
172                 continue;
173             }
174             
175             CEntity enemy = centities.get(en);
176             int enemy_hit_arc =
177                 CEntity.getThreatHitArc(
178                     enemy.current.getFinalCoords(),
179                     enemy.current.getFinalFacing(),
180                     min.getFinalCoords());
181             MoveOption.DamageInfo di = (MoveOption.DamageInfo) min.damageInfos.get(enemy);
182             if (di != null) {
183                 enemy.expected_damage[enemy_hit_arc] += di.min_damage;
184             }
185             if (enemy.expected_damage[enemy_hit_arc] > 0) {
186                 enemy.hasTakenDamage = true;
187             }
188         }
189         if (min.isPhysical) {
190             centities.get(min.getPhysicalTargetId()).isPhysicalTarget = true;
191         }
192         System.out.println(min);
193         min.getCEntity().current = min;
194         min.getCEntity().last = min;
195         this.my_mechs_moved++;
196         min.getCEntity().moved = true;
197 
198         long exit = System.currentTimeMillis();
199         System.out.println("move turn took " + (exit - enter) + " ms");
200         return min;
201     }

206     public void firstPass(CEntity self) {
207         ArrayList enemies = getEnemyEntities();
208         Object[] move_array;
209         if (self.getEntity().isSelectableThisTurn() && !self.moved) {
210             move_array = self.getAllMoves().values().toArray();
211         } else {
212             move_array = new Object[] { self.current };
213         }
214         for (int i = 0; i < move_array.length; i++) {
215             MoveOption option = (MoveOption) move_array[i];
216             option.setState();
217             for (int e = 0; e < enemies.size(); e++) { // for each enemy
218                 Entity en = (Entity) enemies.get(e);
219                 
220                 // ignore loaded units
221                 if (en.getPosition().isValidPosition() == false) {
222                     continue;
223                 }
224                 
225                 CEntity enemy = centities.get(en);
226                 int[] modifiers = option.getModifiers(enemy.getEntity());
227                 if (modifiers[MoveOption.DEFENCE_MOD] == ToHitData.IMPOSSIBLE
228                     && modifiers[MoveOption.ATTACK_MOD] == ToHitData.IMPOSSIBLE) {
229                     continue;
230                 }
231                 int enemy_hit_arc =
232                     CEntity.getThreatHitArc(
233                         enemy.current.getFinalCoords(),
234                         enemy.current.getFinalFacing(),
235                         option.getFinalCoords());
236                 int self_hit_arc =
237                     CEntity.getThreatHitArc(
238                         option.getFinalCoords(),
239                         option.getFinalFacing(),
240                         enemy.current.getFinalCoords());
241                 if (!enemy.getEntity().isImmobile() && modifiers[MoveOption.DEFENCE_MOD] != ToHitData.IMPOSSIBLE) {
242                     self.engaged = true;
243                     int mod = modifiers[MoveOption.DEFENCE_MOD];
244                     double max = option.getMaxModifiedDamage(enemy.current, mod, modifiers[MoveOption.DEFENCE_PC]);
245                     if (en.isSelectableThisTurn()) {
246                         enemy.current.addStep(MovePath.STEP_TURN_RIGHT);
247                         max =
248                             Math.max(
249                                 option.getMaxModifiedDamage(enemy.current, mod + 1, modifiers[MoveOption.DEFENCE_PC]),
250                                 max);
251                         enemy.current.removeLastStep();
252                         enemy.current.addStep(MovePath.STEP_TURN_LEFT);
253                         max =
254                             Math.max(
255                                 option.getMaxModifiedDamage(enemy.current, mod + 1, modifiers[MoveOption.DEFENCE_PC]),
256                                 max);
257                         //return to original facing
258                         enemy.current.removeLastStep();
259                     }
260                     max = self.getThreatUtility(max, self_hit_arc);
261                     if (enemy.getEntity().isProne())
262                         max *= enemy.base_psr_odds;
263                     MoveOption.DamageInfo di = option.getDamageInfo(enemy, true);
264                     di.threat = max;
265                     di.max_threat = max;
266                     option.threat += max;
267                     option.tv.add(max + " Threat " + e + "\n");
268                 }
269                 /*
270                  * As a first approximation, take the maximum to a single
271                  * target
272                  */
273                 if (!option.isPhysical) {
274                     if (modifiers[MoveOption.ATTACK_MOD] != ToHitData.IMPOSSIBLE) {
275                         self.engaged = true;
276                         double max =
277                             enemy.current.getMaxModifiedDamage(option, modifiers[0], modifiers[MoveOption.ATTACK_PC]);
278                         max = enemy.getThreatUtility(max, enemy_hit_arc);
279                         MoveOption.DamageInfo di = option.getDamageInfo(enemy, true);
280                         di.damage = max;
281                         di.min_damage = max;
282                         option.tv.add(max + " Damage " + e + "\n");
283                         option.damage = Math.max(max, option.damage);
284                     }
285                 } else {
286                     CEntity target = centities.get(option.getPhysicalTargetId());
287                     try {
288                         if (target.getEntity().getId() == enemy.getEntity().getId()) {
289                             if (!target.isPhysicalTarget) {
290                                 ToHitData toHit = null;
291                                 double self_threat = 0;
292                                 double damage = 0;
293                                 if (option.isJumping()) {
294                                     self.current.setState();
295                                     toHit =
296                                         DfaAttackAction.toHit(game, option.getEntity().getId(), target.getEntity(), option);
297                                     damage = 2 * DfaAttackAction.getDamageFor(option.getEntity());
298                                     self_threat =
299                                         option.getCEntity().getThreatUtility(
300                                             DfaAttackAction.getDamageTakenBy(option.getEntity()),
301                                     ToHitData.SIDE_REAR)
302                                             * Compute.oddsAbove(toHit.getValue())
303                                             / 100;
304                                     self_threat
305                                         += option.getCEntity().getThreatUtility(
306                                             .1 * self.getEntity().getWeight(),
307                                     ToHitData.SIDE_REAR);
308                                     self_threat *= 100 / option.getCEntity().getEntity().getWeight();
309                                 } else {
310                                     self.current.setState();
311                                     toHit =
312                                         new ChargeAttackAction(option.getEntity(), target.getEntity()).toHit(
313                                             game,
314                                             option);
315                                     damage =
316                                         ChargeAttackAction.getDamageFor(
317                                             option.getEntity(),
318                                             option.getHexesMoved());
319                                     self_threat =
320                                         option.getCEntity().getThreatUtility(
321                                             ChargeAttackAction.getDamageTakenBy(
322                                                 option.getEntity(),
323                                                 target.getEntity()),
324                                     ToHitData.SIDE_FRONT)
325                                             * (Compute.oddsAbove(toHit.getValue()) / 100);
326                                     option.setState();
327                                 }
328                                 damage =
329                                     target.getThreatUtility(damage, toHit.getSideTable())
330                                         * Compute.oddsAbove(toHit.getValue())
331                                         / 100;
332                                 //charging is a good tactic against larger
333                                 // mechs
334                                 if (!option.isJumping())
335                                     damage *= Math.sqrt((double) enemy.bv / (double) self.bv);
336                                 //these are always risky, just don't on 11 or
337                                 // 12
338                                 if (toHit.getValue() > 10)
339                                     damage = 0;
340                                 //7 or less is good
341                                 if (toHit.getValue() < 8)
342                                     damage *= 1.5;
343                                 //this is all you are good for
344                                 if (self.range_damages[CEntity.RANGE_SHORT] < 5)
345                                     damage *= 2;
346                                 MoveOption.DamageInfo di = option.getDamageInfo(enemy, true);
347                                 di.damage = damage;
348                                 di.min_damage = damage;
349                                 option.damage = damage;
350                                 option.movement_threat += self_threat;
351                             } else {
352                                 option.threat += Integer.MAX_VALUE;
353                             }
354                         }
355                     } catch (Exception e1) {
356                         e1.printStackTrace();
357                         option.threat += Integer.MAX_VALUE;
358                     }
359                 }
360             } //-- end while of each enemy
361             self.current.setState();
362         } //-- end while of first pass
363         //top balanced
364         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(1, 1), 100);
365         //top damage
366         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(.5, 1), 100);
367     }

369     public MoveOption[] calculateMove(Entity entity) {
370         java.util.Vector enemy_array = game.getValidTargets(entity);
371         java.util.Vector entities = game.getEntitiesVector();
372         CEntity self = centities.get(entity);
373         Object[] move_array;
374         int friends = entities.size() - enemy_array.size();
375 
376         /***********************************************************************
377          * Second pass, combination moves/firing based only on the present
378          * case, since only one mech moves at a time
379          **********************************************************************/
380         move_array = self.pass.values().toArray();
381         self.pass.clear();
382         for (int j = 0; j < move_array.length && friends > 2; j++) {
383             MoveOption option = (MoveOption) move_array[j];
384             for (int e = 0; e < enemy_array.size(); e++) {
385                 Entity en = (Entity) enemy_array.elementAt(e);
386                 CEntity enemy = centities.get(en);
387                 for (Enumeration f = entities.elements(); f.hasMoreElements();) {
388                     Entity other = (Entity) f.nextElement();
389                     if (other.isEnemyOf(self.entity)) {
390                         continue;
391                     }
392                     MoveOption foption = centities.get(other).current;
393                     double threat_divisor = 1;
394                     MoveOption.DamageInfo di = option.getDamageInfo(enemy, true);
395                     if (foption.getDamageInfo(enemy, false) != null) {
396                         option.damage += (enemy.canMove() ? .1 : .2) * di.damage;
397                         threat_divisor += foption.getCEntity().canMove() ? .4 : .6;
398                     }
399                     option.threat -= di.threat;
400                     di.threat /= threat_divisor;
401                     option.threat += di.threat;
402                 }
403             }
404         }
405         //top balanced
406         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(1, 1), 50);
407         //top damage
408         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(.5, 1), 50);
409         /***********************************************************************
410          * third pass, (not so bad) oppurtunistic planner gives preference to
411          * good ranges/defensive positions based upon the mech characterization
412          **********************************************************************/
413         move_array = self.pass.values().toArray();
414         self.pass.clear();
415 
416         for (int j = 0; j < move_array.length; j++) {
417             MoveOption option = (MoveOption) move_array[j];
418             option.setState();
419             double adjustment = 0;
420             double temp_adjustment = 0;
421             for (int e = 0; e < enemy_array.size(); e++) { // for each enemy
422                 Entity en = (Entity) enemy_array.elementAt(e);
423                 CEntity enemy = centities.get(en);
424                 int current_range = self.current.getFinalCoords().distance(enemy.current.getFinalCoords());
425                 int range = option.getFinalCoords().distance(enemy.current.getFinalCoords());
426                 if (range > self.long_range) {
427                     temp_adjustment += (!(range < enemy.long_range) ? .5 : 1)
428                         * (1 + self.range_damages[self.range])
429                         * (Math.max(range - self.long_range - .5 * Math.max(self.jumpMP, .8 * self.runMP), 0));
430                 }
431                 if ((self.range == CEntity.RANGE_SHORT && (current_range > 5 || range > 9))
432                     || (self.range_damages[CEntity.RANGE_SHORT] < 4 && current_range > 10)) {
433                     temp_adjustment += ((enemy.range > CEntity.RANGE_SHORT) ? .5 : 1)
434                         * (Math.max(1 + self.range_damages[CEntity.RANGE_SHORT], 5))
435                         * Math.max(range - .5 * Math.max(self.jumpMP, .8 * self.runMP), 0);
436                 } else if (self.range == CEntity.RANGE_MEDIUM) {
437                     temp_adjustment += ((current_range < 6 || current_range > 12) ? 1 : .25)
438                         * ((enemy.range > CEntity.RANGE_SHORT) ? .5 : 1)
439                         * (1 + self.range_damages[CEntity.RANGE_MEDIUM])
440                         * Math.abs(range - .5 * Math.max(self.jumpMP, .8 * self.runMP));
441                 } else if (option.damage < .25 * self.range_damages[CEntity.RANGE_LONG]) {
442                     temp_adjustment += ((range < 10) ? .25 : 1)
443                         * (Math.max(1 + self.range_damages[CEntity.RANGE_LONG], 3))
444                         * (1 / (1 + option.threat));
445                 }
446                 adjustment += Math.sqrt(temp_adjustment * enemy.bv / self.bv);
447                 //I would always like to face the opponent
448                 if (!(enemy.getEntity().isProne() || enemy.getEntity().isImmobile())
449                     && CEntity.getThreatHitArc(
450                         option.getFinalCoords(),
451                         option.getFinalFacing(),
452                         enemy.getEntity().getPosition())
453                         != ToHitData.SIDE_FRONT) {
454                     int fa =
455                         CEntity.getFiringAngle(
456                             option.getFinalCoords(),
457                             option.getFinalFacing(),
458                             enemy.getEntity().getPosition());
459                     if (fa > 90 && fa < 270) {
460                         int distance = option.getFinalCoords().distance(enemy.current.getFinalCoords());
461                         double mod = 1;
462                         if (fa > 130 && fa < 240)
463                             mod = 2;
464                         //big formula that says don't do it
465                         mod *= ((Math.max(self.jumpMP, .8 * self.runMP) < 5) ? 2 : 1)
466                             * ((double) self.bv / (double) 50)
467                             * Math.sqrt(((double) self.bv) / enemy.bv)
468                             / ((double) distance / 6 + 1);
469                         option.self_threat += mod;
470                         option.tv.add(mod + " " + fa + " Back to enemy\n");
471                     }
472                 }
473             }
474             adjustment *= self.overall_armor_percent * self.strategy.attack / enemy_array.size();
475             //fix for hiding in level 2 water
476             //To a greedy bot, it always seems nice to stay in here...
477             IHex h = game.getBoard().getHex(option.getFinalCoords());
478             if (h.containsTerrain(Terrains.WATER)
479                 && h.surface() > (self.getEntity().getElevation() + ((option.getFinalProne()) ? 0 : 1))) {
480                 double mod = (self.getEntity().heat + option.getMovementheatBuildup() <= 7) ? 100 : 30;
481                 adjustment += self.bv / mod;
482             }
483             //add them in now, then re-add them later
484             if (self.range > CEntity.RANGE_SHORT) {
485                 int ele_dif =
486                     game.getBoard().getHex(option.getFinalCoords()).getElevation()
487                         - game.getBoard().getHex(self.current.getFinalCoords()).getElevation();
488                 adjustment -= (Math.max(ele_dif, 0) + 1)
489                     * ((double) Compute.getTargetTerrainModifier(game, option.getEntity()).getValue() + 1);
490             }
491 
492             //close the range if nothing else and healthy
493             if (option.damage < .25 * self.range_damages[self.range] && adjustment < self.range_damages[self.range]) {
494                 for (int e = 0; e < enemy_array.size(); e++) {
495                     Entity en = (Entity) enemy_array.elementAt(e);
496                     CEntity enemy = centities.get(en);
497                     int range = option.getFinalCoords().distance(enemy.current.getFinalCoords());
498                     if (range > 5)
499                         adjustment += Math.pow(self.overall_armor_percent, 2)
500                             * Math.sqrt((double) (range - 4) * enemy.bv / (double) self.bv)
501                             / enemy_array.size();
502                 }
503             }
504 
505             if (option.damage < .25 * (1 + self.range_damages[self.range])) {
506                 option.self_threat += 2 * adjustment;
507             } else if (option.damage < .5 * (1 + self.range_damages[self.range])) {
508                 option.self_threat += adjustment;
509             }
510             option.tv.add(option.self_threat + " Initial Damage Adjustment " + "\n");
511         }
512         //top balanced
513         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(1, 1), 30);
514         //top damage
515         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(.5, 1), 30);
516 
517         //reduce self threat, and add bonus for terrain
518         for (Iterator i = self.pass.values().iterator(); i.hasNext();) {
519             MoveOption option = (MoveOption) i.next();
520             option.setState();
521             option.self_damage *= .5;
522             option.self_threat *= .5;
523             //TODO: should scale to the unit bv
524             double terrain = 2 * ((double) Compute.getTargetTerrainModifier(game, option.getEntity()).getValue());
525             option.tv.add(terrain + " Terrain Adjusment " + "\n");
526             option.self_threat -= terrain;
527         }
528 
529         move_array = self.pass.values().toArray();
530         self.pass.clear();
531 
532         //pass should contains 30 ~ 60
533         /***********************************************************************
534          * fourth pass, speculation on top moves use averaging to filter
535          **********************************************************************/
536         for (int e = 0; e < enemy_array.size(); e++) { // for each enemy
537             Entity en = (Entity) enemy_array.elementAt(e);
538             CEntity enemy = centities.get(en);
539             //engage in speculation on "best choices" when you loose iniative
540             if (enemy.canMove()) {
541                 Object[] enemy_move_array = enemy.pass.values().toArray();
542                 Vector to_check = new Vector();
543                 //check some enemy moves
544                 for (int j = 0; j < move_array.length; j++) {
545                     MoveOption option = null;
546                     to_check.clear();
547                     option = (MoveOption) move_array[j];
548                     option.setState();
549                     //check for damning hexes specifically
550                     //could also look at intervening defensive
551                     Vector coord = new Vector();
552                     Coords back = option.getFinalCoords().translated((option.getFinalFacing() + 3) % 6);
553                     coord.add(back);
554                     coord.add(back.translated((option.getFinalFacing() + 2) % 6));
555                     coord.add(back.translated((option.getFinalFacing() + 4) % 6));
556                     coord.add(option.getFinalCoords().translated((option.getFinalFacing())));
557                     coord.add(option.getFinalCoords().translated((option.getFinalFacing() + 1) % 6));
558                     coord.add(option.getFinalCoords().translated((option.getFinalFacing() + 2) % 6));
559                     coord.add(option.getFinalCoords().translated((option.getFinalFacing() + 4) % 6));
560                     coord.add(option.getFinalCoords().translated((option.getFinalFacing() + 5) % 6));
561                     Iterator ci = coord.iterator();
562                     while (ci.hasNext()) {
563                         Coords test = (Coords) ci.next();
564                         List c = enemy.findMoves(test);
565                         if (c.size() != 0)
566                             to_check.addAll(c);
567                     }
568                     int range = option.getFinalCoords().distance(enemy.current.getFinalCoords());
569                     int compare = 0;
570                     if ((enemy.long_range) > range - Math.max(enemy.jumpMP, enemy.runMP)) {
571                         compare = 30;
572                     } else if (enemy.long_range > range) {
573                         compare = 10;
574                     }
575                     double mod = this.enemies_moved / this.getEnemyEntities().size();
576                     compare *= (1 + mod);
577                     for (int k = 0; k <= compare && k < enemy_move_array.length; k++) {
578                         if (enemy_move_array.length < compare) {
579                             to_check.add(enemy_move_array[k]);
580                         } else {
581                             int value = Compute.randomInt(enemy_move_array.length);
582                             if (value % 2 == 1) {
583                                 to_check.add(enemy_move_array[value]);
584                             } else {
585                                 to_check.add(enemy_move_array[k]);
586                             }
587                         }
588                     }
589                     Iterator eo = to_check.iterator();
590                     while (eo.hasNext()) {
591                         MoveOption enemy_option = (MoveOption) eo.next();
592                         double max_threat = 0;
593                         double max_damage = 0;
594                         enemy_option.setState();
595                         int enemy_hit_arc =
596                             CEntity.getThreatHitArc(
597                                 enemy_option.getFinalCoords(),
598                                 enemy_option.getFinalFacing(),
599                                 option.getFinalCoords());
600                         int self_hit_arc =
601                             CEntity.getThreatHitArc(
602                                 enemy_option.getFinalCoords(),
603                                 enemy_option.getFinalFacing(),
604                                 option.getFinalCoords());
605                         if (enemy_option.isJumping()) {
606                             enemy_hit_arc = Compute.ARC_FORWARD;
607                         }
608                         int[] modifiers = option.getModifiers(enemy_option.getEntity());
609                         if (modifiers[1] != ToHitData.IMPOSSIBLE) {
610                             self.engaged = true;
611                             if (!enemy_option.isJumping()) {
612                                 max_threat =
613                                     option.getMaxModifiedDamage(
614                                         enemy_option,
615                                         modifiers[1],
616                                         modifiers[MoveOption.DEFENCE_PC]);
617                             } else {
618                                 max_threat =
619                                     .8
620                                         * enemy.getModifiedDamage(
621                                             (modifiers[MoveOption.DEFENCE_PC] == 1) ? CEntity.TT : ToHitData.SIDE_FRONT,
622                                             enemy_option.getFinalCoords().distance(option.getFinalCoords()),
623                                             modifiers[1]);
624                             }
625                             max_threat = self.getThreatUtility(max_threat, self_hit_arc);
626                         }
627                         if (modifiers[0] != ToHitData.IMPOSSIBLE) {
628                             self.engaged = true;
629                             max_damage =
630                                 enemy_option.getMaxModifiedDamage(
631                                     option,
632                                     modifiers[0],
633                                     modifiers[MoveOption.ATTACK_PC]);
634                             max_damage = enemy.getThreatUtility(max_damage, enemy_hit_arc);
635                             if (option.isPhysical) {
636                                 if (centities.get(option.getPhysicalTargetId()).getEntity().getId()
637                                     == enemy.getEntity().getId()) {
638                                     max_damage = option.getDamage(enemy);
639                                 } else {
640                                     max_damage = 0;
641                                 }
642                             }
643                         }
644                         MoveOption.DamageInfo di = option.getDamageInfo(enemy, true);
645                         di.max_threat = Math.max(max_threat, di.max_threat);
646                         di.min_damage = Math.min(di.min_damage, max_damage);
647                         if (max_threat - max_damage > di.threat - di.damage) {
648                             di.threat = max_threat;
649                             di.damage = max_damage;
650                             option.tv.add(max_threat + " Spec Threat " + e + "\n");
651                             option.tv.add(max_damage + " Spec Damage " + e + "\n");
652                         }
653                     }
654                     //update estimates
655                     option.damage = 0;
656                     option.threat = 0;
657                     for (Iterator i = option.damageInfos.keySet().iterator(); i.hasNext();) {
658                         //my damage is the average of expected and min
659                         CEntity cen = (CEntity) i.next();
660                         //rescale
661                         MoveOption.DamageInfo di = option.getDamageInfo(cen, true);
662                         di.min_damage /= cen.strategy.target;
663                         di.damage /= cen.strategy.target;
664                         option.damage += (di.min_damage + di.damage) / 2;
665 
666                         //my threat is average of absolute worst, and expected
667                         option.threat = Math.max(option.threat, di.max_threat + di.threat) / 2;
668                         di.threat = (di.max_threat + 2 * di.threat) / 3;
669                     }
670                 }
671                 //restore enemy
672                 enemy.current.setState();
673             }
674             self.current.setState();
675         } //--end move speculation
676 
677         //top balanced
678         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(1, 1), 20);
679         //top damage
680         filterMoves(move_array, self.pass, new MoveOption.WeightedComparator(.5, 1), 20);
681 
682         //reduce transient damage estimates
683         for (Iterator i = self.pass.values().iterator(); i.hasNext();) {
684             MoveOption option = (MoveOption) i.next();
685             option.self_threat *= .5;
686             option.self_damage *= .5;
687         }
688 
689         move_array = self.pass.values().toArray();
690         self.pass.clear();
691 
692         //pass should now be 20 ~ 40
693         /***********************************************************************
694          * fourth pass, final damage and threat approximation --prevents moves
695          * that from the previous pass would cause the mech to die
696          **********************************************************************/
697         if (self.engaged) {
698             for (int j = 0; j < move_array.length; j++) {
699                 MoveOption option = (MoveOption) move_array[j];
700                 option.setState();
701                 GAAttack temp = this.bestAttack(option);
702                 if (temp != null) {
703                     option.damage = (option.damage + temp.getFittestChromosomesFitness()) / 2;
704                 } else {
705                     option.damage /= 2;
706                 }
707                 for (int e = 0; e < enemy_array.size(); e++) { // for each
708                     // enemy
709                     Entity en = (Entity) enemy_array.elementAt(e);
710                     CEntity enemy = centities.get(en);
711                     if (!enemy.canMove()) {
712                         option.setThreat(
713                             enemy,
714                             (option.getThreat(enemy) + this.attackUtility(enemy.current, self)) / 2);
715                         option.tv.add(option.getThreat(enemy) + " Revised Threat " + e + " \n");
716                         if (!option.isPhysical) {
717                             if (temp != null) {
718                                 option.setDamage(enemy, (option.getDamage(enemy) + temp.getDamageUtility(enemy)) / 2);
719                             } else {
720                                 //probably zero, but just in case
721                                 option.setDamage(enemy, option.getMinDamage(enemy));
722                             }
723                             option.tv.add(option.getDamage(enemy) + " Revised Damage " + e + " \n");
724                             //this needs to be reworked
725                             if (option.getFinalCoords().distance(enemy.current.getFinalCoords()) == 1) {
726                                 PhysicalOption p =
727                                     pcalc.getBestPhysicalAttack(option.getEntity(), enemy.getEntity(), game);
728                                 if (p != null) {
729                                     option.setDamage(enemy, option.getDamage(enemy) + p.expectedDmg);
730                                     option.tv.add(p.expectedDmg + " Physical Damage " + e + " \n");
731                                 }
732                                 p = pcalc.getBestPhysicalAttack(enemy.getEntity(), option.getEntity(), game);
733                                 if (p != null) {
734                                     option.setThreat(enemy, option.getThreat(enemy) + .5 * p.expectedDmg);
735                                     option.tv.add(.5 * p.expectedDmg + " Physical Threat " + e + " \n");
736                                 }
737                             }
738                         }
739                     } else if (!option.isPhysical) { //enemy can move (not
740                         if (temp != null) {
741                             option.setDamage(enemy, (2 * option.getDamage(enemy) + temp.getDamageUtility(enemy)) / 3);
742                         } else {
743                             option.setDamage(enemy, option.getMinDamage(enemy));
744                         }
745                     } else {
746                         //get a more accurate estimate
747                         option.setDamage(
748                             enemy,
749                             option.getDamage(enemy) / Math.sqrt((double) enemy.bv / (double) self.bv));
750                         option.damage = option.getDamage(enemy);
751                     }
752                 }
753                 option.threat = 0;
754                 for (Iterator i = option.damageInfos.values().iterator(); i.hasNext();) {
755                     option.threat += ((MoveOption.DamageInfo) i.next()).threat;
756                 }
757                 option.tv.add(option.threat + " Revised Threat Utility\n");
758                 option.tv.add(option.damage + " Revised Damage Utility\n");
759             }
760         }
761         Arrays.sort(move_array, new MoveOption.WeightedComparator(1, 1));
762         self.current.setState();
763 
764         /***********************************************************************
765          * Return top twenty moves to the lance algorithm
766          **********************************************************************/
767         MoveOption[] result = new MoveOption[Math.min(move_array.length, 20)];
768         int offset = 0;
769         for (int i = 0; i < Math.min(move_array.length, 20); i++) {
770             MoveOption next = (MoveOption) move_array[i];
771             if (next.isPhysical && self.range_damages[CEntity.RANGE_SHORT] > 5 && next.doomed) {
772                 if (offset + 20 < move_array.length) {
773                     next = (MoveOption) move_array[offset + 20];
774                     offset++;
775                 }
776             }
777             result[i] = next;
778         }
779         return result;
780     }

814     protected Vector calculateWeaponAttacks(Entity en, Mounted mw, boolean best_only) {
815         int from = en.getId();
816         int weaponID = en.getEquipmentNum(mw);
817         Vector result = new Vector();
818         Enumeration ents = game.getValidTargets(en).elements();
819         AttackOption a = null;
820         AttackOption max = new AttackOption(null, null, 0, null);
821         while (ents.hasMoreElements()) {
822             Entity e = (Entity) ents.nextElement();
823             CEntity enemy = centities.get(e);
824             ToHitData th = WeaponAttackAction.toHit(game, from, e, weaponID);
825             if (th.getValue() != ToHitData.IMPOSSIBLE && !(th.getValue() >= 13)) {
826                 double expectedDmg;
827                 // Are we an Infantry platoon?
828                 if (en instanceof Infantry) {
829                     // Get the expected damage, given our current
830                     // manpower level.
831                     Infantry inf = (Infantry) en;
832                     expectedDmg = inf.getDamage(inf.getShootingStrength());
833                 } else {
834                     // Get the expected damage of the weapon.
835                     expectedDmg = CEntity.getExpectedDamage((WeaponType) mw.getType());
836                 }
837 
838                 // Infantry in the open suffer double damage.
839                 if (e instanceof Infantry) {
840                     IHex e_hex = game.getBoard().getHex(e.getPosition());
841                     if (!e_hex.containsTerrain(Terrains.WOODS) && !e_hex.containsTerrain(Terrains.BUILDING)) {
842                         expectedDmg *= 2;
843                     }
844                 }
845 
846                 a = new AttackOption(enemy, mw, expectedDmg, th);
847                 if (a.value > max.value) {
848                     if (best_only) {
849                         max = a;
850                     } else {
851                         result.add(0, a);
852                     }
853                 } else {
854                     result.add(a);
855                 }
856             }
857         }
858         if (best_only && max.target != null) {
859             result.add(max);
860         }
861         if (result.size() > 0) {
862             result.add(new AttackOption(null, mw, 0, null));
863         }
864         return result;
865     }

871     public GAAttack bestAttack(MoveOption es, CEntity target, int search_level) {
872         Entity en = es.getEntity();
873         int attacks[] = new int[3];
874         Vector front = new Vector();
875         Vector left = new Vector();
876         Vector right = new Vector();
877         GAAttack result = null;
878         int o_facing = en.getFacing();
879         for (Enumeration i = en.getWeapons(); i.hasMoreElements();) {
880             Mounted mw = (Mounted) i.nextElement();
881             Vector c = this.calculateWeaponAttacks(en, mw, true);
882             if (c.size() > 0) {
883                 front.add(c);
884                 attacks[0] = Math.max(attacks[0], c.size());
885             }
886             if (!es.getFinalProne() && en.canChangeSecondaryFacing()) {
887                 en.setSecondaryFacing((o_facing + 5) % 6);
888                 c = this.calculateWeaponAttacks(en, mw, true);
889                 if (c.size() > 0) {
890                     left.add(c);
891                     attacks[1] = Math.max(attacks[1], c.size());
892                 }
893                 en.setSecondaryFacing((o_facing + 1) % 6);
894                 c = this.calculateWeaponAttacks(en, mw, true);
895                 if (c.size() > 0) {
896                     right.add(c);
897                     attacks[2] = Math.max(attacks[2], c.size());
898                 }
899             } else {
900                 attacks[1] = 0;
901                 attacks[2] = 0;
902             }
903             en.setSecondaryFacing(o_facing);
904         }
905         Vector arcs = new Vector();
906         arcs.add(front);
907         if (!es.getFinalProne() && en.canChangeSecondaryFacing()) {
908             arcs.add(left);
909             arcs.add(right);
910         }
911         for (int i = 0; i < arcs.size(); i++) {
912             Vector v = (Vector) arcs.elementAt(i);
913             if (v.size() > 0) {
914                 GAAttack test =
915                     new GAAttack(
916                         this,
917                         centities.get(en),
918                         v,
919                         Math.max((v.size() + attacks[i]) * search_level, 20 * search_level),
920                         30 * search_level,
921                         en.isEnemyOf((Entity) getEntitiesOwned().get(0)));
922                 test.setFiringArc(i);
923                 test.evolve();
924                 if (target != null) {
925                     if (result == null || test.getDamageUtility(target) > result.getDamageUtility(target)) {
926                         result = test;
927                     }
928                 } else if (
929                     result == null || test.getFittestChromosomesFitness() > result.getFittestChromosomesFitness()) {
930                     result = test;
931                 }
932             }
933         }
934         return result;
935     }

946     public void calculateFiringTurn() {
947         int first_entity = game.getFirstEntityNum();
948         int entity_num = first_entity;
949         int best_entity = first_entity;
950         double max = java.lang.Double.MIN_VALUE;
951         int[] results = null;
952         Vector winner = null;
953         int arc = 0;
954 
955         if (entity_num == -1) {
956             return;
957         }
958 
959         do {
960             Entity en = game.getEntity(entity_num);
961             CEntity cen = centities.get(en);
962 
963             GAAttack test = bestAttack(cen.current, null, 3);
964 
965             if (test != null && test.getFittestChromosomesFitness() > max) {
966                 max = test.getFittestChromosomesFitness();
967                 results = test.getResultChromosome();
968                 arc = test.getFiringArc();
969                 best_entity = entity_num;
970                 winner = test.getAttack();
971             }
972             entity_num = game.getNextEntityNum(entity_num);
973         } while (entity_num != first_entity && entity_num != -1);
974 
975         java.util.Vector av = new java.util.Vector();
976         //maximum already selected (or default)
977         Entity en = game.getEntity(best_entity);
978         if (results != null) {
979             Entity primary_target = (Entity) game.getEntitiesVector().elementAt(results[results.length - 1]);
980             TreeMap tm = new TreeMap(new AttackOption.Sorter(centities.get(primary_target)));
981             for (int i = 0; i < results.length - 1; i++) {
982                 AttackOption a = (AttackOption) ((Vector) winner.elementAt(i)).elementAt(results[i]);
983                 if (a.target != null) {
984                     a.target.expected_damage[a.toHit.getSideTable()] += a.value;
985                     a.target.hasTakenDamage = true;
986                     tm.put(a, a);
987                 }
988             }
989             Iterator i = tm.values().iterator();
990             while (i.hasNext()) {
991                 AttackOption a = (AttackOption) i.next();
992                 av.addElement(
993                     new WeaponAttackAction(en.getId(), a.target.getEntity().getId(), en.getEquipmentNum(a.weapon)));
994             }
995         }
996         switch (arc) {
997             case 1 :
998                 av.insertElementAt(new TorsoTwistAction(en.getId(), (en.getFacing() + 5) % 6), 0);
999                 break;
1000             case 2 :
1001                 av.insertElementAt(new TorsoTwistAction(en.getId(), (en.getFacing() + 1) % 6), 0);
1002                 break;
1003         }
1004         sendAttackData(best_entity, av);
1005     }

1010     protected void initMovement() {
1011         this.my_mechs_moved = 0;
1012         this.old_moves = null;
1013         this.enemies_moved = 0;
1014         double max_modifier = 1.4;

1016         double num_entities = Math.sqrt(entities.size()) / 100;
1017         Vector friends = new Vector();
1018         Vector foes = new Vector();
1019         double friend_sum = 0;
1020         double foe_sum = 0;
1021         double max_foe_bv = 0;
1022         CEntity max_foe = null;
1023         for (int i = 0; i < entities.size(); i++) {
1024             Entity entity = (Entity) entities.elementAt(i);
1025             CEntity centity = centities.get(entity);
1026             centity.enemy_num = i;
1027             double old_value = centity.bv * (centity.overall_armor_percent + 1);
1028             centity.reset(); //should get fresh values
1029             double new_value = centity.bv * (centity.overall_armor_percent + 1);
1030             double percent = 1 + (new_value - old_value) / old_value;
1031             if (entity.getOwner().equals(getLocalPlayer())) {
1032                 friends.add(centity);
1033                 friend_sum += new_value;
1034                 if (percent < .85) {
1035                     //small retreat
1036                     centity.strategy.attack = .85;
1037                 } else if (percent < .95) {
1038                     centity.strategy.attack = 1;
1039                 } else if (percent <= 1 && centity.strategy.attack < max_modifier) {
1040                     if (percent == 1) {
1041                         if (centity.strategy.attack < 1) {
1042                             centity.strategy.attack = Math.min(1.4 * centity.strategy.attack, 1);
1043                         } else {
1044                             centity.strategy.attack *= (1.0 + num_entities);
1045                         }
1046                     } else {
1047                         centity.strategy.attack *= (1.0 + 2 * num_entities);
1048                     }
1049                 }
1050             } else if (!entity.getOwner().isEnemyOf(getLocalPlayer())) {
1051                 friend_sum += new_value;
1052             } else {
1053                 foes.add(centity);
1054                 foe_sum += new_value;
1055                 if (new_value > max_foe_bv) {
1056                     max_foe_bv = new_value;
1057                     max_foe = centity;
1058                 }
1059                 if (this.getEntitiesOwned().size() > 2) {
1060                     if (centity.strategy.target > 2) {
1061                         centity.strategy.target = 1 + .5 * (centity.strategy.target - 2);
1062                     }
1063                     if (percent < .85 && centity.strategy.target < max_modifier) {
1064                         centity.strategy.target *= (1.0 + 6 * num_entities);
1065                     } else if (percent < .95 && centity.strategy.target < max_modifier) {
1066                         centity.strategy.target *= (1.0 + 4 * num_entities);
1067                     } else if (percent <= 1) {
1068                         if (percent == 1) {
1069                             centity.strategy.target /= (1.0 + 2 * num_entities);
1070                         } else {
1071                             centity.strategy.target /= (1.0 + num_entities);
1072                         }
1073                     }
1074                     //don't go below one
1075                     if (centity.strategy.target < 1)
1076                         centity.strategy.target = 1;
1077                 }
1078             }
1079         }
1080         System.out.println("Us " + friend_sum + " Them " + foe_sum);
1081         //do some more reasoning...
1082         if (this.unit_values.size() == 0) {
1083             this.unit_values.add(new Double(friend_sum));
1084             this.enemy_values.add(new Double(foe_sum));
1085             return;
1086         }
1087         Iterator i = foes.iterator();
1088 
1089         if (friends.size() > 1) {
1090             if ( Strategy.MainTarget == null
1091                  || null == game.getEntity
1092                  (Strategy.MainTarget.getEntity().getId()) ) {
1093                 Strategy.MainTarget = max_foe;
1094             }
1095             // TODO : Handle this better.
1096             if (null == Strategy.MainTarget)
1097                 System.err.println
1098                     ( "TestBot#initMovement() - no main target for bot" );
1099             else if (null == Strategy.MainTarget.strategy)
1100                 System.err.println
1101                     ( "TestBot#initMovement() - no strategy for main target" );
1102             else {
1103                 Strategy.MainTarget.strategy.target += .2;
1104                 while (i.hasNext()) {
1105                     CEntity centity = (CEntity) i.next();
1106                     // good turn, keep up the work, but randomize to reduce
1107                     // predictability
1108                     if (friend_sum - foe_sum
1109                         >= .9
1110                         * (((Double) this.unit_values.getLast()).doubleValue()
1111                            - ((Double) this.enemy_values.getLast()).doubleValue())) {
1112                         if (Compute.randomInt(2) == 1) {
1113                             centity.strategy.target += .3;
1114                         }
1115                         //lost that turn, but still in the fight, just get a
1116                         // little more aggressive
1117                     } else if (friend_sum > .9 * foe_sum) {
1118                         centity.strategy.target += .15;
1119                         //lost that turn and loosing
1120                     } else if (centity.strategy.target < 2) { //go for the gusto
1121                         centity.strategy.target += .3;
1122                     }
1123                     System.out.println(centity.getEntity().getShortName() + " " + centity.strategy.target);
1124                 }
1125             }
1126         }
1127 
1128         double ratio = friend_sum / foe_sum;
1129         double mod = 1;
1130         if (ratio < .9) {
1131             mod = .95;
1132         } else if (ratio < 1) {
1133             //no change
1134         } else { //attack

1136         }
1137         i = friends.iterator();
1138         while (i.hasNext()) {
1139             CEntity centity = (CEntity) i.next();
1140             if (!(mod < 1 && centity.strategy.attack < .6) && !(mod > 1 && centity.strategy.attack >= max_modifier))
1141                 centity.strategy.attack *= mod;
1142         }
1143         System.gc(); //just to make sure
1144     }

1184     protected java.util.Vector deployMinefields(java.util.Vector deployedMinefields, int number, int type) {
1185         for (int i = 0; i < number; i++) {
1186             Coords coords = new Coords(Compute.randomInt(game.getBoard().getWidth()),
1187                     Compute.randomInt(game.getBoard().getHeight()));
1188 
1189             if (game.containsMinefield(coords)) {
1190                 Minefield mf = (Minefield) game.getMinefields(coords).elementAt(0);
1191                 if (mf.getPlayerId() == getLocalPlayer().getId()) {
1192                     i--;
1193                     continue;
1194                 }
1195             } else {
1196                 Minefield mf = null;
1197 
1198                 if (type == 0) {
1199                     mf = Minefield.createConventionalMF(coords, getLocalPlayer().getId());
1200                 } else if (type == 1) {
1201                     mf = Minefield.createCommandDetonatedMF(coords, getLocalPlayer().getId());
1202                 } else if (type == 2) {
1203                     mf = Minefield.createVibrabombMF(coords, getLocalPlayer().getId(), 20);
1204                 }
1205                 deployedMinefields.addElement(mf);
1206             }
1207         }
