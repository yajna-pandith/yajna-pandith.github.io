102     /**
103      * Creates z3 expression from the known value.
104      */
105     private Expr makeLiteral(Object value)
106     {
107         if (value instanceof Boolean)
108         {
109             return context.mkBool((Boolean) value);
110         }
111         else if (value instanceof Byte)
112         {
113             return context.mkBV((Byte) value, 8);
114         }
115         else if (value instanceof Short)
116         {
117             return context.mkBV((Short) value, 16);
118         }
119         else if (value instanceof Integer)
120         {
121             return context.mkBV((Integer) value, 32);
122         }
123         else if (value instanceof Long)
124         {
125             return context.mkBV((Long) value, 64);
126         }
127         else if (value instanceof Character)
128         {
129             return context.mkBV((Character) value, 16);
130         }
131 
132         return null;
133     }

135     /**
136      * Applies cast to the bit-vector expression, returns resulting bit-vector expression.
137      */
138     private BitVecExpr castBV(BitVecExpr bitVec, CtTypeReference<?> fromType, CtTypeReference<?> toType)
139     {
140         if (!isCalculable(toType))
141         {
142             return null;
143         }
144 
145         int newSize = TypeUtils.getPrimitiveTypeSize(toType);
146         int sizeDifference = newSize - bitVec.getSortSize();
147 
148         if (sizeDifference > 0)
149         {
150             // Widening Primitive Conversion
151             if (!isChar(fromType))
152             {
153                 return context.mkSignExt(sizeDifference, bitVec);
154             }
155             else
156             {
157                 return context.mkZeroExt(sizeDifference, bitVec);
158             }
159         }
160         else if (sizeDifference < 0)
161         {
162             // Narrowing Primitive Conversion (signed)
163             return context.mkExtract(newSize - 1, 0, bitVec);
164         }
165 
166         return bitVec;
167     }

169     /**
170      * Applies casts to the expression, returns resulting expression.
171      */
172     private Expr applyCasts(Expr expr, CtTypeReference<?> originalType, List<CtTypeReference<?>> casts)
173     {
174         if (expr == null)
175         {
176             return null;
177         }
178 
179         for (int i = casts.size() - 1; i >= 0; i--)
180         {
181             CtTypeReference<?> castType = casts.get(i);
182             if (expr instanceof BitVecExpr)
183             {
184                 expr = castBV((BitVecExpr) expr, originalType, castType);
185             }
186             else if (expr instanceof RealExpr)
187             {
188                 // We don't actually calculate floats right now => create value from type
189                 expr = makeFreshConstFromType(context, castType);
190             }
191             else if (expr instanceof IntExpr)
192             {
193                 if (castType.isPrimitive())
194                 {
195                     // Unboxing conversion
196                     expr = memory.read(originalType.unbox(), (IntExpr) expr);
197                 }
198             }
199 
200             // Boxing conversion
201             if (!castType.isPrimitive() && originalType.isPrimitive())
202             {
203                 int nextPointer = memory.nextPointer();
204                 IntExpr index =  context.mkInt(nextPointer);
205                 if (isCalculable(castType) && isCalculable(originalType) && expr != null)
206                 {
207                     memory.write(castType.unbox(), index, expr);
208                 }
209                 expr = index;
210             }
211 
212             originalType = castType;
213         }
214 
215         return expr;
216     }

218     /**
219      * This method is equivalent to solver.check(assumption).
220      * We should use it because of this bug: https://github.com/Z3Prover/z3/issues/2107
221      */
222     private Status checkAssumption(BoolExpr assumption)
223     {
224         solver.push();
225         solver.add(assumption);
226         Status status = solver.check();
227         solver.pop();
228         return status;
229     }

231     /**
232      * Adds current conditions and abrupt termination flags to the solver.
233      */
234     private void applyState()
235     {
236         // Apply current conditions
237         if (currentConditions != null)
238         {
239             solver.add(currentConditions);
240         }
241 
242         // Apply information about exit points
243         BoolExpr returnExpr = getReturnExpr();
244         if (returnExpr != null)
245         {
246             BoolExpr reachableExpr = context.mkNot(returnExpr);
247             solver.add(reachableExpr);
248         }
249 
250         // Apply information about breaks
251         BoolExpr breakExpr = getBreakExpr();
252         if (breakExpr != null)
253         {
254             BoolExpr reachableExpr = context.mkNot(breakExpr);
255             solver.add(reachableExpr);
256         }
257 
258         // Apply information about continues
259         BoolExpr continueExpr = getContinueExpr();
260         if (continueExpr != null)
261         {
262             BoolExpr reachableExpr = context.mkNot(continueExpr);
263             solver.add(reachableExpr);
264         }
265 
266         // Apply information about throws
267         BoolExpr throwExpr = getThrowExpr();
268         if (throwExpr != null)
269         {
270             BoolExpr reachableExpr = context.mkNot(throwExpr);
271             solver.add(reachableExpr);
272         }
273     }

275     /**
276      * Checks if the scanner is inside some unreachable condition.
277      * For example: if (false) {...}
278      */
279     private boolean isInsideUnreachableCondition()
280     {
281         if (currentConditions == null)
282         {
283             return false;
284         }
285         solver.push();
286         applyState();
287         Status status = solver.check();
288         solver.pop();
289         return currentConditions != null && status == Status.UNSATISFIABLE;
290     }

292     /**
293      * Checks if some arbitrary condition is always true/false at the moment.
294      */
295     public ConditionStatus checkCond(BoolExpr conditionExpr)
296     {
297         ConditionStatus result = ConditionStatus.OK;
298         if (isInsideUnreachableCondition())
299         {
300             return result;
301         }
302 
303         solver.push();
304 
305         applyState();
306 
307         Status status1 = checkAssumption(conditionExpr);
308         if (status1 == Status.SATISFIABLE)
309         {
310             // To check if formula is valid (i.e., to prove it), we show its negation to be unsatisfiable.
311             Status status2 = checkAssumption(context.mkNot(conditionExpr));
312             if (status2 == Status.UNSATISFIABLE)
313             {
314                 result = ConditionStatus.ALWAYS_TRUE;
315             }
316         }
317         else if (status1 == Status.UNSATISFIABLE)
318         {
319             result = ConditionStatus.ALWAYS_FALSE;
320         }
321 
322         solver.pop();
323         return result;
324     }

326     @Override
327     public void visitCtIf(CtIf ifElement)
328     {
329         BoolExpr conditionExpr = visitCondition(ifElement.getCondition(), false);
330 
331         final boolean hasElseBranch = ifElement.getElseStatement() != null;
332 
333         BranchData thenBranchData = visitBranch(conditionExpr, ifElement.getThenStatement());
334         BranchData elseBranchData = hasElseBranch ? visitBranch(context.mkNot(conditionExpr), ifElement.getElseStatement()) : new BranchData(variablesMap, memory);
335 
336         mergeBranches(conditionExpr, thenBranchData, elseBranchData);
337     }

339     private void visitLoop(CtExpression<Boolean> loopCondition, boolean isPrecondition, CtStatement... loopBody)
340     {
341         BoolExpr oldBreakExpr = getBreakExpr();
342         BoolExpr oldContinueExpr = getContinueExpr();
343 
344         BoolExpr iterationConditionExpr;
345         BranchData iterationBranchData;
346         if (isPrecondition)
347         {
348             if (loopCondition != null)
349             {
350                 // Check if loop condition is always false
351                 visitCondition(loopCondition, true);
352             }
353             ResetOnModificationScanner resetScanner = new ResetOnModificationScanner(context, variablesMap, memory);
354             Arrays.stream(loopBody).forEach(resetScanner::scan);
355             iterationConditionExpr = loopCondition == null ? makeFreshBool(context) : visitCondition(loopCondition, true);
356 
357             iterationBranchData = visitBranch(iterationConditionExpr, loopBody);
358         }
359         else
360         {
361             ResetOnModificationScanner resetScanner = new ResetOnModificationScanner(context, variablesMap, memory);
362             Arrays.stream(loopBody).forEach(resetScanner::scan);
363             iterationBranchData = visitBranch(context.mkTrue(), loopBody);
364             iterationConditionExpr = loopCondition == null ? makeFreshBool(context) : visitCondition(loopCondition, true);
365         }
366 
367         // Save information about the break
368         variablesMap.put(breakFlagReference, iterationBranchData.getVariablesMap().get(breakFlagReference));
369 
370         BoolExpr currentBreakExpr = getBreakExpr();
371 
372         // Reset flow flags after exiting the loop
373         variablesMap.put(breakFlagReference, oldBreakExpr);
374         variablesMap.put(continueFlagReference, oldContinueExpr);
375 
376         // Invert loop condition
377         solver.add(context.mkOr(context.mkNot(iterationConditionExpr), currentBreakExpr));
378     }

417     private BoolExpr visitCondition(CtExpression<Boolean> condition, boolean isLoopCondition)
418     {
419         scan(condition);
420         Expr conditionValue = currentResult;
421         condition.putMetadata("value", conditionValue);
422 
423         checkCondition(condition, isLoopCondition);
424 
425         // Unboxing conversion
426         if (!getActualType(condition).isPrimitive())
427         {
428             conditionValue = memory.read(getActualType(condition).unbox(), (IntExpr) conditionValue);
429         }
430 
431         currentResult = conditionValue;
432         return (BoolExpr) currentResult;
433     }

435     /**
436      * Merges thenMap and elseMap into resultMap via ITE function.
437      */
438     private <T extends Expr> void mergeMaps(BoolExpr cond, Map<CtReference, T> thenMap, Map<CtReference, T> elseMap, Map<CtReference, T> resultMap)
439     {
440         // xNew = ITE(cond, xThen, xElse) for each entry
441         for (Map.Entry<CtReference, T> entry : thenMap.entrySet())
442         {
443             CtReference reference = entry.getKey();
444             T thenBranchValue = entry.getValue();
445             T elseBranchValue = elseMap.get(reference);
446 
447             // Variable was not changed in this if-then-else block
448             if (thenBranchValue == elseBranchValue)
449             {
450                 continue;
451             }
452 
453             if (thenBranchValue != null && elseBranchValue != null)
454             {
455                 T iteExpr = (T) context.mkITE(cond, thenBranchValue, elseBranchValue);
456                 resultMap.put(reference, iteExpr);
457             }
458         }
459     }

461     /**
462      * Merges two branches into the current.
463      */
464     private void mergeBranches(BoolExpr cond, BranchData thenBranchData, BranchData elseBranchData)
465     {
466         mergeMaps(cond, thenBranchData.getVariablesMap(), elseBranchData.getVariablesMap(), variablesMap);
467         mergeMaps(cond, thenBranchData.getMemory().getMemoryMap(), elseBranchData.getMemory().getMemoryMap(), memory.getMemoryMap());
468     }

470     private BranchData visitBranch(BoolExpr branchCond, CtElement... branchBody)
471     {
472         // Values before entering the branch
473         Map<CtReference, Expr> oldValues = new HashMap<>(variablesMap);
474         BoolExpr oldConditions = currentConditions;
475         Memory oldMemory = new Memory(memory);
476 
477         currentConditions = currentConditions == null ? branchCond : context.mkAnd(currentConditions, branchCond);
478         for (CtElement element : branchBody)
479         {
480             scan(element);
481         }
482         Map<CtReference, Expr> newValues = variablesMap;
483         Memory newMemory = memory;
484 
485         currentConditions = oldConditions;
486         variablesMap = oldValues;
487         memory = oldMemory;
488 
489         return new BranchData(newValues, newMemory);
490     }

492     @Override
493     public <T> void visitCtConditional(CtConditional<T> conditional)
494     {
495         BoolExpr conditionExpr = visitCondition(conditional.getCondition(), false);
496 
497         BranchData thenBranchData = visitBranch(conditionExpr, conditional.getThenExpression());
498         Expr thenExpr = currentResult;
499         solver.push();
500         solver.add(conditionExpr);
501         checkConditionalThenExpression(conditional.getThenExpression());
502         solver.pop();
503 
504         BranchData elseBranchData = visitBranch(context.mkNot(conditionExpr), conditional.getElseExpression());
505         Expr elseExpr = currentResult;
506         solver.push();
507         solver.add(context.mkNot(conditionExpr));
508         checkConditionalElseExpression(conditional.getElseExpression());
509         solver.pop();
510 
511         mergeBranches(conditionExpr, thenBranchData, elseBranchData);
512 
513         currentResult = context.mkITE(conditionExpr, thenExpr, elseExpr);
514         conditional.putMetadata("value", currentResult);
515         checkConditionalResult(conditional);
516     }

518     @Override
519     public <S> void visitCtSwitch(CtSwitch<S> switchStatement)
520     {
521         Expr breakExpr = variablesMap.get(breakFlagReference);
522 
523         // The type of the selector must be char, byte, short, int, Character, Byte, Short, Integer, String, or an enum type.
524         CtExpression<S> selector = switchStatement.getSelector();
525         CtTypeReference<?> selectorType = getActualType(selector);
526         scan(selector);
527         Expr selectorValue = currentResult;
528 
529         // Unboxing conversion
530         if (!getActualType(selector).isPrimitive())
531         {
532             selectorValue = memory.read(getActualType(selector).unbox(), (IntExpr) selectorValue);
533         }
534 
535         BoolExpr commonConditionExpr = null;
536         List<CtCase<? super S>> cases = switchStatement.getCases();
537         for (CtCase<? super S> aCase : cases)
538         {
539             CtExpression<?> caseExpression = aCase.getCaseExpression();
540             BoolExpr branchExpr;
541             if (caseExpression == null)
542             {
543                 // Default label
544                 branchExpr = (commonConditionExpr == null) ? context.mkTrue() : context.mkNot(commonConditionExpr);
545             }
546             else
547             {
548                 // Case label
549                 CtTypeReference<?> caseType = getActualType(caseExpression);
550                 scan(caseExpression);
551                 Expr caseValue = currentResult;
552 
553                 // Binary Numeric Promotion
554                 if (selectorValue instanceof BitVecExpr && caseValue instanceof BitVecExpr)
555                 {
556                     Expr[] result = promoteNumericValues(context, selectorValue, selectorType, caseValue, caseType);
557                     selectorValue = result[0];
558                     caseValue = result[1];
559                 }
560 
561                 BoolExpr conditionExpr = context.mkEq(selectorValue, caseValue);
562 
563                 // Handle cases as conditions
564                 caseExpression.putMetadata("value", conditionExpr);
565                 checkCondition(caseExpression, false);
566 
567                 // Connect cases with OR
568                 commonConditionExpr = (commonConditionExpr == null) ? conditionExpr : context.mkOr(commonConditionExpr, conditionExpr);
569 
570                 branchExpr = context.mkAnd(commonConditionExpr, context.mkNot(getBreakExpr()));
571             }
572 
573             BranchData thenBranchData = visitBranch(branchExpr, aCase);
574             BranchData elseBranchData = new BranchData(variablesMap, memory);
575 
576             mergeBranches(branchExpr, thenBranchData, elseBranchData);
577         }
578 
579         variablesMap.put(breakFlagReference, breakExpr);
580     }

630     @Override
631     public <T> void visitCtConstructorCall(CtConstructorCall<T> constructorCall)
632     {
633         int nextPointer = memory.nextPointer();
634 
635         // Test code for Integer(x) constructor
636         if (constructorCall.getType().getQualifiedName().equals("java.lang.Integer")
637             && constructorCall.getArguments().size() == 1)
638         {
639             CtExpression<?> arg1 = constructorCall.getArguments().get(0);
640             scan(arg1);
641             Expr arg1Value = currentResult;
642 
643             // Cast argument type to the parameter type
644             CtTypeReference<?> arg1Type = getActualType(arg1);
645             CtTypeReference<?> arg1SignatureType = constructorCall.getExecutable().getParameters().get(0);
646             arg1Value = applyCasts(arg1Value, arg1Type, Collections.singletonList(arg1SignatureType));
647 
648             memory.write(constructorCall.getType().unbox(), context.mkInt(nextPointer), arg1Value);
649         }
650         else
651         {
652             constructorCall.getArguments().forEach(this::scan);
653         }
654 
655         IntExpr constructorCallValue = context.mkInt(nextPointer);
656         currentResult = applyCasts(constructorCallValue, constructorCall.getType(), constructorCall.getTypeCasts());
657         constructorCall.putMetadata("value", currentResult);
658     }

672     @Override
673     public <T> void visitCtNewArray(CtNewArray<T> newArray)
674     {
675         int nextPointer = memory.nextPointer();
676         IntExpr arrayValue = context.mkInt(nextPointer);
677 
678         for (CtExpression<Integer> dimensionExpression : newArray.getDimensionExpressions())
679         {
680             scan(dimensionExpression);
681             // TODO: Set array.length equal to dimension size
682             // It seems that spoon does not provide a way to get reference to the array.length property so far.
683             // The code should be something like this:
684             // memory.write(lengthFieldReference, arrayValue, dimensionLengthExpr);
685         }
686 
687         CtTypeReference<?> componentType = ((CtArrayTypeReference) (newArray.getType())).getComponentType();
688 
689         int i = 0;
690         for (CtExpression<?> arrayElement : newArray.getElements())
691         {
692             scan(arrayElement);
693             Expr arrayElementExpr = currentResult;
694             if (arrayElementExpr instanceof BitVecExpr)
695             {
696                 arrayElementExpr = castBV((BitVecExpr) arrayElementExpr, getActualType(arrayElement), componentType);
697             }
698             memory.writeArray((CtArrayTypeReference) newArray.getType(), arrayValue, context.mkBV(i, 32), arrayElementExpr);
699             i++;
700         }
701 
702         currentResult = applyCasts(arrayValue, newArray.getType(), newArray.getTypeCasts());
703     }

705     /**
706      * Makes targetExpr not null (if something is dereferenced => it is not null).
707      */
708     private void visitDereference(Expr targetExpr)
709     {
710         if (targetExpr != null)
711         {
712             BoolExpr notNullExpr = context.mkDistinct(targetExpr, context.mkInt(Memory.nullPointer()));
713             currentConditions = currentConditions == null ? notNullExpr : context.mkAnd(currentConditions, notNullExpr);
714         }
715     }

717     @Override
718     public <T> void visitCtInvocation(CtInvocation<T> invocation)
719     {
720         // Reset arguments
721         List<CtExpression<?>> arguments = invocation.getArguments();
722         for (CtExpression<?> argument : arguments)
723         {
724             scan(argument);
725             Expr argumentExpr = currentResult;
726             if (argumentExpr != null)
727             {
728                 CtTypeReference<?> argumentType = getActualType(argument);
729                 if (!argumentType.isPrimitive() && !isImmutable(argumentType))
730                 {
731                     memory.resetObject(argumentType, (IntExpr) argumentExpr);
732                 }
733             }
734         }
735 
736         // Reset target
737         CtExpression<?> target = invocation.getTarget();
738         Expr targetExpr = null;
739         boolean dereferenceTarget = false;
740         if (target != null)
741         {
742             scan(target);
743             targetExpr = currentResult;
744             if (targetExpr != null)
745             {
746                 CtTypeReference<?> targetType = getActualType(target);
747                 if (!(target instanceof CtTypeAccess))
748                 {
749                     dereferenceTarget = true;
750                     if (!isImmutable(targetType))
751                     {
752                         memory.resetObject(targetType, (IntExpr) targetExpr);
753                     }
754                 }
755             }
756         }
757 
758         CtTypeReference<?> returnType = invocation.getType();
759         if (!isVoid(returnType))
760         {
761             // In the most general case the return value of a function is unknown
762             Expr returnValue = makeFreshConstFromType(context, returnType);
763             currentResult = applyCasts(returnValue, returnType, invocation.getTypeCasts());
764             invocation.putMetadata("value", currentResult);
765         }
766 
767         checkInvocation(invocation);
768 
769         if (dereferenceTarget)
770         {
771             visitDereference(targetExpr);
772         }
773     }

936     private void visitAssignment(CtExpression<?> left, Expr leftValue, CtTypeReference<?> leftType,
937                                  Expr rightValue, CtTypeReference<?> rightType)
938     {
939         CtReference leftReference;
940         if (left instanceof CtArrayWrite)
941         {
942             leftReference = ((CtArrayWrite<?>) left).getTarget().getType();
943         }
944         else
945         {
946             leftReference = ((CtVariableWrite<?>) left).getVariable();
947         }
948 
949         if (leftType.isPrimitive())
950         {
951             if (isCalculable(leftType) && rightValue != null)
952             {
953                 // Unboxing conversion
954                 if (!rightType.isPrimitive())
955                 {
956                     rightValue = memory.read(rightType.unbox(), (IntExpr) rightValue);
957                 }
958 
959                 if (rightValue instanceof BitVecExpr)
960                 {
961                     rightValue = castBV((BitVecExpr) rightValue, rightType, leftType);
962                 }
963             }
964             else
965             {
966                 // left or right is double or float => create value from type
967                 Sort sort = TypeUtils.getTypeSort(context, leftType);
968                 rightValue = context.mkFreshConst("", sort);
969             }
970         }
971         else
972         {
973             if (rightType.isPrimitive())
974             {
975                 // Boxing conversion
976                 int nextPointer = memory.nextPointer();
977                 IntExpr index =  context.mkInt(nextPointer);
978                 if (isCalculable(leftType) && rightValue != null)
979                 {
980                     if (rightValue instanceof BitVecExpr)
981                     {
982                         rightValue = castBV((BitVecExpr) rightValue, rightType, leftType);
983                     }
984                     memory.write(leftType.unbox(), index, rightValue);
985                 }
986                 rightValue = index;
987             }
988         }
989 
990         if (left instanceof CtFieldWrite)
991         {
992             // Update memory
993             memory.write(leftReference, (IntExpr) leftValue, rightValue);
994         }
995 
996         if (left instanceof CtArrayWrite)
997         {
998             // Update memory
999             CtExpression<Integer> index = ((CtArrayWrite<?>) left).getIndexExpression();
1000             CtTypeReference<?> indexType = getActualType(index);
1001             Expr indexExpr = (Expr) index.getMetadata("value");
1002             indexExpr = promoteNumericValue(context, indexExpr, indexType);
1003             memory.writeArray((CtArrayTypeReference) leftReference, (IntExpr) leftValue, indexExpr, rightValue);
1004         }
1005 
1006         variablesMap.put(leftReference, rightValue);
1007         currentResult = rightValue; // Assignment returns its value
1008     }

1031     @Override
1032     public <T, A extends T> void visitCtOperatorAssignment(CtOperatorAssignment<T, A> assignment)
1033     {
1034         // A compound assignment expression of the form E1 op= E2 is equivalent to E1 = (T) ((E1) op (E2)),
1035         // where T is the type of E1, except that E1 is evaluated only once.
1036 
1037         CtExpression<T> left = assignment.getAssigned();
1038         CtExpression<A> right = assignment.getAssignment();
1039 
1040         CtTypeReference<?> leftType = getActualType(left);
1041         CtTypeReference<?> rightType = getActualType(right);
1042 
1043         scan(left);
1044         Expr leftValue = currentResult;
1045         scan(right);
1046         Expr rightValue = currentResult;
1047 
1048         Expr leftData = leftValue;
1049         if (left instanceof CtFieldWrite)
1050         {
1051              leftData = memory.read(((CtFieldWrite<T>) left).getVariable(), (IntExpr) leftValue);
1052         }
1053 
1054         if (left instanceof CtArrayWrite)
1055         {
1056             CtArrayWrite arrayWrite = (CtArrayWrite) left;
1057             CtExpression index = arrayWrite.getIndexExpression();
1058             Expr arrayIndex = (Expr) index.getMetadata("value");
1059             leftData = memory.readArray((CtArrayTypeReference) arrayWrite.getTarget().getType(), (IntExpr) leftValue, arrayIndex);
1060         }
1061 
1062         Expr resExpr = calcBinaryOperator(leftData, leftType, rightValue, rightType, assignment.getKind());
1063         applyCasts(resExpr, assignment.getAssignment().getType(), Collections.singletonList(assignment.getAssigned().getType()));
1064         rightValue = resExpr;
1065         rightType = assignment.getAssigned().getType().unbox(); // Binary operator unboxes its operands
1066 
1067         visitAssignment(left, leftValue, leftType, rightValue, rightType);
1068         assignment.putMetadata("value", currentResult);
1069     }

1200     /**
1201      * Calculates the value of the binary operation.
1202      */
1203     private Expr calcBinaryOperator(Expr leftValue, CtTypeReference<?> leftType,
1204                                     Expr rightValue, CtTypeReference<?> rightType,
1205                                     BinaryOperatorKind kind)
1206     {
1207         final boolean compareReferences =
1208             (kind == BinaryOperatorKind.EQ || kind == BinaryOperatorKind.NE)
1209             && !leftType.isPrimitive() && !rightType.isPrimitive();
1210 
1211         if (!compareReferences)
1212         {
1213             // Handle unknown types
1214             if (!isCalculable(leftType) || leftValue == null)
1215             {
1216                 return isBooleanOperatorKind(kind) ? context.mkFreshConst("", context.getBoolSort()) : null;
1217             }
1218 
1219             // Unboxing conversion
1220             if (!leftType.isPrimitive())
1221             {
1222                 leftValue = memory.read(leftType.unbox(), (IntExpr) leftValue);
1223             }
1224         }
1225 
1226         if (!compareReferences)
1227         {
1228             // Handle unknown types
1229             if (!isCalculable(rightType) || rightValue == null)
1230             {
1231                 return isBooleanOperatorKind(kind) ? context.mkFreshConst("", context.getBoolSort()) : null;
1232             }
1233 
1234             // Unboxing conversion
1235             if (!rightType.isPrimitive())
1236             {
1237                 rightValue = memory.read(rightType.unbox(), (IntExpr) rightValue);
1238             }
1239         }
1240 
1241         if (!compareReferences)
1242         {
1243             // Binary Numeric Promotion
1244             if (leftValue instanceof BitVecExpr && rightValue instanceof BitVecExpr)
1245             {
1246                 Expr[] result = promoteNumericValues(context, leftValue, leftType, rightValue, rightType);
1247                 leftValue = result[0];
1248                 rightValue = result[1];
1249             }
1250         }
1251 
1252         switch (kind)
1253         {
1254             case AND:
1255                 return context.mkAnd((BoolExpr) leftValue, (BoolExpr) rightValue);
1256             case OR:
1257                 return context.mkOr((BoolExpr) leftValue, (BoolExpr) rightValue);
1258             case BITOR:
1259                 if (leftValue instanceof BitVecExpr && rightValue instanceof BitVecExpr)
1260                 {
1261                     return context.mkBVOR((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1262                 }
1263                 else
1264                 {
1265                     return context.mkOr((BoolExpr) leftValue, (BoolExpr) rightValue);
1266                 }
1267             case BITXOR:
1268                 if (leftValue instanceof BitVecExpr && rightValue instanceof BitVecExpr)
1269                 {
1270                     return context.mkBVXOR((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1271                 }
1272                 else
1273                 {
1274                     return context.mkDistinct(leftValue, rightValue);
1275                 }
1276             case BITAND:
1277                 if (leftValue instanceof BitVecExpr && rightValue instanceof BitVecExpr)
1278                 {
1279                     return context.mkBVAND((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1280                 }
1281                 else
1282                 {
1283                     return context.mkAnd((BoolExpr) leftValue, (BoolExpr) rightValue);
1284                 }
1285             case EQ:
1286                 return context.mkEq(leftValue, rightValue);
1287             case NE:
1288                 return context.mkDistinct(leftValue, rightValue);
1289             case LT:
1290                 return context.mkBVSLT((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1291             case LE:
1292                 return context.mkBVSLE((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1293             case GT:
1294                 return context.mkBVSGT((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1295             case GE:
1296                 return context.mkBVSGE((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1297             case SL:
1298                 return context.mkBVSHL((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1299             case SR:
1300                 return context.mkBVASHR((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1301             case USR:
1302                 return context.mkBVLSHR((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1303             case PLUS:
1304                 return context.mkBVAdd((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1305             case MINUS:
1306                 return context.mkBVSub((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1307             case MUL:
1308                 return context.mkBVMul((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1309             case DIV:
1310                 return context.mkBVSDiv((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1311             case MOD:
1312                 return context.mkBVSRem((BitVecExpr) leftValue, (BitVecExpr) rightValue);
1313             default:
1314                 throw new RuntimeException("Unexpected binary operator");
1315         }
1316     }

1318     @Override
1319     public <T> void visitCtBinaryOperator(CtBinaryOperator<T> operator)
1320     {
1321         CtTypeReference<?> leftType = getActualType(operator.getLeftHandOperand());
1322         CtTypeReference<?> rightType = getActualType(operator.getRightHandOperand());
1323         BinaryOperatorKind kind = operator.getKind();
1324 
1325         scan(operator.getLeftHandOperand());
1326         Expr leftValue = currentResult;
1327         checkBinaryOperatorLeft(kind, operator.getLeftHandOperand());
1328 
1329         // Short circuit evaluation:
1330         // When visiting right operand of AND, we should add leftValue to the current conditions;
1331         // When visiting right operand of OR, we should add NOT leftValue to the current conditions;
1332         BoolExpr prev = currentConditions;
1333         if (kind == BinaryOperatorKind.AND || kind == BinaryOperatorKind.OR)
1334         {
1335             Expr predicateValue = leftValue;
1336             if (!leftType.isPrimitive()) // Unboxing conversion
1337             {
1338                 predicateValue = memory.read(leftType.unbox(), (IntExpr) predicateValue);
1339             }
1340             BoolExpr res = kind == BinaryOperatorKind.OR ? context.mkNot((BoolExpr) predicateValue) : (BoolExpr) predicateValue;
1341             currentConditions = currentConditions == null ? res : context.mkAnd(currentConditions, res);
1342         }
1343         scan(operator.getRightHandOperand());
1344         Expr rightValue = currentResult;
1345         checkBinaryOperatorRight(kind, operator.getRightHandOperand());
1346         currentConditions = prev;
1347 
1348         currentResult = calcBinaryOperator(leftValue, leftType, rightValue, rightType, kind);
1349         currentResult = applyCasts(currentResult, operator.getType(), operator.getTypeCasts());
1350         operator.putMetadata("value", currentResult);
1351         checkBinaryOperatorResult(operator);
1352     }

1354     @Override
1355     public <T> void visitCtUnaryOperator(CtUnaryOperator<T> operator)
1356     {
1357         scan(operator.getOperand());
1358         Expr operandValue = currentResult;
1359 
1360         UnaryOperatorKind kind = operator.getKind();
1361         CtTypeReference<?> operandType = getActualType(operator.getOperand());
1362 
1363         // Handle unknown types
1364         if (!isCalculable(operandType) || operandValue == null)
1365         {
1366             currentResult = isBooleanOperatorKind(kind) ? context.mkFreshConst("", context.getBoolSort()) : null;
1367             operator.putMetadata("value", currentResult);
1368             return;
1369         }
1370 
1371         // Unary Numeric Promotion
1372         if (operandValue instanceof BitVecExpr)
1373         {
1374             operandValue = promoteNumericValue(context, operandValue, operandType);
1375         }
1376 
1377         // Unboxing conversion
1378         if (!operandType.isPrimitive())
1379         {
1380             operandValue = memory.read(operandType.unbox(), (IntExpr) operandValue);
1381         }
1382 
1383         switch (kind)
1384         {
1385             case NOT:
1386                 currentResult = context.mkNot((BoolExpr) operandValue);
1387                 break;
1388             case NEG:
1389                 currentResult = context.mkBVNeg((BitVecExpr) operandValue);
1390                 break;
1391             case POS:
1392                 currentResult = operandValue;
1393                 break;
1394             case COMPL:
1395                 int size = TypeUtils.isLong(operandType) ? 64 : 32;
1396                 currentResult = context.mkBVSub(context.mkBVNeg((BitVecExpr) operandValue), context.mkBV(1, size));
1397                 break;
1398             case POSTINC:
1399             case PREINC:
1400             case POSTDEC:
1401             case PREDEC:
1402                 CtExpression<T> operand = operator.getOperand();
1403                 Expr prevExpr;
1404                 if (operand instanceof CtArrayWrite)
1405                 {
1406                     IntExpr targetExpr = getTargetValue(context, variablesMap, memory, ((CtArrayWrite<T>) operand).getTarget());
1407                     CtExpression<Integer> index = ((CtArrayWrite<T>) operand).getIndexExpression();
1408                     CtTypeReference<?> indexType = getActualType(index);
1409                     Expr indexExpr = (Expr) index.getMetadata("value");
1410                     indexExpr = promoteNumericValue(context, indexExpr, indexType);
1411                     CtTypeReference<?> arrayType = ((CtArrayWrite<T>) operand).getTarget().getType();
1412                     prevExpr = memory.readArray((CtArrayTypeReference) arrayType, targetExpr, indexExpr);
1413                 }
1414                 else
1415                 {
1416                     CtVariableReference<?> variable = ((CtVariableWrite<?>) operand).getVariable();
1417                     prevExpr = variablesMap.get(variable);
1418                 }
1419
