129 	public void compileTest() throws Exception {
130 		// contract: the modified version of classes is the one that is compiled to binary code
131 		final Launcher launcher = new Launcher();
132 		launcher.addInputResource("./src/test/resources/noclasspath/Simple.java");
133 		File outputBinDirectory = new File("./target/class-simple");
134 		if (!outputBinDirectory.exists()) {
135 			outputBinDirectory.mkdirs();
136 		}
137 		launcher.setBinaryOutputDirectory(outputBinDirectory);
138 		launcher.getEnvironment().setShouldCompile(true);
139 		launcher.buildModel();
140 
141 		Factory factory = launcher.getFactory();
142 		CoreFactory core = factory.Core();
143 		CodeFactory code = factory.Code();
144 
145 		CtClass simple = factory.Class().get("Simple");
146 
147 		CtMethod method = core.createMethod();
148 		method.addModifier(ModifierKind.PUBLIC);
149 		method.setType(factory.Type().integerPrimitiveType());
150 		method.setSimpleName("m");
151 
152 		CtBlock block = core.createBlock();
153 		CtReturn aReturn = core.createReturn();
154 
155 		CtBinaryOperator binaryOperator = code.createBinaryOperator(
156 				code.createLiteral(10),
157 				code.createLiteral(32),
158 				BinaryOperatorKind.PLUS);
159 		aReturn.setReturnedExpression(binaryOperator);
160 
161 		// return 10 + 32;
162 		block.addStatement(aReturn);
163 		method.setBody(block);
164 
165 		simple.addMethod(method);
166 
167 		launcher.getModelBuilder().compile();
168 
169 		final URLClassLoader urlClassLoader = new URLClassLoader(new URL[]{outputBinDirectory.toURL()});
170 
171 		Class<?> aClass = urlClassLoader.loadClass("Simple");
172 		Method m = aClass.getMethod("m");
173 		assertEquals(42, m.invoke(aClass.newInstance()));
174 	}

202 	public void testNewInstance() {
203 		// contract: a ctclass can be instantiated, and each modification results in a new valid object
204 		Factory factory = new Launcher().getFactory();
205 		CtClass<Ifoo> c = factory.Code().createCodeSnippetStatement(
206 				"class X implements spoon.test.compilation.testclasses.Ifoo { public int foo() {int i=0; return i;} }").compile();
207 		c.addModifier(ModifierKind.PUBLIC); // required otherwise java.lang.IllegalAccessException at runtime when instantiating
208 
209 		CtBlock body = c.getElements(new TypeFilter<>(CtBlock.class)).get(1);
210 		Ifoo o = c.newInstance();
211 		assertEquals(0, o.foo());
212 		for (int i = 1; i <= 10; i++) {
213 			body.getStatement(0).replace(factory.Code().createCodeSnippetStatement("int i = " + i + ";"));
214 			o = c.newInstance();
215 			// each time this is a new class
216 			// each time the behavior has changed!
217 			assertEquals(i, o.foo());
218 		}
219 
220 	}

301 		// without precompile
302 		Launcher l = new Launcher();
303 		l.setArgs(new String[]{"--noclasspath", "-i", "src/test/resources/compilation/"});
304 		l.buildModel();
305 		CtClass klass = l.getFactory().Class().get("compilation.Bar");
306 		// without precompile, actualClass does not exist (an exception is thrown)
307 		try {
308 			klass.getSuperInterfaces().toArray(new CtTypeReference[0])[0].getActualClass();
309 			fail();
310 		} catch (SpoonClassNotFoundException ignore) { }
311 
312 		// with precompile
313 		Launcher l2 = new Launcher();
314 		l2.setArgs(new String[]{"--precompile", "--noclasspath", "-i", "src/test/resources/compilation/"});
315 		l2.buildModel();
316 		CtClass klass2 = l2.getFactory().Class().get("compilation.Bar");
317 		// with precompile, actualClass is not null
318 		Class actualClass = klass2.getSuperInterfaces().toArray(new CtTypeReference[0])[0].getActualClass();
319 		assertNotNull(actualClass);
320 		assertEquals("IBar", actualClass.getSimpleName());
321 
322 		// precompile can be used to compile processors on the fly
323 		Launcher l3 = new Launcher();
324 		l3.setArgs(new String[]{"--precompile", "--noclasspath", "-i", "src/test/resources/compilation/", "-p", "compilation.SimpleProcessor"});
325 		l3.run();

354 	public void testSingleClassLoader() throws Exception {
355 		/*
356 		 *  contract: the environment exposes a classloader configured by the spoonclass path,
357 		 *  there is one class loader, so the loaded classes are compatible
358 		 */
359 		Launcher launcher = new Launcher();
360 		launcher.addInputResource(new FileSystemFolder("./src/test/resources/classloader-test"));
361 		File outputBinDirectory = new File("./target/classloader-test");
362 		if (!outputBinDirectory.exists()) {
363 			outputBinDirectory.mkdirs();
364 		}
365 		launcher.setBinaryOutputDirectory(outputBinDirectory);
366 		launcher.getModelBuilder().build();
367 
368 		CtTypeReference<?> mIFoo = launcher.getFactory().Type().createReference("spoontest.IFoo");
369 		CtTypeReference<?> mFoo = launcher.getFactory().Type().createReference("spoontest.Foo");
370 		assertTrue("Foo subtype of IFoo", mFoo.isSubtypeOf(mIFoo));
371 
372 		launcher.getModelBuilder().compile(SpoonModelBuilder.InputType.FILES);
373 
374 		//Create new launcher which uses classes compiled by previous launcher.
375 		//It simulates the classes without sources, which has to be accessed using reflection
376 		launcher = new Launcher();
377 
378 		// not in the classpath
379 		try {
380 			Class.forName("spoontest.IFoo");
381 			fail();
382 		} catch (ClassNotFoundException expected) {
383 		}
384 
385 		// not in the spoon classpath before setting it
386 		try {
387 			launcher.getEnvironment().getInputClassLoader().loadClass("spoontest.IFoo");
388 			fail();
389 		} catch (ClassNotFoundException expected) {
390 		}
391 
392 		launcher.getEnvironment().setSourceClasspath(new String[]{outputBinDirectory.getAbsolutePath()});
393 
394 		mIFoo = launcher.getFactory().Type().createReference("spoontest.IFoo");
395 		mFoo = launcher.getFactory().Type().createReference("spoontest.Foo");
396 		//if it fails then it is because each class is loaded by different class loader
397 		assertTrue("Foo subtype of IFoo", mFoo.isSubtypeOf(mIFoo));
398 
399 		// not in the spoon classpath before setting it
400 		Class<?> ifoo = launcher.getEnvironment().getInputClassLoader().loadClass("spoontest.IFoo");
401 		Class<?> foo = launcher.getEnvironment().getInputClassLoader().loadClass("spoontest.Foo");
402 
403 		assertTrue(ifoo.isAssignableFrom(foo));
404 		assertSame(ifoo.getClassLoader(), foo.getClassLoader());
405 	}

408 	public void testExoticClassLoader() {
409 		// contract: Spoon uses the exotic class loader
410 
411 		final List<String> l = new ArrayList<>();
412 		class MyClassLoader extends ClassLoader {
413 			@Override
414 			protected Class<?> loadClass(String name, boolean resolve) throws ClassNotFoundException {
415 				l.add(name);
416 				return super.loadClass(name, resolve);
417 			}
418 		}
419 
420 		Launcher launcher = new Launcher();
421 		launcher.getEnvironment().setInputClassLoader(new MyClassLoader());
422 		launcher.getEnvironment().setNoClasspath(true);
423 		launcher.addInputResource("src/test/resources/reference-test/Foo.java");
424 		launcher.buildModel();
425 		launcher.getModel().getRootPackage().accept(new CtScanner() {
426 			@Override
427 			public <T> void visitCtTypeReference(CtTypeReference<T> reference) {
428 				try {
429 					// forcing loading it
430 					reference.getTypeDeclaration();
431 				} catch (SpoonClassNotFoundException ignore) { }
432 			}
433 		});
434 
435 		//JDK 9 has implicit constructor, while JDK 8 has not
436 		assertTrue(l.size() >= 2);
437 		assertTrue(l.contains("KJHKY"));
438 		assertSame(MyClassLoader.class, launcher.getEnvironment().getInputClassLoader().getClass());
439 	}
