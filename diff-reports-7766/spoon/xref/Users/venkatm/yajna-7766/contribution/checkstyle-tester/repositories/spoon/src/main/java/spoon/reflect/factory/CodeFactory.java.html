146 	 *
147 	 * @param <T>
148 	 * 		the actual type of the accessed class if available
149 	 * @param type
150 	 * 		a type reference to the accessed class
151 	 * @return the class access expression.
152 	 */
153 	public <T> CtFieldAccess<Class<T>> createClassAccess(CtTypeReference<T> type) {
154 		@SuppressWarnings({ "rawtypes", "unchecked" }) CtTypeReference<Class<T>> classType = (CtTypeReference) factory.Type().createReference(Class.class);
155 		CtTypeAccess<T> typeAccess = factory.Code().createTypeAccess(type);
156 
157 		CtFieldReference<Class<T>> fieldReference = factory.Core().createFieldReference();
158 		fieldReference.setSimpleName("class");
159 		fieldReference.setType(classType);
160 		fieldReference.setDeclaringType(type);
161 
162 		CtFieldRead<Class<T>> fieldRead = factory.Core().createFieldRead();
163 		fieldRead.setType(classType.clone());
164 		fieldRead.setVariable(fieldReference);
165 		fieldRead.setTarget(typeAccess);
166 		return fieldRead;
167 	}

169 	/**
170 	 * Creates a constructor call. The correct constructor is inferred based on parameters
171 	 *
172 	 * @param type the decelerating type of the constructor
173 	 * @param parameters the arguments of the constructor call
174 	 * @param <T> the actual type of the decelerating type of the constructor if available
175 	 * @return the constructor call
176 	 */
177 	public <T> CtConstructorCall<T> createConstructorCall(CtTypeReference<T> type, CtExpression<?>...parameters) {
178 		CtConstructorCall<T> constructorCall = factory.Core()
179 				.createConstructorCall();
180 		CtExecutableReference<T> executableReference = factory.Core()
181 				.createExecutableReference();
182 		executableReference.setType(type);
183 		executableReference.setDeclaringType(type == null ? type : type.clone());
184 		executableReference.setSimpleName(CtExecutableReference.CONSTRUCTOR_NAME);
185 		List<CtTypeReference<?>> typeReferences = new ArrayList<>();
186 		for (CtExpression<?> parameter : parameters) {
187 			typeReferences.add(parameter.getType());
188 		}
189 		executableReference.setParameters(typeReferences);
190 		constructorCall.setArguments(Arrays.asList(parameters));
191 		constructorCall.setExecutable(executableReference);
192 		return constructorCall;
193 	}

195 	/**
196 	 * Creates a new anonymous class.
197 	 */
198 	public <T> CtNewClass<T> createNewClass(CtType<T> superClass, CtExpression<?>...parameters) {
199 		CtNewClass<T> ctNewClass = factory.Core().createNewClass();
200 		CtConstructor<T> constructor = ((CtClass) superClass).getConstructor(Arrays.stream(parameters).map(x -> x.getType()).toArray(CtTypeReference[]::new));
201 		if (constructor == null) {
202 			throw new SpoonException("no appropriate constructor for these parameters " + Arrays.toString(parameters));
203 		}
204 		CtExecutableReference<T> executableReference = constructor.getReference();
205 		ctNewClass.setArguments(Arrays.asList(parameters));
206 		ctNewClass.setExecutable(executableReference);
207 		CtClass c = superClass.getFactory().createClass();
208 		c.setSuperclass(superClass.getReference());
209 		c.setSimpleName("0");
210 		ctNewClass.setAnonymousClass(c);
211 		return ctNewClass;
212 	}

270 	/**
271 	 * Creates a one-dimension array that must only contain literals.
272 	 */
273 	@SuppressWarnings("unchecked")
274 	public <T> CtNewArray<T[]> createLiteralArray(T[] value) {
275 		if (!value.getClass().isArray()) {
276 			throw new RuntimeException("value is not an array");
277 		}
278 		if (value.getClass().getComponentType().isArray()) {
279 			throw new RuntimeException("can only create one-dimension arrays");
280 		}
281 		final CtTypeReference<T> componentTypeRef = factory.Type().createReference((Class<T>) value.getClass().getComponentType());
282 		final CtArrayTypeReference<T[]> arrayReference = factory.Type().createArrayReference(componentTypeRef);
283 		CtNewArray<T[]> array = factory.Core().<T[]>createNewArray().setType(arrayReference);
284 		for (T e : value) {
285 			CtLiteral<T> l = factory.Core().createLiteral();
286 			l.setValue(e);
287 			array.addElement(l);
288 		}
289 		return array;
290 	}

578 		}
579 		CtTypeReference<T> typeReference = factory.Core().<T>createTypeReference();
580 		typeReference.setSimpleName(originalClass.getSimpleName());

585 			// the inner class reference does not have package
586 			return typeReference.setDeclaringType(createCtTypeReference(originalClass.getDeclaringClass()));
587 		}
588 		return typeReference.setPackage(createCtPackageReference(originalClass.getPackage()));
589 	}

704 	/**
705 	 * Creates a javadoc tag
706 	 *
707 	 * @param content The content of the javadoc tag with a possible paramater
708 	 * @param type The tag type
709 	 * @return a new CtJavaDocTag
710 	 */
711 	public CtJavaDocTag createJavaDocTag(String content, CtJavaDocTag.TagType type) {
712 		if (content == null) {
713 			content = "";
714 		}
715 		CtJavaDocTag docTag = factory.Core().createJavaDocTag();
716 		if (type != null && type.hasParam()) {
717 			int firstWord = content.indexOf(' ');
718 			int firstLine = content.indexOf('\n');
719 			if (firstLine < firstWord && firstLine >= 0) {
720 				firstWord = firstLine;
721 			}
722 			if (firstWord == -1) {
723 				firstWord = content.length();
724 			}
725 			String param = content.substring(0, firstWord);
726 			content = content.substring(firstWord);
727 			docTag.setParam(param);
728 		}
729 		return docTag.setContent(content.trim()).setType(type);
730 	}
