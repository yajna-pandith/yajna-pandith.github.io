48  	@Test
49  	public void testTypeErasure() throws Exception {
50  		//contract: the type erasure computed by getTypeErasure is same as the one computed by the Java compiler
51  		CtClass<?> ctModel = (CtClass<?>) ModelUtils.buildClass(ErasureModelA.class);
52  		//visit all methods of type ctModel
53  		//visit all inner types and their methods recursively `getTypeErasure` returns expected value
54  		//for each formal type parameter or method parameter check if
55  		checkType(ctModel);
56  	}

58  	private void checkType(CtType<?> type) throws NoSuchFieldException, SecurityException {
59  		List<CtTypeParameter> formalTypeParameters = type.getFormalCtTypeParameters();
60  		for (CtTypeParameter ctTypeParameter : formalTypeParameters) {
61  			checkTypeParamErasureOfType(ctTypeParameter, type.getActualClass());
62  		}
63  
64  		for (CtTypeMember member : type.getTypeMembers()) {
65  			if (member instanceof CtFormalTypeDeclarer) {
66  				CtFormalTypeDeclarer ftDecl = (CtFormalTypeDeclarer) member;
67  				formalTypeParameters = ftDecl.getFormalCtTypeParameters();
68  				if (member instanceof CtExecutable<?>) {
69  					CtExecutable<?> exec = (CtExecutable<?>) member;
70  					for (CtTypeParameter ctTypeParameter : formalTypeParameters) {
71  						checkTypeParamErasureOfExecutable(ctTypeParameter);
72  					}
73  					for (CtParameter<?> param : exec.getParameters()) {
74  						checkParameterErasureOfExecutable(param);
75  					}
76  				} else if (member instanceof CtType<?>) {
77  					CtType<?> nestedType = (CtType<?>) member;
78  					// recursive call for nested type
79  					checkType(nestedType);
80  				}
81  			}
82  		}
83  	}

85  	private void checkTypeParamErasureOfType(CtTypeParameter typeParam, Class<?> clazz) throws NoSuchFieldException, SecurityException {
86  		Field field = clazz.getDeclaredField("param" + typeParam.getSimpleName());
87  		assertEquals("TypeErasure of type param " + getTypeParamIdentification(typeParam), field.getType().getName(), typeParam.getTypeErasure().getQualifiedName());
88  	}
89  
90  	private void checkTypeParamErasureOfExecutable(CtTypeParameter typeParam) throws SecurityException {
91  		CtExecutable<?> exec = (CtExecutable<?>) typeParam.getParent();
92  		CtParameter<?> param = exec.filterChildren(new NamedElementFilter<>(CtParameter.class, "param" + typeParam.getSimpleName())).first();
93  		assertNotNull("Missing param" + typeParam.getSimpleName() + " in " + exec.getSignature(), param);
94  		int paramIdx = exec.getParameters().indexOf(param);
95  		Class declClass = exec.getParent(CtType.class).getActualClass();
96  		Executable declExec;
97  		if (exec instanceof CtConstructor) {
98  			declExec = declClass.getDeclaredConstructors()[0];
99  		} else {
100 			declExec = getMethodByName(declClass, exec.getSimpleName());
101 		}
102 		Class<?> paramType = declExec.getParameterTypes()[paramIdx];
103 		// contract the type erasure given with Java reflection is the same as the one computed by spoon
104 		assertEquals("TypeErasure of executable param " + getTypeParamIdentification(typeParam), paramType.getTypeName(), param.getType().getTypeErasure().toString());
105 	}

107 	private void checkParameterErasureOfExecutable(CtParameter<?> param) {
108 		CtExecutable<?> exec = param.getParent();
109 		CtTypeReference<?> typeErasure = param.getType().getTypeErasure();
110 		int paramIdx = exec.getParameters().indexOf(param);
111 		Class declClass = exec.getParent(CtType.class).getActualClass();
112 		Executable declExec;
113 		if (exec instanceof CtConstructor) {
114 			declExec = declClass.getDeclaredConstructors()[0];
115 		} else {
116 			declExec = getMethodByName(declClass, exec.getSimpleName());
117 		}
118 		Class<?> paramType = declExec.getParameterTypes()[paramIdx];
119 		assertEquals(0, typeErasure.getActualTypeArguments().size());
120 		// contract the type erasure of the method parameter given with Java reflection is the same as the one computed by spoon
121 		assertEquals("TypeErasure of executable " + exec.getSignature() + " parameter " + param.getSimpleName(), paramType.getTypeName(), typeErasure.getQualifiedName());
122 	}

124 	private Executable getMethodByName(Class declClass, String simpleName) {
125 		for (Method method : declClass.getDeclaredMethods()) {
126 			if (method.getName().equals(simpleName)) {
127 				return method;
128 			}
129 		}
130 		fail("Method " + simpleName + " not found in " + declClass.getName());
131 		return null;
132 	}

134 	private String getTypeParamIdentification(CtTypeParameter typeParam) {
135 		String result = "<" + typeParam.getSimpleName() + ">";
136 		CtFormalTypeDeclarer l_decl = typeParam.getParent(CtFormalTypeDeclarer.class);
137 		if (l_decl instanceof CtType) {
138 			return ((CtType) l_decl).getQualifiedName() + result;
139 		}
140 		if (l_decl instanceof CtExecutable) {
141 			CtExecutable exec = (CtExecutable) l_decl;
142 			if (exec instanceof CtMethod) {
143 				result = exec.getSignature() + result;
144 			}
145 			return exec.getParent(CtType.class).getQualifiedName() + "#" + result;
146 		}
147 		throw new AssertionError();
148 	}

150 	@Test
151 	public void testTypeSame() throws Exception {
152 		CtClass<?> ctModel = (CtClass<?>) ModelUtils.buildClass(ErasureModelA.class);
153 		CtTypeParameter tpA = ctModel.getFormalCtTypeParameters().get(0);
154 		CtTypeParameter tpB = ctModel.getFormalCtTypeParameters().get(1);
155 		CtTypeParameter tpC = ctModel.getFormalCtTypeParameters().get(2);
156 		CtTypeParameter tpD = ctModel.getFormalCtTypeParameters().get(3);
157 
158 		CtConstructor<?> ctModelCons = ctModel.getConstructors().iterator().next();
159 		CtMethod<?> ctModelMethod = ctModel.getMethodsByName("method").get(0);
160 		CtMethod<?> ctModelMethod2 = ctModel.getMethodsByName("method2").get(0);
161 
162 		CtClass<?> ctModelB = ctModel.filterChildren(new NamedElementFilter<>(CtClass.class, "ModelB")).first();
163 		CtTypeParameter tpA2 = ctModelB.getFormalCtTypeParameters().get(0);
164 		CtTypeParameter tpB2 = ctModelB.getFormalCtTypeParameters().get(1);
165 		CtTypeParameter tpC2 = ctModelB.getFormalCtTypeParameters().get(2);
166 		CtTypeParameter tpD2 = ctModelB.getFormalCtTypeParameters().get(3);
167 
168 		CtConstructor<?> ctModelBCons = ctModelB.getConstructors().iterator().next();
169 		CtMethod<?> ctModelBMethod = ctModelB.getMethodsByName("method").get(0);
170 
171 		//the type parameters of ErasureModelA and ErasureModelA$ModelB are same if they are on the same position.
172 		checkIsSame(ctModel.getFormalCtTypeParameters(), ctModelB.getFormalCtTypeParameters(), true);
173 
174 		//the type parameters of ErasureModelA#constructor and ErasureModelA$ModelB constructor are same, because constructors has same formal type parameters
175 		//https://docs.oracle.com/javase/specs/jls/se8/html/jls-8.html#jls-8.4.4
176 		checkIsSame(ctModelCons.getFormalCtTypeParameters(), ctModelBCons.getFormalCtTypeParameters(), true);
177 		//the type parameters of ctModel ErasureModelA#method and ErasureModelA$ModelB#method are same if they are on the same position.
178 		checkIsSame(ctModelMethod.getFormalCtTypeParameters(), ctModelBMethod.getFormalCtTypeParameters(), true);
179 
180 		//the type parameters of ctModel ErasureModelA#constructor and ErasureModelA$ModelB#method are never same, because they have different type of scope (Method!=Constructor)
181 		checkIsSame(ctModelCons.getFormalCtTypeParameters(), ctModelBMethod.getFormalCtTypeParameters(), false);
182 		//the type parameters of ctModel ErasureModelA#method and ErasureModelA#method2 are same, because they have same formal type parameters
183 		//https://docs.oracle.com/javase/specs/jls/se8/html/jls-8.html#jls-8.4.4
184 		checkIsSame(ctModelMethod.getFormalCtTypeParameters(), ctModelMethod2.getFormalCtTypeParameters(), true);
185 
186 		CtClass<?> ctModelC = ctModel.filterChildren(new NamedElementFilter<>(CtClass.class, "ModelC")).first();
187 	}

189 	/**
190 	 * checks that parameters on the same position are same and parameters on other positions are not same
191 	 *
192 	 * @param isSameOnSameIndex TODO
193 	 */
194 	private void checkIsSame(List<CtTypeParameter> tps1, List<CtTypeParameter> tps2, boolean isSameOnSameIndex) {
195 		for (int i = 0; i < tps1.size(); i++) {
196 			CtTypeParameter tp1 = tps1.get(i);
197 			for (int j = 0; j < tps2.size(); j++) {
198 				CtTypeParameter tp2 = tps2.get(j);
199 				if (i == j && isSameOnSameIndex) {
200 					checkIsSame(tp1, tp2);
201 				} else {
202 					checkIsNotSame(tp1, tp2);
203 				}
204 			}
205 		}
206 	}
207 
208 	private void checkIsSame(CtTypeParameter tp1, CtTypeParameter tp2) {
209 		assertTrue(isSame(tp1, tp2, false, true) || isSame(tp2, tp1, false, true));
210 	}
211 
212 	private void checkIsNotSame(CtTypeParameter tp1, CtTypeParameter tp2) {
213 		assertFalse(isSame(tp1, tp2, false, true) || isSame(tp2, tp1, false, true));
214 	}
215 
216 	private static boolean isSame(CtTypeParameter thisType, CtTypeParameter thatType, boolean canTypeErasure, boolean checkMethodOverrides) {
217 		CtTypeReference<?> thatAdaptedType = thisType.getFactory().Type().createTypeAdapter(thisType.getTypeParameterDeclarer()).adaptType(thatType);
218 		if (thatAdaptedType == null) {
219 			return false;
220 		}
221 		return thisType.getQualifiedName().equals(thatAdaptedType.getQualifiedName());
222 	}
