61  	@Test
62  	public void testPackage() throws Exception {
63  		final String classFilePath = "./src/test/java/spoon/test/pkg/name/PackageTestClass.java";
64  		final String packageInfoFilePath = "./src/test/java/spoon/test/pkg/package-info.java";
65  		final File packageInfoFile = new File(packageInfoFilePath);
66  
67  		Launcher spoon = new Launcher();
68  		Factory factory = spoon.createFactory();
69  		factory.getEnvironment().setCommentEnabled(true);
70  		spoon.createCompiler(factory, SpoonResourceHelper.resources(classFilePath, packageInfoFilePath)).build();
71  
72  		CtClass<?> clazz = factory.Class().get(PackageTestClass.class);
73  		assertSame(PackageTestClass.class, clazz.getActualClass());
74  
75  		CtPackage ctPackage = clazz.getPackage();
76  		assertEquals("spoon.test.pkg.name", ctPackage.getQualifiedName());
77  		assertEquals("", ctPackage.getDocComment());
78  		assertTrue(CtPackage.class.isAssignableFrom(ctPackage.getParent().getClass()));
79  		assertFalse(ctPackage.hasPackageInfo());
80  		assertFalse(ctPackage.isEmpty());
81  
82  		ctPackage = (CtPackage) ctPackage.getParent();
83  		assertEquals("spoon.test.pkg", ctPackage.getQualifiedName());
84  		assertNotNull(ctPackage.getPosition());
85  		assertEquals(packageInfoFile.getCanonicalPath(), ctPackage.getPosition().getFile().getCanonicalPath());
86  		assertEquals(1, ctPackage.getPosition().getLine());
87  		assertEquals(0, ctPackage.getPosition().getSourceStart());
88  		assertEquals(71, ctPackage.getPosition().getSourceEnd());
89  		assertEquals(1, ctPackage.getAnnotations().size());
90  		assertEquals("This is test\nJavaDoc.", ctPackage.getComments().get(0).getContent());
91  		assertTrue(ctPackage.hasPackageInfo());
92  
93  		CtAnnotation<?> annotation = ctPackage.getAnnotations().get(0);
94  		assertSame(Deprecated.class, annotation.getAnnotationType().getActualClass());
95  		assertEquals(packageInfoFile.getCanonicalPath(), annotation.getPosition().getFile().getCanonicalPath());
96  		assertEquals(5, annotation.getPosition().getLine());
97  
98  		assertTrue(CtPackage.class.isAssignableFrom(ctPackage.getParent().getClass()));
99  
100 		ctPackage = (CtPackage) ctPackage.getParent();
101 		assertEquals("spoon.test", ctPackage.getQualifiedName());
102 		assertEquals("", ctPackage.getDocComment());
103 
104 		//contract: a freshly created package is empty
105 		assertTrue(factory.createPackage().isEmpty());
106 	}

165 		final Launcher spoon = new Launcher();
166 		spoon.addInputResource("./src/test/java/spoon/test/pkg/testclasses/Foo.java");
167 		spoon.buildModel();

254 	@Test
255 	public void testAddAnnotationToPackage() throws Exception {
256 		// contract: Created package-info should used imports in auto-import
257 		final Launcher spoon = new Launcher();
258 		spoon.addInputResource("./src/test/java/spoon/test/pkg/testclasses/Foo.java");
259 		spoon.getEnvironment().setAutoImports(true);
260 		File outputDir = new File("./target/spoon-packageinfo");
261 		spoon.getEnvironment().setSourceOutputDirectory(outputDir);
262 
263 		spoon.buildModel();
264 
265 		CtAnnotationType annotationType = (CtAnnotationType)spoon.getFactory().Annotation().get(GlobalAnnotation.class);
266 		CtAnnotation annotation = spoon.getFactory().Core().createAnnotation();
267 		annotation.setAnnotationType(annotationType.getReference());
268 		CtPackage ctPackage = spoon.getFactory().Package().get("spoon.test.pkg.testclasses");
269 		ctPackage.addAnnotation(annotation);
270 
271 
272 		JavaOutputProcessor outputProcessor = spoon.createOutputWriter();
273 		outputProcessor.process(ctPackage);
274 
275 		File packageInfo = new File(outputDir, "spoon/test/pkg/testclasses/package-info.java");
276 		assertTrue(packageInfo.exists());
277 
278 		canBeBuilt(packageInfo, 8);
279 
280 		List<String> lines = Files.readAllLines(packageInfo.toPath());
281 
282 		assertFalse(lines.isEmpty());
283 
284 		for (String s : lines) {
285 			if (s.trim().startsWith("import")) {
286 				assertEquals("import spoon.test.annotation.testclasses.GlobalAnnotation;", s.trim());
287 			}
288 			if (s.trim().startsWith("@")) {
289 				assertEquals("@GlobalAnnotation", s.trim());
290 			}
291 		}
292 	}
