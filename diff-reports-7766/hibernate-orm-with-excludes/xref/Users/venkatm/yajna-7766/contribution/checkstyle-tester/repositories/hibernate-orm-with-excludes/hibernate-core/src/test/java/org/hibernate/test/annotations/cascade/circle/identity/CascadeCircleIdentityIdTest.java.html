24  package org.hibernate.test.annotations.cascade.circle.identity;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.testing.DialectChecks;
30  import org.hibernate.testing.RequiresDialectFeature;
31  import org.hibernate.testing.TestForIssue;
32  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
33  
34  @RequiresDialectFeature(DialectChecks.SupportsIdentityColumns.class)
35  public class CascadeCircleIdentityIdTest extends BaseCoreFunctionalTestCase {
36  	@Test
37  	@TestForIssue( jiraKey = "HHH-5472" )
38  	public void testCascade() {
39  		A a = new A();
40  		B b = new B();
41  		C c = new C();
42  		D d = new D();
43  		E e = new E();
44  		F f = new F();
45  		G g = new G();
46  		H h = new H();
47  
48  		a.getBCollection().add(b);
49  		b.setA(a);
50  		
51  		a.getCCollection().add(c);
52  		c.setA(a);
53  		
54  		b.getCCollection().add(c);
55  		c.setB(b);
56  		
57  		a.getDCollection().add(d);
58  		d.getACollection().add(a);
59  		
60  		d.getECollection().add(e);
61  		e.setF(f);
62  		
63  		f.getBCollection().add(b);
64  		b.setF(f);
65  		
66  		c.setG(g);
67  		g.getCCollection().add(c);
68  		
69  		f.setH(h);
70  		h.setG(g);
71  		
72  		Session s;
73  		s = openSession();
74  		s.getTransaction().begin();
75  		try {
76  			// Fails: says that C.b is null (even though it isn't). Doesn't fail if you persist c, g or h instead of a
77  			s.persist(a);
78  			s.flush();
79  		} finally {
80  			s.getTransaction().rollback();
81  			s.close();
82  		}
83  	}
84  
85  	@Override
86  	protected Class[] getAnnotatedClasses() {
87  		return new Class[]{
88  				A.class,
89  				B.class,
90  				C.class,
91  				D.class,
92  				E.class,
93  				F.class,
94  				G.class,
95  				H.class
96  		};
97  	}
98  
99  }
