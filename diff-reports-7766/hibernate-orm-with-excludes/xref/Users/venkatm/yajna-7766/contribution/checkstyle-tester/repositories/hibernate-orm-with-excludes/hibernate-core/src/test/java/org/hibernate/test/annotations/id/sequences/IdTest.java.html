244 	@Test
245 	public void testIdClass() throws Exception {
246 		Session s;
247 		Transaction tx;
248 		s = openSession();
249 		tx = s.beginTransaction();
250 		Footballer fb = new Footballer( "David", "Beckam", "Arsenal" );
251 		GoalKeeper keeper = new GoalKeeper( "Fabien", "Bartez", "OM" );
252 		s.persist( fb );
253 		s.persist( keeper );
254 		tx.commit();
255 		s.clear();
256 
257 		// lookup by id
258 		tx = s.beginTransaction();
259 		FootballerPk fpk = new FootballerPk( "David", "Beckam" );
260 		fb = ( Footballer ) s.get( Footballer.class, fpk );
261 		FootballerPk fpk2 = new FootballerPk( "Fabien", "Bartez" );
262 		keeper = ( GoalKeeper ) s.get( GoalKeeper.class, fpk2 );
263 		assertNotNull( fb );
264 		assertNotNull( keeper );
265 		assertEquals( "Beckam", fb.getLastname() );
266 		assertEquals( "Arsenal", fb.getClub() );
267 		assertEquals(
268 				1, s.createQuery(
269 						"from Footballer f where f.firstname = 'David'"
270 				).list().size()
271 		);
272 		tx.commit();
273 
274 		// reattach by merge
275 		tx = s.beginTransaction();
276 		fb.setClub( "Bimbo FC" );
277 		s.merge( fb );
278 		tx.commit();
279 
280 		// reattach by saveOrUpdate
281 		tx = s.beginTransaction();
282 		fb.setClub( "Bimbo FC SA" );
283 		s.saveOrUpdate( fb );
284 		tx.commit();
285 
286 		// clean up
287 		s.clear();
288 		tx = s.beginTransaction();
289 		fpk = new FootballerPk( "David", "Beckam" );
290 		fb = ( Footballer ) s.get( Footballer.class, fpk );
291 		assertEquals( "Bimbo FC SA", fb.getClub() );
292 		s.delete( fb );
293 		s.delete( keeper );
294 		tx.commit();
295 		s.close();
296 	}

322 	@Test
323 	public void testLowAllocationSize() throws Exception {
324 		Session s;
325 		Transaction tx;
326 		s = openSession();
327 		tx = s.beginTransaction();
328 		int size = 4;
329 		BreakDance[] bds = new BreakDance[size];
330 		for ( int i = 0; i < size; i++ ) {
331 			bds[i] = new BreakDance();
332 			s.persist( bds[i] );
333 		}
334 		s.flush();
335 		for ( int i = 0; i < size; i++ ) {
336 			assertEquals( i + 1, bds[i].id.intValue() );
337 		}
338 		tx.rollback();
339 		s.close();
340 	}
