48  	@Test
49  	public void testManyToOneFromNonPk() throws Exception {
50  		Session s = openSession();
51  		Transaction tx = s.beginTransaction();
52  		Menu menu = new Menu();
53  		menu.setOrderNbr( "123" );
54  		menu.setDefault( "F" );
55  		s.persist( menu );
56  		FoodItem foodItem = new FoodItem();
57  		foodItem.setItem( "Mouse" );
58  		foodItem.setOrder( menu );
59  		s.persist( foodItem );
60  		s.flush();
61  		s.clear();
62  		foodItem = ( FoodItem ) s.get( FoodItem.class, foodItem.getId() );
63  		assertNotNull( foodItem.getOrder() );
64  		assertEquals( "123", foodItem.getOrder().getOrderNbr() );
65  		tx.rollback();
66  		s.close();
67  	}

118 	@Test
119 	public void testReferencedColumnNameBelongsToEmbeddedIdOfReferencedEntity() throws Exception {
120 		Session session = openSession();
121 		Transaction tx = session.beginTransaction();
122 
123 		Integer companyCode = 10;
124 		Integer mfgCode = 100;
125 		String contractNumber = "NSAR97841";
126 		ContractId contractId = new ContractId(companyCode, 12457l, 1);
127 
128 		Manufacturer manufacturer = new Manufacturer(new ManufacturerId(
129 				companyCode, mfgCode), "FORD");
130 
131 		Model model = new Model(new ModelId(companyCode, mfgCode, "FOCUS"),
132 				"FORD FOCUS");
133 
134 		session.persist(manufacturer);
135 		session.persist(model);
136 
137 		Contract contract = new Contract();
138 		contract.setId(contractId);
139 		contract.setContractNumber(contractNumber);
140 		contract.setManufacturer(manufacturer);
141 		contract.setModel(model);
142 
143 		session.persist(contract);
144 
145 		session.flush();
146 		session.clear();
147 
148 		contract = (Contract) session.load(Contract.class, contractId);
149 		assertEquals("NSAR97841", contract.getContractNumber());
150 		assertEquals("FORD", contract.getManufacturer().getName());
151 		assertEquals("FORD FOCUS", contract.getModel().getName());
152 
153 		tx.commit();
154 		session.close();
155 	}

157 	@Test
158 	@SkipForDialects( {
159 			@SkipForDialect( value = { HSQLDialect.class }, comment = "The used join conditions does not work in HSQLDB. See HHH-4497." ), 
160 			@SkipForDialect( value = { SQLServer2005Dialect.class } ),
161 			@SkipForDialect( value = { Oracle8iDialect.class }, comment = "Oracle/DB2 do not support 'substring' function" ),
162 			@SkipForDialect( value = { DB2Dialect.class }, comment = "Oracle/DB2 do not support 'substring' function" ) } )
163 	public void testManyToOneFromNonPkToNonPk() throws Exception {
164 		// also tests usage of the stand-alone @JoinFormula annotation (i.e. not wrapped within @JoinColumnsOrFormulas)
165 		Session s = openSession();
166 		Transaction tx = s.beginTransaction();
167 
168         Product kit = new Product();
169         kit.id = 1;
170         kit.productIdnf = "KIT";
171         kit.description = "Kit";
172         s.persist(kit);
173 
174         Product kitkat = new Product();
175         kitkat.id = 2;
176         kitkat.productIdnf = "KIT_KAT";
177         kitkat.description = "Chocolate";
178         s.persist(kitkat);
179 
180         s.flush();
181         s.clear();
182 
183         kit = (Product) s.get(Product.class, 1);
184         kitkat = (Product) s.get(Product.class, 2);
185         System.out.println(kitkat.description);
186         assertNotNull(kitkat);
187         assertEquals(kit, kitkat.getProductFamily());
188         assertEquals(kit.productIdnf, kitkat.getProductFamily().productIdnf);
189         assertEquals("KIT_KAT", kitkat.productIdnf.trim());
190         assertEquals("Chocolate", kitkat.description.trim());
191 
192         tx.rollback();
193 		s.close();
194     }
