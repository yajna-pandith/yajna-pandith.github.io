102 		Transaction tx = s.beginTransaction();
103 
104 		Ball b = new Ball();
105 		Dog d = new Dog();
106 		Computer c = new Computer();

122 		tx.commit();
123 		s.close();

237 	@Test
238 	public void testIdClass() throws Exception {
239 		Session s;
240 		Transaction tx;
241 		s = openSession();
242 		tx = s.beginTransaction();
243 		Footballer fb = new Footballer("David", "Beckam", "Arsenal");
244 		GoalKeeper keeper = new GoalKeeper("Fabien", "Bartez", "OM");
245 		s.persist(fb);
246 		s.persist(keeper);
247 		tx.commit();
248 		s.clear();
249 
250 		// lookup by id
251 		tx = s.beginTransaction();
252 		FootballerPk fpk = new FootballerPk("David", "Beckam");
253 		fb = (Footballer) s.get(Footballer.class, fpk);
254 		FootballerPk fpk2 = new FootballerPk("Fabien", "Bartez");
255 		keeper = (GoalKeeper) s.get(GoalKeeper.class, fpk2);
256 		assertNotNull(fb);
257 		assertNotNull(keeper);
258 		assertEquals("Beckam", fb.getLastname());
259 		assertEquals("Arsenal", fb.getClub());
260 		assertEquals(1, s.createQuery(
261 				"from Footballer f where f.firstname = 'David'").list().size());
262 		tx.commit();
263 
264 		// reattach by merge
265 		tx = s.beginTransaction();
266 		fb.setClub("Bimbo FC");
267 		s.merge(fb);
268 		tx.commit();
269 
270 		// reattach by saveOrUpdate
271 		tx = s.beginTransaction();
272 		fb.setClub("Bimbo FC SA");
273 		s.saveOrUpdate(fb);
274 		tx.commit();
275 
276 		// clean up
277 		s.clear();
278 		tx = s.beginTransaction();
279 		fpk = new FootballerPk("David", "Beckam");
280 		fb = (Footballer) s.get(Footballer.class, fpk);
281 		assertEquals("Bimbo FC SA", fb.getClub());
282 		s.delete(fb);
283 		s.delete(keeper);
284 		tx.commit();
285 		s.close();
286 	}

295 	@Test
296 	public void testLowAllocationSize() throws Exception {
297 		Session s;
298 		Transaction tx;
299 		s = openSession();
300 		tx = s.beginTransaction();
301 		int size = 4;
302 		BreakDance[] bds = new BreakDance[size];
303 		for (int i = 0; i < size; i++) {
304 			bds[i] = new BreakDance();
305 			s.persist(bds[i]);
306 		}
307 		s.flush();
308 		for (int i = 0; i < size; i++) {
309 			assertEquals(i + 1, bds[i].id.intValue());
310 		}
311 		tx.rollback();
312 		s.close();
313 	}
