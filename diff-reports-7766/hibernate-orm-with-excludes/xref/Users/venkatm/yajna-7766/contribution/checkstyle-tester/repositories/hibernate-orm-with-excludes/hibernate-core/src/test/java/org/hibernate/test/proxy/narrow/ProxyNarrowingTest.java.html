21  package org.hibernate.test.proxy.narrow;
22  
23  import static org.junit.Assert.assertTrue;
24  
25  import org.hibernate.Hibernate;
26  import org.hibernate.Session;
27  import org.hibernate.Transaction;
28  import org.hibernate.proxy.HibernateProxy;
29  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
30  import org.junit.Test;
31  
32  /**
33   * @author Yoann Rodi√®re
34   * @author Guillaume Smet
35   */
36  public class ProxyNarrowingTest extends BaseCoreFunctionalTestCase {
37  
38  	@Override
39  	protected Class<?>[] getAnnotatedClasses() {
40  		return new Class<?>[] { AbstractEntity.class, ConcreteEntity.class, LazyAbstractEntityReference.class };
41  	}
42  
43  	@Test
44  	public void testNarrowedProxyIsInitializedIfOriginalProxyIsInitialized() {
45  		Session session = openSession();
46  
47  		Integer entityReferenceId = null;
48  
49  		// Populate the database
50  		try {
51  			Transaction t = session.beginTransaction();
52  
53  			ConcreteEntity entity = new ConcreteEntity();
54  			session.save( entity );
55  
56  			LazyAbstractEntityReference reference = new LazyAbstractEntityReference( entity );
57  			session.save( reference );
58  			entityReferenceId = reference.getId();
59  
60  			session.flush();
61  			t.commit();
62  		}
63  		finally {
64  			session.close();
65  		}
66  
67  		session = openSession();
68  
69  		try {
70  			// load a proxified version of the entity into the session: the proxy is based on the AbstractEntity class
71  			// as the reference class property is of type AbstractEntity.
72  			LazyAbstractEntityReference reference = (LazyAbstractEntityReference) session.get( LazyAbstractEntityReference.class, entityReferenceId );
73  			AbstractEntity abstractEntityProxy = reference.getEntity();
74  
75  			assertTrue( ( abstractEntityProxy instanceof HibernateProxy ) && !Hibernate.isInitialized( abstractEntityProxy ) );
76  			Hibernate.initialize( abstractEntityProxy );
77  			assertTrue( Hibernate.isInitialized( abstractEntityProxy ) );
78  
79  			// load the concrete class via session.load to trigger the StatefulPersistenceContext.narrowProxy code
80  			ConcreteEntity concreteEntityProxy = (ConcreteEntity) session.load( ConcreteEntity.class, abstractEntityProxy.getId() );
81  
82  			// the new proxy created should be initialized
83  			assertTrue( Hibernate.isInitialized( concreteEntityProxy ) );
84  			assertTrue( session.contains( concreteEntityProxy ) );
85  		}
86  		finally {
87  			session.close();
88  		}
89  	}
90  
91  }
