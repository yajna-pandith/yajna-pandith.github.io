38  	@Test
39  	public void testDefaultAnyAssociation() {
40  		Session s = openSession();
41  		Transaction t = s.beginTransaction();
42  
43  		PropertySet set1 = new PropertySet( "string" );
44  		Property property = new StringProperty( "name", "Alex" );
45  		set1.setSomeProperty( property );
46  		set1.addGeneratedProperty( property );
47  		s.save( set1 );
48  
49  		PropertySet set2 = new PropertySet( "integer" );
50  		property = new IntegerProperty( "age", 33 );
51  		set2.setSomeProperty( property );
52  		set2.addGeneratedProperty( property );
53  		s.save( set2 );
54  
55  		s.flush();
56  		s.clear();
57  
58  		Query q = s
59  				.createQuery( "select s from PropertySet s where name = :name" );
60  		q.setString( "name", "string" );
61  		PropertySet result = (PropertySet) q.uniqueResult();
62  
63  		assertNotNull( result );
64  		assertNotNull( result.getSomeProperty() );
65  		assertTrue( result.getSomeProperty() instanceof StringProperty );
66  		assertEquals( "Alex", result.getSomeProperty().asString() );
67  		assertNotNull( result.getGeneralProperties() );
68  		assertEquals( 1, result.getGeneralProperties().size() );
69  		assertEquals( "Alex", result.getGeneralProperties().get( 0 ).asString() );
70  
71  		q.setString( "name", "integer" );
72  		result = (PropertySet) q.uniqueResult();
73  		assertNotNull( result );
74  		assertNotNull( result.getSomeProperty() );
75  		assertTrue( result.getSomeProperty() instanceof IntegerProperty );
76  		assertEquals( "33", result.getSomeProperty().asString() );
77  		assertNotNull( result.getGeneralProperties() );
78  		assertEquals( 1, result.getGeneralProperties().size() );
79  		assertEquals( "33", result.getGeneralProperties().get( 0 ).asString() );
80  
81  		t.rollback();
82  		s.close();
83  	}

85  	@Test
86  	public void testManyToAnyWithMap() throws Exception {
87  
88  		Session s = openSession();
89  		Transaction t = s.beginTransaction();
90  
91  		PropertyMap map = new PropertyMap( "sample" );
92  		map.getProperties().put( "name", new StringProperty( "name", "Alex" ) );
93  		map.getProperties().put( "age", new IntegerProperty( "age", 33 ) );
94  
95  		s.save( map );
96  
97  		s.flush();
98  		s.clear();
99  
100 		Query q = s
101 				.createQuery( "SELECT map FROM PropertyMap map WHERE map.name = :name" );
102 		q.setString( "name", "sample" );
103 		PropertyMap actualMap = (PropertyMap) q.uniqueResult();
104 
105 		assertNotNull( actualMap );
106 		assertNotNull( actualMap.getProperties() );
107 
108 		Property property = actualMap.getProperties().get( "name" );
109 		assertNotNull( property );
110 		assertTrue( property instanceof StringProperty );
111 		assertEquals( "Alex", property.asString() );
112 
113 		property = actualMap.getProperties().get( "age" );
114 		assertNotNull( property );
115 		assertTrue( property instanceof IntegerProperty );
116 		assertEquals( "33", property.asString() );
117 
118 		t.rollback();
119 		s.close();
120 
121 	}

123 	@Test
124 	public void testMetaDataUseWithManyToAny() throws Exception {
125 		Session s = openSession();
126 		Transaction t = s.beginTransaction();
127 
128 		PropertyList list = new PropertyList( "sample" );
129 		StringProperty stringProperty = new StringProperty( "name", "Alex" );
130 		IntegerProperty integerProperty = new IntegerProperty( "age", 33 );
131 		LongProperty longProperty = new LongProperty( "distance", 121L );
132 		CharProperty charProp = new CharProperty( "Est", 'E' );
133 
134 		list.setSomeProperty( longProperty );
135 
136 		list.addGeneratedProperty( stringProperty );
137 		list.addGeneratedProperty( integerProperty );
138 		list.addGeneratedProperty( longProperty );
139 		list.addGeneratedProperty( charProp );
140 
141 		s.save( list );
142 
143 		s.flush();
144 		s.clear();
145 
146 		Query q = s
147 				.createQuery( "SELECT list FROM PropertyList list WHERE list.name = :name" );
148 		q.setString( "name", "sample" );
149 		PropertyList<Property> actualList = (PropertyList<Property>) q
150 				.uniqueResult();
151 
152 		assertNotNull( actualList );
153 		assertNotNull( actualList.getGeneralProperties() );
154 		assertEquals( 4, actualList.getGeneralProperties().size() );
155 
156 		Property property = actualList.getSomeProperty();
157 		assertNotNull( property );
158 		assertTrue( property instanceof LongProperty );
159 		assertEquals( "121", property.asString() );
160 
161 		assertEquals( "Alex", actualList.getGeneralProperties().get( 0 )
162 				.asString() );
163 		assertEquals( "33", actualList.getGeneralProperties().get( 1 ).asString() );
164 		assertEquals( "121", actualList.getGeneralProperties().get( 2 ).asString() );
165 		assertEquals( "E", actualList.getGeneralProperties().get( 3 ).asString() );
166 
167 		t.rollback();
168 		s.close();
169 	}
