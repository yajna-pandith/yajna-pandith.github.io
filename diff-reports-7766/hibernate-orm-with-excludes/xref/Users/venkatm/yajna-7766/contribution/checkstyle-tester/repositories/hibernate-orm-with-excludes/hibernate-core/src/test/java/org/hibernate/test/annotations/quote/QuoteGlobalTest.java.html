54  	@Test
55  	@TestForIssue(jiraKey = "HHH-7890")
56  	public void testQuotedUniqueConstraint() {
57  		Iterator<UniqueKey> itr = configuration().getClassMapping( Person.class.getName() )
58  				.getTable().getUniqueKeyIterator();
59  		while ( itr.hasNext() ) {
60  			UniqueKey uk = itr.next();
61  			assertEquals( uk.getColumns().size(), 1 );
62  			assertEquals( uk.getColumn( 0 ).getName(),  "name");
63  			return;
64  		}
65  		fail( "GLOBALLY_QUOTED_IDENTIFIERS caused the unique key creation to fail." );
66  	}
67  	
68  	@Test
69  	public void testQuoteManytoMany() {
70  		Session s = openSession();
71  		Transaction tx = s.beginTransaction();
72  		User u = new User();
73  		s.persist( u );
74  		Role r = new Role();
75  		s.persist( r );
76  		u.getRoles().add( r );
77  		s.flush();
78  		s.clear();
79  		u = (User) s.get( User.class, u.getId() );
80  		assertEquals( 1, u.getRoles().size() );
81  		tx.rollback();
82  		String role = User.class.getName() + ".roles";
83  		assertEquals( "User_Role", configuration().getCollectionMapping( role ).getCollectionTable().getName() );
84  		s.close();
85  	}
86  
87  	@Test
88  	@TestForIssue(jiraKey = "HHH-7927")
89  	public void testTableGeneratorQuoting() {
90  		Configuration configuration = constructAndConfigureConfiguration();
91  		configuration.addAnnotatedClass(TestEntity.class);
92  		new SchemaExport(configuration).execute( false, true, false, true );
93  		try {
94  			new SchemaValidator(configuration).validate();
95  		}
96  		catch (HibernateException e) {
97  			fail( "The identifier generator table should have validated.  " + e.getMessage() );
98  		}
99  	}
