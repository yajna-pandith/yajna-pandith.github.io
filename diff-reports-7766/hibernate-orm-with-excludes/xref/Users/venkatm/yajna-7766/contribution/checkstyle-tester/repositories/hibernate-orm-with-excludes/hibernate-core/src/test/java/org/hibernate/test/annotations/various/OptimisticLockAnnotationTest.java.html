24  package org.hibernate.test.annotations.various;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
30  
31  import static org.junit.Assert.assertEquals;
32  
33  /**
34   * Test for the @OptimisticLock annotation.
35   *
36   * @author Emmanuel Bernard
37   */
38  public class OptimisticLockAnnotationTest extends BaseCoreFunctionalTestCase {
39  	@Test
40  	public void testOptimisticLockExcludeOnNameProperty() throws Exception {
41  		Conductor c = new Conductor();
42  		c.setName( "Bob" );
43  		Session s = openSession();
44  		s.getTransaction().begin();
45  		s.persist( c );
46  		s.flush();
47  
48  		s.clear();
49  
50  		c = ( Conductor ) s.get( Conductor.class, c.getId() );
51  		Long version = c.getVersion();
52  		c.setName( "Don" );
53  		s.flush();
54  
55  		s.clear();
56  
57  		c = ( Conductor ) s.get( Conductor.class, c.getId() );
58  		assertEquals( version, c.getVersion() );
59  
60  		s.getTransaction().rollback();
61  		s.close();
62  	}
63  
64  	@Override
65  	protected Class[] getAnnotatedClasses() {
66  		return new Class[] {
67  				Conductor.class
68  		};
69  	}
70  }
