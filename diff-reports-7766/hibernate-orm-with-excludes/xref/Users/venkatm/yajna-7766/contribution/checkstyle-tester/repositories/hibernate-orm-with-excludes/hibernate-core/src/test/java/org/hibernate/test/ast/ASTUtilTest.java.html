24  package org.hibernate.test.ast;
25  
26  import antlr.ASTFactory;
27  import antlr.collections.AST;
28  import org.junit.Test;
29  
30  import org.hibernate.hql.internal.ast.util.ASTUtil;
31  import org.hibernate.testing.junit4.BaseUnitTestCase;
32  
33  import static org.junit.Assert.assertEquals;
34  import static org.junit.Assert.assertNotNull;
35  import static org.junit.Assert.assertNull;
36  import static org.junit.Assert.assertSame;
37  
38  /**
39   * Unit test for ASTUtil.
40   */
41  public class ASTUtilTest extends BaseUnitTestCase {
42  	private ASTFactory factory = new ASTFactory();
43  
44  	@Test
45  	public void testCreate() throws Exception {
46  		AST n = ASTUtil.create( factory, 1, "one");
47  		assertNull( n.getFirstChild() );
48  		assertEquals( "one", n.getText() );
49  		assertEquals( 1, n.getType() );
50  	}
51  
52  	@Test
53  	public void testCreateTree() throws Exception {
54  		AST[] tree = new AST[4];
55  		AST grandparent = tree[0] = ASTUtil.create(factory, 1, "grandparent");
56  		AST parent = tree[1] = ASTUtil.create(factory,2,"parent");
57  		AST child = tree[2] = ASTUtil.create(factory,3,"child");
58  		AST baby = tree[3] = ASTUtil.create(factory,4,"baby");
59  		AST t = ASTUtil.createTree( factory, tree);
60  		assertSame(t,grandparent);
61  		assertSame(parent,t.getFirstChild());
62  		assertSame(child,t.getFirstChild().getFirstChild());
63  		assertSame( baby, t.getFirstChild().getFirstChild().getFirstChild() );
64  	}
65  
66  	@Test
67  	public void testFindPreviousSibling() throws Exception {
68  		AST child1 = ASTUtil.create(factory,2, "child1");
69  		AST child2 = ASTUtil.create(factory,3, "child2");
70  		AST n = factory.make( new AST[] {
71  			ASTUtil.create(factory, 1, "parent"),
72  			child1,
73  			child2,
74  		});
75  		assertSame(child1,ASTUtil.findPreviousSibling( n,child2));
76  		Exception e = null;
77  		try {
78  			ASTUtil.findPreviousSibling(child1,null);
79  		}
80  		catch (Exception x) {
81  			e = x;
82  		}
83  		assertNotNull(e);
84  	}
85  }
