24  package org.hibernate.test.annotations.id.sequences;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.Transaction;
30  import org.hibernate.test.annotations.id.sequences.entities.Location;
31  import org.hibernate.test.annotations.id.sequences.entities.Tower;
32  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
33  
34  import static org.junit.Assert.assertNotNull;
35  
36  /**
37   * @author Emmanuel Bernard
38   */
39  @SuppressWarnings("unchecked")
40  public class IdClassTest extends BaseCoreFunctionalTestCase {
41  	@Test
42  	public void testIdClassInSuperclass() throws Exception {
43  		Tower tower = new Tower();
44  		tower.latitude = 10.3;
45  		tower.longitude = 45.4;
46  		Session s = openSession();
47  		Transaction tx = s.beginTransaction();
48  		s.persist( tower );
49  		s.flush();
50  		s.clear();
51  		Location loc = new Location();
52  		loc.latitude = tower.latitude;
53  		loc.longitude = tower.longitude;
54  		assertNotNull( s.get( Tower.class, loc ) );
55  		tx.rollback();
56  		s.close();
57  	}
58  
59  	@Override
60  	protected Class[] getAnnotatedClasses() {
61  		return new Class[] { Tower.class };
62  	}
63  }
