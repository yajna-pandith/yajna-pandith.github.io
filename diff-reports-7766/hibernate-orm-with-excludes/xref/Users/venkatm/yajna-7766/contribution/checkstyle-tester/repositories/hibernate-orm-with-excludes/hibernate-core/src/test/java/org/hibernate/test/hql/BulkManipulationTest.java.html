359 	protected boolean supportsBulkInsertIdGeneration(Class entityClass) {
360 		EntityPersister persister = sessionFactory().getEntityPersister( entityClass.getName() );
361 		IdentifierGenerator generator = persister.getIdentifierGenerator();
362 		return BulkInsertionCapableIdentifierGenerator.class.isInstance( generator )
363 				&& BulkInsertionCapableIdentifierGenerator.class.cast( generator ).supportsBulkInsertionIdentifierGeneration();
364 	}
