3   
4   import org.hibernate.Hibernate;

12   * @author Steve Ebersole

14  public class TestFetchAllExecutable extends AbstractExecutable {
15  	public void execute() {
16  		Session s = getFactory().openSession();
17  		Transaction t = s.beginTransaction();
18  		Owner o = new Owner();
19  		Document doc = new Document();
20  		Folder fol = new Folder();
21  		o.setName("gavin");
22  		doc.setName("Hibernate in Action");
23  		doc.setSummary("blah");
24  		doc.updateText("blah blah");
25  		fol.setName("books");
26  		doc.setOwner(o);
27  		doc.setFolder(fol);
28  		fol.getDocuments().add(doc);
29  		s.persist(o);
30  		s.persist(fol);
31  		t.commit();
32  		s.close();
33  
34  		s = getFactory().openSession();
35  		t = s.beginTransaction();
36  		doc = (Document) s.createQuery("from Document fetch all properties").uniqueResult();
37  		Assert.assertTrue( Hibernate.isPropertyInitialized( doc, "summary" ) );
38  		Assert.assertTrue( Hibernate.isPropertyInitialized( doc, "upperCaseName" ) );
39  		Assert.assertTrue( Hibernate.isPropertyInitialized( doc, "owner" ) );
40  		Assert.assertEquals( doc.getSummary(), "blah" );
41  		s.delete(doc);
42  		s.delete( doc.getOwner() );
43  		s.delete( doc.getFolder() );
44  		t.commit();
45  		s.close();
46  	}
47  }
