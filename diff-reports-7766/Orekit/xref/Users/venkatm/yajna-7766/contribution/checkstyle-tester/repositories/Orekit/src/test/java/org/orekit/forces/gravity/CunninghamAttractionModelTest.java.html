109     private static class SpotStepHandler implements OrekitFixedStepHandler {
110 
111         public SpotStepHandler(AbsoluteDate date, double mu) throws OrekitException {
112             sun       = CelestialBodyFactory.getSun();
113             previous  = Double.NaN;
114         }
115 
116         private PVCoordinatesProvider sun;
117         private double previous;
118         public void init(SpacecraftState s0, AbsoluteDate t) {
119         }
120         public void handleStep(SpacecraftState currentState, boolean isLast)
121             throws OrekitException {
122 
123 
124             AbsoluteDate current = currentState.getDate();
125             Vector3D sunPos = sun.getPVCoordinates(current , FramesFactory.getEME2000()).getPosition();
126             Vector3D normal = currentState.getPVCoordinates().getMomentum();
127             double angle = Vector3D.angle(sunPos , normal);
128             if (! Double.isNaN(previous)) {
129                 Assert.assertEquals(previous, angle, 0.0013);
130             }
131             previous = angle;
132         }

175     private static class EckStepHandler implements OrekitFixedStepHandler {
176 
177         /** Body mu */
178         private static final double mu =  3.986004415e+14;
179 
180         private EckStepHandler(Orbit initialOrbit, double ae,
181                                double c20, double c30, double c40, double c50, double c60)
182         throws FileNotFoundException, OrekitException {
183             referencePropagator =
184                 new EcksteinHechlerPropagator(initialOrbit,
185                                               ae, mu, c20, c30, c40, c50, c60);
186         }
187 
188         private EcksteinHechlerPropagator referencePropagator;
189         public void handleStep(SpacecraftState currentState, boolean isLast) throws OrekitException {
190             SpacecraftState EHPOrbit   = referencePropagator.propagate(currentState.getDate());
191             Vector3D posEHP  = EHPOrbit.getPVCoordinates().getPosition();
192             Vector3D posDROZ = currentState.getPVCoordinates().getPosition();
193             Vector3D velEHP  = EHPOrbit.getPVCoordinates().getVelocity();
194             Vector3D dif     = posEHP.subtract(posDROZ);
195 
196             Vector3D T = new Vector3D(1 / velEHP.getNorm(), velEHP);
197             Vector3D W = EHPOrbit.getPVCoordinates().getMomentum().normalize();
198             Vector3D N = Vector3D.crossProduct(W, T);
199 
200             Assert.assertTrue(dif.getNorm() < 111);
201             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(dif, T)) < 111);
202             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(dif, N)) <  54);
203             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(dif, W)) <  12);
204         }
205 
206     }

262 
263         Utils.setDataRoot("regular-data:potential/grgs-format");

266         // pos-vel (from a ZOOM ephemeris reference)

268         final Vector3D vel = new Vector3D(2.14718074509906819e+03, 7.38239351251748485e+03, -1.14097953925384523e+01);
269         final SpacecraftState spacecraftState =
270                 new SpacecraftState(new CartesianOrbit(new PVCoordinates(pos, vel),
271                                                        FramesFactory.getGCRF(),
272                                                        new AbsoluteDate(2005, 3, 5, 0, 24, 0.0, TimeScalesFactory.getTAI()),
273                                                        GravityFieldFactory.getUnnormalizedProvider(1, 1).getMu()));
274 
275         AccelerationRetriever accelerationRetriever = new AccelerationRetriever();
276         for (int i = 2; i <= 69; i++) {
277             // perturbing force (ITRF2008 central body frame)
278             final ForceModel cunModel =
279                     new CunninghamAttractionModel(FramesFactory.getITRF(IERSConventions.IERS_2010, true),
280                                                   GravityFieldFactory.getUnnormalizedProvider(i, i), 1.0);
281             final ForceModel droModel =
282                     new DrozinerAttractionModel(FramesFactory.getITRF(IERSConventions.IERS_2010, true),
283                                                 GravityFieldFactory.getUnnormalizedProvider(i, i), 1.0);
284 
285             /**
286              * Compute acceleration
287              */
288             cunModel.addContribution(spacecraftState, accelerationRetriever);
289             final Vector3D cunGamma = accelerationRetriever.getAcceleration();
290             droModel.addContribution(spacecraftState, accelerationRetriever);
291             final Vector3D droGamma = accelerationRetriever.getAcceleration();

293

298     @Test
299     public void testTimeDependentField() throws IOException, ParseException, OrekitException {
300 
301         Utils.setDataRoot("regular-data:potential/icgem-format");
302         GravityFieldFactory.addPotentialCoefficientsReader(new ICGEMFormatReader("eigen-6s-truncated", true));
303 
304         final Vector3D pos = new Vector3D(6.46885878304673824e+06, -1.88050918456274318e+06, -1.32931592294715829e+04);
305         final Vector3D vel = new Vector3D(2.14718074509906819e+03, 7.38239351251748485e+03, -1.14097953925384523e+01);
306         final SpacecraftState spacecraftState =
307                 new SpacecraftState(new CartesianOrbit(new PVCoordinates(pos, vel),
308                                                        FramesFactory.getGCRF(),
309                                                        new AbsoluteDate(2005, 3, 5, 0, 24, 0.0, TimeScalesFactory.getTAI()),
310                                                        GravityFieldFactory.getUnnormalizedProvider(1, 1).getMu()));
311 
312         double dP = 0.1;
313         double duration = 3 * Constants.JULIAN_DAY;
314         BoundedPropagator fixedFieldEphemeris   = createEphemeris(dP, spacecraftState, duration,
315                                                                   GravityFieldFactory.getConstantUnnormalizedProvider(8, 8));
316         BoundedPropagator varyingFieldEphemeris = createEphemeris(dP, spacecraftState, duration,
317                                                                   GravityFieldFactory.getUnnormalizedProvider(8, 8));
318 
319         double step = 60.0;
320         double maxDeltaT = 0;
321         double maxDeltaN = 0;
322         double maxDeltaW = 0;
323         for (AbsoluteDate date = fixedFieldEphemeris.getMinDate();
324              date.compareTo(fixedFieldEphemeris.getMaxDate()) < 0;
325              date = date.shiftedBy(step)) {
326             PVCoordinates pvFixedField   = fixedFieldEphemeris.getPVCoordinates(date, FramesFactory.getGCRF());
327             PVCoordinates pvVaryingField = varyingFieldEphemeris.getPVCoordinates(date, FramesFactory.getGCRF());
328             Vector3D t = pvFixedField.getVelocity().normalize();
329             Vector3D w = pvFixedField.getMomentum().normalize();
330             Vector3D n = Vector3D.crossProduct(w, t);
331             Vector3D delta = pvVaryingField.getPosition().subtract(pvFixedField.getPosition());
332             maxDeltaT = FastMath.max(maxDeltaT, FastMath.abs(Vector3D.dotProduct(delta, t)));
333             maxDeltaN = FastMath.max(maxDeltaN, FastMath.abs(Vector3D.dotProduct(delta, n)));
334             maxDeltaW = FastMath.max(maxDeltaW, FastMath.abs(Vector3D.dotProduct(delta, w)));
335         }
336         Assert.assertTrue(maxDeltaT > 0.15);
337         Assert.assertTrue(maxDeltaT < 0.25);
338         Assert.assertTrue(maxDeltaN > 0.01);
339         Assert.assertTrue(maxDeltaN < 0.02);
340         Assert.assertTrue(maxDeltaW > 0.05);
341         Assert.assertTrue(maxDeltaW < 0.10);
342 
343     }

345     private BoundedPropagator createEphemeris(double dP, SpacecraftState initialState, double duration,
346                                               UnnormalizedSphericalHarmonicsProvider provider)
347         throws OrekitException {
348         double[][] tol = NumericalPropagator.tolerances(dP, initialState.getOrbit(), OrbitType.CARTESIAN);
349         AbstractIntegrator integrator =
350                 new DormandPrince853Integrator(0.001, 120.0, tol[0], tol[1]);
351         NumericalPropagator propagator = new NumericalPropagator(integrator);
352         propagator.setEphemerisMode();
353         propagator.setOrbitType(OrbitType.CARTESIAN);
354         propagator.addForceModel(new CunninghamAttractionModel(FramesFactory.getITRF(IERSConventions.IERS_2010, true), provider, 1.0));
355         propagator.setInitialState(initialState);
356         propagator.propagate(initialState.getDate().shiftedBy(duration));
357         return propagator.getGeneratedEphemeris();
358     }
