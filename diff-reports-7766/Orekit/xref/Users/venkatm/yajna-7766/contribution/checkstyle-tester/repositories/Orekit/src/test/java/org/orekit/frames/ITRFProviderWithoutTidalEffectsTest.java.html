39      @Test
40      public void testRoughRotation() throws OrekitException {
41  
42          AbsoluteDate date1 = new AbsoluteDate(new DateComponents(2006, 02, 24),
43                                                new TimeComponents(15, 38, 00),
44                                                TimeScalesFactory.getUTC());
45          Frame itrf2008 = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
46          Transform t0 = itrf2008.getTransformTo(FramesFactory.getEME2000(), date1);
47  
48          double dt = 10.0;
49          AbsoluteDate date2 = date1.shiftedBy(dt);
50          Transform t1 = itrf2008.getTransformTo(FramesFactory.getEME2000(), date2);
51          Transform evolution = new Transform(date2, t0.getInverse(), t1);
52  
53          Vector3D p = new Vector3D(6000,6000,0);
54          Assert.assertEquals(0.0, evolution.transformPosition(Vector3D.ZERO).getNorm(), 1.0e-15);
55          Assert.assertEquals(0.0, evolution.transformVector(p).getZ(), 0.003);
56          Assert.assertEquals(2 * FastMath.PI * dt / 86164,
57                              Vector3D.angle(t0.transformVector(p), t1.transformVector(p)),
58                              1.0e-9);
59  
60      }

62      @Test
63      public void testRoughOrientation() throws OrekitException {
64  
65          AbsoluteDate date = new AbsoluteDate(2001, 03, 21, 0, 4, 0, TimeScalesFactory.getUTC());
66          Frame itrf2008 = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
67  
68          Vector3D u = itrf2008.getTransformTo(FramesFactory.getEME2000(), date).transformVector(Vector3D.PLUS_I);
69          Assert.assertTrue(Vector3D.angle(u, Vector3D.MINUS_I) < FastMath.toRadians(0.5));
70  
71          date = date.shiftedBy(6 * 3600);
72          u = itrf2008.getTransformTo(FramesFactory.getEME2000(), date).transformVector(Vector3D.PLUS_I);
73          Assert.assertTrue(Vector3D.angle(u, Vector3D.MINUS_J) < FastMath.toRadians(0.5));
74  
75          date = date.shiftedBy(6 * 3600);
76          u = itrf2008.getTransformTo(FramesFactory.getEME2000(), date).transformVector(Vector3D.PLUS_I);
77          Assert.assertTrue(Vector3D.angle(u, Vector3D.PLUS_I) < FastMath.toRadians(0.5));
78  
79          date = date.shiftedBy(6 * 3600);
80          u = itrf2008.getTransformTo(FramesFactory.getEME2000(), date).transformVector(Vector3D.PLUS_I);
81          Assert.assertTrue(Vector3D.angle(u, Vector3D.PLUS_J) < FastMath.toRadians(0.5));
82  
83      }

121     @Test
122     public void testMSLIBTransformJ2000_TerRef() throws OrekitException {
123 
124         AbsoluteDate t0 = new AbsoluteDate(new DateComponents(2003, 10, 14),
125                                            new TimeComponents(02, 00, 00),
126                                            TimeScalesFactory.getUTC());
127         Frame itrf = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
128         Transform trans = FramesFactory.getEME2000().getTransformTo(itrf, t0);
129 
130         // Coordinates in EME2000
131         PVCoordinates pvEME2000 =
132             new PVCoordinates(new Vector3D(6500000.0, -1234567.0, 4000000.0),
133                               new Vector3D(3609.28229, 3322.88979, -7083.950661));
134 
135         // Reference coordinates in ITRF
136         PVCoordinates pvRef =
137             new PVCoordinates(new Vector3D(3011113.971820046, -5889827.854375269, 4002158.938875904),
138                               new Vector3D(4410.393506651586, -1033.61784235127, -7082.633883124906));
139 
140 
141         // tests using direct transform
142         checkPV(pvRef, trans.transformPVCoordinates(pvEME2000), 0.594, 1.79e-4);
143 
144         // compute local evolution using finite differences
145         double h = 1.0;
146         Rotation r0 = trans.getRotation();
147         AbsoluteDate date = t0.shiftedBy(-2 * h);
148         Rotation evoM2h = FramesFactory.getEME2000().getTransformTo(itrf, date).getRotation().compose(r0.revert(),
149                                                                                                       RotationConvention.VECTOR_OPERATOR);
150         double alphaM2h = -evoM2h.getAngle();
151         Vector3D axisM2h = evoM2h.getAxis(RotationConvention.VECTOR_OPERATOR);
152         date = t0.shiftedBy(-h);
153         Rotation evoM1h = FramesFactory.getEME2000().getTransformTo(itrf, date).getRotation().compose(r0.revert(),
154                                                                                                       RotationConvention.VECTOR_OPERATOR);
155         double alphaM1h = -evoM1h.getAngle();
156         Vector3D axisM1h = evoM1h.getAxis(RotationConvention.VECTOR_OPERATOR);
157         date = t0.shiftedBy(h);
158         Rotation evoP1h = FramesFactory.getEME2000().getTransformTo(itrf, date).getRotation().compose(r0.revert(),
159                                                                                                       RotationConvention.VECTOR_OPERATOR);
160         double alphaP1h =  evoP1h.getAngle();
161         Vector3D axisP1h = evoP1h.getAxis(RotationConvention.VECTOR_OPERATOR).negate();
162         date = t0.shiftedBy(2 * h);
163         Rotation evoP2h = FramesFactory.getEME2000().getTransformTo(itrf, date).getRotation().compose(r0.revert(),
164                                                                                                       RotationConvention.VECTOR_OPERATOR);
165         double alphaP2h =  evoP2h.getAngle();
166         Vector3D axisP2h = evoP2h.getAxis(RotationConvention.VECTOR_OPERATOR).negate();
167         double w = (8 * (alphaP1h - alphaM1h) - (alphaP2h - alphaM2h)) / (12 * h);
168         Vector3D axis = axisM2h.add(axisM1h).add(axisP1h.add(axisP2h)).normalize();
169         Transform finiteDiffTransform = new Transform(t0, trans.getRotation() , new Vector3D(w ,axis));
170 
171         checkPV(pvRef, finiteDiffTransform.transformPVCoordinates(pvEME2000), 0.594, 1.78e-4);
172 
173     }

189     private void checkPV(PVCoordinates reference, PVCoordinates result,
190                          double expectedPositionError, double expectedVelocityError) {
191 
192         Vector3D dP = result.getPosition().subtract(reference.getPosition());
193         Vector3D dV = result.getVelocity().subtract(reference.getVelocity());
194         Assert.assertEquals(expectedPositionError, dP.getNorm(), 0.01 * expectedPositionError);
195         Assert.assertEquals(expectedVelocityError, dV.getNorm(), 0.01 * expectedVelocityError);
196     }
