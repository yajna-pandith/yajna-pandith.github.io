109     @Test
110     public void testCircularToCartesian() {
111 
112         double ix = 1.200e-04;
113         double iy = -1.16e-04;
114         double i  = 2 * FastMath.asin(FastMath.sqrt((ix * ix + iy * iy) / 4));
115         double raan = FastMath.atan2(iy, ix);
116         double cosRaan = FastMath.cos(raan);
117         double sinRaan = FastMath.sin(raan);
118         double exTilde = -7.900e-6;
119         double eyTilde = 1.100e-4;
120         double ex = exTilde * cosRaan + eyTilde * sinRaan;
121         double ey = eyTilde * cosRaan - exTilde * sinRaan;
122 
123         CircularOrbit circ=
124             new CircularOrbit(42166.712, ex, ey, i, raan,
125                                    5.300 - raan, PositionAngle.MEAN,
126                                    FramesFactory.getEME2000(), date, mu);
127         Vector3D pos = circ.getPVCoordinates().getPosition();
128         Vector3D vel = circ.getPVCoordinates().getVelocity();
129 
130         // check 1/a = 2/r  - V2/mu
131         double r = pos.getNorm();
132         double v = vel.getNorm();
133         Assert.assertEquals(2 / r - v * v / mu, 1 / circ.getA(), 1.0e-7);
134 
135         Assert.assertEquals( 0.233745668678733e+05, pos.getX(), Utils.epsilonTest * r);
136         Assert.assertEquals(-0.350998914352669e+05, pos.getY(), Utils.epsilonTest * r);
137         Assert.assertEquals(-0.150053723123334e+01, pos.getZ(), Utils.epsilonTest * r);
138 
139         Assert.assertEquals(0.809135038364960e+05, vel.getX(), Utils.epsilonTest * v);
140         Assert.assertEquals(0.538902268252598e+05, vel.getY(), Utils.epsilonTest * v);
141         Assert.assertEquals(0.158527938296630e+02, vel.getZ(), Utils.epsilonTest * v);

188     public void testAnomalyEll() {
189 
190         // elliptic orbit
191         Vector3D position = new Vector3D(7.0e6, 1.0e6, 4.0e6);
192         Vector3D velocity = new Vector3D(-500.0, 8000.0, 1000.0);
193 
194         PVCoordinates pvCoordinates = new PVCoordinates( position, velocity);
195 
196         CircularOrbit  p   = new CircularOrbit(pvCoordinates, FramesFactory.getEME2000(), date, mu);
197         KeplerianOrbit kep = new KeplerianOrbit(p);
198 
199         double e       = p.getE();
200         double eRatio  = FastMath.sqrt((1 - e) / (1 + e));
201         double raan    = kep.getRightAscensionOfAscendingNode();
202         double paPraan = kep.getPerigeeArgument() + raan;
203 
204         double lv = 1.1;
205         // formulations for elliptic case
206         double lE = 2 * FastMath.atan(eRatio * FastMath.tan((lv - paPraan) / 2)) + paPraan;
207         double lM = lE - e * FastMath.sin(lE - paPraan);
208 
209         p = new CircularOrbit(p.getA() , p.getCircularEx(), p.getCircularEy(),
210                                    p.getRightAscensionOfAscendingNode(),
211                                    p.getAlphaV(), lv - raan, PositionAngle.TRUE, p.getFrame(), date, mu);
212         Assert.assertEquals(p.getAlphaV() + raan, lv, Utils.epsilonAngle * FastMath.abs(lv));
213         Assert.assertEquals(p.getAlphaE() + raan, lE, Utils.epsilonAngle * FastMath.abs(lE));
214         Assert.assertEquals(p.getAlphaM() + raan, lM, Utils.epsilonAngle * FastMath.abs(lM));
215         p = new CircularOrbit(p.getA() , p.getCircularEx(), p.getCircularEy(),
216                                    p.getRightAscensionOfAscendingNode(),
217                                    p.getAlphaV(), 0, PositionAngle.TRUE, p.getFrame(), date, mu);
218 
219 
220         p = new CircularOrbit(p.getA() , p.getCircularEx(), p.getCircularEy(),
221                                    p.getRightAscensionOfAscendingNode(),
222                                    p.getAlphaV(), lE - raan, PositionAngle.ECCENTRIC, p.getFrame(), date, mu);
223         Assert.assertEquals(p.getAlphaV() + raan, lv, Utils.epsilonAngle * FastMath.abs(lv));
224         Assert.assertEquals(p.getAlphaE() + raan, lE, Utils.epsilonAngle * FastMath.abs(lE));
225         Assert.assertEquals(p.getAlphaM() + raan, lM, Utils.epsilonAngle * FastMath.abs(lM));
226         p = new CircularOrbit(p.getA() , p.getCircularEx(), p.getCircularEy(),
227                                    p.getRightAscensionOfAscendingNode(),
228                                    p.getAlphaV(), 0, PositionAngle.TRUE, p.getFrame(), date, mu);

231                                    p.getRightAscensionOfAscendingNode(),
232                                    p.getAlphaV(), lM - raan, PositionAngle.MEAN, p.getFrame(), date, mu);
233         Assert.assertEquals(p.getAlphaV() + raan, lv, Utils.epsilonAngle * FastMath.abs(lv));

286     @Test
287     public void testPositionVelocityNormsEll() {
288 
289         // elliptic and non equatorial (i retrograde) orbit
290         double hx =  1.2;
291         double hy =  2.1;
292         double i  = 2 * FastMath.atan(FastMath.sqrt(hx * hx + hy * hy));
293         double raan = FastMath.atan2(hy, hx);
294         CircularOrbit p =
295             new CircularOrbit(42166.712, 0.5, -0.5, i, raan,
296                                    0.67 - raan, PositionAngle.TRUE,
297                                    FramesFactory.getEME2000(), date, mu);
298 
299         double ex = p.getEquinoctialEx();
300         double ey = p.getEquinoctialEy();
301         double lv = p.getLv();
302         double ksi     = 1 + ex * FastMath.cos(lv) + ey * FastMath.sin(lv);
303         double nu      = ex * FastMath.sin(lv) - ey * FastMath.cos(lv);
304         double epsilon = FastMath.sqrt(1 - ex * ex - ey * ey);
305 
306         double a  = p.getA();
307         double na = FastMath.sqrt(mu / a);
308 
309         Assert.assertEquals(a * epsilon * epsilon / ksi,
310                      p.getPVCoordinates().getPosition().getNorm(),
311                      Utils.epsilonTest * FastMath.abs(p.getPVCoordinates().getPosition().getNorm()));
312         Assert.assertEquals(na * FastMath.sqrt(ksi * ksi + nu * nu) / epsilon,
313                      p.getPVCoordinates().getVelocity().getNorm(),
314                      Utils.epsilonTest * FastMath.abs(p.getPVCoordinates().getVelocity().getNorm()));
315 
316     }

376 
377         // elliptic and non equatorial (i retrograde) orbit
378         double hx =  1.2;
379         double hy =  2.1;
380         double i  = 2 * FastMath.atan(FastMath.sqrt(hx * hx + hy * hy));
381         double raan = FastMath.atan2(hy, hx);
382         CircularOrbit p =
383             new CircularOrbit(42166.712, 0.5, -0.5, i, raan,
384                                    0.67 - raan, PositionAngle.TRUE,
385                                    FramesFactory.getEME2000(), date, mu);
386 
387         Vector3D position = p.getPVCoordinates().getPosition();
388         Vector3D velocity = p.getPVCoordinates().getVelocity();
389         Vector3D momentum = p.getPVCoordinates().getMomentum().normalize();
390 
391         double apogeeRadius  = p.getA() * (1 + p.getE());
392         double perigeeRadius = p.getA() * (1 - p.getE());
393 
394         for (double alphaV = 0; alphaV <= 2 * FastMath.PI; alphaV += 2 * FastMath.PI/100.) {
395             p = new CircularOrbit(p.getA() , p.getCircularEx(), p.getCircularEy(), p.getI(),
396                                        p.getRightAscensionOfAscendingNode(),
397                                        alphaV, PositionAngle.TRUE, p.getFrame(), date, mu);
398             position = p.getPVCoordinates().getPosition();
399             // test if the norm of the position is in the range [perigee radius, apogee radius]
400             // Warning: these tests are without absolute value by choice
401             Assert.assertTrue((position.getNorm() - apogeeRadius)  <= (  apogeeRadius * Utils.epsilonTest));
402             Assert.assertTrue((position.getNorm() - perigeeRadius) >= (- perigeeRadius * Utils.epsilonTest));
403 
404             position= position.normalize();
405             velocity = p.getPVCoordinates().getVelocity();
406             velocity= velocity.normalize();
407 
408             // at this stage of computation, all the vectors (position, velocity and momemtum) are normalized here
409 
410             // test of orthogonality between position and momentum
411             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(position, momentum)) < Utils.epsilonTest);
412             // test of orthogonality between velocity and momentum
413             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(velocity, momentum)) < Utils.epsilonTest);
414         }

619     private double[][] finiteDifferencesJacobian(PositionAngle type, CircularOrbit orbit, double hP)
620         throws OrekitException {
621         double[][] jacobian = new double[6][6];
622         for (int i = 0; i < 6; ++i) {
623             fillColumn(type, i, orbit, hP, jacobian);
624         }

628     private void fillColumn(PositionAngle type, int i, CircularOrbit orbit, double hP, double[][] jacobian) {
629 
630         // at constant energy (i.e. constant semi major axis), we have dV = -mu dP / (V * r^2)
631         // we use this to compute a velocity step size from the position step size
632         Vector3D p = orbit.getPVCoordinates().getPosition();
633         Vector3D v = orbit.getPVCoordinates().getVelocity();
634         double hV = orbit.getMu() * hP / (v.getNorm() * p.getNormSq());
635 
636         double h;
637         Vector3D dP = Vector3D.ZERO;
638         Vector3D dV = Vector3D.ZERO;
639         switch (i) {
640         case 0:
641             h = hP;
642             dP = new Vector3D(hP, 0, 0);
643             break;
644         case 1:
645             h = hP;
646             dP = new Vector3D(0, hP, 0);
647             break;
648         case 2:
649             h = hP;
650             dP = new Vector3D(0, 0, hP);
651             break;
652         case 3:
653             h = hV;
654             dV = new Vector3D(hV, 0, 0);
655             break;
656         case 4:
657             h = hV;
658             dV = new Vector3D(0, hV, 0);
659             break;
660         default:
661             h = hV;
662             dV = new Vector3D(0, 0, hV);
663             break;
664         }
665 
666         CircularOrbit oM4h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, -4, dP), new Vector3D(1, v, -4, dV)),
667                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
668         CircularOrbit oM3h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, -3, dP), new Vector3D(1, v, -3, dV)),
669                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
670         CircularOrbit oM2h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, -2, dP), new Vector3D(1, v, -2, dV)),
671                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
672         CircularOrbit oM1h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, -1, dP), new Vector3D(1, v, -1, dV)),
673                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
674         CircularOrbit oP1h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, +1, dP), new Vector3D(1, v, +1, dV)),
675                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
676         CircularOrbit oP2h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, +2, dP), new Vector3D(1, v, +2, dV)),
677                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
678         CircularOrbit oP3h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, +3, dP), new Vector3D(1, v, +3, dV)),
679                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
680         CircularOrbit oP4h = new CircularOrbit(new PVCoordinates(new Vector3D(1, p, +4, dP), new Vector3D(1, v, +4, dV)),
681                                                orbit.getFrame(), orbit.getDate(), orbit.getMu());
682 
683         jacobian[0][i] = (-3 * (oP4h.getA()                             - oM4h.getA()) +
684                           32 * (oP3h.getA()                             - oM3h.getA()) -
685                          168 * (oP2h.getA()                             - oM2h.getA()) +
686                          672 * (oP1h.getA()                             - oM1h.getA())) / (840 * h);
687         jacobian[1][i] = (-3 * (oP4h.getCircularEx()                    - oM4h.getCircularEx()) +
688                           32 * (oP3h.getCircularEx()                    - oM3h.getCircularEx()) -
689                          168 * (oP2h.getCircularEx()                    - oM2h.getCircularEx()) +
690                          672 * (oP1h.getCircularEx()                    - oM1h.getCircularEx())) / (840 * h);
691         jacobian[2][i] = (-3 * (oP4h.getCircularEy()                    - oM4h.getCircularEy()) +
692                           32 * (oP3h.getCircularEy()                    - oM3h.getCircularEy()) -
693                          168 * (oP2h.getCircularEy()                    - oM2h.getCircularEy()) +
694                          672 * (oP1h.getCircularEy()                    - oM1h.getCircularEy())) / (840 * h);
695         jacobian[3][i] = (-3 * (oP4h.getI()                             - oM4h.getI()) +
696                           32 * (oP3h.getI()                             - oM3h.getI()) -
697                          168 * (oP2h.getI()                             - oM2h.getI()) +
698                          672 * (oP1h.getI()                             - oM1h.getI())) / (840 * h);
699         jacobian[4][i] = (-3 * (oP4h.getRightAscensionOfAscendingNode() - oM4h.getRightAscensionOfAscendingNode()) +
700                           32 * (oP3h.getRightAscensionOfAscendingNode() - oM3h.getRightAscensionOfAscendingNode()) -
701                          168 * (oP2h.getRightAscensionOfAscendingNode() - oM2h.getRightAscensionOfAscendingNode()) +
702                          672 * (oP1h.getRightAscensionOfAscendingNode() - oM1h.getRightAscensionOfAscendingNode())) / (840 * h);
703         jacobian[5][i] = (-3 * (oP4h.getAlpha(type)                     - oM4h.getAlpha(type)) +
704                           32 * (oP3h.getAlpha(type)                     - oM3h.getAlpha(type)) -
705                          168 * (oP2h.getAlpha(type)                     - oM2h.getAlpha(type)) +
706                          672 * (oP1h.getAlpha(type)                     - oM1h.getAlpha(type))) / (840 * h);
707 
708     }

710     @Test
711     public void testInterpolation() throws OrekitException {
712 
713         final double ehMu  = 3.9860047e14;
714         final double ae  = 6.378137e6;
715         final double c20 = -1.08263e-3;
716         final double c30 = 2.54e-6;
717         final double c40 = 1.62e-6;
718         final double c50 = 2.3e-7;
719         final double c60 = -5.5e-7;
720 
721         final AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(584.);
722         final Vector3D position = new Vector3D(3220103., 69623., 6449822.);
723         final Vector3D velocity = new Vector3D(6414.7, -2006., -3180.);
724         final CircularOrbit initialOrbit = new CircularOrbit(new PVCoordinates(position, velocity),
725                                                              FramesFactory.getEME2000(), date, ehMu);
726 
727         EcksteinHechlerPropagator propagator =
728                 new EcksteinHechlerPropagator(initialOrbit, ae, ehMu, c20, c30, c40, c50, c60);
729 
730         // set up a 5 points sample
731         List<Orbit> sample = new ArrayList<Orbit>();
732         for (double dt = 0; dt < 300.0; dt += 60.0) {
733             sample.add(propagator.propagate(date.shiftedBy(dt)).getOrbit());
734         }
735 
736         // well inside the sample, interpolation should be much better than Keplerian shift
737         double maxShiftError = 0;
738         double maxInterpolationError = 0;
739         for (double dt = 0; dt < 241.0; dt += 1.0) {
740             AbsoluteDate t        = initialOrbit.getDate().shiftedBy(dt);
741             Vector3D shifted      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
742             Vector3D interpolated = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
743             Vector3D propagated   = propagator.propagate(t).getPVCoordinates().getPosition();
744             maxShiftError = FastMath.max(maxShiftError, shifted.subtract(propagated).getNorm());
745             maxInterpolationError = FastMath.max(maxInterpolationError, interpolated.subtract(propagated).getNorm());
746         }
747         Assert.assertTrue(maxShiftError         > 390.0);
748         Assert.assertTrue(maxInterpolationError < 0.04);
749 
750         // slightly past sample end, interpolation should quickly increase, but remain reasonable
751         maxShiftError = 0;
752         maxInterpolationError = 0;
753         for (double dt = 240; dt < 300.0; dt += 1.0) {
754             AbsoluteDate t        = initialOrbit.getDate().shiftedBy(dt);
755             Vector3D shifted      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
756             Vector3D interpolated = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
757             Vector3D propagated   = propagator.propagate(t).getPVCoordinates().getPosition();
758             maxShiftError = FastMath.max(maxShiftError, shifted.subtract(propagated).getNorm());
759             maxInterpolationError = FastMath.max(maxInterpolationError, interpolated.subtract(propagated).getNorm());
760         }
761         Assert.assertTrue(maxShiftError         <  610.0);
762         Assert.assertTrue(maxInterpolationError <    1.3);
763 
764         // far past sample end, interpolation should become really wrong
765         // (in this test case, break even occurs at around 863 seconds, with a 3.9 km error)
766         maxShiftError = 0;
767         maxInterpolationError = 0;
768         for (double dt = 300; dt < 1000; dt += 1.0) {
769             AbsoluteDate t        = initialOrbit.getDate().shiftedBy(dt);
770             Vector3D shifted      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
771             Vector3D interpolated = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
772             Vector3D propagated   = propagator.propagate(t).getPVCoordinates().getPosition();
773             maxShiftError = FastMath.max(maxShiftError, shifted.subtract(propagated).getNorm());
774             maxInterpolationError = FastMath.max(maxInterpolationError, interpolated.subtract(propagated).getNorm());
775         }
776         Assert.assertTrue(maxShiftError         < 5000.0);
777         Assert.assertTrue(maxInterpolationError > 8800.0);
778 
779     }
