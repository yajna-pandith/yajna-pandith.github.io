73      @Test
74      public void testCloseEventsFirstOneIsReset() throws OrekitException {
75          // setup
76          // a fairly rare state to reproduce this bug. Two dates, d1 < d2, that
77          // are very close. Event triggers on d1 will reset state to break out of
78          // event handling loop in AbstractIntegrator.acceptStep(). At this point
79          // detector2 has g0Positive == true but the event time is set to just
80          // before the event so g(t0) is negative. Now on processing the
81          // next step the root solver checks the sign of the start, midpoint,
82          // and end of the interval so we need another event less than half a max
83          // check interval after d2 so that the g function will be negative at
84          // all three times. Then we get a non bracketing exception.
85          Propagator propagator = getPropagator(10.0);
86  
87          double t1 = 49, t2 = t1 + 1e-15, t3 = t1 + 4.9;
88          List<Event<EventDetector>> events = new ArrayList<>();
89          TimeDetector detector1 = new TimeDetector(t1)
90                  .withHandler(new Handler<>(events, Action.RESET_DERIVATIVES))
91                  .withMaxCheck(10)
92                  .withThreshold(1e-9);
93          propagator.addEventDetector(detector1);
94          TimeDetector detector2 = new TimeDetector(t2, t3)
95                  .withHandler(new RecordAndContinue<>(events))
96                  .withMaxCheck(11)
97                  .withThreshold(1e-9);
98          propagator.addEventDetector(detector2);
99  
100         // action
101         propagator.propagate(epoch.shiftedBy(60));
102 
103         // verify
104         Assert.assertEquals(1, events.size());
105         Assert.assertEquals(t1, events.get(0).getState().getDate().durationFrom(epoch), 0.0);
106         Assert.assertEquals(detector1, events.get(0).getDetector());
107     }

1722         private ContinuousDetector(double newMaxCheck,
1723                                    double newThreshold,
1724                                    int newMaxIter,
1725                                    EventHandler<? super ContinuousDetector> newHandler,
1726                                    List<AbsoluteDate> eventDates) {
1727             super(newMaxCheck, newThreshold, newMaxIter, newHandler);
1728             this.eventTs = eventDates;
1729         }
1730 
1731         @Override
1732         public double g(SpacecraftState s) throws OrekitException {
1733             final AbsoluteDate t = s.getDate();
1734             int i = 0;
1735             while (i < eventTs.size() && t.compareTo(eventTs.get(i)) > 0) {
1736                 i++;
1737             }
1738             i--;
1739             if (i < 0) {
1740                 return t.durationFrom(eventTs.get(0));
1741             } else if (i < eventTs.size() - 1) {
1742                 int sign = (i % 2) * 2 - 1;
1743                 return -sign * (t.durationFrom(eventTs.get(i)))
1744                         * (eventTs.get(i + 1).durationFrom(t));
1745             } else {
1746                 int sign = (i % 2) * 2 - 1;
1747                 return -sign * (t.durationFrom(eventTs.get(i)));
1748             }
1749         }

1753                 double newMaxCheck,
1754                 double newThreshold,
1755                 int newMaxIter,
1756                 EventHandler<? super ContinuousDetector> newHandler) {
1757             return new ContinuousDetector(
1758                     newMaxCheck, newThreshold, newMaxIter, newHandler, eventTs);
