66      @Test
67      public void testShiftError()
68          throws ParseException, OrekitException {
69  
70  
71          // polynomial models for interpolation error in position, velocity, acceleration and attitude
72          // these models grow as follows
73          //   interpolation time (s)    position error (m)   velocity error (m/s)   acceleration error (m/s²)  attitude error (°)
74          //           60                       30                    1                     0.014               0.00002
75          //          120                      100                    2                     0.013               0.00009
76          //          300                      600                    4                     0.011               0.0009
77          //          600                     2000                    6                     0.006               0.006
78          //          900                     4000                    6                     0.002               0.02
79          // the expected maximum residuals with respect to these models are about 0.2m, 0.8mm/s, 7.9μm/s² and 2.8e-7°
80          PolynomialFunction pModel = new PolynomialFunction(new double[] {
81              -0.16513714130703838,    0.008052836586593358,  0.007306374155052651,
82              -1.9942719771217313E-6, -1.8063354449344768E-9, 9.042229494006868E-13,
83          });
84          PolynomialFunction vModel = new PolynomialFunction(new double[] {
85              -6.205041765231733E-4,   0.014820651821078279,  -7.458097665912488E-6,
86              -1.5914983761563204E-9, -1.7936013181449342E-12, 2.187916095307246E-15,
87          });
88          PolynomialFunction aModel = new PolynomialFunction(new double[] {
89              0.014788789776214587,   -1.418490913753379E-5,  1.947898940800356E-9,
90              2.6179181901457335E-12, -7.603278343088821E-15, 2.968153861123117E-18,
91          });
92          PolynomialFunction rModel = new PolynomialFunction(new double[] {
93              -2.7689062182403017E-6, 1.7406542555507358E-7,  2.510979532481025E-9,
94               2.039932266627844E-11, 9.912634888010535E-15, -3.5015638902258456E-18,
95          });
96  
97          AbsoluteDate centerDate = orbit.getDate().shiftedBy(100.0);
98          SpacecraftState centerState = propagator.propagate(centerDate);
99          double maxResidualP = 0;
100         double maxResidualV = 0;
101         double maxResidualA = 0;
102         double maxResidualR = 0;
103         for (double dt = 0; dt < 900.0; dt += 5) {
104             SpacecraftState shifted = centerState.shiftedBy(dt);
105             SpacecraftState propagated = propagator.propagate(centerDate.shiftedBy(dt));
106             PVCoordinates dpv = new PVCoordinates(propagated.getPVCoordinates(), shifted.getPVCoordinates());
107             double residualP = pModel.value(dt) - dpv.getPosition().getNorm();
108             double residualV = vModel.value(dt) - dpv.getVelocity().getNorm();
109             double residualA = aModel.value(dt) - dpv.getAcceleration().getNorm();
110             double residualR = rModel.value(dt) -
111                                FastMath.toDegrees(Rotation.distance(shifted.getAttitude().getRotation(),
112                                                                     propagated.getAttitude().getRotation()));
113             maxResidualP = FastMath.max(maxResidualP, FastMath.abs(residualP));
114             maxResidualV = FastMath.max(maxResidualV, FastMath.abs(residualV));
115             maxResidualA = FastMath.max(maxResidualA, FastMath.abs(residualA));
116             maxResidualR = FastMath.max(maxResidualR, FastMath.abs(residualR));
117         }
118         Assert.assertEquals(0.2,    maxResidualP, 0.1);
119         Assert.assertEquals(0.0008, maxResidualV, 0.0001);
120         Assert.assertEquals(7.9e-6, maxResidualA, 1.0e-7);
121         Assert.assertEquals(2.8e-6, maxResidualR, 1.0e-1);
122     }

124     @Test
125     public void testInterpolation()
126         throws ParseException, OrekitException {
127         checkInterpolationError( 2,  106.46533, 0.40709287, 169847806.33e-9, 0.0, 450 * 450);
128         checkInterpolationError( 3,    0.00353, 0.00003250,    189886.01e-9, 0.0, 0.0);
129         checkInterpolationError( 4,    0.00002, 0.00000023,       232.25e-9, 0.0, 0.0);

132     private void checkInterpolationError(int n, double expectedErrorP, double expectedErrorV,
133                                          double expectedErrorA, double expectedErrorM, double expectedErrorQ)
134         throws OrekitException {
135         AbsoluteDate centerDate = orbit.getDate().shiftedBy(100.0);
136         SpacecraftState centerState = propagator.propagate(centerDate).addAdditionalState("quadratic", 0);
137         List<SpacecraftState> sample = new ArrayList<SpacecraftState>();
138         for (int i = 0; i < n; ++i) {
139             double dt = i * 900.0 / (n - 1);
140             SpacecraftState state = propagator.propagate(centerDate.shiftedBy(dt));
141             state = state.addAdditionalState("quadratic", dt * dt);
142             sample.add(state);
143         }
144         double maxErrorP = 0;
145         double maxErrorV = 0;
146         double maxErrorA = 0;
147         double maxErrorM = 0;
148         double maxErrorQ = 0;
149         for (double dt = 0; dt < 900.0; dt += 5) {
150             SpacecraftState interpolated = centerState.interpolate(centerDate.shiftedBy(dt), sample);
151             SpacecraftState propagated = propagator.propagate(centerDate.shiftedBy(dt));
152             PVCoordinates dpv = new PVCoordinates(propagated.getPVCoordinates(), interpolated.getPVCoordinates());
153             maxErrorP = FastMath.max(maxErrorP, dpv.getPosition().getNorm());
154             maxErrorV = FastMath.max(maxErrorV, dpv.getVelocity().getNorm());
155             maxErrorA = FastMath.max(maxErrorA, FastMath.toDegrees(Rotation.distance(interpolated.getAttitude().getRotation(),
156                                                                                                   propagated.getAttitude().getRotation())));
157             maxErrorM = FastMath.max(maxErrorM, FastMath.abs(interpolated.getMass() - propagated.getMass()));
158             maxErrorQ = FastMath.max(maxErrorQ, FastMath.abs(interpolated.getAdditionalState("quadratic")[0] - dt * dt));
159         }
160         Assert.assertEquals(expectedErrorP, maxErrorP, 1.0e-3);
161         Assert.assertEquals(expectedErrorV, maxErrorV, 1.0e-6);
162         Assert.assertEquals(expectedErrorA, maxErrorA, 4.0e-10);
163         Assert.assertEquals(expectedErrorM, maxErrorM, 1.0e-15);
164         Assert.assertEquals(expectedErrorQ, maxErrorQ, 2.0e-10);
165     }

173     /**
174      * Check orbit and attitude dates can be off by a few ulps. I see this when using
175      * FixedRate attitude provider.
176      */
177     @Test
178     public void testDateConsistencyClose() throws OrekitException {
179         //setup
180         Orbit orbit10Shifts = orbit;
181         for (int i = 0; i < 10; i++) {
182             orbit10Shifts = orbit10Shifts.shiftedBy(0.1);
183         }
184         final Orbit orbit1Shift = orbit.shiftedBy(1);
185         Attitude shiftedAttitude = attitudeLaw
186                 .getAttitude(orbit1Shift, orbit1Shift.getDate(), orbit.getFrame());
187 
188         //verify dates are very close, but not equal
189         Assert.assertNotEquals(shiftedAttitude.getDate(), orbit10Shifts.getDate());
190         Assert.assertEquals(
191                 shiftedAttitude.getDate().durationFrom(orbit10Shifts.getDate()),
192                 0, Precision.EPSILON);
193 
194         //action + verify no exception is thrown
195         new SpacecraftState(orbit10Shifts, shiftedAttitude);
196     }

206     @Test
207     public void testTransform()
208         throws ParseException, OrekitException {
209 
210         double maxDP = 0;
211         double maxDV = 0;
212         double maxDA = 0;
213         for (double t = 0; t < orbit.getKeplerianPeriod(); t += 60) {
214             final SpacecraftState state = propagator.propagate(orbit.getDate().shiftedBy(t));
215             final Transform transform = state.toTransform().getInverse();
216             PVCoordinates pv = transform.transformPVCoordinates(PVCoordinates.ZERO);
217             PVCoordinates dPV = new PVCoordinates(pv, state.getPVCoordinates());
218             Vector3D mZDirection = transform.transformVector(Vector3D.MINUS_K);
219             double alpha = Vector3D.angle(mZDirection, state.getPVCoordinates().getPosition());
220             maxDP = FastMath.max(maxDP, dPV.getPosition().getNorm());
221             maxDV = FastMath.max(maxDV, dPV.getVelocity().getNorm());
222             maxDA = FastMath.max(maxDA, FastMath.toDegrees(alpha));
223         }
224         Assert.assertEquals(0.0, maxDP, 1.0e-6);
225         Assert.assertEquals(0.0, maxDV, 1.0e-9);
226         Assert.assertEquals(0.0, maxDA, 1.0e-12);
227 
228     }

230     @Test
231     public void testAdditionalStates() throws OrekitException {
232         final SpacecraftState state = propagator.propagate(orbit.getDate().shiftedBy(60));
233         final SpacecraftState extended =
234                 state.
235                  addAdditionalState("test-1", new double[] { 1.0, 2.0 }).
236                   addAdditionalState("test-2", 42.0);
237         Assert.assertEquals(0, state.getAdditionalStates().size());
238         Assert.assertFalse(state.hasAdditionalState("test-1"));
239         try {
240             state.getAdditionalState("test-1");
241             Assert.fail("an exception should have been thrown");
242         } catch (OrekitException oe) {
243             Assert.assertEquals(oe.getSpecifier(), OrekitMessages.UNKNOWN_ADDITIONAL_STATE);
244             Assert.assertEquals(oe.getParts()[0], "test-1");
245         }
246         try {
247             state.ensureCompatibleAdditionalStates(extended);
248             Assert.fail("an exception should have been thrown");
249         } catch (OrekitException oe) {
250             Assert.assertEquals(oe.getSpecifier(), OrekitMessages.UNKNOWN_ADDITIONAL_STATE);
251             Assert.assertTrue(oe.getParts()[0].toString().startsWith("test-"));
252         }
253         try {
254             extended.ensureCompatibleAdditionalStates(state);
255             Assert.fail("an exception should have been thrown");
256         } catch (OrekitException oe) {
257             Assert.assertEquals(oe.getSpecifier(), OrekitMessages.UNKNOWN_ADDITIONAL_STATE);
258             Assert.assertTrue(oe.getParts()[0].toString().startsWith("test-"));
259         }
260         try {
261             extended.ensureCompatibleAdditionalStates(extended.addAdditionalState("test-2", new double[7]));
262             Assert.fail("an exception should have been thrown");
263         } catch (MathIllegalStateException mise) {
264             Assert.assertEquals(LocalizedCoreFormats.DIMENSIONS_MISMATCH, mise.getSpecifier());
265             Assert.assertEquals(7, ((Integer) mise.getParts()[0]).intValue());
266         }
267         Assert.assertEquals(2, extended.getAdditionalStates().size());
268         Assert.assertTrue(extended.hasAdditionalState("test-1"));
269         Assert.assertTrue(extended.hasAdditionalState("test-2"));
270         Assert.assertEquals( 1.0, extended.getAdditionalState("test-1")[0], 1.0e-15);
271         Assert.assertEquals( 2.0, extended.getAdditionalState("test-1")[1], 1.0e-15);
272         Assert.assertEquals(42.0, extended.getAdditionalState("test-2")[0], 1.0e-15);
273 
274         // test various constructors
275         Map<String, double[]> map = new HashMap<String, double[]>();
276         map.put("test-3", new double[] { -6.0 });
277         SpacecraftState sO = new SpacecraftState(state.getOrbit(), map);
278         Assert.assertEquals(-6.0, sO.getAdditionalState("test-3")[0], 1.0e-15);
279         SpacecraftState sOA = new SpacecraftState(state.getOrbit(), state.getAttitude(), map);
280         Assert.assertEquals(-6.0, sOA.getAdditionalState("test-3")[0], 1.0e-15);
281         SpacecraftState sOM = new SpacecraftState(state.getOrbit(), state.getMass(), map);
282         Assert.assertEquals(-6.0, sOM.getAdditionalState("test-3")[0], 1.0e-15);
283         SpacecraftState sOAM = new SpacecraftState(state.getOrbit(), state.getAttitude(), state.getMass(), map);
284         Assert.assertEquals(-6.0, sOAM.getAdditionalState("test-3")[0], 1.0e-15);
285 
286     }

288     @Test
289     public void testSerialization()
290             throws IOException, ClassNotFoundException, NoSuchFieldException, IllegalAccessException, OrekitException {
291 
292         propagator.resetInitialState(propagator.getInitialState().
293                                      addAdditionalState("p1", 12.25).
294                                      addAdditionalState("p2", 1, 2, 3));
295         SpacecraftState state = propagator.propagate(orbit.getDate().shiftedBy(123.456));
296 
297         Assert.assertEquals(2, state.getAdditionalStates().size());
298         Assert.assertEquals(1, state.getAdditionalState("p1").length);
299         Assert.assertEquals(12.25, state.getAdditionalState("p1")[0], 1.0e-15);
300         Assert.assertEquals(3, state.getAdditionalState("p2").length);
301         Assert.assertEquals(1.0, state.getAdditionalState("p2")[0], 1.0e-15);
302         Assert.assertEquals(2.0, state.getAdditionalState("p2")[1], 1.0e-15);
303         Assert.assertEquals(3.0, state.getAdditionalState("p2")[2], 1.0e-15);
304 
305         ByteArrayOutputStream bos = new ByteArrayOutputStream();
306         ObjectOutputStream    oos = new ObjectOutputStream(bos);
307         oos.writeObject(state);
308 
309         Assert.assertTrue(bos.size() > 700);
310         Assert.assertTrue(bos.size() < 800);
311 
312         ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
313         ObjectInputStream     ois = new ObjectInputStream(bis);
314         SpacecraftState deserialized  = (SpacecraftState) ois.readObject();
315         Assert.assertEquals(0.0,
316                             Vector3D.distance(state.getPVCoordinates().getPosition(),
317                                               deserialized.getPVCoordinates().getPosition()),
318                             1.0e-10);
319         Assert.assertEquals(0.0,
320                             Vector3D.distance(state.getPVCoordinates().getVelocity(),
321                                               deserialized.getPVCoordinates().getVelocity()),
322                             1.0e-10);
323         Assert.assertEquals(0.0,
324                             Rotation.distance(state.getAttitude().getRotation(),
325                                               deserialized.getAttitude().getRotation()),
326                             1.0e-10);
327         Assert.assertEquals(0.0,
328                             Vector3D.distance(state.getAttitude().getSpin(),
329                                               deserialized.getAttitude().getSpin()),
330                             1.0e-10);
331         Assert.assertEquals(state.getDate(), deserialized.getDate());
332         Assert.assertEquals(state.getMu(), deserialized.getMu(), 1.0e-10);
333         Assert.assertEquals(state.getFrame().getName(), deserialized.getFrame().getName());
334         Assert.assertEquals(2, deserialized.getAdditionalStates().size());
335         Assert.assertEquals(1, deserialized.getAdditionalState("p1").length);
336         Assert.assertEquals(12.25, deserialized.getAdditionalState("p1")[0], 1.0e-15);
337         Assert.assertEquals(3, deserialized.getAdditionalState("p2").length);
338         Assert.assertEquals(1.0, deserialized.getAdditionalState("p2")[0], 1.0e-15);
339         Assert.assertEquals(2.0, deserialized.getAdditionalState("p2")[1], 1.0e-15);
340         Assert.assertEquals(3.0, deserialized.getAdditionalState("p2")[2], 1.0e-15);
341 
342     }
