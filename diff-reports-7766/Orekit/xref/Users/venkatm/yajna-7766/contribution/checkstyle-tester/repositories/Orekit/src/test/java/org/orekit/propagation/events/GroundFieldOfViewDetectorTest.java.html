47  /**
48   * Unit tests for {@link GroundFieldOfViewDetector}.
49   *
50   * @author Evan Ward
51   */
52  public class GroundFieldOfViewDetectorTest {
53  
54      @Before
55      public void setUp() {
56          Utils.setDataRoot("regular-data");
57      }
58  
59      /**
60       * Check FoV detector is similar to {@link ElevationDetector} when using
61       * zenith pointing.
62       *
63       * @throws OrekitException on error.
64       */
65      @Test
66      public void testCaseSimilarToElevationDetector() throws OrekitException {
67          //setup
68          double pi = FastMath.PI;
69          AbsoluteDate date = AbsoluteDate.J2000_EPOCH; //arbitrary date
70          AbsoluteDate endDate = date.shiftedBy(Constants.JULIAN_DAY);
71          Frame eci = FramesFactory.getGCRF();
72          Frame ecef = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
73          BodyShape earth = new OneAxisEllipsoid(
74                  Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
75                  Constants.WGS84_EARTH_FLATTENING,
76                  ecef);
77          GeodeticPoint gp = new GeodeticPoint(
78                  FastMath.toRadians(39), FastMath.toRadians(77), 0);
79          TopocentricFrame topo = new TopocentricFrame(earth, gp, "topo");
80          //iss like orbit
81          KeplerianOrbit orbit = new KeplerianOrbit(
82                  6378137 + 400e3, 0, FastMath.toRadians(51.65), 0, 0, 0,
83                  PositionAngle.TRUE, eci, date, Constants.EGM96_EARTH_MU);
84          Propagator prop = new KeplerianPropagator(orbit);
85  
86          //compute expected result
87          ElevationDetector elevationDetector =
88                  new ElevationDetector(topo).withConstantElevation(pi / 6)
89                          .withMaxCheck(5.0);
90          EventsLogger logger = new EventsLogger();
91          prop.addEventDetector(logger.monitorDetector(elevationDetector));
92          prop.propagate(endDate);
93          List<LoggedEvent> expected = logger.getLoggedEvents();
94  
95          //action
96          //construct similar FoV based detector
97          //half width of 60 deg pointed along +Z in antenna frame
98          //not a perfect small circle b/c FoV makes a polygon with great circles
99          FieldOfView fov =
100                 new FieldOfView(Vector3D.PLUS_K, Vector3D.PLUS_I, pi / 3, 16, 0);
101         //simple case for fixed pointing to be similar to elevation detector.
102         //could define new frame with varying rotation for slewing antenna.
103         GroundFieldOfViewDetector fovDetector =
104                 new GroundFieldOfViewDetector(topo, fov)
105                         .withMaxCheck(5.0);
106         Assert.assertSame(topo, fovDetector.getFrame());
107         Assert.assertSame(fov, fovDetector.getFieldOfView());
108         logger = new EventsLogger();
109 
110         prop = new KeplerianPropagator(orbit);
111         prop.addEventDetector(logger.monitorDetector(fovDetector));
112         prop.propagate(endDate);
113         List<LoggedEvent> actual = logger.getLoggedEvents();
114 
115         //verify
116         Assert.assertEquals(2, expected.size());
117         Assert.assertEquals(2, actual.size());
118         for (int i = 0; i < 2; i++) {
119             AbsoluteDate expectedDate = expected.get(i).getState().getDate();
120             AbsoluteDate actualDate = actual.get(i).getState().getDate();
121             // same event times to within 1s.
122             Assert.assertEquals(expectedDate.durationFrom(actualDate), 0.0, 1.0);
123         }
124 
125     }
