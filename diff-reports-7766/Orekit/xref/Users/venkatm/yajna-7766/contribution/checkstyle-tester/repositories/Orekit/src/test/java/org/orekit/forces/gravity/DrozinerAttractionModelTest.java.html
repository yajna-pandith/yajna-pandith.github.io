104     private static class SpotStepHandler implements OrekitFixedStepHandler {
105 
106         public SpotStepHandler() throws OrekitException {
107             sun       = CelestialBodyFactory.getSun();
108             previous  = Double.NaN;
109         }
110 
111         private PVCoordinatesProvider sun;
112         private double previous;
113         public void handleStep(SpacecraftState currentState, boolean isLast)
114             throws OrekitException {
115 
116             AbsoluteDate current = currentState.getDate();
117             Vector3D sunPos = sun.getPVCoordinates(current , FramesFactory.getEME2000()).getPosition();
118             Vector3D normal = currentState.getPVCoordinates().getMomentum();
119             double angle = Vector3D.angle(sunPos , normal);
120             if (! Double.isNaN(previous)) {
121                 Assert.assertEquals(previous, angle, 0.0013);
122             }
123             previous = angle;
124         }
125 
126     }

168     private static class EckStepHandler implements OrekitFixedStepHandler {
169 
170         private EckStepHandler(Orbit initialOrbit, double ae, double mu,
171                                double c20, double c30, double c40, double c50, double c60)
172         throws FileNotFoundException, OrekitException {
173             referencePropagator =
174                 new EcksteinHechlerPropagator(initialOrbit,
175                                               ae, mu, c20, c30, c40, c50, c60);
176         }
177 
178         private EcksteinHechlerPropagator referencePropagator;
179 
180         public void handleStep(SpacecraftState currentState, boolean isLast) throws OrekitException {
181 
182             SpacecraftState EHPOrbit   = referencePropagator.propagate(currentState.getDate());
183             Vector3D posEHP  = EHPOrbit.getPVCoordinates().getPosition();
184             Vector3D posDROZ = currentState.getPVCoordinates().getPosition();
185             Vector3D velEHP  = EHPOrbit.getPVCoordinates().getVelocity();
186             Vector3D dif     = posEHP.subtract(posDROZ);
187 
188             Vector3D T = new Vector3D(1 / velEHP.getNorm(), velEHP);
189             Vector3D W = EHPOrbit.getPVCoordinates().getMomentum().normalize();
190             Vector3D N = Vector3D.crossProduct(W, T);
191 
192             Assert.assertTrue(dif.getNorm() < 111);
193             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(dif, T)) < 111);
194             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(dif, N)) <  54);
195             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(dif, W)) <  12);
196 
197         }
198 
199     }

201     // test the difference with the Holmes-Featherstone model
202     @Test
203     public void testTesserealWithHolmesFeaterstoneReference()
204         throws OrekitException, IOException, ParseException {
205 
206         Utils.setDataRoot("regular-data:potential/grgs-format");
207         GravityFieldFactory.addPotentialCoefficientsReader(new GRGSFormatReader("grim4s4_gr", true));
208         UnnormalizedSphericalHarmonicsProvider unnormalized =
209                 GravityFieldFactory.getUnnormalizedProvider(10, 10);
210         NormalizedSphericalHarmonicsProvider normalized =
211                 GravityFieldFactory.getNormalizedProvider(10, 10);
212 
213         //  initialization
214         AbsoluteDate date = new AbsoluteDate(new DateComponents(2000, 07, 01),
215                                              new TimeComponents(13, 59, 27.816),
216                                              TimeScalesFactory.getUTC());
217         double i     = FastMath.toRadians(98.7);
218         double omega = FastMath.toRadians(93.0);
219         double OMEGA = FastMath.toRadians(15.0 * 22.5);
220         Orbit orbit = new KeplerianOrbit(7201009.7124401, 1e-3, i , omega, OMEGA,
221                                          0, PositionAngle.MEAN, FramesFactory.getEME2000(), date, mu);
222         propagator = new NumericalPropagator(new ClassicalRungeKuttaIntegrator(100));
223         propagator.addForceModel(new HolmesFeatherstoneAttractionModel(itrf2008, normalized));
224         propagator.setInitialState(new SpacecraftState(orbit));
225         SpacecraftState hfOrb = propagator.propagate(date.shiftedBy(Constants.JULIAN_DAY));
226 
227         propagator.removeForceModels();
228         propagator.addForceModel(new DrozinerAttractionModel(itrf2008, unnormalized, 1.0));
229                                                              
230 
231         propagator.setInitialState(new SpacecraftState(orbit));
232         SpacecraftState drozOrb = propagator.propagate(date.shiftedBy(Constants.JULIAN_DAY));
233 
234         Vector3D dif = hfOrb.getPVCoordinates().getPosition().subtract(drozOrb.getPVCoordinates().getPosition());
235         Assert.assertEquals(0, dif.getNorm(), 3.1e-7);
236         Assert.assertTrue(propagator.getCalls() < 3500);
237 
238     }

264     @Test
265     public void testStateJacobian()
266         throws OrekitException {
267 
268         Utils.setDataRoot("regular-data:potential/grgs-format");
269         GravityFieldFactory.addPotentialCoefficientsReader(new GRGSFormatReader("grim4s4_gr", true));
270 
271         // initialization
272         AbsoluteDate date = new AbsoluteDate(new DateComponents(2000, 07, 01),
273                                              new TimeComponents(13, 59, 27.816),
274                                              TimeScalesFactory.getUTC());
275         double i     = FastMath.toRadians(98.7);
276         double omega = FastMath.toRadians(93.0);
277         double OMEGA = FastMath.toRadians(15.0 * 22.5);
278         Orbit orbit = new KeplerianOrbit(7201009.7124401, 1e-3, i , omega, OMEGA,
279                                          0, PositionAngle.MEAN, FramesFactory.getEME2000(), date, mu);
280         OrbitType integrationType = OrbitType.CARTESIAN;
281         double[][] tolerances = NumericalPropagator.tolerances(0.01, orbit, integrationType);
282 
283         propagator = new NumericalPropagator(new DormandPrince853Integrator(1.0e-3, 120,
284                                                                             tolerances[0], tolerances[1]));
285         propagator.setOrbitType(integrationType);
286         DrozinerAttractionModel drModel =
287                 new DrozinerAttractionModel(itrf2008, GravityFieldFactory.getUnnormalizedProvider(50, 50), 1.0);
288         Assert.assertEquals(TideSystem.UNKNOWN, drModel.getTideSystem());
289         propagator.addForceModel(drModel);
290         SpacecraftState state0 = new SpacecraftState(orbit);
291         propagator.setInitialState(state0);
292         checkStateJacobian(propagator, state0, date.shiftedBy(3.5 * 3600.0),
293                            40000, tolerances[0], 7.9e-6);
294
