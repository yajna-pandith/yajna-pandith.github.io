233     public void testPositionVelocityNorms() {
234         double mu = 3.9860047e14;
235 
236         // elliptic and non equatorial orbit
237         KeplerianOrbit p =
238             new KeplerianOrbit(24464560.0, 0.7311, 2.1, 3.10686, 1.00681,
239                                     0.67, PositionAngle.TRUE,
240                                     FramesFactory.getEME2000(), date, mu);
241 
242         double e       = p.getE();
243         double v       = p.getTrueAnomaly();
244         double ksi     = 1 + e * FastMath.cos(v);
245         double nu      = e * FastMath.sin(v);
246         double epsilon = FastMath.sqrt((1 - e) * (1 + e));
247 
248         double a  = p.getA();
249         double na = FastMath.sqrt(mu / a);
250 
251         // validation of: r = a .(1 - e2) / (1 + e.cos(v))
252         Assert.assertEquals(a * epsilon * epsilon / ksi,
253                      p.getPVCoordinates().getPosition().getNorm(),
254                      Utils.epsilonTest * FastMath.abs(p.getPVCoordinates().getPosition().getNorm()));
255 
256         // validation of: V = sqrt(mu.(1+2e.cos(v)+e2)/a.(1-e2) )
257         Assert.assertEquals(na * FastMath.sqrt(ksi * ksi + nu * nu) / epsilon,
258                      p.getPVCoordinates().getVelocity().getNorm(),
259                      Utils.epsilonTest * FastMath.abs(p.getPVCoordinates().getVelocity().getNorm()));
260 
261 
262         //  circular and equatorial orbit
263         KeplerianOrbit pCirEqua =
264             new KeplerianOrbit(24464560.0, 0.1e-10, 0.1e-8, 3.10686, 1.00681,

275         na = FastMath.sqrt(mu / a);
276 
277         // validation of: r = a .(1 - e2) / (1 + e.cos(v))

279                      pCirEqua.getPVCoordinates().getPosition().getNorm(),
280                      Utils.epsilonTest * FastMath.abs(pCirEqua.getPVCoordinates().getPosition().getNorm()));
281 
282         // validation of: V = sqrt(mu.(1+2e.cos(v)+e2)/a.(1-e2) )
283         Assert.assertEquals(na * FastMath.sqrt(ksi * ksi + nu * nu) / epsilon,
284                      pCirEqua.getPVCoordinates().getVelocity().getNorm(),

289     public void testGeometry() {
290         double mu = 3.9860047e14;
291 
292         // elliptic and non equatorial orbit
293         KeplerianOrbit p =
294             new KeplerianOrbit(24464560.0, 0.7311, 2.1, 3.10686, 1.00681,

297 
298         Vector3D position = p.getPVCoordinates().getPosition();
299         Vector3D velocity = p.getPVCoordinates().getVelocity();
300         Vector3D momentum = p.getPVCoordinates().getMomentum().normalize();
301 
302         double apogeeRadius  = p.getA() * (1 + p.getE());
303         double perigeeRadius = p.getA() * (1 - p.getE());
304 
305         for (double lv = 0; lv <= 2 * FastMath.PI; lv += 2 * FastMath.PI/100.) {
306             p = new KeplerianOrbit(p.getA(),p.getE(), p.getI(), p.getPerigeeArgument(),
307                                         p.getRightAscensionOfAscendingNode(), lv , PositionAngle.TRUE,
308                                         p.getFrame(), p.getDate(), p.getMu());
309             position = p.getPVCoordinates().getPosition();
310 
311             // test if the norm of the position is in the range [perigee radius, apogee radius]
312             Assert.assertTrue((position.getNorm() - apogeeRadius)  <= (  apogeeRadius * Utils.epsilonTest));
313             Assert.assertTrue((position.getNorm() - perigeeRadius) >= (- perigeeRadius * Utils.epsilonTest));
314 
315             position = position.normalize();
316             velocity = p.getPVCoordinates().getVelocity();
317             velocity = velocity.normalize();
318 
319             // at this stage of computation, all the vectors (position, velocity and momemtum) are normalized here
320 
321             // test of orthogonality between position and momentum
322             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(position, momentum)) < Utils.epsilonTest);
323             // test of orthogonality between velocity and momentum
324             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(velocity, momentum)) < Utils.epsilonTest);
325 
326         }
327 
328         // apsides
329         p = new KeplerianOrbit(p.getA(),p.getE(), p.getI(), p.getPerigeeArgument(),
330                                     p.getRightAscensionOfAscendingNode(), 0 , PositionAngle.TRUE, p.getFrame(), p.getDate(), p.getMu());

334                                     p.getRightAscensionOfAscendingNode(), FastMath.PI , PositionAngle.TRUE, p.getFrame(), p.getDate(), p.getMu());

341                                     p.getFrame(), p.getDate(), p.getMu());
342         Assert.assertTrue(FastMath.abs(p.getPVCoordinates().getPosition().getZ()) < p.getPVCoordinates().getPosition().getNorm() * Utils.epsilonTest);
343         Assert.assertTrue(p.getPVCoordinates().getVelocity().getZ() < 0);

347                                     p.getRightAscensionOfAscendingNode(),2.0 * FastMath.PI - p.getPerigeeArgument() , PositionAngle.TRUE,
348                                     p.getFrame(), p.getDate(), p.getMu());
349         Assert.assertTrue(FastMath.abs(p.getPVCoordinates().getPosition().getZ()) < p.getPVCoordinates().getPosition().getNorm() * Utils.epsilonTest);

369                                                pCirEqua.getRightAscensionOfAscendingNode(), lv, PositionAngle.TRUE,
370                                                pCirEqua.getFrame(), pCirEqua.getDate(), pCirEqua.getMu());
371             position = pCirEqua.getPVCoordinates().getPosition();
372

785         // we use this to compute a velocity step size from the position step size
786         Vector3D p = orbit.getPVCoordinates().getPosition();
787         Vector3D v = orbit.getPVCoordinates().getVelocity();
788         double hV = orbit.getMu() * hP / (v.getNorm() * p.getNormSq());
789 
790         double h;
791         Vector3D dP = Vector3D.ZERO;
792         Vector3D dV = Vector3D.ZERO;
793         switch (i) {
794         case 0:
795             h = hP;
796             dP = new Vector3D(hP, 0, 0);
797             break;
798         case 1:
799             h = hP;
800             dP = new Vector3D(0, hP, 0);
801             break;
802         case 2:

805             break;

807             h = hV;

809             break;
810         case 4:
811             h = hV;
812             dV = new Vector3D(0, hV, 0);

817             break;
818         }

866 
867         final double ehMu  = 3.9860047e14;
868         final double ae  = 6.378137e6;
869         final double c20 = -1.08263e-3;
870         final double c30 = 2.54e-6;
871         final double c40 = 1.62e-6;
872         final double c50 = 2.3e-7;
873         final double c60 = -5.5e-7;
874 
875         final AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(584.);
876         final Vector3D position = new Vector3D(3220103., 69623., 6449822.);
877         final Vector3D velocity = new Vector3D(6414.7, -2006., -3180.);
878         final KeplerianOrbit initialOrbit = new KeplerianOrbit(new PVCoordinates(position, velocity),
879                                                                FramesFactory.getEME2000(), date, ehMu);
880 
881         EcksteinHechlerPropagator propagator =
882                 new EcksteinHechlerPropagator(initialOrbit, ae, ehMu, c20, c30, c40, c50, c60);
883 
884         // set up a 5 points sample
885         List<Orbit> sample = new ArrayList<Orbit>();
886         for (double dt = 0; dt < 300.0; dt += 60.0) {
887             sample.add(propagator.propagate(date.shiftedBy(dt)).getOrbit());
888         }
889 
890         // well inside the sample, interpolation should be slightly better than Keplerian shift
891         // the relative bad behaviour here is due to eccentricity, which cannot be
892         // accurately interpolated with a polynomial in this case
893         double maxShiftPositionError = 0;
894         double maxInterpolationPositionError = 0;
895         double maxShiftEccentricityError = 0;
896         double maxInterpolationEccentricityError = 0;
897         for (double dt = 0; dt < 241.0; dt += 1.0) {
898             AbsoluteDate t         = initialOrbit.getDate().shiftedBy(dt);
899             Vector3D shiftedP      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
900             Vector3D interpolatedP = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
901             Vector3D propagatedP   = propagator.propagate(t).getPVCoordinates().getPosition();
902             double shiftedE        = initialOrbit.shiftedBy(dt).getE();
903             double interpolatedE   = initialOrbit.interpolate(t, sample).getE();
904             double propagatedE     = propagator.propagate(t).getE();
905             maxShiftPositionError = FastMath.max(maxShiftPositionError, shiftedP.subtract(propagatedP).getNorm());
906             maxInterpolationPositionError = FastMath.max(maxInterpolationPositionError, interpolatedP.subtract(propagatedP).getNorm());
907             maxShiftEccentricityError = FastMath.max(maxShiftEccentricityError, FastMath.abs(shiftedE - propagatedE));
908             maxInterpolationEccentricityError = FastMath.max(maxInterpolationEccentricityError, FastMath.abs(interpolatedE - propagatedE));
909         }
910         Assert.assertTrue(maxShiftPositionError             > 390.0);
911         Assert.assertTrue(maxInterpolationPositionError     < 62.0);
912         Assert.assertTrue(maxShiftEccentricityError         > 4.5e-4);
913         Assert.assertTrue(maxInterpolationEccentricityError < 2.6e-5);
914 
915         // slightly past sample end, bad eccentricity interpolation shows up
916         // (in this case, interpolated eccentricity exceeds 1.0 btween 1900
917         // and 1910s, while semi-majaxis remains positive, so this is not
918         // even a proper hyperbolic orbit...)
919         maxShiftPositionError = 0;
920         maxInterpolationPositionError = 0;
921         maxShiftEccentricityError = 0;
922         maxInterpolationEccentricityError = 0;
923         for (double dt = 240; dt < 600; dt += 1.0) {
924             AbsoluteDate t         = initialOrbit.getDate().shiftedBy(dt);
925             Vector3D shiftedP      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
926             Vector3D interpolatedP = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
927             Vector3D propagatedP   = propagator.propagate(t).getPVCoordinates().getPosition();
928             double shiftedE        = initialOrbit.shiftedBy(dt).getE();
929             double interpolatedE   = initialOrbit.interpolate(t, sample).getE();
930             double propagatedE     = propagator.propagate(t).getE();
931             maxShiftPositionError = FastMath.max(maxShiftPositionError, shiftedP.subtract(propagatedP).getNorm());
932             maxInterpolationPositionError = FastMath.max(maxInterpolationPositionError, interpolatedP.subtract(propagatedP).getNorm());
933             maxShiftEccentricityError = FastMath.max(maxShiftEccentricityError, FastMath.abs(shiftedE - propagatedE));
934             maxInterpolationEccentricityError = FastMath.max(maxInterpolationEccentricityError, FastMath.abs(interpolatedE - propagatedE));
935         }
936         Assert.assertTrue(maxShiftPositionError             <  2200.0);
937         Assert.assertTrue(maxInterpolationPositionError     > 72000.0);
938         Assert.assertTrue(maxShiftEccentricityError         <  1.2e-3);
939         Assert.assertTrue(maxInterpolationEccentricityError >  3.8e-3);
