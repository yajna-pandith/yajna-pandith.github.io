292     @Test
293     public void testPositionVelocityNorms() {
294 
295         // elliptic and non equatorial (i retrograde) orbit
296         EquinoctialOrbit p =
297             new EquinoctialOrbit(42166.712, 0.5, -0.5, 1.200, 2.1,
298                                       0.67, PositionAngle.TRUE,
299                                       FramesFactory.getEME2000(), date, mu);
300 
301         double ex = p.getEquinoctialEx();
302         double ey = p.getEquinoctialEy();
303         double lv = p.getLv();
304         double ksi = 1 + ex * FastMath.cos(lv) + ey * FastMath.sin(lv);
305         double nu = ex * FastMath.sin(lv) - ey * FastMath.cos(lv);
306         double epsilon = FastMath.sqrt(1 - ex * ex - ey * ey);
307 
308         double a = p.getA();
309         double na = FastMath.sqrt(p.getMu() / a);
310 
311         Assert.assertEquals(a * epsilon * epsilon / ksi, p.getPVCoordinates().getPosition().getNorm(),
312                      Utils.epsilonTest * FastMath.abs(p.getPVCoordinates().getPosition().getNorm()));
313         Assert.assertEquals(na * FastMath.sqrt(ksi * ksi + nu * nu) / epsilon, p
314                      .getPVCoordinates().getVelocity().getNorm(), Utils.epsilonTest
315                      * FastMath.abs(p.getPVCoordinates().getVelocity().getNorm()));
316 
317         // circular and equatorial orbit
318         EquinoctialOrbit pCirEqua =
319             new EquinoctialOrbit(42166.712, 0.1e-8, 0.1e-8, 0.1e-8, 0.1e-8,
320                                       0.67, PositionAngle.TRUE,
321                                       FramesFactory.getEME2000(), date, mu);
322 
323         ex = pCirEqua.getEquinoctialEx();
324         ey = pCirEqua.getEquinoctialEy();
325         lv = pCirEqua.getLv();
326         ksi = 1 + ex * FastMath.cos(lv) + ey * FastMath.sin(lv);
327         nu = ex * FastMath.sin(lv) - ey * FastMath.cos(lv);
328         epsilon = FastMath.sqrt(1 - ex * ex - ey * ey);
329 
330         a = pCirEqua.getA();
331         na = FastMath.sqrt(pCirEqua.getMu() / a);
332 
333         Assert.assertEquals(a * epsilon * epsilon / ksi, pCirEqua.getPVCoordinates().getPosition()
334                      .getNorm(), Utils.epsilonTest
335                      * FastMath.abs(pCirEqua.getPVCoordinates().getPosition().getNorm()));
336         Assert.assertEquals(na * FastMath.sqrt(ksi * ksi + nu * nu) / epsilon, pCirEqua
337                      .getPVCoordinates().getVelocity().getNorm(), Utils.epsilonTest
338                      * FastMath.abs(pCirEqua.getPVCoordinates().getVelocity().getNorm()));
339     }

342     public void testGeometry() {
343 
344         // elliptic and non equatorial (i retrograde) orbit
345         EquinoctialOrbit p =
346             new EquinoctialOrbit(42166.712, 0.5, -0.5, 1.200, 2.1,

349 
350         Vector3D position = p.getPVCoordinates().getPosition();
351         Vector3D velocity = p.getPVCoordinates().getVelocity();
352         Vector3D momentum = p.getPVCoordinates().getMomentum().normalize();
353 
354         double apogeeRadius = p.getA() * (1 + p.getE());
355         double perigeeRadius = p.getA() * (1 - p.getE());
356 
357         for (double lv = 0; lv <= 2 * FastMath.PI; lv += 2 * FastMath.PI / 100.) {
358             p = new EquinoctialOrbit(p.getA() ,p.getEquinoctialEx(),
359                                           p.getEquinoctialEy() , p.getHx(), p.getHy() , lv , PositionAngle.TRUE,
360                                           p.getFrame(), p.getDate(), p.getMu());
361             position = p.getPVCoordinates().getPosition();
362 
363             // test if the norm of the position is in the range [perigee radius,
364             // apogee radius]
365             // Warning: these tests are without absolute value by choice
366             Assert.assertTrue((position.getNorm() - apogeeRadius) <= (apogeeRadius * Utils.epsilonTest));
367             Assert.assertTrue((position.getNorm() - perigeeRadius) >= (-perigeeRadius * Utils.epsilonTest));
368 
369             position = position.normalize();
370             velocity = p.getPVCoordinates().getVelocity();
371             velocity = velocity.normalize();
372 
373             // at this stage of computation, all the vectors (position, velocity and
374             // momemtum) are normalized here
375 
376             // test of orthogonality between position and momentum
377             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(position, momentum)) < Utils.epsilonTest);
378             // test of orthogonality between velocity and momentum
379             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(velocity, momentum)) < Utils.epsilonTest);
380         }
381 
382         // circular and equatorial orbit
383         EquinoctialOrbit pCirEqua =
384             new EquinoctialOrbit(42166.712, 0.1e-8, 0.1e-8, 0.1e-8, 0.1e-8,
385                                       0.67, PositionAngle.TRUE,
386                                       FramesFactory.getEME2000(), date, mu);
387 
388         position = pCirEqua.getPVCoordinates().getPosition();
389         velocity = pCirEqua.getPVCoordinates().getVelocity();
390 
391         momentum = Vector3D.crossProduct(position, velocity).normalize();
392 
393         apogeeRadius = pCirEqua.getA() * (1 + pCirEqua.getE());
394         perigeeRadius = pCirEqua.getA() * (1 - pCirEqua.getE());
395         // test if apogee equals perigee
396         Assert.assertEquals(perigeeRadius, apogeeRadius, 1.e+4 * Utils.epsilonTest
397                      * apogeeRadius);
398 
399         for (double lv = 0; lv <= 2 * FastMath.PI; lv += 2 * FastMath.PI / 100.) {
400             pCirEqua = new EquinoctialOrbit(pCirEqua.getA() ,pCirEqua.getEquinoctialEx(),
401                                                  pCirEqua.getEquinoctialEy() , pCirEqua.getHx(), pCirEqua.getHy() , lv , PositionAngle.TRUE,
402                                                  pCirEqua.getFrame(), p.getDate(), p.getMu());
403             position = pCirEqua.getPVCoordinates().getPosition();
404 
405             // test if the norm pf the position is in the range [perigee radius,
406             // apogee radius]
407             Assert.assertTrue((position.getNorm() - apogeeRadius) <= (apogeeRadius * Utils.epsilonTest));
408             Assert.assertTrue((position.getNorm() - perigeeRadius) >= (-perigeeRadius * Utils.epsilonTest));
409 
410             position = position.normalize();
411             velocity = pCirEqua.getPVCoordinates().getVelocity();
412             velocity = velocity.normalize();
413 
414             // at this stage of computation, all the vectors (position, velocity and
415             // momemtum) are normalized here
416 
417             // test of orthogonality between position and momentum
418             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(position, momentum)) < Utils.epsilonTest);
419             // test of orthogonality between velocity and momentum
420             Assert.assertTrue(FastMath.abs(Vector3D.dotProduct(velocity, momentum)) < Utils.epsilonTest);
421         }

425     @Test
426     public void testRadiusOfCurvature() {
427 
428         // elliptic and non equatorial (i retrograde) orbit
429         EquinoctialOrbit p =
430             new EquinoctialOrbit(42166.712, 0.5, -0.5, 1.200, 2.1,
431                                       0.67, PositionAngle.TRUE,
432                                       FramesFactory.getEME2000(), date, mu);
433 
434         // arbitrary orthogonal vectors in the orbital plane
435         Vector3D u = p.getPVCoordinates().getMomentum().orthogonal();
436         Vector3D v = Vector3D.crossProduct(p.getPVCoordinates().getMomentum(), u).normalize();
437 
438         // compute radius of curvature in the orbital plane from Cartesian coordinates
439         double xDot    = Vector3D.dotProduct(p.getPVCoordinates().getVelocity(),     u);
440         double yDot    = Vector3D.dotProduct(p.getPVCoordinates().getVelocity(),     v);
441         double xDotDot = Vector3D.dotProduct(p.getPVCoordinates().getAcceleration(), u);
442         double yDotDot = Vector3D.dotProduct(p.getPVCoordinates().getAcceleration(), v);
443         double dot2    = xDot * xDot + yDot * yDot;
444         double rCart   = dot2 * FastMath.sqrt(dot2) /
445                          FastMath.abs(xDot * yDotDot - yDot * xDotDot);
446 
447         // compute radius of curvature in the orbital plane from orbital parameters
448         double ex   = p.getEquinoctialEx();
449         double ey   = p.getEquinoctialEy();
450         double f    = ex * FastMath.cos(p.getLE()) + ey * FastMath.sin(p.getLE());
451         double oMf2 = 1 - f * f;
452         double rOrb = p.getA() * oMf2 * FastMath.sqrt(oMf2 / (1 - (ex * ex + ey * ey)));
453 
454         // both methods to compute radius of curvature should match
455         Assert.assertEquals(rCart, rOrb, 1.0e-15 * p.getA());
456 
457         // at this place for such an eccentric orbit,
458         // the radius of curvature is much smaller than semi major axis
459         Assert.assertEquals(0.8477 * p.getA(), rCart, 1.0e-4 * p.getA());
460 
461     }

510                                           FastMath.toRadians(40.), PositionAngle.MEAN,
511                                           FramesFactory.getEME2000(), dateTca, mu);
512 
513         // the following reference values have been computed using the free software
514         // version 6.2 of the MSLIB fortran library by the following program:
515         //         program equ_jacobian
516         //
517         //         use mslib
518         //         implicit none
519         //
520         //         integer, parameter :: nb = 11
521         //         integer :: i,j
522         //         type(tm_code_retour)      ::  code_retour
523         //
524         //         real(pm_reel), parameter :: mu= 3.986004415e+14_pm_reel
525         //         real(pm_reel),dimension(3)::vit_car,pos_car
526         //         type(tm_orb_cir_equa)::cir_equa
527         //         real(pm_reel), dimension(6,6)::jacob
528         //         real(pm_reel)::norme,hx,hy,f,dix,diy
529         //         intrinsic sqrt
530         //
531         //         cir_equa%a=7000000_pm_reel
532         //         cir_equa%ex=0.01_pm_reel
533         //         cir_equa%ey=-0.02_pm_reel
534         //
535         //         ! mslib cir-equ parameters use ix = 2 sin(i/2) cos(gom) and iy = 2 sin(i/2) sin(gom)
536         //         ! equinoctial parameters use hx = tan(i/2) cos(gom) and hy = tan(i/2) sin(gom)
537         //         ! the conversions between these parameters and their differentials can be computed
538         //         ! from the ratio f = 2cos(i/2) which can be found either from (ix, iy) or (hx, hy):
539         //         !   f = sqrt(4 - ix^2 - iy^2) =  2 / sqrt(1 + hx^2 + hy^2)
540         //         !  hx = ix / f,  hy = iy / f
541         //         !  ix = hx * f, iy = hy *f
542         //         ! dhx = ((1 + hx^2) / f) dix + (hx hy / f) diy, dhy = (hx hy / f) dix + ((1 + hy^2) /f) diy
543         //         ! dix = ((1 - ix^2 / 4) f dhx - (ix iy / 4) f dhy, diy = -(ix iy / 4) f dhx + (1 - iy^2 / 4) f dhy
544         //         hx=1.2_pm_reel
545         //         hy=2.1_pm_reel
546         //         f=2_pm_reel/sqrt(1+hx*hx+hy*hy)
547         //         cir_equa%ix=hx*f
548         //         cir_equa%iy=hy*f
549         //
550         //         cir_equa%pso_M=40_pm_reel*pm_deg_rad
551         //
552         //         call mv_cir_equa_car(mu,cir_equa,pos_car,vit_car,code_retour)
553         //         write(*,*)code_retour%valeur
554         //         write(*,1000)pos_car,vit_car
555         //
556         //
557         //         call mu_norme(pos_car,norme,code_retour)
558         //         write(*,*)norme
559         //
560         //         call mv_car_cir_equa (mu, pos_car, vit_car, cir_equa, code_retour, jacob)
561         //         write(*,*)code_retour%valeur
562         //
563         //         f=sqrt(4_pm_reel-cir_equa%ix*cir_equa%ix-cir_equa%iy*cir_equa%iy)
564         //         hx=cir_equa%ix/f
565         //         hy=cir_equa%iy/f
566         //         write(*,*)"ix = ", cir_equa%ix, ", iy = ", cir_equa%iy

568         //
569         //         do j = 1,6

576         //         do i = 1,6

582         Vector3D pRef = new Vector3D(2004367.298657628707588, 6575317.978060320019722, -1518024.843913963763043);

586             {  9.52434720041122055E-008,  9.49704503778007296E-008,  4.46607520107935678E-008,  1.69704446323098610E-004,  7.05603505855828105E-005,  1.14825140460141970E-004 },

588             {  1.96680305426455816E-007, -1.12388745957974467E-007, -2.27118924123407353E-007,  2.06472886488132167E-004, -1.17984506564646906E-004, -2.38427023682723818E-004 },

636     private double[][] finiteDifferencesJacobian(PositionAngle type, EquinoctialOrbit orbit, double hP)
637         throws OrekitException {
638         double[][] jacobian = new double[6][6];
639         for (int i = 0; i < 6; ++i) {
640             fillColumn(type, i, orbit, hP, jacobian);
641         }
642         return jacobian;

645     private void fillColumn(PositionAngle type, int i, EquinoctialOrbit orbit, double hP, double[][] jacobian) {
646 
647         // at constant energy (i.e. constant semi major axis), we have dV = -mu dP / (V * r^2)
648         // we use this to compute a velocity step size from the position step size
649         Vector3D p = orbit.getPVCoordinates().getPosition();
650         Vector3D v = orbit.getPVCoordinates().getVelocity();
651         double hV = orbit.getMu() * hP / (v.getNorm() * p.getNormSq());
652 
653         double h;
654         Vector3D dP = Vector3D.ZERO;
655         Vector3D dV = Vector3D.ZERO;
656         switch (i) {
657         case 0:
658             h = hP;
659             dP = new Vector3D(hP, 0, 0);
660             break;
661         case 1:
662             h = hP;
663             dP = new Vector3D(0, hP, 0);
664             break;
665         case 2:
666             h = hP;
667             dP = new Vector3D(0, 0, hP);
668             break;
669         case 3:
670             h = hV;
671             dV = new Vector3D(hV, 0, 0);
672             break;
673         case 4:
674             h = hV;
675             dV = new Vector3D(0, hV, 0);
676             break;
677         default:
678             h = hV;
679             dV = new Vector3D(0, 0, hV);
680             break;
681         }
682 
683         EquinoctialOrbit oM4h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, -4, dP), new Vector3D(1, v, -4, dV)),
684                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
685         EquinoctialOrbit oM3h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, -3, dP), new Vector3D(1, v, -3, dV)),
686                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
687         EquinoctialOrbit oM2h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, -2, dP), new Vector3D(1, v, -2, dV)),
688                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
689         EquinoctialOrbit oM1h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, -1, dP), new Vector3D(1, v, -1, dV)),
690                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
691         EquinoctialOrbit oP1h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, +1, dP), new Vector3D(1, v, +1, dV)),
692                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
693         EquinoctialOrbit oP2h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, +2, dP), new Vector3D(1, v, +2, dV)),
694                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
695         EquinoctialOrbit oP3h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, +3, dP), new Vector3D(1, v, +3, dV)),
696                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
697         EquinoctialOrbit oP4h = new EquinoctialOrbit(new PVCoordinates(new Vector3D(1, p, +4, dP), new Vector3D(1, v, +4, dV)),
698                                                      orbit.getFrame(), orbit.getDate(), orbit.getMu());
699 
700         jacobian[0][i] = (-3 * (oP4h.getA()             - oM4h.getA()) +
701                           32 * (oP3h.getA()             - oM3h.getA()) -
702                          168 * (oP2h.getA()             - oM2h.getA()) +
703                          672 * (oP1h.getA()             - oM1h.getA())) / (840 * h);
704         jacobian[1][i] = (-3 * (oP4h.getEquinoctialEx() - oM4h.getEquinoctialEx()) +
705                           32 * (oP3h.getEquinoctialEx() - oM3h.getEquinoctialEx()) -
706                          168 * (oP2h.getEquinoctialEx() - oM2h.getEquinoctialEx()) +
707                          672 * (oP1h.getEquinoctialEx() - oM1h.getEquinoctialEx())) / (840 * h);
708         jacobian[2][i] = (-3 * (oP4h.getEquinoctialEy() - oM4h.getEquinoctialEy()) +
709                           32 * (oP3h.getEquinoctialEy() - oM3h.getEquinoctialEy()) -
710                          168 * (oP2h.getEquinoctialEy() - oM2h.getEquinoctialEy()) +
711                          672 * (oP1h.getEquinoctialEy() - oM1h.getEquinoctialEy())) / (840 * h);
712         jacobian[3][i] = (-3 * (oP4h.getHx()            - oM4h.getHx()) +
713                           32 * (oP3h.getHx()            - oM3h.getHx()) -
714                          168 * (oP2h.getHx()            - oM2h.getHx()) +
715                          672 * (oP1h.getHx()            - oM1h.getHx())) / (840 * h);
716         jacobian[4][i] = (-3 * (oP4h.getHy()            - oM4h.getHy()) +
717                           32 * (oP3h.getHy()            - oM3h.getHy()) -
718                          168 * (oP2h.getHy()            - oM2h.getHy()) +
719                          672 * (oP1h.getHy()            - oM1h.getHy())) / (840 * h);
720         jacobian[5][i] = (-3 * (oP4h.getL(type)         - oM4h.getL(type)) +
721                           32 * (oP3h.getL(type)         - oM3h.getL(type)) -
722                          168 * (oP2h.getL(type)         - oM2h.getL(type)) +
723                          672 * (oP1h.getL(type)         - oM1h.getL(type))) / (840 * h);
724 
725     }

727     @Test
728     public void testInterpolation() throws OrekitException {
729 
730         final double ehMu  = 3.9860047e14;
731         final double ae  = 6.378137e6;
732         final double c20 = -1.08263e-3;
733         final double c30 = 2.54e-6;
734         final double c40 = 1.62e-6;
735         final double c50 = 2.3e-7;
736         final double c60 = -5.5e-7;
737 
738         final AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(584.);
739         final Vector3D position = new Vector3D(3220103., 69623., 6449822.);
740         final Vector3D velocity = new Vector3D(6414.7, -2006., -3180.);
741         final EquinoctialOrbit initialOrbit = new EquinoctialOrbit(new PVCoordinates(position, velocity),
742                                                                    FramesFactory.getEME2000(), date, ehMu);
743 
744         EcksteinHechlerPropagator propagator =
745                 new EcksteinHechlerPropagator(initialOrbit, ae, ehMu, c20, c30, c40, c50, c60);
746 
747         // set up a 5 points sample
748         List<Orbit> sample = new ArrayList<Orbit>();
749         for (double dt = 0; dt < 300.0; dt += 60.0) {
750             sample.add(propagator.propagate(date.shiftedBy(dt)).getOrbit());
751         }
752 
753         // well inside the sample, interpolation should be much better than Keplerian shift
754         double maxShiftError = 0;
755         double maxInterpolationError = 0;
756         for (double dt = 0; dt < 241.0; dt += 1.0) {
757             AbsoluteDate t        = initialOrbit.getDate().shiftedBy(dt);
758             Vector3D shifted      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
759             Vector3D interpolated = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
760             Vector3D propagated   = propagator.propagate(t).getPVCoordinates().getPosition();
761             maxShiftError = FastMath.max(maxShiftError, shifted.subtract(propagated).getNorm());
762             maxInterpolationError = FastMath.max(maxInterpolationError, interpolated.subtract(propagated).getNorm());
763         }
764         Assert.assertTrue(maxShiftError         > 390.0);
765         Assert.assertTrue(maxInterpolationError < 0.04);
766 
767         // slightly past sample end, interpolation should quickly increase, but remain reasonable
768         maxShiftError = 0;
769         maxInterpolationError = 0;
770         for (double dt = 240; dt < 300.0; dt += 1.0) {
771             AbsoluteDate t        = initialOrbit.getDate().shiftedBy(dt);
772             Vector3D shifted      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
773             Vector3D interpolated = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
774             Vector3D propagated   = propagator.propagate(t).getPVCoordinates().getPosition();
775             maxShiftError = FastMath.max(maxShiftError, shifted.subtract(propagated).getNorm());
776             maxInterpolationError = FastMath.max(maxInterpolationError, interpolated.subtract(propagated).getNorm());
777         }
778         Assert.assertTrue(maxShiftError         <  610.0);
779         Assert.assertTrue(maxInterpolationError < 1.3);
780 
781         // far past sample end, interpolation should become really wrong
782         // (in this test case, break even occurs at around 863 seconds, with a 3.9 km error)
783         maxShiftError = 0;
784         maxInterpolationError = 0;
785         for (double dt = 300; dt < 1000; dt += 1.0) {
786             AbsoluteDate t        = initialOrbit.getDate().shiftedBy(dt);
787             Vector3D shifted      = initialOrbit.shiftedBy(dt).getPVCoordinates().getPosition();
788             Vector3D interpolated = initialOrbit.interpolate(t, sample).getPVCoordinates().getPosition();
789             Vector3D propagated   = propagator.propagate(t).getPVCoordinates().getPosition();
790             maxShiftError = FastMath.max(maxShiftError, shifted.subtract(propagated).getNorm());
791             maxInterpolationError = FastMath.max(maxInterpolationError, interpolated.subtract(propagated).getNorm());
792         }
793         Assert.assertTrue(maxShiftError         < 5000.0);
794         Assert.assertTrue(maxInterpolationError > 8800.0);
795 
796     }
