121     @Test
122     public void testEphemerisDates() throws OrekitException {
123         //setup
124         TimeScale tai = TimeScalesFactory.getTAI();
125         AbsoluteDate initialDate = new AbsoluteDate("2015-07-01", tai);
126         AbsoluteDate startDate = new AbsoluteDate("2015-07-03", tai).shiftedBy(-0.1);
127         AbsoluteDate endDate = new AbsoluteDate("2015-07-04", tai);
128         Frame eci = FramesFactory.getGCRF();
129         KeplerianOrbit orbit = new KeplerianOrbit(
130                 600e3 + Constants.WGS84_EARTH_EQUATORIAL_RADIUS, 0, 0, 0, 0, 0,
131                 PositionAngle.TRUE, eci, initialDate, Constants.EIGEN5C_EARTH_MU);
132         double[][] tol = DSSTPropagator
133                 .tolerances(1, orbit);
134         Propagator prop = new DSSTPropagator(
135                 new DormandPrince853Integrator(0.1, 500, tol[0], tol[1]));
136         prop.resetInitialState(new SpacecraftState(new CartesianOrbit(orbit)));
137 
138         //action
139         prop.setEphemerisMode();
140         prop.propagate(startDate, endDate);
141         BoundedPropagator ephemeris = prop.getGeneratedEphemeris();
142 
143         //verify
144         TimeStampedPVCoordinates actualPV = ephemeris.getPVCoordinates(startDate, eci);
145         TimeStampedPVCoordinates expectedPV = orbit.getPVCoordinates(startDate, eci);
146         MatcherAssert.assertThat(actualPV.getPosition(),
147                 OrekitMatchers.vectorCloseTo(expectedPV.getPosition(), 1.0));
148         MatcherAssert.assertThat(actualPV.getVelocity(),
149                 OrekitMatchers.vectorCloseTo(expectedPV.getVelocity(), 1.0));
150         MatcherAssert.assertThat(ephemeris.getMinDate().durationFrom(startDate),
151                 OrekitMatchers.closeTo(0, 0));
152         MatcherAssert.assertThat(ephemeris.getMaxDate().durationFrom(endDate),
153                 OrekitMatchers.closeTo(0, 0));
154         //test date
155         AbsoluteDate date = endDate.shiftedBy(-0.11);
156         Assert.assertEquals(
157                 ephemeris.propagate(date).getDate().durationFrom(date), 0, 0);
158     }

234     @Test
235     public void testImpulseManeuver() throws OrekitException {
236         final Orbit initialOrbit = new KeplerianOrbit(24532000.0, 0.72, 0.3, FastMath.PI, 0.4, 2.0, PositionAngle.MEAN, FramesFactory.getEME2000(), new AbsoluteDate(new DateComponents(2008, 06, 23), new TimeComponents(14, 18, 37), TimeScalesFactory.getUTC()), 3.986004415e14);
237         final double a = initialOrbit.getA();
238         final double e = initialOrbit.getE();
239         final double i = initialOrbit.getI();
240         final double mu = initialOrbit.getMu();
241         final double vApo = FastMath.sqrt(mu * (1 - e) / (a * (1 + e)));
242         double dv = 0.99 * FastMath.tan(i) * vApo;
243 
244         // Set propagator with state
245         setDSSTProp(new SpacecraftState(initialOrbit));
246 
247         // Add impulse maneuver
248         dsstProp.setAttitudeProvider(new LofOffset(initialOrbit.getFrame(), LOFType.VVLH));
249         dsstProp.addEventDetector(new ImpulseManeuver<NodeDetector>(new NodeDetector(initialOrbit, FramesFactory.getEME2000()), new Vector3D(dv, Vector3D.PLUS_J), 400.0));
250         SpacecraftState propagated = dsstProp.propagate(initialOrbit.getDate().shiftedBy(8000));
251 
252         Assert.assertEquals(0.0028257, propagated.getI(), 1.0e-6);
253     }

446 
447         // Central Body geopotential 2x0
448         final UnnormalizedSphericalHarmonicsProvider provider = GravityFieldFactory.getUnnormalizedProvider(2, 0);
449         DSSTForceModel zonal    = new DSSTZonal(provider, 2, 1, 5);
450         DSSTForceModel tesseral = new DSSTTesseral(CelestialBodyFactory.getEarth().getBodyOrientedFrame(),
451                                                    Constants.WGS84_EARTH_ANGULAR_VELOCITY,
452                                                    provider, 2, 0, 0, 2, 2, 0, 0);
453

457

459         final AbsoluteDate initDate = new AbsoluteDate(2003, 9, 16, 0, 0, 00.000,

469                                                initDate,

471
