273 	@Deprecated
274 	public void testServletContextPropertyPlaceholderConfigurerWithAttributes() {
275 		MockServletContext sc = new MockServletContext();
276 		sc.addInitParameter("key4", "mykey4");
277 
278 		StaticWebApplicationContext wac = new StaticWebApplicationContext();
279 		wac.setServletContext(sc);
280 
281 		MutablePropertyValues pvs = new MutablePropertyValues();
282 		pvs.add("age", "${age}");
283 		pvs.add("name", "name${var}${var}${");
284 		pvs.add("spouse", new RuntimeBeanReference("${ref}"));
285 		wac.registerSingleton("tb1", TestBean.class, pvs);
286 
287 		ConstructorArgumentValues cas = new ConstructorArgumentValues();
288 		cas.addIndexedArgumentValue(1, "${age}");
289 		cas.addGenericArgumentValue("${var}name${age}");
290 
291 		pvs = new MutablePropertyValues();
292 		List<Object> friends = new ManagedList<Object>();
293 		friends.add("na${age}me");
294 		friends.add(new RuntimeBeanReference("${ref}"));
295 		pvs.add("friends", friends);
296 
297 		Set<Object> someSet = new ManagedSet<Object>();
298 		someSet.add("na${age}me");
299 		someSet.add(new RuntimeBeanReference("${ref}"));
300 		pvs.add("someSet", someSet);
301 
302 		Map<String, Object> someMap = new ManagedMap<String, Object>();
303 		someMap.put("key1", new RuntimeBeanReference("${ref}"));
304 		someMap.put("key2", "${age}name");
305 		MutablePropertyValues innerPvs = new MutablePropertyValues();
306 		innerPvs.add("touchy", "${os.name}");
307 		RootBeanDefinition innerBd = new RootBeanDefinition(TestBean.class);
308 		innerBd.setPropertyValues(innerPvs);
309 		someMap.put("key3", innerBd);
310 		MutablePropertyValues innerPvs2 = new MutablePropertyValues(innerPvs);
311 		someMap.put("${key4}", new BeanDefinitionHolder(new ChildBeanDefinition("tb1", innerPvs2), "child"));
312 		pvs.add("someMap", someMap);
313 
314 		RootBeanDefinition bd = new RootBeanDefinition(TestBean.class, cas, pvs);
315 		wac.getDefaultListableBeanFactory().registerBeanDefinition("tb2", bd);
316 
317 		pvs = new MutablePropertyValues();
318 		pvs.add("properties", "var=${m}var\nref=tb2\nm=my");
319 		pvs.add("searchContextAttributes", Boolean.TRUE);
320 		wac.registerSingleton("configurer", ServletContextPropertyPlaceholderConfigurer.class, pvs);
321 		sc.setAttribute("age", new Integer(98));
322 
323 		wac.refresh();
324 
325 		TestBean tb1 = (TestBean) wac.getBean("tb1");
326 		TestBean tb2 = (TestBean) wac.getBean("tb2");
327 		assertEquals(98, tb1.getAge());
328 		assertEquals(98, tb2.getAge());
329 		assertEquals("namemyvarmyvar${", tb1.getName());
330 		assertEquals("myvarname98", tb2.getName());
331 		assertEquals(tb2, tb1.getSpouse());
332 		assertEquals(2, tb2.getFriends().size());
333 		assertEquals("na98me", tb2.getFriends().iterator().next());
334 		assertEquals(tb2, tb2.getFriends().toArray()[1]);
335 		assertEquals(2, tb2.getSomeSet().size());
336 		assertTrue(tb2.getSomeSet().contains("na98me"));
337 		assertTrue(tb2.getSomeSet().contains(tb2));
338 		assertEquals(4, tb2.getSomeMap().size());
339 		assertEquals(tb2, tb2.getSomeMap().get("key1"));
340 		assertEquals("98name", tb2.getSomeMap().get("key2"));
341 		TestBean inner1 = (TestBean) tb2.getSomeMap().get("key3");
342 		TestBean inner2 = (TestBean) tb2.getSomeMap().get("mykey4");
343 		assertEquals(0, inner1.getAge());
344 		assertEquals(null, inner1.getName());
345 		assertEquals(System.getProperty("os.name"), inner1.getTouchy());
346 		assertEquals(98, inner2.getAge());
347 		assertEquals("namemyvarmyvar${", inner2.getName());
348 		assertEquals(System.getProperty("os.name"), inner2.getTouchy());
349 	}

364 		paths.add("/WEB-INF/context1.xml");
365 		paths.add("/WEB-INF/context2.xml");
366 
367 		MockServletContext sc = new MockServletContext("classpath:org/springframework/web/context") {
368 			@Override
369 			public Set<String> getResourcePaths(String path) {
370 				if ("/WEB-INF/".equals(path)) {
371 					return paths;
372 				}
373 				return null;
374 			}
375 		};
376 
377 		ServletContextResourcePatternResolver rpr = new ServletContextResourcePatternResolver(sc);
378 		Resource[] found = rpr.getResources("/WEB-INF/*.xml");
379 		Set<String> foundPaths = new HashSet<String>();
380 		for (int i = 0; i < found.length; i++) {
381 			foundPaths.add(((ServletContextResource) found[i]).getPath());
382 		}
383 		assertEquals(2, foundPaths.size());

385 		assertTrue(foundPaths.contains("/WEB-INF/context2.xml"));
