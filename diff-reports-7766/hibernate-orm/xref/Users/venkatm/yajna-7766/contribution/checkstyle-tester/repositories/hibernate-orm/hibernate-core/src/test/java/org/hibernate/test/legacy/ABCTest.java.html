44  	@Test
45  	public void testFormulaAssociation() throws Throwable {
46  		Session s = openSession();
47  		Transaction t = s.beginTransaction();
48  		Long did = Long.valueOf(12);
49  		D d = new D( did );
50  		s.save(d);
51  		A a = new A();
52  		a.setName("a");
53  		s.save( a );
54  		d.setReverse( a );
55  		d.inverse = a;
56  		t.commit();
57  		s.close();
58  		
59  		s = openSession();
60  		t = s.beginTransaction();
61  		d = (D) s.get(D.class, did);
62  		assertNotNull( d.getReverse() );
63  		s.clear();
64  		sessionFactory().getCache().evictEntityRegion( D.class );
65  		sessionFactory().getCache().evictEntityRegion(A.class);
66  		d = (D) s.get(D.class, did);
67  		assertNotNull( d.inverse );
68  		assertTrue(d.inverse.getName().equals("a"));
69  		s.clear();
70  		sessionFactory().getCache().evictEntityRegion( D.class );
71  		sessionFactory().getCache().evictEntityRegion( A.class );
72  		assertTrue( s.createQuery( "from D d join d.reverse r join d.inverse i where i = r" ).list().size()==1 );
73  		t.commit();
74  		s.close();
75  	}

91  	@Test
92  	public void testSubclassing() throws Exception {
93  		Session s = openSession();
94  		Transaction t = s.beginTransaction();
95  		C1 c1 = new C1();
96  		D d = new D();
97  		d.setAmount(213.34f);
98  		c1.setAddress("foo bar");
99  		c1.setCount(23432);
100 		c1.setName("c1");
101 		c1.setBName("a funny name");
102 		c1.setD(d);
103 		s.save(c1);
104 		d.setId( c1.getId() );
105 		s.save(d);
106 
107 		assertTrue( s.createQuery( "from C2 c where 1=1 or 1=1" ).list().size()==0 );
108 
109 		t.commit();
110 		s.close();
111 
112 		sessionFactory().getCache().evictEntityRegion( A.class );
113 		
114 		s = openSession();
115 		t = s.beginTransaction();
116 		c1 = (C1) s.get( A.class, c1.getId() );
117 		assertTrue(
118 			c1.getAddress().equals("foo bar") &&
119 			(c1.getCount()==23432) &&
120 			c1.getName().equals("c1") &&
121 			c1.getD().getAmount()>213.3f
122 		);
123 		assertEquals( "a funny name", c1.getBName() );
124 		t.commit();
125 		s.close();
126 		
127 		sessionFactory().getCache().evictEntityRegion( A.class );
128 
129 		s = openSession();
130 		t = s.beginTransaction();
131 		c1 = (C1) s.get( B.class, c1.getId() );
132 		assertTrue(
133 			c1.getAddress().equals("foo bar") &&
134 			(c1.getCount()==23432) &&
135 			c1.getName().equals("c1") &&
136 			c1.getD().getAmount()>213.3f
137 		);
138 		assertEquals( "a funny name", c1.getBName() );
139 		t.commit();
140 		s.close();
141 
142 		s = openSession();
143 		t = s.beginTransaction();
144 		c1 = (C1) s.load( C1.class, c1.getId() );
145 		assertTrue(
146 			c1.getAddress().equals("foo bar") &&
147 			(c1.getCount()==23432) &&
148 			c1.getName().equals("c1") &&
149 			c1.getD().getAmount()>213.3f
150 		);
151 		t.commit();
152 		s.close();
153 
154 		s = openSession();
155 		t = s.beginTransaction();
156 		List bs = s.createQuery("from B").list();
157 		for (int i=0; i<bs.size(); i++) {
158 			C1 b = (C1) bs.get(i);
159 			s.delete(b);
160 			s.delete( b.getD() );
161 		}
162 		t.commit();
163 		s.close();
164 	}
