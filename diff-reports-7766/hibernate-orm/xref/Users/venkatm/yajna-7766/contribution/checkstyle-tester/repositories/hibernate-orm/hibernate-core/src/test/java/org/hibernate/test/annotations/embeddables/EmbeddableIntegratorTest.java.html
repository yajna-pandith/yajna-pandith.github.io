54  @RequiresDialect(H2Dialect.class)
55  public class EmbeddableIntegratorTest extends BaseUnitTestCase {
56  
57  	/**
58  	 * Throws a mapping exception because DollarValue is not mapped
59  	 */
60  	@Test(expected=GenericJDBCException.class)
61  	public void testWithoutIntegrator() {
62  		
63  		ServiceRegistry reg = new ServiceRegistryBuilder(new BootstrapServiceRegistryImpl())
64  		.buildServiceRegistry();
65  		
66  		SessionFactory sf = new Configuration()
67  		.addAnnotatedClass( Investor.class )
68  
69  		.buildSessionFactory(reg);
70  		
71  		Session sess = sf.openSession();
72  		Investor myInv = getInvestor();
73  		myInv.setId(1L);
74  		
75  		sess.save(myInv);
76  		sess.flush();
77  		sess.clear();
78  		
79  		Investor inv = (Investor) sess.get(Investor.class, 1L);
80  		assertEquals(new BigDecimal("100"), inv.getInvestments().get(0).getAmount().getAmount());
81  		
82  		sess.close();
83  	}
84  
85  	@Test
86  	public void testWithIntegrator() {
87  		
88  		LinkedHashSet<Integrator> providedIntegrators = new LinkedHashSet<Integrator>();
89  		providedIntegrators.add(new InvestorIntegrator());
90  		ClassLoaderService classLoaderService = new ClassLoaderServiceImpl();
91  		IntegratorService integratorService = new IntegratorServiceImpl(providedIntegrators, classLoaderService);
92  		
93  		ServiceRegistry reg = new ServiceRegistryBuilder(new BootstrapServiceRegistryImpl(
94  				classLoaderService,
95  				integratorService)).buildServiceRegistry();
96  		
97  		SessionFactory sf = new Configuration()
98  		.addAnnotatedClass( Investor.class )
99  
100 		.setProperty("hibernate.hbm2ddl.auto", "create-drop")
101 		.buildSessionFactory(reg);
102 		
103 		Session sess = sf.openSession();
104 		Investor myInv = getInvestor();
105 		myInv.setId(2L);
106 		
107 		sess.save(myInv);
108 		sess.flush();
109 		sess.clear();
110 		
111 		Investor inv = (Investor) sess.get(Investor.class, 2L);
112 		assertEquals(new BigDecimal("100"), inv.getInvestments().get(0).getAmount().getAmount());
113 		
114 		sess.close();
115 	}
116 	
117 	private Investor getInvestor() {
118 		Investor i = new Investor();
119 		List<Investment> investments = new ArrayList<Investment>();
120 		Investment i1 = new Investment();
121 		i1.setAmount(new DollarValue(new BigDecimal("100")));
122 		i1.setDate(new MyDate(new Date()));
123 		i1.setDescription("Test Investment");
124 		investments.add(i1);
125 		i.setInvestments(investments);
126 		
127 		return i;
128 	}
