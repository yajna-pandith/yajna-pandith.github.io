35  import static org.junit.Assert.assertFalse;
36  import static org.junit.Assert.assertNotNull;

39  /**
40   * @author Gavin King
41   */
42  public class InterfaceProxyTest extends BaseCoreFunctionalTestCase {
43  	@Override
44  	public String[] getMappings() {
45  		return new String[] { "interfaceproxy/Item.hbm.xml" };
46  	}
47  
48  	@Override
49  	public String getCacheConcurrencyStrategy() {
50  		return null;
51  	}
52  
53  	@Test
54  	@RequiresDialectFeature(
55  			value = DialectChecks.SupportsExpectedLobUsagePattern.class,
56  			comment = "database/driver does not support expected LOB usage pattern"
57  	)
58  	public void testInterfaceProxies() {
59  		Session s = openSession( new DocumentInterceptor() );
60  		Transaction t = s.beginTransaction();
61  		Document d = new DocumentImpl();
62  		d.setName("Hibernate in Action");
63  		d.setContent( s.getLobHelper().createBlob( "blah blah blah".getBytes() ) );
64  		Long did = (Long) s.save(d);
65  		SecureDocument d2 = new SecureDocumentImpl();
66  		d2.setName("Secret");
67  		d2.setContent( s.getLobHelper().createBlob( "wxyz wxyz".getBytes() ) );
68  		// SybaseASE15Dialect only allows 7-bits in a byte to be inserted into a tinyint 
69  		// column (0 <= val < 128)		
70  		d2.setPermissionBits( (byte) 127 );
71  		d2.setOwner("gavin");
72  		Long d2id = (Long) s.save(d2);
73  		t.commit();
74  		s.close();
75  
76  		s = openSession( new DocumentInterceptor() );
77  		t = s.beginTransaction();
78  		d = (Document) s.load(ItemImpl.class, did);
79  		assertEquals( did, d.getId() );
80  		assertEquals( "Hibernate in Action", d.getName() );
81  		assertNotNull( d.getContent() );
82  		
83  		d2 = (SecureDocument) s.load(ItemImpl.class, d2id);
84  		assertEquals( d2id, d2.getId() );
85  		assertEquals( "Secret", d2.getName() );
86  		assertNotNull( d2.getContent() );
87  		
88  		s.clear();
89  		
90  		d = (Document) s.load(DocumentImpl.class, did);
91  		assertEquals( did, d.getId() );
92  		assertEquals( "Hibernate in Action", d.getName() );
93  		assertNotNull( d.getContent() );
94  		
95  		d2 = (SecureDocument) s.load(SecureDocumentImpl.class, d2id);
96  		assertEquals( d2id, d2.getId() );
97  		assertEquals( "Secret", d2.getName() );
98  		assertNotNull( d2.getContent() );
99  		assertEquals( "gavin", d2.getOwner() );
100 		
101 		//s.clear();
102 		
103 		d2 = (SecureDocument) s.load(SecureDocumentImpl.class, did);
104 		assertEquals( did, d2.getId() );
105 		assertEquals( "Hibernate in Action", d2.getName() );
106 		assertNotNull( d2.getContent() );
107 		
108 		try {
109 			d2.getOwner(); //CCE
110 			assertFalse(true);
111 		}
112 		catch (ClassCastException cce) {
113 			//correct
114 		}
115 
116 		s.createQuery( "delete ItemImpl" ).executeUpdate();
117 		t.commit();
118 		s.close();
119 	}
120 }
