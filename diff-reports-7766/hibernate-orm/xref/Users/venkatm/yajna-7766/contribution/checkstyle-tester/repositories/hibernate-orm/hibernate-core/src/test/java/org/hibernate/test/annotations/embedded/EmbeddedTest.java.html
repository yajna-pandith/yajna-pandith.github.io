211 		shortSwap.setFixedLeg( shortFixed );
212 		shortSwap.setFloatLeg( shortFloating );
213 		// Create medium swap
214 		Swap swap = new Swap();
215 		swap.setTenor( 7 );

217 		fixed.setPaymentFrequency( Frequency.MONTHLY );
218 		fixed.setRate( 7.6 );

222 		floating.setRateSpread( 0.8 );
223 		swap.setFixedLeg( fixed );
224 		swap.setFloatLeg( floating );
225 		// Create long swap
226 		Swap longSwap = new Swap();
227 		longSwap.setTenor( 7 );
228 		FixedLeg longFixed = new FixedLeg();
229 		longFixed.setPaymentFrequency( Frequency.MONTHLY );
230 		longFixed.setRate( 7.6 );
231 		FloatLeg longFloating = new FloatLeg();
232 		longFloating.setPaymentFrequency( Frequency.MONTHLY );
233 		longFloating.setRateIndex( RateIndex.TIBOR );
234 		longFloating.setRateSpread( 0.8 );
235 		longSwap.setFixedLeg( longFixed );
236 		longSwap.setFloatLeg( longFloating );
237 		// Compose a curve spread deal
238 		SpreadDeal deal = new SpreadDeal();
239 		deal.setId( "FX45632" );
240 		deal.setNotional( 450000.0 );
241 		deal.setShortSwap( shortSwap );
242 		deal.setSwap( swap );
243 		deal.setLongSwap( longSwap );
244 		s.persist( deal );
245

261 		assertNotNull( "Fixed leg is null.", deal.getSwap().getFixedLeg() );
262 		assertNotNull( "Floating leg is null.", deal.getSwap().getFloatLeg() );
