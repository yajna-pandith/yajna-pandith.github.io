73  	@Override
74  	protected void cleanupTestData() throws Exception {
75  		Session s = openSession();
76  		s.beginTransaction();
77  		List<Alias> aliases = s.createQuery( "from Alias" ).list();
78  		for ( Alias alias : aliases ) {
79  			for ( Character character : alias.getCharacters() ) {
80  				character.getAliases().clear();
81  			}
82  			alias.getCharacters().clear();
83  		}
84  		s.flush();
85  		s.createQuery( "delete Alias" ).executeUpdate();
86  		s.createQuery( "delete Character" ).executeUpdate();
87  		s.getTransaction().commit();
88  		s.close();
89  	}

216 	protected void checkListener(
217 			int eventIndex,
218 			Class<? extends AbstractCollectionEvent> expectedEventType,
219 			Identifiable expectedOwner,
220 			List<? extends Identifiable> expectedCollectionEntrySnapshot) {
221 		final AggregatedCollectionEventListener.EventEntry eventEntry
222 				= collectionListenerIntegrator.getListener().getEventEntryList().get( eventIndex );
223 		final AbstractCollectionEvent event = eventEntry.getEvent();
224 
225 		assertTyping( expectedEventType, event );
226 
227 // because of the merge graphs, the instances are likely different.  just base check on type and id
228 //		assertEquals( expectedOwner, event.getAffectedOwnerOrNull() );
229 		assertEquals( expectedOwner.getClass().getName(), event.getAffectedOwnerEntityName() );
230 		assertEquals( expectedOwner.getId(), event.getAffectedOwnerIdOrNull() );
231 
232 		if ( event instanceof PreCollectionUpdateEvent
233 				|| event instanceof PreCollectionRemoveEvent
234 				|| event instanceof PostCollectionRecreateEvent ) {
235 			List<Identifiable> snapshot = (List) eventEntry.getSnapshotAtTimeOfEventHandling();
236 			assertEquals( expectedCollectionEntrySnapshot.size(), snapshot.size() );
237 			for ( int i = 0; i < expectedCollectionEntrySnapshot.size(); i++ ) {
238 				Identifiable expected = expectedCollectionEntrySnapshot.get( i );
239 				Identifiable found = snapshot.get( i );
240 				assertEquals( expected.getClass().getName(), found.getClass().getName() );
241 				assertEquals( expected.getId(), found.getId() );
242 			}
243 		}
244 	}

246 	private <T> List<T> copy(List<T> source) {
247 		ArrayList<T> copy = new ArrayList<T>( source.size() );
248 		copy.addAll( source );
249 		return copy;
