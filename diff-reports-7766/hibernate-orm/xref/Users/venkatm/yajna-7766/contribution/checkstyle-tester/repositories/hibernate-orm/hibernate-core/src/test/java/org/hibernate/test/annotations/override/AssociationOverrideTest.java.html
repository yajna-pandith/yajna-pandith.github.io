28  		Trip trip = new Trip();
29  		trip.setFrom( paris );
30  		//trip.setTo( atlanta );
31  		Session s = openSession();
32  		Transaction tx = s.beginTransaction();
33  		s.persist( paris );
34  		s.persist( atlanta );
35  		try {
36  			s.persist( trip );
37  			s.flush();
38  			fail( "Should be non nullable" );
39  		}
40  		catch (HibernateException e) {
41  			//success
42  		}
43  		finally {
44  			tx.rollback();
45  			s.close();
46  		}
47  	}

49  	@Test
50  	public void testDottedNotation() throws Exception {
51  		assertTrue( SchemaUtil.isTablePresent( "Employee", configuration() ) );
52  		assertTrue( "Overridden @JoinColumn fails",
53  				SchemaUtil.isColumnPresent( "Employee", "fld_address_fk", configuration() ) );
54  
55  		assertTrue( "Overridden @JoinTable name fails", SchemaUtil.isTablePresent( "tbl_empl_sites", configuration() ) );
56  		assertTrue( "Overridden @JoinTable with default @JoinColumn fails",
57  				SchemaUtil.isColumnPresent( "tbl_empl_sites", "employee_id", configuration() ) );
58  		assertTrue( "Overridden @JoinTable.inverseJoinColumn fails",
59  				SchemaUtil.isColumnPresent( "tbl_empl_sites", "to_website_fk", configuration() ) );
60  
61  		Session s = openSession();
62  		Transaction tx = s.beginTransaction();
63  		ContactInfo ci = new ContactInfo();
64  		Addr address = new Addr();
65  		address.setCity("Boston");
66  		address.setCountry("USA");
67  		address.setState("MA");
68  		address.setStreet("27 School Street");
69  		address.setZipcode("02108");
70  		ci.setAddr(address);
71  		List<PhoneNumber> phoneNumbers = new ArrayList();
72  		PhoneNumber num = new PhoneNumber();
73  		num.setNumber(5577188);
74  		Employee e = new Employee();
75  		Collection employeeList = new ArrayList();
76  		employeeList.add(e);
77  		e.setContactInfo(ci);
78  		num.setEmployees(employeeList);
79  		phoneNumbers.add(num);
80  		ci.setPhoneNumbers(phoneNumbers);
81  		SocialTouchPoints socialPoints = new SocialTouchPoints();
82  		List<SocialSite> sites = new ArrayList<SocialSite>();
83  		SocialSite site = new SocialSite();
84  		site.setEmployee(employeeList);
85  		site.setWebsite("www.jboss.org");
86  		sites.add(site);
87  		socialPoints.setWebsite(sites);
88  		ci.setSocial(socialPoints);
89  		s.persist(e);
90  		tx.commit();
91  
92  		tx = s.beginTransaction();
93  		s.clear();
94  		e = (Employee) s.get(Employee.class,e.getId());
95  		tx.commit();
96  		s.close();
97  	}
