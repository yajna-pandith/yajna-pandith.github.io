58  		NativeSQLQueryScalarReturn aliasTypeLong =  new NativeSQLQueryScalarReturn( "abc",sessionFactory().getTypeResolver().basic( "long" ) );
59  		NativeSQLQueryScalarReturn aliasTypeLongClass =  new NativeSQLQueryScalarReturn( "abc",sessionFactory().getTypeResolver().basic( Long.class.getName() ) );

63  		check( false, typeNoAlias, aliasNoType );
64  		check( false, typeNoAlias, aliasTypeInt );
65  		check( false, typeNoAlias, aliasTypeLong );
66  		check( false, typeNoAlias, aliasTypeLongClass );
67  		check( false, typeNoAlias, aliasTypeString );
68  		check( false, typeNoAlias, aliasTypeStringClass );
69  
70  		check( false, aliasNoType, aliasTypeInt );
71  		check( false, aliasNoType, aliasTypeLong );
72  		check( false, aliasNoType, aliasTypeLongClass );
73  		check( false, aliasNoType, aliasTypeString );
74  		check( false, aliasNoType, aliasTypeStringClass );
75  
76  		check( false, aliasTypeInt, aliasTypeLong );
77  		check( false, aliasTypeInt, aliasTypeLongClass );
78  		check( false, aliasTypeInt, aliasTypeString );
79  		check( false, aliasTypeInt, aliasTypeStringClass );
80  
81  		check( true, aliasTypeLong, aliasTypeLongClass );
82  		check( false, aliasTypeLong, aliasTypeString );

250 		NativeSQLQueryScalarReturn r1 = new NativeSQLQueryScalarReturn( "a",sessionFactory().getTypeResolver().basic( "int" ) );
251 		NativeSQLQueryRootReturn r2 = new NativeSQLQueryRootReturn( "a", "b", LockMode.NONE );
252 		NativeSQLQueryJoinReturn r3 = new NativeSQLQueryJoinReturn( "a", "b", "c", Collections.singletonMap( "key", "value" ), LockMode.NONE );
253 		NativeSQLQueryCollectionReturn r4 = new NativeSQLQueryCollectionReturn( "a", "b", "c", Collections.singletonMap( "key", "value" ), LockMode.NONE);
254 
255 		check( false, r1, r2 );
256 		check( false, r1, r3 );
257 		check( false, r1, r4 );
258

261 
262 		check( false, r3, r4 );

264 
265 	private void check(boolean expectedEquals, NativeSQLQueryReturn queryReturn1, NativeSQLQueryReturn queryReturn2) {
266 		if ( expectedEquals ) {
267 			assertTrue( queryReturn1.equals( queryReturn2 ) );
268 			assertTrue( queryReturn2.equals( queryReturn1 ) );
269 			assertTrue( queryReturn1.hashCode() == queryReturn2.hashCode() );
270 		}
271 		else {
272 			assertFalse( queryReturn1.equals( queryReturn2 ) );
273 			assertFalse( queryReturn2.equals( queryReturn1 ) );
274 			assertFalse( queryReturn1.hashCode() == queryReturn2.hashCode() );
275 		}
