50  /**
51   * The thing that handles actual class enhancement in regards to
52   * intercepting field accesses.

60  	private static final String EACH_READ_METHOD_PREFIX = "$javassist_read_";
61  
62  	private static final String EACH_WRITE_METHOD_PREFIX = "$javassist_write_";
63  
64  	private static final String FIELD_HANDLED_TYPE_NAME = FieldHandled.class
65  			.getName();
66  
67  	private static final String HANDLER_FIELD_NAME = "$JAVASSIST_READ_WRITE_HANDLER";
68  
69  	private static final String FIELD_HANDLER_TYPE_NAME = FieldHandler.class
70  			.getName();
71  
72  	private static final String HANDLER_FIELD_DESCRIPTOR = 'L' + FIELD_HANDLER_TYPE_NAME
73  			.replace('.', '/') + ';';

75  	private static final String GETFIELDHANDLER_METHOD_NAME = "getFieldHandler";
76  
77  	private static final String SETFIELDHANDLER_METHOD_NAME = "setFieldHandler";
78  
79  	private static final String GETFIELDHANDLER_METHOD_DESCRIPTOR = "()"
80  	                                                                + HANDLER_FIELD_DESCRIPTOR;
81  
82  	private static final String SETFIELDHANDLER_METHOD_DESCRIPTOR = "("
83  	                                                                + HANDLER_FIELD_DESCRIPTOR + ")V";

106 	public void transform(File file) throws Exception {
107 		DataInputStream in = new DataInputStream(new FileInputStream(file));
108 		ClassFile classfile = new ClassFile(in);
109 		transform(classfile);
110 		DataOutputStream out = new DataOutputStream(new FileOutputStream(file));
111 		try {
112 			classfile.write(out);
113 		} finally {
114 			out.close();
115 		}
116 	}
117 
118 	public void transform(ClassFile classfile) throws Exception {
119 		if (classfile.isInterface()) {
120 			return;
121 		}
122 		try {
123 			addFieldHandlerField(classfile);
124 			addGetFieldHandlerMethod(classfile);
125 			addSetFieldHandlerMethod(classfile);
126 			addFieldHandledInterface(classfile);
127 			addReadWriteMethods(classfile);
128 			transformInvokevirtualsIntoPutAndGetfields(classfile);
129 		} catch (CannotCompileException e) {
130 			throw new RuntimeException(e.getMessage(), e);
131 		}
132 	}
133 
134 	private void addFieldHandlerField(ClassFile classfile)
135 			throws CannotCompileException {
136 		ConstPool cp = classfile.getConstPool();
137 		FieldInfo finfo = new FieldInfo(cp, HANDLER_FIELD_NAME,
138 		                                HANDLER_FIELD_DESCRIPTOR);
139 		finfo.setAccessFlags(AccessFlag.PRIVATE | AccessFlag.TRANSIENT);
140 		classfile.addField(finfo);
141 	}

143 	private void addGetFieldHandlerMethod(ClassFile classfile)
144 			throws CannotCompileException, BadBytecode {
145 		ConstPool cp = classfile.getConstPool();
146 		int this_class_index = cp.getThisClassInfo();
147 		MethodInfo minfo = new MethodInfo(cp, GETFIELDHANDLER_METHOD_NAME,
148 		                                  GETFIELDHANDLER_METHOD_DESCRIPTOR);
149 		/* local variable | this | */
150 		Bytecode code = new Bytecode(cp, 2, 1);
151 		// aload_0 // load this
152 		code.addAload(0);
153 		// getfield // get field "$JAVASSIST_CALLBACK" defined already
154 		code.addOpcode(Opcode.GETFIELD);
155 		int field_index = cp.addFieldrefInfo(this_class_index,
156 		                                     HANDLER_FIELD_NAME, HANDLER_FIELD_DESCRIPTOR);
157 		code.addIndex(field_index);
158 		// areturn // return the value of the field
159 		code.addOpcode(Opcode.ARETURN);
160 		minfo.setCodeAttribute(code.toCodeAttribute());
161 		minfo.setAccessFlags(AccessFlag.PUBLIC);
162 		CodeAttribute codeAttribute = minfo.getCodeAttribute();
163 		if (codeAttribute != null) {
164 			StackMapTable smt = MapMaker.make(classPool, minfo);
165 			codeAttribute.setAttribute(smt);
166 		}
167 		classfile.addMethod(minfo);
168 	}

170 	private void addSetFieldHandlerMethod(ClassFile classfile)
171 			throws CannotCompileException, BadBytecode {
172 		ConstPool cp = classfile.getConstPool();
173 		int this_class_index = cp.getThisClassInfo();
174 		MethodInfo minfo = new MethodInfo(cp, SETFIELDHANDLER_METHOD_NAME,
175 		                                  SETFIELDHANDLER_METHOD_DESCRIPTOR);
176 		/* local variables | this | callback | */
177 		Bytecode code = new Bytecode(cp, 3, 3);
178 		// aload_0 // load this
179 		code.addAload(0);
180 		// aload_1 // load callback
181 		code.addAload(1);
182 		// putfield // put field "$JAVASSIST_CALLBACK" defined already
183 		code.addOpcode(Opcode.PUTFIELD);
184 		int field_index = cp.addFieldrefInfo(this_class_index,
185 		                                     HANDLER_FIELD_NAME, HANDLER_FIELD_DESCRIPTOR);
186 		code.addIndex(field_index);
187 		// return
188 		code.addOpcode(Opcode.RETURN);
189 		minfo.setCodeAttribute(code.toCodeAttribute());
190 		minfo.setAccessFlags(AccessFlag.PUBLIC);
191 		CodeAttribute codeAttribute = minfo.getCodeAttribute();
192 		if (codeAttribute != null) {
193 			StackMapTable smt = MapMaker.make(classPool, minfo);
194 			codeAttribute.setAttribute(smt);
195 		}
196 		classfile.addMethod(minfo);
197 	}

199 	private void addFieldHandledInterface(ClassFile classfile) {
200 		String[] interfaceNames = classfile.getInterfaces();
201 		String[] newInterfaceNames = new String[interfaceNames.length + 1];
202 		System.arraycopy(interfaceNames, 0, newInterfaceNames, 0,
203 		                 interfaceNames.length);
204 		newInterfaceNames[newInterfaceNames.length - 1] = FIELD_HANDLED_TYPE_NAME;
205 		classfile.setInterfaces(newInterfaceNames);
206 	}

208 	private void addReadWriteMethods(ClassFile classfile)
209 			throws CannotCompileException, BadBytecode {
210 		List fields = classfile.getFields();
211 		for (Iterator field_iter = fields.iterator(); field_iter.hasNext();) {
212 			FieldInfo finfo = (FieldInfo) field_iter.next();
213 			if ((finfo.getAccessFlags() & AccessFlag.STATIC) == 0
214 			    && (!finfo.getName().equals(HANDLER_FIELD_NAME))) {
215 				// case of non-static field
216 				if (filter.handleRead(finfo.getDescriptor(), finfo
217 						.getName())) {
218 					addReadMethod(classfile, finfo);
219 				}
220 				if (filter.handleWrite(finfo.getDescriptor(), finfo
221 						.getName())) {
222 					addWriteMethod(classfile, finfo);
223 				}
224 			}
225 		}
226 	}

228 	private void addReadMethod(ClassFile classfile, FieldInfo finfo)
229 			throws CannotCompileException, BadBytecode {
230 		ConstPool cp = classfile.getConstPool();
231 		int this_class_index = cp.getThisClassInfo();
232 		String desc = "()" + finfo.getDescriptor();
233 		MethodInfo minfo = new MethodInfo(cp, EACH_READ_METHOD_PREFIX
234 		                                      + finfo.getName(), desc);
235 		/* local variables | target obj | each oldvalue | */
236 		Bytecode code = new Bytecode(cp, 5, 3);
237 		// aload_0
238 		code.addAload(0);
239 		// getfield // get each field
240 		code.addOpcode(Opcode.GETFIELD);
241 		int base_field_index = cp.addFieldrefInfo(this_class_index, finfo
242 				.getName(), finfo.getDescriptor());
243 		code.addIndex(base_field_index);
244 		// aload_0
245 		code.addAload(0);
246 		// invokeinterface // invoke Enabled.getInterceptFieldCallback()
247 		int enabled_class_index = cp.addClassInfo(FIELD_HANDLED_TYPE_NAME);
248 		code.addInvokeinterface(enabled_class_index,
249 		                        GETFIELDHANDLER_METHOD_NAME, GETFIELDHANDLER_METHOD_DESCRIPTOR,
250 		                        1);
251 		// ifnonnull
252 		code.addOpcode(Opcode.IFNONNULL);
253 		code.addIndex(4);
254 		// *return // each type
255 		addTypeDependDataReturn(code, finfo.getDescriptor());
256 		// *store_1 // each type
257 		addTypeDependDataStore(code, finfo.getDescriptor(), 1);
258 		// aload_0
259 		code.addAload(0);
260 		// invokeinterface // invoke Enabled.getInterceptFieldCallback()
261 		code.addInvokeinterface(enabled_class_index,
262 		                        GETFIELDHANDLER_METHOD_NAME, GETFIELDHANDLER_METHOD_DESCRIPTOR,
263 		                        1);
264 		// aload_0
265 		code.addAload(0);
266 		// ldc // name of the field
267 		code.addLdc(finfo.getName());
268 		// *load_1 // each type
269 		addTypeDependDataLoad(code, finfo.getDescriptor(), 1);
270 		// invokeinterface // invoke Callback.read*() // each type
271 		addInvokeFieldHandlerMethod(classfile, code, finfo.getDescriptor(),
272 		                            true);
273 		// *return // each type
274 		addTypeDependDataReturn(code, finfo.getDescriptor());
275 
276 		minfo.setCodeAttribute(code.toCodeAttribute());
277 		minfo.setAccessFlags(AccessFlag.PUBLIC);
278 		CodeAttribute codeAttribute = minfo.getCodeAttribute();
279 		if (codeAttribute != null) {
280 			StackMapTable smt = MapMaker.make(classPool, minfo);
281 			codeAttribute.setAttribute(smt);
282 		}
283 		classfile.addMethod(minfo);
284 	}

286 	private void addWriteMethod(ClassFile classfile, FieldInfo finfo)
287 			throws CannotCompileException, BadBytecode {
288 		ConstPool cp = classfile.getConstPool();
289 		int this_class_index = cp.getThisClassInfo();
290 		String desc = "(" + finfo.getDescriptor() + ")V";
291 		MethodInfo minfo = new MethodInfo(cp, EACH_WRITE_METHOD_PREFIX
292 		                                      + finfo.getName(), desc);
293 		/* local variables | target obj | each oldvalue | */
294 		Bytecode code = new Bytecode(cp, 6, 3);
295 		// aload_0
296 		code.addAload(0);
297 		// invokeinterface // enabled.getInterceptFieldCallback()
298 		int enabled_class_index = cp.addClassInfo(FIELD_HANDLED_TYPE_NAME);
299 		code.addInvokeinterface(enabled_class_index,
300 		                        GETFIELDHANDLER_METHOD_NAME, GETFIELDHANDLER_METHOD_DESCRIPTOR,
301 		                        1);
302 		// ifnonnull (label1)
303 		code.addOpcode(Opcode.IFNONNULL);
304 		code.addIndex(9);
305 		// aload_0
306 		code.addAload(0);
307 		// *load_1
308 		addTypeDependDataLoad(code, finfo.getDescriptor(), 1);
309 		// putfield
310 		code.addOpcode(Opcode.PUTFIELD);
311 		int base_field_index = cp.addFieldrefInfo(this_class_index, finfo
312 				.getName(), finfo.getDescriptor());
313 		code.addIndex(base_field_index);
314 		code.growStack(-Descriptor.dataSize(finfo.getDescriptor()));
315 		// return ;
316 		code.addOpcode(Opcode.RETURN);
317 		// aload_0
318 		code.addAload(0);
319 		// dup
320 		code.addOpcode(Opcode.DUP);
321 		// invokeinterface // enabled.getInterceptFieldCallback()
322 		code.addInvokeinterface(enabled_class_index,
323 		                        GETFIELDHANDLER_METHOD_NAME, GETFIELDHANDLER_METHOD_DESCRIPTOR,
324 		                        1);
325 		// aload_0
326 		code.addAload(0);
327 		// ldc // field name
328 		code.addLdc(finfo.getName());
329 		// aload_0
330 		code.addAload(0);
331 		// getfield // old value of the field
332 		code.addOpcode(Opcode.GETFIELD);
333 		code.addIndex(base_field_index);
334 		code.growStack(Descriptor.dataSize(finfo.getDescriptor()) - 1);
335 		// *load_1
336 		addTypeDependDataLoad(code, finfo.getDescriptor(), 1);
337 		// invokeinterface // callback.write*(..)
338 		addInvokeFieldHandlerMethod(classfile, code, finfo.getDescriptor(),
339 		                            false);
340 		// putfield // new value of the field
341 		code.addOpcode(Opcode.PUTFIELD);
342 		code.addIndex(base_field_index);
343 		code.growStack(-Descriptor.dataSize(finfo.getDescriptor()));
344 		// return
345 		code.addOpcode(Opcode.RETURN);
346 
347 		minfo.setCodeAttribute(code.toCodeAttribute());
348 		minfo.setAccessFlags(AccessFlag.PUBLIC);
349 		CodeAttribute codeAttribute = minfo.getCodeAttribute();
350 		if (codeAttribute != null) {
351 			StackMapTable smt = MapMaker.make(classPool, minfo);
352 			codeAttribute.setAttribute(smt);
353 		}
354 		classfile.addMethod(minfo);
355 	}

357 	private void transformInvokevirtualsIntoPutAndGetfields(ClassFile classfile)
358 			throws CannotCompileException, BadBytecode {
359 		List methods = classfile.getMethods();
360 		for (Iterator method_iter = methods.iterator(); method_iter.hasNext();) {
361 			MethodInfo minfo = (MethodInfo) method_iter.next();
362 			String methodName = minfo.getName();
363 			if (methodName.startsWith(EACH_READ_METHOD_PREFIX)
364 			    || methodName.startsWith(EACH_WRITE_METHOD_PREFIX)
365 			    || methodName.equals(GETFIELDHANDLER_METHOD_NAME)
366 			    || methodName.equals(SETFIELDHANDLER_METHOD_NAME)) {
367 				continue;
368 			}
369 			CodeAttribute codeAttr = minfo.getCodeAttribute();
370 			if (codeAttr == null) {
371 				continue;
372 			}
373 			CodeIterator iter = codeAttr.iterator();
374 			while (iter.hasNext()) {
375 				int pos = iter.next();
376 				pos = transformInvokevirtualsIntoGetfields(classfile, iter, pos);
377 				pos = transformInvokevirtualsIntoPutfields(classfile, iter, pos);
378 			}
379 			StackMapTable smt = MapMaker.make(classPool, minfo);
380 			codeAttr.setAttribute(smt);
381 		}
382 	}

384 	private int transformInvokevirtualsIntoGetfields(ClassFile classfile, CodeIterator iter, int pos) {
385 		ConstPool cp = classfile.getConstPool();
386 		int c = iter.byteAt(pos);
387 		if (c != Opcode.GETFIELD) {
388 			return pos;
389 		}
390 		int index = iter.u16bitAt(pos + 1);
391 		String fieldName = cp.getFieldrefName(index);
392 		String className = cp.getFieldrefClassName(index);
393 		if ( !filter.handleReadAccess( className, fieldName ) ) {
394 			return pos;
395 		}
396 		String desc = "()" + cp.getFieldrefType( index );
397 		int read_method_index = cp.addMethodrefInfo(
398 				cp.getThisClassInfo(),
399 				EACH_READ_METHOD_PREFIX + fieldName,
400 				desc
401 		);
402 		iter.writeByte(Opcode.INVOKEVIRTUAL, pos);
403 		iter.write16bit(read_method_index, pos + 1);
404 		return pos;
405 	}
406 
407 	private int transformInvokevirtualsIntoPutfields(
408 			ClassFile classfile,
409 			CodeIterator iter, int pos) {
410 		ConstPool cp = classfile.getConstPool();
411 		int c = iter.byteAt(pos);
412 		if (c != Opcode.PUTFIELD) {
413 			return pos;
414 		}
415 		int index = iter.u16bitAt(pos + 1);
416 		String fieldName = cp.getFieldrefName(index);
417 		String className = cp.getFieldrefClassName(index);
418 		if ( !filter.handleWriteAccess( className, fieldName ) ) {
419 			return pos;
420 		}
421 		String desc = "(" + cp.getFieldrefType( index ) + ")V";
422 		int write_method_index = cp.addMethodrefInfo(
423 				cp.getThisClassInfo(),
424 				EACH_WRITE_METHOD_PREFIX + fieldName,
425 				desc
426 		);
427 		iter.writeByte(Opcode.INVOKEVIRTUAL, pos);
428 		iter.write16bit(write_method_index, pos + 1);
429 		return pos;
430 	}

432 	private static void addInvokeFieldHandlerMethod(ClassFile classfile,
433 	                                                Bytecode code, String typeName, boolean isReadMethod) {
434 		ConstPool cp = classfile.getConstPool();
435 		// invokeinterface
436 		int callback_type_index = cp.addClassInfo(FIELD_HANDLER_TYPE_NAME);
437 		if ((typeName.charAt(0) == 'L')
438 		    && (typeName.charAt(typeName.length() - 1) == ';')
439 		    || (typeName.charAt(0) == '[')) {
440 			// reference type
441 			int indexOfL = typeName.indexOf('L');
442 			String type;
443 			if (indexOfL == 0) {
444 				// not array
445 				type = typeName.substring(1, typeName.length() - 1);
446 				type = type.replace('/', '.');
447 			} else if (indexOfL == -1) {
448 				// array of primitive type
449 				// do nothing
450 				type = typeName;
451 			} else {
452 				// array of reference type
453 				type = typeName.replace('/', '.');
454 			}
455 			if (isReadMethod) {
456 				code
457 						.addInvokeinterface(
458 								callback_type_index,
459 								"readObject",
460 								"(Ljava/lang/Object;Ljava/lang/String;Ljava/lang/Object;)Ljava/lang/Object;",
461 								4);
462 				// checkcast
463 				code.addCheckcast(type);
464 			} else {
465 				code
466 						.addInvokeinterface(
467 								callback_type_index,
468 								"writeObject",
469 								"(Ljava/lang/Object;Ljava/lang/String;Ljava/lang/Object;Ljava/lang/Object;)Ljava/lang/Object;",
470 								5);
471 				// checkcast
472 				code.addCheckcast(type);
473 			}
474 		} else if (typeName.equals("Z")) {
475 			// boolean
476 			if (isReadMethod) {
477 				code.addInvokeinterface(callback_type_index, "readBoolean",
478 				                        "(Ljava/lang/Object;Ljava/lang/String;Z)Z", 4);
479 			} else {
480 				code.addInvokeinterface(callback_type_index, "writeBoolean",
481 				                        "(Ljava/lang/Object;Ljava/lang/String;ZZ)Z", 5);
482 			}
483 		} else if (typeName.equals("B")) {
484 			// byte
485 			if (isReadMethod) {
486 				code.addInvokeinterface(callback_type_index, "readByte",
487 				                        "(Ljava/lang/Object;Ljava/lang/String;B)B", 4);
488 			} else {
489 				code.addInvokeinterface(callback_type_index, "writeByte",
490 				                        "(Ljava/lang/Object;Ljava/lang/String;BB)B", 5);
491 			}
492 		} else if (typeName.equals("C")) {
493 			// char
494 			if (isReadMethod) {
495 				code.addInvokeinterface(callback_type_index, "readChar",
496 				                        "(Ljava/lang/Object;Ljava/lang/String;C)C", 4);
497 			} else {
498 				code.addInvokeinterface(callback_type_index, "writeChar",
499 				                        "(Ljava/lang/Object;Ljava/lang/String;CC)C", 5);
500 			}
501 		} else if (typeName.equals("I")) {
502 			// int
503 			if (isReadMethod) {
504 				code.addInvokeinterface(callback_type_index, "readInt",
505 				                        "(Ljava/lang/Object;Ljava/lang/String;I)I", 4);
506 			} else {
507 				code.addInvokeinterface(callback_type_index, "writeInt",
508 				                        "(Ljava/lang/Object;Ljava/lang/String;II)I", 5);
509 			}
510 		} else if (typeName.equals("S")) {
511 			// short
512 			if (isReadMethod) {
513 				code.addInvokeinterface(callback_type_index, "readShort",
514 				                        "(Ljava/lang/Object;Ljava/lang/String;S)S", 4);
515 			} else {
516 				code.addInvokeinterface(callback_type_index, "writeShort",
517 				                        "(Ljava/lang/Object;Ljava/lang/String;SS)S", 5);
518 			}
519 		} else if (typeName.equals("D")) {
520 			// double
521 			if (isReadMethod) {
522 				code.addInvokeinterface(callback_type_index, "readDouble",
523 				                        "(Ljava/lang/Object;Ljava/lang/String;D)D", 5);
524 			} else {
525 				code.addInvokeinterface(callback_type_index, "writeDouble",
526 				                        "(Ljava/lang/Object;Ljava/lang/String;DD)D", 7);
527 			}
528 		} else if (typeName.equals("F")) {
529 			// float
530 			if (isReadMethod) {
531 				code.addInvokeinterface(callback_type_index, "readFloat",
532 				                        "(Ljava/lang/Object;Ljava/lang/String;F)F", 4);
533 			} else {
534 				code.addInvokeinterface(callback_type_index, "writeFloat",
535 				                        "(Ljava/lang/Object;Ljava/lang/String;FF)F", 5);
536 			}
537 		} else if (typeName.equals("J")) {
538 			// long
539 			if (isReadMethod) {
540 				code.addInvokeinterface(callback_type_index, "readLong",
541 				                        "(Ljava/lang/Object;Ljava/lang/String;J)J", 5);
542 			} else {
543 				code.addInvokeinterface(callback_type_index, "writeLong",
544 				                        "(Ljava/lang/Object;Ljava/lang/String;JJ)J", 7);
545 			}
546 		} else {
547 			// bad type
548 			throw new RuntimeException("bad type: " + typeName);
549 		}
550 	}

552 	private static void addTypeDependDataLoad(Bytecode code, String typeName,
553 	                                          int i) {
554 		if ((typeName.charAt(0) == 'L')
555 		    && (typeName.charAt(typeName.length() - 1) == ';')
556 		    || (typeName.charAt(0) == '[')) {
557 			// reference type
558 			code.addAload(i);
559 		} else if (typeName.equals("Z") || typeName.equals("B")
