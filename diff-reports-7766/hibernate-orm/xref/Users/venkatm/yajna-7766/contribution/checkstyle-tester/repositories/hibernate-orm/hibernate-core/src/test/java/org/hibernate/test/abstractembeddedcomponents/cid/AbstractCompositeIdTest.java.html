41  	@Test
42  	public void testEmbeddedCompositeIdentifierOnAbstractClass() {
43  		MyInterfaceImpl myInterface = new MyInterfaceImpl();
44  		myInterface.setKey1( "key1" );
45  		myInterface.setKey2( "key2" );
46  		myInterface.setName( "test" );
47  
48  		Session s = openSession();
49  		Transaction t = s.beginTransaction();
50  		s.save( myInterface );
51  		s.flush();
52  
53  		s.createQuery( "from MyInterface" ).list();
54  
55  		s.delete( myInterface );
56  		t.commit();
57  		s.close();
58  
59  	}
