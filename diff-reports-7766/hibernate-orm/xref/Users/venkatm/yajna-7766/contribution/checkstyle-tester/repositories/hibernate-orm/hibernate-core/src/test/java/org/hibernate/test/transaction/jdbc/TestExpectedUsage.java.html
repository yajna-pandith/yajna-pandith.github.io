24  package org.hibernate.test.transaction.jdbc;

26  import static org.junit.Assert.assertEquals;
27  import static org.junit.Assert.assertFalse;
28  import static org.junit.Assert.assertTrue;
29  import static org.junit.Assert.fail;

31  import java.sql.PreparedStatement;
32  import java.sql.SQLException;
33  import java.sql.Statement;

35  import org.hibernate.ConnectionReleaseMode;
36  import org.hibernate.engine.jdbc.spi.JdbcCoordinator;
37  import org.hibernate.engine.jdbc.spi.LogicalConnectionImplementor;
38  import org.hibernate.engine.transaction.internal.TransactionCoordinatorImpl;
39  import org.hibernate.engine.transaction.spi.TransactionContext;
40  import org.hibernate.engine.transaction.spi.TransactionImplementor;
41  import org.hibernate.service.ServiceRegistryBuilder;
42  import org.hibernate.service.internal.StandardServiceRegistryImpl;
43  import org.hibernate.test.common.JournalingTransactionObserver;
44  import org.hibernate.test.common.TransactionContextImpl;
45  import org.hibernate.test.common.TransactionEnvironmentImpl;
46  import org.hibernate.testing.env.ConnectionProviderBuilder;
47  import org.hibernate.testing.junit4.BaseUnitTestCase;
48  import org.junit.After;
49  import org.junit.Before;
50  import org.junit.Test;

52  /**
53   * @author Steve Ebersole
54   */
55  public class TestExpectedUsage extends BaseUnitTestCase {
56  	private StandardServiceRegistryImpl serviceRegistry;
57  
58  	@Before
59  	public void setUp() throws Exception {
60  		serviceRegistry = (StandardServiceRegistryImpl) new ServiceRegistryBuilder()
61  				.applySettings( ConnectionProviderBuilder.getConnectionProviderProperties() )
62  				.buildServiceRegistry();
63  	}
64  
65  	@After
66  	public void tearDown() throws Exception {
67  		serviceRegistry.destroy();
68  	}
69  
70  	@Test
71  	public void testBasicUsage() {
72  		final TransactionContext transactionContext = new TransactionContextImpl( new TransactionEnvironmentImpl( serviceRegistry ) ) {
73  			@Override
74  			public ConnectionReleaseMode getConnectionReleaseMode() {
75  				return ConnectionReleaseMode.AFTER_TRANSACTION;
76  			}
77  		};
78  
79  		TransactionCoordinatorImpl transactionCoordinator = new TransactionCoordinatorImpl( null, transactionContext );
80  		JournalingTransactionObserver observer = new JournalingTransactionObserver();
81  		transactionCoordinator.addObserver( observer );
82  
83  		JdbcCoordinator jdbcCoordinator = transactionCoordinator.getJdbcCoordinator();
84  		LogicalConnectionImplementor logicalConnection = jdbcCoordinator.getLogicalConnection();
85  
86  		// set up some tables to use
87  		Statement statement = jdbcCoordinator.getStatementPreparer().createStatement();
88  		jdbcCoordinator.getResultSetReturn().execute( statement, "drop table SANDBOX_JDBC_TST if exists" );
89  		jdbcCoordinator.getResultSetReturn().execute( statement, "create table SANDBOX_JDBC_TST ( ID integer, NAME varchar(100) )" );
90  		assertTrue( jdbcCoordinator.hasRegisteredResources() );
91  		assertTrue( logicalConnection.isPhysicallyConnected() );
92  		jdbcCoordinator.release( statement );
93  		assertFalse( jdbcCoordinator.hasRegisteredResources() );
94  		assertTrue( logicalConnection.isPhysicallyConnected() ); // after_transaction specified
95  
96  		// ok, now we can get down to it...
97  		TransactionImplementor txn = transactionCoordinator.getTransaction();  // same as Session#getTransaction
98  		txn.begin();
99  		assertEquals( 1, observer.getBegins() );
100 		try {
101 			PreparedStatement ps = jdbcCoordinator.getStatementPreparer().prepareStatement( "insert into SANDBOX_JDBC_TST( ID, NAME ) values ( ?, ? )" );
102 			ps.setLong( 1, 1 );
103 			ps.setString( 2, "name" );
104 			jdbcCoordinator.getResultSetReturn().execute( ps );
105 			assertTrue( jdbcCoordinator.hasRegisteredResources() );
106 			jdbcCoordinator.release( ps );
107 			assertFalse( jdbcCoordinator.hasRegisteredResources() );
108 
109 			ps = jdbcCoordinator.getStatementPreparer().prepareStatement( "select * from SANDBOX_JDBC_TST" );
110 			jdbcCoordinator.getResultSetReturn().extract( ps );
111 			ps = jdbcCoordinator.getStatementPreparer().prepareStatement( "delete from SANDBOX_JDBC_TST" );
112 			jdbcCoordinator.getResultSetReturn().execute( ps );
113 			// lets forget to close these...
114 			assertTrue( jdbcCoordinator.hasRegisteredResources() );
115 
116 			// and commit the transaction...
117 			txn.commit();
118 
119 			// we should now have:
120 			//		1) no resources because of after_transaction release mode
121 			assertFalse( jdbcCoordinator.hasRegisteredResources() );
122 			//		2) non-physically connected logical connection, again because of after_transaction release mode
123 			assertFalse( logicalConnection.isPhysicallyConnected() );
124 			//		3) transaction observer callbacks
125 			assertEquals( 1, observer.getBeforeCompletions() );
126 			assertEquals( 1, observer.getAfterCompletions() );
127 		}
128 		catch ( SQLException sqle ) {
129 			fail( "incorrect exception type : SQLException" );
130 		}
131 		finally {
132 			logicalConnection.close();
133 		}
134 	}
