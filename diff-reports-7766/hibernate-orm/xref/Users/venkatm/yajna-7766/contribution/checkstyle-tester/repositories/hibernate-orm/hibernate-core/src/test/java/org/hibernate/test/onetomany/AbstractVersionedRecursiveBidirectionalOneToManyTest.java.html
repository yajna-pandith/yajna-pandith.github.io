24  package org.hibernate.test.onetomany;
25  
26  import org.hibernate.Session;
27  import org.hibernate.Transaction;
28  
29  import static org.junit.Assert.assertEquals;
30  
31  /**
32   * @author Burkhard Graves
33   * @author Gail Badner
34   */
35  public abstract class AbstractVersionedRecursiveBidirectionalOneToManyTest extends AbstractRecursiveBidirectionalOneToManyTest {
36  	@Override
37  	public String[] getMappings() {
38  		return new String[] { "onetomany/VersionedNode.hbm.xml" };
39  	}
40  
41  	@Override
42  	@SuppressWarnings( {"UnnecessaryBoxing"})
43  	void check(boolean simplePropertyUpdated) {
44  		super.check( simplePropertyUpdated );
45  		Session s = openSession();
46  		Transaction tx = s.beginTransaction();
47  		Node node1 = ( Node ) s.get( Node.class, Integer.valueOf( 1 ) );
48  		Node node2 = ( Node ) s.get( Node.class, Integer.valueOf( 2 ) );
49  		Node node3 = ( Node ) s.get( Node.class, Integer.valueOf( 3 ) );
50  		assertEquals( 1, node1.getVersion() );
51  		assertEquals( 1, node2.getVersion() );
52  		assertEquals( 1, node3.getVersion() );
53  		tx.commit();
54  		s.close();
55  	}
56  }
