60  	@Test
61  	public void testDiscriminatorSubclass() {
62  		Session s = openSession();
63  		Transaction t = s.beginTransaction();
64  
65  		Employee mark = new Employee();
66  		mark.setId( 1 );
67  		mark.setName( "Mark" );
68  		mark.setTitle( "internal sales" );
69  		mark.setSex( 'M' );
70  
71  		Customer joe = new Customer();
72  		joe.setId( 2 );
73  		joe.setName( "Joe" );
74  		joe.setComments( "Very demanding" );
75  		joe.setSex( 'M' );
76  
77  		Person yomomma = new Person();
78  		yomomma.setId( 3 );
79  		yomomma.setName("mum");
80  		yomomma.setSex('F');
81  
82  		s.save(yomomma);
83  		s.save(mark);
84  		s.save(joe);
85  
86  		assertEquals( s.createQuery("from java.io.Serializable").list().size(), 0 );
87  
88  		assertEquals( s.createQuery("from org.hibernate.test.discriminator.Person").list().size(), 3 );
89  		assertEquals( s.createQuery("from org.hibernate.test.discriminator.Person p where p.class = org.hibernate.test.discriminator.Person").list().size(), 1 );
90  		assertEquals( s.createQuery("from org.hibernate.test.discriminator.Person p where p.class = org.hibernate.test.discriminator.Customer").list().size(), 1 );
91  		assertEquals( s.createQuery("from org.hibernate.test.discriminator.Person p where type(p) = :who").setParameter("who", Person.class).list().size(), 1 );
92  		assertEquals( s.createQuery("from org.hibernate.test.discriminator.Person p where type(p) in :who").setParameterList("who", new Class[] {Customer.class, Person.class}).list().size(), 2 );
93  		s.clear();
94  
95  		List customers = s.createQuery("from org.hibernate.test.discriminator.Customer").list();
96  		for ( Object customer : customers ) {
97  			Customer c = (Customer) customer;
98  			assertEquals( "Very demanding", c.getComments() );
99  		}
100 		assertEquals( customers.size(), 1 );
101 		s.clear();
102 
103 		mark = (Employee) s.get( Employee.class, mark.getId() );
104 		joe = (Customer) s.get( Customer.class, joe.getId() );
105 
106  		s.delete(mark);
107 		s.delete(joe);
108 		s.delete(yomomma);
109 		assertTrue( s.createQuery("from org.hibernate.test.discriminator.Person").list().isEmpty() );
110 		t.commit();
111 		s.close();
112 	}

191 	@Test
192 	public void testLoadSuperclassProxyPolymorphicAccess() {
193 		Session s = openSession();
194 		s.beginTransaction();
195 		Employee e = new Employee();
196 		e.setId( 7 );
197 		e.setName( "Steve" );
198 		e.setSex( 'M' );
199 		e.setTitle( "grand poobah" );
200 		s.save( e );
201 		s.getTransaction().commit();
202 		s.close();
203 
204 		s = openSession();
205 		s.beginTransaction();
206 		// load the superclass proxy.
207 		Person pLoad = ( Person ) s.load( Person.class, new Long( e.getId() ) );
208 		assertTrue( pLoad instanceof HibernateProxy);
209 		Person pGet = ( Person ) s.get( Person.class, e.getId());
210 		Person pQuery = ( Person ) s.createQuery( "from org.hibernate.test.discriminator.Person where id = :id" )
211 				.setLong( "id", e.getId() )
212 				.uniqueResult();
213 		Person pCriteria = ( Person ) s.createCriteria( Person.class )
214 				.add( Restrictions.idEq( e.getId() ) )
215 				.uniqueResult();
216 		// assert that executing the queries polymorphically returns the same proxy
217 		assertSame( pLoad, pGet );
218 		assertSame( pLoad, pQuery );
219 		assertSame( pLoad, pCriteria );
220 
221 		// assert that the proxy is not an instance of Employee
222 		assertFalse( pLoad instanceof Employee );
223 
224 		s.getTransaction().commit();
225 		s.close();
226 
227 		s = openSession();
228 		s.beginTransaction();
229 		s.delete( e );
230 		s.getTransaction().commit();
231 		s.close();
232 	}

234 	@Test
235 	public void testLoadSuperclassProxyEvictPolymorphicAccess() {
236 		Session s = openSession();
237 		s.beginTransaction();
238 		Employee e = new Employee();
239 		e.setId( 8 );
240 		e.setName( "Steve" );
241 		e.setSex( 'M' );
242 		e.setTitle( "grand poobah" );
243 		s.save( e );
244 		s.getTransaction().commit();
245 		s.close();
246 
247 		s = openSession();
248 		s.beginTransaction();
249 		// load the superclass proxy.
250 		Person pLoad = ( Person ) s.load( Person.class, new Long( e.getId() ) );
251 		assertTrue( pLoad instanceof HibernateProxy);
252 		// evict the proxy
253 		s.evict( pLoad );
254 		Employee pGet = ( Employee ) s.get( Person.class, e.getId() );
255 		Employee pQuery = ( Employee ) s.createQuery( "from org.hibernate.test.discriminator.Person where id = :id" )
256 				.setLong( "id", e.getId() )
257 				.uniqueResult();
258 		Employee pCriteria = ( Employee ) s.createCriteria( Person.class )
259 				.add( Restrictions.idEq( e.getId() ) )
260 				.uniqueResult();
261 		// assert that executing the queries polymorphically returns the same Employee instance
262 		assertSame( pGet, pQuery );
263 		assertSame( pGet, pCriteria );
264 		s.getTransaction().commit();
265 		s.close();
266 
267 		s = openSession();
268 		s.beginTransaction();
269 		s.delete( e );
270 		s.getTransaction().commit();
271 		s.close();
272 	}
