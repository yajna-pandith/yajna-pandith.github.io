522     private boolean updateInstitutionCode(Dataset oldCode, Dataset newCode) {
523         if (oldCode == null ? newCode == null 
524                 : oldCode.equals(newCode)) {
525             return false;
526         }
527         try {
528             setInstitutionCode(newCode == null ? null
529                     : CodeBean.valueOf(codeHome, newCode));
530         } catch (CreateException e) {
531             throw new EJBException(e);
532         } catch (FinderException e) {
533             throw new EJBException(e);
534         }
535         return true;
536     }

538     /**
539      * Update series attributes and SeriesRequest.
540      * <p>
541      * Deletes SeriesRequest objects if RequestAttributesSeq in newAttrs is empty. 
542      *  
543      * @param ds Dataset with series attributes.
544      * @throws CreateException 
545      * 
546      * @ejb.interface-method
547      */
548     public boolean updateAttributes( Dataset newAttrs, 
549             boolean overwriteReqAttrSQ, Dataset modifiedAttrs ) {
550         Dataset oldAttrs = getAttributes(false);
551         updateSeriesRequest(oldAttrs, newAttrs, overwriteReqAttrSQ);
552         if ( oldAttrs == null ) {
553             updateInstitutionCode(null,
554                     newAttrs.getItem(Tags.InstitutionCodeSeq));
555             setAttributes( newAttrs );
556         } else {
557             updateInstitutionCode(oldAttrs.getItem(Tags.InstitutionCodeSeq),
558                     newAttrs.getItem(Tags.InstitutionCodeSeq));
559             AttributeFilter filter = AttributeFilter.getSeriesAttributeFilter();
560             if (!AttrUtils.updateAttributes(oldAttrs, filter.filter(newAttrs),
561                     modifiedAttrs, log) )
562                 return false;
563             setAttributes(oldAttrs);
564         }
565         return true;
566     }
567     
568     private boolean updateSeriesRequest(Dataset oldAttrs, Dataset newAttrs,
569             boolean overwriteReqAttrSQ) {
570         DcmElement newReqAttrSQ = newAttrs.get(Tags.RequestAttributesSeq);
571         if (newReqAttrSQ == null) {
572             return false;
573         }
574         if (oldAttrs != null
575                 && newReqAttrSQ.equals(oldAttrs.get(Tags.RequestAttributesSeq))) {
576             return false;
577         }
578         Collection c = getRequestAttributes();
579         if ( overwriteReqAttrSQ && ! c.isEmpty() ) {
580             oldAttrs.remove(Tags.RequestAttributesSeq);//remove to force update of RequestAttributesSeq
581             SeriesRequestLocal[] srls = new SeriesRequestLocal[c.size()];
582             srls = (SeriesRequestLocal[]) c.toArray(srls);

775     public Collection ejbHomeListPriorOfPatient(PatientLocal pat, String ignoreStudyIUID, Timestamp createdAfter, int minAvail, 
776             String[] modalities) throws FinderException {
777         StringBuffer jbossQl = new StringBuffer(
778                 "SELECT OBJECT(s) FROM Series s");
779         jbossQl.append(" WHERE s.study.patient = ?1");
780         ArrayList params = new ArrayList();
781         params.add(pat);
782         int idx = 2;
783         if (ignoreStudyIUID != null) {
784             jbossQl.append(" AND s.study.studyIuid <> ?").append(idx++);
785             params.add(ignoreStudyIUID);
786         }
787         if (createdAfter != null) {
788             jbossQl.append(" AND s.study.createdTime > ?").append(idx++);
789             params.add(createdAfter);
790         }
791         if (minAvail < Availability.UNAVAILABLE) {
792             jbossQl.append(" AND s.availability <= ?").append(idx++);
793             params.add(minAvail);
794         }
795         if (modalities != null && modalities.length > 0) {
796             jbossQl.append(" AND s.modality");
797             idx = addIN(jbossQl, params, idx, modalities);
798         }
799         log.debug("Execute JBossQL: " + jbossQl);
800         return ejbSelectGeneric(jbossQl.toString(), params.toArray());
801     }

818     private int addIN(StringBuffer jbossQl, List params, int idx, String[] values) {
819         params.add(values[0]);
820         if (values.length > 1) {
821             jbossQl.append(" IN ( ");
822             for (int i = 1; i < values.length; i++) {
823                 jbossQl.append("?").append(idx++).append(", ");
824                 params.add(values[i]);
825             }
826             jbossQl.append("?").append(idx++).append(")");
827         } else {
828             jbossQl.append(" = ?").append(idx++);
829         }
830         return idx;
831     }

1125                 setAttributes(attrs);
1126             }
1127         }

1129                 attrs.getItem(Tags.InstitutionCodeSeq));
