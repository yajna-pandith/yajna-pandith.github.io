146     public String getOrderControlOperationMap() {
147         StringBuffer sb = new StringBuffer();
148         for (int i = 0; i < ops.length; i++) {
149             sb.append(orderControls.get(i)).append(':')
150                 .append(OP_CODES[ops[i]]).append("\r\n");
151         }
152         return sb.toString();
153     }
154 
155     public void setOrderControlOperationMap(String s) {
156         StringTokenizer stk = new StringTokenizer(s, " \r\n\t,;");
157         int lines = stk.countTokens();
158         int[] newops = new int[lines];
159         List<String> newocs = new ArrayList<String>(lines);
160         for (int i = 0; i < lines; i++) {
161             String[] ocop = StringUtils.split(stk.nextToken(), ':');
162             if (ocop.length != 2
163                     || (newops[i] = OP_CODES_LIST.indexOf(ocop[1])) == -1) {
164                 throw new IllegalArgumentException(s);
165             }
166             newocs.add(ocop[0]);            
167         }
168         ops = newops;
169         orderControls = newocs;
170     }

232     private void process(int op[], Document msg, String[] xslSubdirs) throws HL7Exception {
233         try {
234             Dataset ds = xslt(msg, xslPath, xslSubdirs);
235             final String pid = ds.getString(Tags.PatientID);
236             if (pid == null)
237                 throw new HL7Exception("AR",
238                         "Missing required PID-3: Patient ID (Internal ID)");
239             final String pname = ds.getString(Tags.PatientName);
240             if (pname == null)
241                 throw new HL7Exception("AR",
242                         "Missing required PID-5: Patient Name");
243             mergeProtocolCodes(ds, op);
244             ds = addScheduledStationInfo(ds);
245             MWLManager mwlManager = getMWLManager();
246             DcmElement spsSq = ds.remove(Tags.SPSSeq);
247             Dataset sps;
248             int opIdx;
249             for (int i = 0, n = spsSq.countItems(); i < n; ++i) {
250                 sps = spsSq.getItem(i);
251                 ds.putSQ(Tags.SPSSeq).addItem(sps);
252                 adjustAttributes(ds);
253                 opIdx = op.length == 1 ? 0 : i;
254                 switch (op[opIdx]) {
255                 case NW:
256                 	processNW(ds, mwlManager);
257                     break;
258                 case XO:
259                 	processXO(ds, mwlManager);
260                 	break;
261                 case CA:
262                 	processCA(ds, mwlManager);
263                 	break;
264                 case NOOP:
265                     log("NOOP", ds);
266                     break;
267                 default:
268                     sps.putCS(Tags.SPSStatus, SPSStatus.toString(op[opIdx]-SC_OFF));
269                     updateSPSStatus(ds, mwlManager, createMissingOrderOnStatusChange);
270                     break;
271                 }
272             }
273         } catch (HL7Exception e) {
274             throw e;
275         } catch (PatientMismatchException e) {
276             throw new HL7Exception("AR", e.getMessage(), e);
277         } catch (PatientMergedException e) {
278             throw new HL7Exception("AR", e.getMessage(), e);
279         } catch (DuplicateMWLItemException e) {
280             throw new HL7Exception("AR", e.getMessage(), e);
281         } catch (Exception e) {
282             throw new HL7Exception("AE", e.getMessage(), e);
283         }
284     }

286     protected void processNW(Dataset ds, MWLManager mwlManager) throws Exception {
287         addMissingAttributes(ds);
288         log("Schedule", ds);
289         logDataset("Insert MWL Item:", ds);
290         mwlManager.addWorklistItem(ds, patientMatching);
291         updateRequestAttributes(ds, mwlManager);
292     }
293 
294     protected void processXO(Dataset ds, MWLManager mwlManager) throws Exception {
295         log("Update", ds);
296         logDataset("Update MWL Item:", ds);
297         if (!mwlManager.updateWorklistItem(ds, patientMatching)) {
298             log("No Such ", ds);
299             addMissingAttributes(ds);
300             log("->Schedule New ", ds);
301             logDataset("Insert MWL Item:", ds);
302             mwlManager.addWorklistItem(ds, patientMatching);
303         }
304         updateRequestAttributes(ds, mwlManager);
305     }
306 
307     protected void processCA(Dataset ds, MWLManager mwlManager) throws Exception {
308         log("Cancel", ds);
309         if (mwlManager.removeWorklistItem(ds, patientMatching) == null) {
310             log("No Such ", ds);
311         } else {
312             ds.getItem(Tags.SPSSeq).putCS(Tags.SPSStatus, "DISCONTINUED");
313             updateRequestAttributes(ds, mwlManager);
314         }
315     }

317     @SuppressWarnings("unchecked")
318     private void updateRequestAttributes(Dataset mwlitem,
319             MWLManager mwlManager) throws Exception {
320         MPPSManager mppsManager = getMPPSManager();
321         List<Dataset> mppsList = 
322             mppsManager.updateScheduledStepAttributes(mwlitem,
323                     patientMatching, updateDifferentPatientOfExistingStudy);
324         if (!mppsList.isEmpty()) {
325             if ("DISCONTINUED".equals(mwlitem.getItem(Tags.SPSSeq).getString(Tags.SPSStatus))) {
326                 HashSet<String> seriesIuids = new HashSet<String>();
327                 for (Dataset mpps : mppsList) {
328                     DcmElement perfSeriesSq = mpps.get(Tags.PerformedSeriesSeq);
329                     if (perfSeriesSq != null && !perfSeriesSq.isEmpty()) {
330                         for (int i = 0, n = perfSeriesSq.countItems(); i < n; i++) {
331                             seriesIuids.add(perfSeriesSq.getItem(i).getString(Tags.SeriesInstanceUID));
332                         }
333                     }
334                 }
335                 mppsManager.removeRequestAttributesInSeries(mwlitem, seriesIuids);
336             } else {
337                 updateSPSStatus(mwlitem, mppsList.get(0), mwlManager);
338                 updateRequestAttributesInSeries(mwlitem, mppsList, mppsManager);
339             }
340         }
341     }

343     private void updateRequestAttributesInSeries(Dataset mwlitem,
344             List<Dataset> mppsList, MPPSManager mppsManager) throws Exception {
345         for (Dataset mpps : mppsList) {
346             String aet = mpps.getString(Tags.PerformedStationAET);
347             Templates xslt = null;
348             try {
349                 if ( aeTemplates.getTemplatesServiceName() == null) {
350                     aeTemplates.setTemplatesServiceName(this.getTemplatesServiceName());
351                 }
352                 xslt = aeTemplates.getTemplatesForAET(aet, MWL2STORE_XSL);
353             } catch (Exception x) {
354                 log.error("Internal error to get template for "+MWL2STORE_XSL, x);
355             }
356             if (xslt == null) {
357                 log.warn("Failed to find or load stylesheet "
358                             + MWL2STORE_XSL
359                             + " for "
360                             + aet
361                             + ". Cannot update object attributes with request information.");
362                 continue;
363             }
364             Dataset rqAttrs = DcmObjectFactory.getInstance().newDataset();
365             XSLTUtils.xslt(mwlitem, xslt, rqAttrs);
366             DcmElement perfSeriesSq = mpps.get(Tags.PerformedSeriesSeq);
367             if (perfSeriesSq != null && !perfSeriesSq.isEmpty()) {
368                 for (int i = 0, n = perfSeriesSq.countItems(); i < n; i++) {
369                     String uid = perfSeriesSq.getItem(i)
370                             .getString(Tags.SeriesInstanceUID);
371                     updateSeriesAttributes(mppsManager, uid , rqAttrs, i == 0);
372                 }
373             }
374         }
375     }

386     protected void updateSPSStatus(Dataset mwlitem, Dataset mpps,
387             MWLManager mwlManager) throws PatientMismatchException,
388             RemoteException {
389         String spsStatus;
390         Dataset sps = mwlitem.get(Tags.SPSSeq).getItem();
391         switch (PPSStatus.toInt(mpps.getString(Tags.PPSStatus))) {
392         case PPSStatus.IN_PROGRESS:
393             spsStatus = "STARTED";
394             break;
395         case PPSStatus.COMPLETED:
396             spsStatus = "COMPLETED";
397             break;
398         default: // PPSStatus.DISCONTINUED
399             spsStatus = "DISCONTINUED";
400             break;
401         }
402         sps.putCS(Tags.SPSStatus, spsStatus);
403         updateSPSStatus(mwlitem, mwlManager, false);
404     }

406     protected void updateSPSStatus(Dataset ds, MWLManager mwlManager, boolean createMissingOrder)
407             throws PatientMismatchException, RemoteException {
408         log("Change SPS status of MWL Item:", ds);
409         if (!mwlManager.updateSPSStatus(ds, patientMatching)) {
410             log("No Such ", ds);
411             if (createMissingOrder) {
412                 addMissingAttributes(ds);
413                 log("->Create new MWL Item with status "+ds.getString(Tags.SPSStatus)+":", ds);
414                 logDataset("Insert MWL Item:", ds);
415                 try {
416                     mwlManager.addWorklistItem(ds, patientMatching);
417                     updateRequestAttributes(ds, mwlManager);
418                 } catch (Exception x) {
419                     log.warn("Create missing order on Status Change (SC) failed!", x);
420                 }
421             }
422         }
423     }

449     @SuppressWarnings("unchecked")
450     private int[] toOp(Document msg) throws HL7Exception {
451         Element rootElement = msg.getRootElement();
452         List<Element> orcs = rootElement.elements("ORC");
453         List<Element> obrs = rootElement.elements("OBR");
454         if (orcs.isEmpty()) {
455             throw new HL7Exception("AR", "Missing ORC Segment");                             
456         }
457         int[] op = new int[orcs.size()];
458         for (int i = 0; i < op.length; i++) {           
459             List<Element> orc = orcs.get(i).elements("field");
460             String orderControl = getText(orc, 0);
461             String orderStatus = getText(orc, 4);
462             if (orderStatus.length() == 0 && obrs.size() > i) {
463                 // use Result Status (OBR-25), if no Order Status (ORC-5);
464                 List<Element> obr = obrs.get(i).elements("field");
465                 orderStatus = getText(obr, 24);
466             }
467             op[i] = toOp(orderControl, orderStatus);
468         }
469         return op;
470     }
471 
472     private int toOp(String orderControl, String orderStatus)
473             throws HL7Exception {
474         int opIndex = orderControls.indexOf(orderControl + "(" + orderStatus + ")");
475         if (opIndex == -1) {
476             opIndex = orderControls.indexOf(orderControl);
477             if (opIndex == -1) {
478                 throw new HL7Exception("AR", "Illegal Order Control Code ORC-1:"
479                         + orderControl);
480             }
481         }
482         return ops[opIndex];
483     }

485     private String getText(List<Element> fields, int i) throws HL7Exception {
486         try {
487             return fields.get(i).getText();
488         } catch (IndexOutOfBoundsException e) {
489             return "";
490         }
491     }

499     private void addMissingAttributes(Dataset ds) {
500         Dataset sps = ds.getItem(Tags.SPSSeq);
501         if (!sps.containsValue(Tags.ScheduledStationAET)) {
502             log.info("No Scheduled Station AET - use default: "
503                     + defaultStationAET);
504             sps.putAE(Tags.ScheduledStationAET, defaultStationAET);
505         }
506         if (!sps.containsValue(Tags.ScheduledStationName)) {
507             log.info("No Scheduled Station Name - use default: "
508                     + defaultStationName);
509             sps.putSH(Tags.ScheduledStationName, defaultStationName);
510         }
511         if (!sps.containsValue(Tags.Modality)) {
512             log.info("No Modality - use default: " + defaultModality);
513             sps.putCS(Tags.Modality, defaultModality);
514         }
515         if (!sps.containsValue(Tags.SPSStartDate)) {
516             log.info("No SPS Start Date - use current date/time");
517             Date now = new Date();
518             sps.putDA(Tags.SPSStartDate, now);
519             sps.putTM(Tags.SPSStartTime, now);
520         }
521     }

523     private void adjustAttributes(Dataset ds) {
524         Dataset sps = ds.getItem(Tags.SPSSeq);
525         String val;
526         Dataset code;
527         if ((val = sps.getString(Tags.RequestingPhysician)) != null) {
528             log.info("Detect Requesting Physician on SPS Level");
529             ds.putPN(Tags.RequestingPhysician, val);
530             sps.remove(Tags.RequestingPhysician);
531         }
532         if ((val = sps.getString(Tags.RequestingService)) != null) {
533             log.info("Detect Requesting Service on SPS Level");
534             ds.putLO(Tags.RequestingService, val);
535             sps.remove(Tags.RequestingService);
536         }
537         if ((val = sps.getString(Tags.StudyInstanceUID)) != null) {
538             log.info("Detect Study Instance UID on SPS Level");
539             ds.putUI(Tags.StudyInstanceUID, val);
540             sps.remove(Tags.StudyInstanceUID);
541         }
542         if ((val = sps.getString(Tags.AccessionNumber)) != null) {
543             log.info("Detect Accession Number on SPS Level");
544             ds.putSH(Tags.AccessionNumber, val);
545             sps.remove(Tags.AccessionNumber);
546         }
547         if ((val = sps.getString(Tags.RequestedProcedurePriority)) != null) {
548             log.info("Detect Requested Procedure Priority on SPS Level");
549             ds.putCS(Tags.RequestedProcedurePriority, val);
550             sps.remove(Tags.RequestedProcedurePriority);
551         }
552         if ((val = sps.getString(Tags.RequestedProcedureID)) != null) {
553             log.info("Detect Requested Procedure ID on SPS Level");
554             ds.putSH(Tags.RequestedProcedureID, val);
555             sps.remove(Tags.RequestedProcedureID);
556         }
557         if ((val = sps.getString(Tags.RequestedProcedureDescription)) != null) {
558             log.info("Detect Requested Procedure Description on SPS Level");
559             ds.putLO(Tags.RequestedProcedureDescription, val);
560             sps.remove(Tags.RequestedProcedureDescription);
561         }
562         if ((code = sps.getItem(Tags.RequestedProcedureCodeSeq)) != null) {
563             log.info("Detect Requested Procedure Code on SPS Level");
564             ds.putSQ(Tags.RequestedProcedureCodeSeq).addItem(code);
565             sps.remove(Tags.RequestedProcedureCodeSeq);
566         }
567     }

569     private void mergeProtocolCodes(Dataset orm, int[] op) {
570         DcmElement prevSpsSq = orm.remove(Tags.SPSSeq);
571         DcmElement newSpsSq = orm.putSQ(Tags.SPSSeq);
572         HashMap<String,DcmElement> spcSqMap = new HashMap<String,DcmElement>();
573         DcmElement spcSq0, spcSqI;
574         Dataset sps;
575         String spsid;
576         for (int i = 0, j = 0, n = prevSpsSq.countItems(); i < n; ++i) {
577             sps = prevSpsSq.getItem(i);
578             spsid = sps.getString(Tags.SPSID);
579             spcSqI = sps.get(Tags.ScheduledProtocolCodeSeq);
580             spcSq0 = spcSqMap.get(spsid);
581             if (spcSq0 != null) {
582                 spcSq0.addItem(spcSqI.getItem());
583             } else {
584                 spcSqMap.put(spsid, spcSqI);
585                 newSpsSq.addItem(sps);
586                 if ( op.length > 1 ) 
587                     op[j++] = op[i];
588             }
589         }
590     }
