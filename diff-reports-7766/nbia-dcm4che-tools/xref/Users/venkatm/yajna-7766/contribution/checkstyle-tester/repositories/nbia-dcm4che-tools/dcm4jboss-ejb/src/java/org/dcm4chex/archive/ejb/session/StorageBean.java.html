195     /**
196      * @ejb.interface-method
197      */
198     public org.dcm4che.data.Dataset store(org.dcm4che.data.Dataset ds,
199             long fspk, java.lang.String fileid, long size, byte[] md5, int fileStatus,
200             boolean updateStudyAccessTime, boolean clearExternalRetrieveAET, PatientMatching matching,
201             boolean canRollback) throws DcmServiceException, NonUniquePatientIDException {
202         FileMetaInfo fmi = ds.getFileMetaInfo();
203         final String iuid = fmi.getMediaStorageSOPInstanceUID();
204         final String tsuid = fmi.getTransferSyntaxUID();
205         log.info("inserting instance " + fmi);
206         try {
207             Dataset coercedElements = DcmObjectFactory.getInstance().newDataset();
208             InstanceLocal instance;
209             int prevAvailability = Availability.UNAVAILABLE;
210             try {
211                 instance = instHome.findBySopIuid(iuid);
212                 prevAvailability = instance.getAvailabilitySafe();
213                 coerceInstanceIdentity(instance, ds, coercedElements);
214                 if (clearExternalRetrieveAET && instance.getExternalRetrieveAET() != null) {
215                     log.info("Clear ExternalRetrieveAET of instance "+instance.getSopIuid());
216                     instance.setExternalRetrieveAET(null);
217                 }
218             } catch (ObjectNotFoundException onfe) {
219                 instance = instHome.create(ds,
220                         getSeries(matching, ds, coercedElements));
221             }
222             if (fspk != -1) {
223                 FileSystemLocal fs = fileSystemHome.findByPrimaryKey(new Long(fspk));
224                 fileHome.create(fileid, tsuid, size, md5, fileStatus, instance, fs);
225                 instance.setAvailability(Math.min(fs.getAvailability(), prevAvailability));
226                 instance.addRetrieveAET(fs.getRetrieveAET());
227                 if (updateStudyAccessTime) {
228                     touchStudyOnFileSystem(ds.getString(Tags.StudyInstanceUID), fs);
229                 }
230             } else {
231                 instance.setAvailability(
232                         Availability.toInt(ds.getString(Tags.InstanceAvailability)));
233                 instance.setExternalRetrieveAET(ds.getString(Tags.RetrieveAET));
234             }
235             if (fileStatus == FileStatus.ARCHIVED)
236                 instance.setArchived(true);
237             instance.setInstanceStatus(RECEIVED);
238             instance.getSeries().setSeriesStatus(RECEIVED);
239             log.info("inserted records for instance[uid=" + iuid + "]");
240             return coercedElements;
241         } catch (Exception e) {
242             log.warn("inserting records for instance[uid=" + iuid
243                     + "] failed: " + e.getMessage());
244             if (canRollback) sessionCtx.setRollbackOnly(); 
245             if (e instanceof NonUniquePatientIDException) {
246                 throw (NonUniquePatientIDException) e;
247             } else if (e instanceof DcmServiceException){
248                 throw (DcmServiceException) e;
249             } else {
250                 throw new DcmServiceException(Status.ProcessingFailure, e);
251             }
252         }
253     }

255     private void touchStudyOnFileSystem(String siud, FileSystemLocal fs)
256             throws FinderException, CreateException {
257         String dirPath = fs.getDirectoryPath();
258         try {
259             sofHome.findByStudyAndFileSystem(siud, dirPath).touch();
260         } catch (ObjectNotFoundException e) {
261             try {
262                 sofHome.create(studyHome.findByStudyIuid(siud), fs);
263             } catch (Exception ignore) {
264                 log.info("Create StudyOnFS failed! Check if concurrent create.", ignore);
265                 sofHome.findByStudyAndFileSystem(siud, dirPath).touch();
266             }
267         }

278     /**
279      * @ejb.interface-method
280      */
281     public void commitSeriesStored(SeriesStored seriesStored)
282             throws FinderException {
283         Dataset ian = seriesStored.getIAN();
284         Dataset refSeries = ian.get(Tags.RefSeriesSeq).getItem(0);
285         DcmElement refSOPs = refSeries.get(Tags.RefSOPSeq);
286         int numI = refSOPs.countItems();
287         HashSet iuids = new HashSet(numI * 4 / 3 + 1);
288         for (int i = 0; i < numI; i++) {
289             iuids.add(refSOPs.getItem(i).getString(Tags.RefSOPInstanceUID));
290         }
291         String seriuid = refSeries.getString(Tags.SeriesInstanceUID);
292         SeriesLocal series = findBySeriesIuid(seriuid);
293         Collection c = series.getInstances();
294         int remaining = 0;
295         for (Iterator iter = c.iterator(); iter.hasNext();) {
296             InstanceLocal inst = (InstanceLocal) iter.next();
297             if (inst.getInstanceStatus() != RECEIVED) {
298                 continue;
299             }
300             if (iuids.remove(inst.getSopIuid())) {
301                 inst.setInstanceStatus(STORED);
302             } else {
303                 ++remaining;
304             }
305         }
306         if (remaining == 0) {
307             series.setSeriesStatus(STORED);
308         }
309    }

320     /**
321      * @ejb.interface-method
322      */
323     public Collection<Long> getPksOfPendingSeries(Map<String, Long> delays)
324             throws FinderException {
325         Long minDelay = null;
326         for (Long delay : delays.values()) {
327             if (minDelay == null || delay < minDelay) 
328                 minDelay = delay;
329         }
330         long ts = System.currentTimeMillis();
331         Timestamp updatedBefore = new Timestamp(ts - minDelay);
332         Collection<SeriesLocal> c = seriesHome.getSeriesWithStatusAndUpdatedBefore(RECEIVED, updatedBefore);
333         String aet;
334         Long delay;
335         ArrayList<Long> pks = new ArrayList<Long>();
336         for (SeriesLocal s: c) {
337             aet = s.getSourceAET();
338             delay = delays.get(aet);
339             if (delay == null) {
340                 delay = delays.get(null);
341                 if (delay == null) {
342                     log.warn("No default delay and delay for sourceAET "+aet+" is not configured! Skipped!");
343                     continue;
344                 }
345             }
346             if (s.getUpdatedTime().getTime() < ts - delay) {
347                 pks.add(s.getPk());
348             }
349         }
350         return pks;
351     }

364     private SeriesStored makeSeriesStored(SeriesLocal series)
365             throws FinderException {
366         StudyLocal study = series.getStudy();
367         PatientLocal pat = study.getPatient();
368         if ( pat == null ) {
369             log.warn("Failed: SeriesStored for series "+series.getSeriesIuid()+" (pk="+series.getPk()+
370                     ") Reason: Missing reference to a patient! Study iuid:"+study.getStudyIuid());
371             return null;
372         }
373         UpdateDerivedFieldsUtils.updateDerivedFieldsOf(series);
374         UpdateDerivedFieldsUtils.updateDerivedFieldsOf(study);
375         Dataset ian = DcmObjectFactory.getInstance().newDataset();
376         ian.putUI(Tags.StudyInstanceUID, study.getStudyIuid());
377         Dataset refSeries = ian.putSQ(Tags.RefSeriesSeq).addNewItem();
378         DcmElement refSOPs = refSeries.putSQ(Tags.RefSOPSeq);
379         refSeries.putUI(Tags.SeriesInstanceUID, series.getSeriesIuid());
380         HashSet commonRetrieveAETs = null;
381         Collection c = series.getInstances();
382         for (Iterator iter = c.iterator(); iter.hasNext();) {
383             InstanceLocal inst = (InstanceLocal) iter.next();
384             if (inst.getInstanceStatus() != RECEIVED) {
385                 continue;
386             }
387             String[] retrieveAETs = StringUtils.split(
388                     inst.getRetrieveAETs(), '\\');
389             if ( retrieveAETs == null ) {
390                 log.debug("Use SeriesStored without local file for series "+series.getSeriesIuid()+" (pk="+series.getPk()+
391                         ") Reason: Found Instance with status RECEIVED but without Retrieve AET! iuid:"+inst.getSopIuid());
392                 if (commonRetrieveAETs == null)
393                     commonRetrieveAETs = new HashSet();
394             } else if (commonRetrieveAETs == null) {
395                 commonRetrieveAETs = new HashSet();
396                 commonRetrieveAETs.addAll(Arrays.asList(retrieveAETs));
397             } else {
398                 commonRetrieveAETs.retainAll(Arrays.asList(retrieveAETs));
399             }
400             Dataset refSOP = refSOPs.addNewItem();           
401             refSOP.putUI(Tags.RefSOPClassUID, inst.getSopCuid());
402             refSOP.putUI(Tags.RefSOPInstanceUID, inst.getSopIuid());
403             refSOP.putAE(Tags.RetrieveAET, retrieveAETs);
404             refSOP.putCS(Tags.InstanceAvailability,
405                     Availability.toString(inst.getAvailabilitySafe()));
406         }
407         if (commonRetrieveAETs == null) {
408             return null;
409         }
410         Dataset patAttrs = pat.getAttributes(false);
411         Dataset studyAttrs = study.getAttributes(false);
412         Dataset seriesAttrs = series.getAttributes(false);
413         Dataset pps = seriesAttrs.getItem(Tags.RefPPSSeq);
414         DcmElement refPPSSeq = ian.putSQ(Tags.RefPPSSeq);
415         if (pps != null) {
416             if (!pps.contains(Tags.PerformedWorkitemCodeSeq)) {
417                 pps.putSQ(Tags.PerformedWorkitemCodeSeq);
418             }
419             refPPSSeq.addItem(pps);
420         }
421         SeriesStored seriesStored = new SeriesStored(series.getSourceAET(),
422                 commonRetrieveAETs.isEmpty() ? null
423                         : (String) commonRetrieveAETs.iterator().next(),
424                 patAttrs, studyAttrs, seriesAttrs, ian);
425         return seriesStored;
426     }

451     private SeriesLocal getSeries(PatientMatching matching, Dataset ds,
452             Dataset coercedElements) throws Exception {
453         final String uid = ds.getString(Tags.SeriesInstanceUID);
454         SeriesLocal series;
455         try {
456             series = findBySeriesIuid(uid);
457         } catch (ObjectNotFoundException onfe) {
458             try {
459                 return seriesHome.create(ds,
460                         getStudy(matching, ds, coercedElements));
461             } catch (CreateException e1) {
462                 // check if Series record was inserted by concurrent thread
463                 try {
464                     series = findBySeriesIuid(uid);
465                 } catch (Exception e2) {
466                     throw e1;
467                 }
468             }
469         }
470         coerceSeriesIdentity(series, ds, coercedElements);
471         return series;
472     }

497     private PatientLocal getPatient(PatientMatching matching, Dataset ds,
498             Dataset coercedElements) throws Exception {
499         PatientLocal pat;
500         try {
501             pat = patHome.selectPatient(ds, matching, true);
502         } catch (ObjectNotFoundException onfe) {
503             try {
504                 pat = patHome.create(ds);
505                 // Check if patient record was also inserted by concurrent thread
506                 try {
507                     return patHome.selectPatient(ds, matching, true);
508                 } catch (NonUniquePatientException nupe) {
509                     pat.remove();
510                     pat = patHome.selectPatient(ds, matching, true);
511                 } catch (ObjectNotFoundException onfe2) {
512                     // Just inserted Patient not found because of missing value
513                     // of attribute configured as required for Patient Matching
514                     return pat;
515                 }
516              } catch (CreateException ce) {
517                 // Check if patient record was inserted by concurrent thread
518                 // with unique index on (pat_id, pat_id_issuer)
519                  try {
520                      pat = patHome.selectPatient(ds, matching, true);
521                  } catch (ObjectNotFoundException onfe2) {
522                      throw ce;
523                  }
524             }
525         } catch (NonUniquePatientException nupe) {
526             return patHome.create(ds);
527         }
528         coercePatientIdentity(pat, ds, coercedElements);
529         return pat;
530     }

543     private PatientLocal getPatient(StudyLocal study, Dataset ds)
544             throws Exception{
545         String pid = ds.getString(Tags.PatientID);
546         String issuer = ds.getString(Tags.IssuerOfPatientID);
547         PatientLocal priorPat = study.getPatient();
548         if (priorPat.getIssuerOfPatientId() == null && issuer != null) {
549             try {
550                 PatientLocal dominantPat =
551                         patHome.selectPatient(pid, issuer);
552                 if (!dominantPat.isIdentical(priorPat)) {
553                     log.info("Detect duplicate Patient Record: "
554                             + dominantPat.asString());
555                     dominantPat.getStudies().addAll(priorPat.getStudies());
556                     dominantPat.getMpps().addAll(priorPat.getMpps());
557                     dominantPat.getMwlItems().addAll(priorPat.getMwlItems());
558                     dominantPat.getGsps().addAll(priorPat.getGsps());
559                     dominantPat.getGppps().addAll(priorPat.getGppps());
560                     dominantPat.getUPS().addAll(priorPat.getUPS());
561                     dominantPat.getMerged().addAll(priorPat.getMerged());
562                     priorPat.remove();
563                     return dominantPat;
564                 }
565             } catch (ObjectNotFoundException e) {}
566         }
567         return priorPat;
568     }

593     public void commited(Dataset stgCmtResult) throws FinderException {
594         DcmElement refSOPSeq = stgCmtResult.get(Tags.RefSOPSeq);
595         if (refSOPSeq == null) return;
596         HashSet seriesSet = new HashSet();
597         HashSet studySet = new HashSet();
598         final String aet0 = stgCmtResult.getString(Tags.RetrieveAET);
599         for (int i = 0, n = refSOPSeq.countItems(); i < n; ++i) {
600             final Dataset refSOP = refSOPSeq.getItem(i);
601             final String iuid = refSOP.getString(Tags.RefSOPInstanceUID);
602             final String aet = refSOP.getString(Tags.RetrieveAET, aet0);
603             if (iuid != null && aet != null)
604                 commited(seriesSet, studySet, iuid, aet);
605         }
606         for (Iterator series = seriesSet.iterator(); series.hasNext();) {
607             final SeriesLocal ser = findBySeriesIuid((String) series.next());
608             ser.updateExternalRetrieveAET();
609         }
610         for (Iterator studies = studySet.iterator(); studies.hasNext();) {
611             final StudyLocal study = studyHome.findByStudyIuid((String) studies.next());
612             study.updateExternalRetrieveAET();
613         }
614     }

617             final String iuid, final String aet) throws FinderException {
618         InstanceLocal inst = instHome.findBySopIuid(iuid);

620         SeriesLocal series = inst.getSeries();
621         seriesSet.add(series.getSeriesIuid());
622         StudyLocal study = series.getStudy();
623         studySet.add(study.getStudyIuid());
624     }

690     private SeriesLocal findBySeriesIuid(String uid)
691             throws javax.ejb.FinderException {
692         Long pk = (Long)seriesPkCache.get(uid);
693         if (pk != null) {
694             try { 
695                 return seriesHome.findByPrimaryKey(pk);
696             } catch ( ObjectNotFoundException x ) {
697                 log.warn("Series "+uid+" not found with cached pk! Cache entry removed!");
698                 seriesPkCache.remove(uid);
699             }
700         }
701         SeriesLocal ser = seriesHome.findBySeriesIuid(uid);
702         seriesPkCache.put(uid, ser.getPk());
703         return ser;
704     }

723     /**
724      * @ejb.interface-method
725      */
726     public List<String> getSopIuidsForRejectionNote(Dataset rejNote, String srcAet)
727             throws FinderException{
728         List<String> iuids = new ArrayList<String>();
729         DcmElement sq = rejNote.get(Tags.CurrentRequestedProcedureEvidenceSeq);
730         Dataset seriesItem;
731         SeriesLocal series;
732         String seriesIuid;
733         for (int i = 0, iLen = sq.countItems(); i < iLen; i++) {
734             DcmElement refSerSq = sq.getItem(i).get(Tags.RefSeriesSeq);
735             for (int j = 0, jLen = refSerSq.countItems(); j < jLen; j++) {
736                 seriesItem = refSerSq.getItem(j);
737                 seriesIuid = seriesItem.getString(Tags.SeriesInstanceUID);
738                 try {
739                     series = seriesHome.findBySeriesIuid(seriesIuid);
740                     if (!srcAet.equals(series.getSourceAET()) ) {
741                         log.info("CallingAET ("+srcAet+") of RejectionNote request is different to SourceAET ("+
742                                 series.getSourceAET()+") of Series "+seriesIuid+"! Ignore this series in RejectionNote!");
743                     } else {
744                         DcmElement refSopSq = seriesItem.get(Tags.RefSOPSeq);
745                         for (int k = 0, kLen = refSopSq.countItems(); k < kLen; k++) {
746                             iuids.add(refSopSq.getItem(k).getString(Tags.RefSOPInstanceUID));
747                         }
748                     }
749                 } catch (FinderException x) {
750                     log.warn("Series "+seriesIuid+" not found. Ignore this series in RejectionNote!");
751                 }
752             }
753         }
754         return iuids;
755     }
