315     public int forward(byte[] hl7msg) {
316         XMLReader xmlReader = new HL7XMLReader();
317         SAXContentHandler hl7in = new SAXContentHandler();
318         xmlReader.setContentHandler(hl7in);
319         try {
320             InputSource in = new InputSource(new InputStreamReader(
321                     new ByteArrayInputStream(hl7msg), getCharsetName()));
322             xmlReader.parse(in);
323         } catch (Exception e) {
324             log.error("Failed to parse HL7 message", e);
325             return -1;
326         }
327         return forward(hl7msg, hl7in.getDocument());
328     }

330     private int forward(byte[] hl7msg, Document msg) {
331         MSH msh = new MSH(msg);
332         Map<String,String[]> param = new HashMap<String,String[]>();
333         String sending = msh.sendingApplication + '^' + msh.sendingFacility;
334         param.put("sending", new String[] { sending });
335         param.put("receiving", new String[] { msh.receivingApplication + '^'
336                 + msh.receivingFacility });
337         param.put("msgtype", new String[] { msh.messageType + '^'
338                 + msh.triggerEvent });
339         String[] dests = forwardingRules.getForwardDestinationsFor(param);
340         if (dests.length > 0) {
341             hl7msg = preprocessForward(hl7msg, msg, msh, sending);
342         }
343         int count = 0;
344         for (int i = 0; i < dests.length; i++) {
345             HL7SendOrder order = new HL7SendOrder(hl7msg, dests[i]);
346             try {
347                 order.processOrderProperties(msh);
348             } catch (Exception e) {
349                 log.error("Failed to process order properties for " + order, e);
350             }
351             try {
352                 log.info("Scheduling " + order);
353                 jmsDelegate.queue(queueName, order, Message.DEFAULT_PRIORITY,
354                         0L);
355                 ++count;
356             } catch (Exception e) {
357                 log.error("Failed to schedule " + order, e);
358             }
359         }
360         return count;
361     }

363     private byte[] preprocessForward(byte[] hl7msg, Document msg, MSH msh, String sending) {
364         String[] variations = new String[] {"_"+msh.messageType+"^"+msh.triggerEvent, "_"+msh.messageType, "" };
365         Templates xslt = templates.findTemplates(new String[]{sending}, FORWARD_XSL, variations, XSL_EXT);
366         if (xslt != null) {
367             log.info("Transform HL7 message with hl7forward stylesheet!");
368             try {
369                 Transformer t = xslt.newTransformer();
370                 ByteArrayOutputStream bos = new ByteArrayOutputStream(hl7msg.length);
371                 XMLWriter xmlWriter = new HL7XMLWriter(
372                         new OutputStreamWriter(bos, getCharsetName()));
373                 t.transform(new DocumentSource(msg), new SAXResult(xmlWriter.getContentHandler()));
374                 hl7msg = bos.toByteArray();
375             } catch (Exception x) {
376                 log.error("Can not apply hl7forward stylesheet!", x);
377             }
378             
379         }
380         return hl7msg;
381     }

383     public void onMessage(Message message) {
384         ObjectMessage om = (ObjectMessage) message;
385         try {
386             HL7SendOrder order = (HL7SendOrder) om.getObject();
387             try {
388             	log.info("Start processing " + order);
389                 sendTo(order.getHL7Message(), order.getReceiving());
390                 log.info("Finished processing " + order);
391             } catch (Exception e) {
392                 order.setThrowable(e);
393                 final int failureCount = order.getFailureCount() + 1;
394                 order.setFailureCount(failureCount);
395                 final long delay = retryIntervalls.getIntervall(failureCount);
396                 if (delay == -1L) {
397                     log.error("Give up to process " + order);
398                     jmsDelegate.fail(queueName,order);
399                 } else {
400                     log.warn("Failed to process " + order
401                             + ". Scheduling retry.", e);
402                     jmsDelegate.queue(queueName, order, 0, System
403                             .currentTimeMillis()
404                             + delay);
405                 }
406             }
407         } catch (JMSException e) {
408             log.error("jms error during processing message: " + message, e);
409         } catch (Throwable e) {
410             log.error("unexpected error during processing message: " + message,
411                     e);
412         }
413     }

415     public Document invoke(byte[] message, String receiver) throws Exception {
416         AEDTO localAE = new AEDTO();
417         localAE.setTitle(receiver);
418         localAE.setHostName("127.0.0.1");
419         localAE.setPort(getLocalHL7Port());
420         AEDTO remoteAE = LOCAL_HL7_AET.equals(receiver) 
421                 ? localAE : aeMgt().findByAET(receiver);
422         Socket s = tlsConfig.createSocket(localAE, remoteAE);
423         try {
424             MLLPDriver mllpDriver = new MLLPDriver(s.getInputStream(), s
425                     .getOutputStream(), true);
426             writeMessage(message, receiver, mllpDriver.getOutputStream());
427             mllpDriver.turn();
428             if (acTimeout > 0) {
429                 s.setSoTimeout(acTimeout);
430             }
431             if (!mllpDriver.hasMoreInput()) {
432                 throw new IOException("Receiver " + receiver
433                         + " closed socket " + s
434                         + " during waiting on response.");
435             }
436             return readMessage(mllpDriver.getInputStream());
437         } finally {
438             if (soCloseDelay > 0)
439                 try {
440                     Thread.sleep(soCloseDelay);
441                 } catch (InterruptedException ignore) {
442                 }
443             s.close();
444         }
445     }

447     public void sendTo(byte[] message, String receiver) throws Exception {
448         Document rsp = invoke(message, receiver);
449         checkResponse(rsp);
450     }
451 
452     private void checkResponse(Document rsp) throws HL7Exception {
453         MSH msh = new MSH(rsp);
454         if ("ACK".equals(msh.messageType)) {
455             ACK ack = new ACK(rsp);
456             if (!("AA".equals(ack.acknowledgmentCode)
457                     || "CA".equals(ack.acknowledgmentCode)))
458                 throw new HL7Exception(ack.acknowledgmentCode, ack.textMessage);
459         } else {
460             log.warn("Unsupport response message type: " + msh.messageType
461                     + '^' + msh.triggerEvent
462                     + ". Assume successful message forward.");
463         }
464     }

479     private Document readMessage(InputStream mllpIn) throws IOException,
480             SAXException {
481         InputSource in = new InputSource(mllpIn);
482         in.setEncoding(getCharsetName());
483         XMLReader xmlReader = new HL7XMLReader();
484         SAXContentHandler hl7in = new SAXContentHandler();
485         xmlReader.setContentHandler(hl7in);
486         xmlReader.parse(in);
487         Document msg = hl7in.getDocument();
488         return msg;
489     }

491     private void writeMessage(byte[] message, String receiving, OutputStream out)
492             throws UnsupportedEncodingException, IOException {
493         final String charsetName = getCharsetName();
494         int offs = writePartTo(out, message, '|', 0, 4); //write MSH 1-4
495         final int delim = receiving.indexOf('^');
496         out.write(receiving.substring(0, delim).getBytes(charsetName));
497         out.write('|');
498         out.write(receiving.substring(delim + 1).getBytes(charsetName));
499         while (message[++offs] != '|') {} //skip MSH-5 receiving application
500         while (message[++offs] != '|') {} //skip MSH-6 receiving facility
501         // write remaining message
502         out.write(message, offs, message.length - offs);
503     }
504     
505     private int writePartTo(OutputStream out, byte[] ba, char b, int offs, int count) throws IOException {
506         for ( int i = offs ; i < ba.length ; i++) {
507             if (ba[i]==b) {
508                 count--;
509                 if (count == 0) {
510                     out.write(ba, offs, i-offs+1);
511                     return i;
512                 }
513             }
514         }
515         return -1;
516     }

537             String sending, String receiving, boolean useForward)

554                 sendTo(sb.toString().getBytes(charsetName), receiving);
555             }
556         }

560     /**
561      * Sends a PDQ query with the parameters being contains in the input Map, and the output
562      * being contained in a List of Maps of Strings to Strings - the data being the result of the
563      * query.  The last list element is a continuation element if count is non-zero.
564      */
565     public List<Map<String,String>> sendQBP_Q22(String pdqManager, Map<String,String> query, String domain, int count, String continuation) 
566     throws Exception {
567         String timestamp = new SimpleDateFormat(DATETIME_FORMAT).format(new Date());
568         StringBuffer sb = makeMSH(timestamp, "QBP^Q22", null, pdqManager, ++msgCtrlid, "2.5");
569         String qpd = makeQPD_Q22(query,domain);
570         sb.append('\r').append(qpd).append("\rRCP|I||||||\r");
571         String s = sb.toString();
572         log.info("Query PDQ Manager " + pdqManager + ":\n"
573                 + s.replace('\r', '\n'));
574         final String charsetName = getCharsetName();
575         Document msg = invoke(s.getBytes(charsetName), pdqManager);
576         log.info("PDQ Query returns:");
577         logMessage(msg);
578         List<Map<String,String>> ret = parsePDQ(msg);
579         log.info("Returning PDQ response now:"+toString(ret));
580         return ret;
581     }

584     public static final String toString(List<Map<String,String>> lst) {        
585         StringBuffer sb = new StringBuffer("[");
586         boolean listFirst = true;
587         for(Map<String,String> map : lst) {
588             if(listFirst) listFirst = false;
589             else sb.append(",\n  ");
590             sb.append("{");
591             boolean first = true;
592             for(Map.Entry<String,String> me : map.entrySet()) {
593                 if( first ) first = false;
594                 else sb.append(", ");
595                 sb.append(me.getKey()).append(":\"").append(me.getValue()).append("\"");
596             }
597             sb.append("}");
598         }
599         sb.append(']');
600         return sb.toString();
601     }

655     /** Parse the PID entries into a list of maps */
656     public static final List<Map<String,String>> parsePDQ(Document msg) {
657         List<Map<String,String>> ret = new ArrayList<Map<String,String>>();
658         Element root = msg.getRootElement();
659         List<?> content = root.content();
660         for(Object c : content) {
661             if( !(c instanceof Element) ) continue;
662             Element e = (Element) c;
663             if( e.getName().equals("PID") ) {
664                 Map<String,String> pid = new HashMap<String,String>();
665                 pid.put("Type", "Patient");
666                 List<?> fields = e.elements(HL7XMLLiterate.TAG_FIELD);
667                 
668                 int pidNo = 0;
669                 for(Object f : fields) {
670                     pidNo++;
671                     if( pidNo>=FIELD_NAMES.length ) continue;
672                     String fieldName = FIELD_NAMES[pidNo];
673                     if( fieldName==null ) continue;
674                     Element field = (Element) f;
675                     if( field.isTextOnly() ) {
676                         String txt = field.getText();
677                         if( txt==null || txt.length() == 0 ) continue;
678                         pid.put(fieldName, txt);
679                         continue;
680                     }
681                     if( pidNo==3 ) {
682                         List<?> comps = field.elements(HL7XMLLiterate.TAG_COMPONENT);
683                         if (comps.size() < 3) {
684                             throw new IllegalArgumentException("Missing Authority in PID-3");           
685                         }
686                         Element authority = (Element) comps.get(2);
687                         List<?> authorityUID = authority.elements(HL7XMLLiterate.TAG_SUBCOMPONENT);
688                         pid.put("PatientID",field.getText());
689                         StringBuffer issuer = new StringBuffer(authority.getText());
690                         for (int i = 0; i < authorityUID.size(); i++) {
691                             issuer.append("&").append(((Element) authorityUID.get(i)).getText());
692                         }
693                         pid.put("IssuerOfPatientID", issuer.toString());
694                         continue;
695                     }
696                     if( pidNo==5 ) {
697                         String name = field.getText() + "^" + field.elementText(HL7XMLLiterate.TAG_COMPONENT);
698                         pid.put(fieldName,name);
699                         continue;
700                     }
701                     pid.put(fieldName, field.asXML());
702                 }
703                 ret.add(pid);
704             }
705         }
706         return ret;
707     }

741     @SuppressWarnings("unchecked")
742     public List<String[]> sendQBP_Q23(String pixManager, String pixQueryName,
743             String patientID, String issuer, String[] domains)
744             throws Exception {
745         String timestamp = new SimpleDateFormat(DATETIME_FORMAT)
746                 .format(new Date());
747         long msgCtrlid = ++HL7SendService.msgCtrlid;
748         long queryTag = ++HL7SendService.queryTag;
749         StringBuffer sb = makeMSH(timestamp, "QBP^Q23^QBP_Q21", null, pixManager, msgCtrlid, "2.5");
750         String qpd = makeQPD(pixQueryName, queryTag, patientID, issuer, domains);
751         sb.append('\r').append(qpd).append("\rRCP|I||||||\r");
752         String s = sb.toString();
753         log.info("Query PIX Manager " + pixManager + ":\n"
754                 + s.replace('\r', '\n'));
755         final String charsetName = getCharsetName();
756         try {
757             Document msg = invoke(s.getBytes(charsetName), pixManager);
758             log.info("PIX Query returns:");
759             logMessage(msg);
760             MSH msh = new MSH(msg);
761             if (!"RSP".equals(msh.messageType) || !"K23".equals(msh.triggerEvent)) {
762                 String prompt = "Unsupport response message type: "
763                         + msh.messageType + '^' + msh.triggerEvent;
764                 log.error(prompt);
765                 throw new IOException(prompt);
766             }
767             RSP rsp = new RSP(msg);
768             if ("AE".equals(rsp.acknowledgmentCode)) {
769                 ERR err = new ERR(msg);
770                 if (err.isUnknownKeyIdentifier()) return null;
771             }
772             if (!"AA".equals(rsp.acknowledgmentCode)) {
773                 log.error("PIX Query fails with code " + rsp.acknowledgmentCode
774                         + " - " + rsp.textMessage);
775                 throw new HL7Exception(rsp.acknowledgmentCode, rsp.textMessage);
776             }
777             if (auditPIXQuery) {
778                 auditPIXQuery(pixManager, patientID, issuer, msgCtrlid, queryTag, qpd, null);
779             }
780             return rsp.getPatientIDs();
781         } catch (Exception e) {
782             if (auditPIXQuery) {
783                 auditPIXQuery(pixManager, patientID, issuer, msgCtrlid, queryTag, qpd, e);
784             }
785             throw e;
786         }
787     }

789     private String makeQPD(String pixQueryName, long queryTag, 
790             String patientID, String issuer, String[] domains) {
791         StringBuffer sb = new StringBuffer("QPD|");
792         sb.append(pixQueryName).append('|');
793         sb.append(queryTag).append('|');
794         sb.append(patientID).append("^^^").append(issuer);
795         if (domains != null && domains.length > 0) {
796             sb.append("|^^^").append(domains[0]);
797             for (int i = 1; i < domains.length; i++) {
798                 sb.append("~^^^").append(domains[i]);// ~ is repeat delimiter
799                 // used in makeMSH
800             }
801         }
802         return sb.toString();
803     }

805     private String makeQPD_Q22(Map<String,String> params,String domain) {
806         StringBuffer sb = new StringBuffer("QPD|IHE PDQ Query|");
807         sb.append((++queryTag)).append('|');
808         Iterator<Entry<String, String>> it = params.entrySet().iterator();
809         String sep = null;
810         while(it.hasNext()) {
811             Map.Entry<String,String> me = it.next();
812             if( sep==null ) sep = "~";
813             else sb.append(sep);
814             sb.append('@').append(me.getKey()).append("^").append(me.getValue());
815         }
816         sb.append("|||||");
817         if( domain!=null ) sb.append("^^^").append(domain);
818         return sb.toString();
819     }

913     private StringBuffer makeMSH(String timestamp, String msgType,
914             String sending, String receiving, long msgCtrlid, String version) {
915         StringBuffer sb = new StringBuffer();
916         sb.append("MSH|^~\\&|");
917         int delim;
918         if (sending == null || sending.trim().length() == 0) {
919             sb.append(getSendingApplication()).append("|");
920             sb.append(getSendingFacility()).append("|");
921         } else {
922             delim = sending.indexOf('^');
923             sb.append(sending.substring(0, delim)).append("|");
924             sb.append(sending.substring(delim + 1)).append("|");
925         }
926         delim = receiving.indexOf('^');
927         sb.append(receiving.substring(0, delim)).append("|");
928         sb.append(receiving.substring(delim + 1)).append("|");
929         sb.append(timestamp).append("||");
930         sb.append(msgType).append("|");
931         sb.append(msgCtrlid).append("|P|");
932         sb.append(version).append("||||||||");
933         return sb;
934     }

965     // concerns different order of name suffix, prefix in HL7 XPN compared to
966     // DICOM PN
967     private void addPersonName(StringBuffer sb, final String patName) {
968         StringTokenizer stk = new StringTokenizer(patName, "^", true);
969         for (int i = 0; i < 6 && stk.hasMoreTokens(); ++i) {
970             sb.append(stk.nextToken());
971         }
972         if (stk.hasMoreTokens()) {
973             String prefix = stk.nextToken();
974             if (stk.hasMoreTokens()) {
975                 stk.nextToken(); // skip delim
976                 if (stk.hasMoreTokens()) {
977                     sb.append(stk.nextToken()); // name suffix
978                 }
979             }
980             sb.append('^').append(prefix);
981         }
982     }

1022     public boolean sendHl7FromDataset( Dataset ds, Templates tpl, String sender, String receiver) {
1023         Socket s = null;
1024         try {
1025             AEDTO localAE = new AEDTO();
1026             localAE.setTitle(receiver);
1027             localAE.setHostName("127.0.0.1");
1028             localAE.setPort(getLocalHL7Port());
1029             AEDTO remoteAE = LOCAL_HL7_AET.equals(receiver) 
1030                     ? localAE : aeMgt().findByAET(receiver);
1031             s = tlsConfig.createSocket(localAE, remoteAE);
1032             MLLPDriver mllpDriver = new MLLPDriver(s.getInputStream(), s
1033                     .getOutputStream(), true);
1034             OutputStream out = mllpDriver.getOutputStream();
1035             writeDatasetAsHL7msg(ds, tpl, sender, receiver, out);         
1036             mllpDriver.turn();
1037             if (acTimeout > 0) {
1038                 s.setSoTimeout(acTimeout);
1039             }
1040             if (!mllpDriver.hasMoreInput()) {
1041                 throw new IOException("Receiver " + receiver
1042                         + " closed socket " + s
1043                         + " during waiting on response.");
1044             }
1045             Document rsp = readMessage(mllpDriver.getInputStream());
1046             checkResponse(rsp);
1047             return true;
1048         } catch (Exception x) {
1049             log.warn("Sending HL7 message from Dataset failed! Reason:"+x.getMessage(), x);
1050             final long delay = retryIntervalls.getIntervall(1);
1051             if (delay != -1L) {
1052                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
1053                 try {
1054                     writeDatasetAsHL7msg(ds, tpl, sender, receiver, baos);
1055                     byte[] msg = baos.toByteArray();
1056                     if ( msg.length > 3 && msg[0]=='M' && msg[1]=='S' && msg[2]=='H') {
1057                         HL7SendOrder order = new HL7SendOrder(msg, receiver);
1058                         order.setFailureCount(1);
1059                         jmsDelegate.queue(queueName, order, 0, System.currentTimeMillis() + delay);
1060                     } else {
1061                         log.error("Message generated from Dataset is not valid (Does not start with MSH!) and will not be scheduled for retry!");
1062                         log.debug("Dataset:");log.debug(ds);
1063                     }
1064                 } catch (Exception e) {
1065                     log.error("Cannot schedule 'retry' order for failed 'send HL7 message from Dataset'!",e);
1066                     return false;
1067                 }         
1068             }
1069             return false;
1070         } finally {
1071             if ( s != null ) {
1072                 if (soCloseDelay > 0)
1073                     try {
1074                         Thread.sleep(soCloseDelay);
1075                     } catch (InterruptedException ignore) {
1076                     }
1077                 try {
1078                     s.close();
1079                 } catch (IOException ignore) {
1080                 }
1081             }
1082         }
1083     }

1093     private TransformerHandler getTransformHandler(Templates tpl, String sender, String receiver)
1094             throws TransformerConfigurationException {
1095         TransformerHandler th = XSLTUtils.transformerFactory.newTransformerHandler(tpl);
1096         XSLTUtils.setDateParameters(th);
1097         Transformer t = th.getTransformer();
1098         final SimpleDateFormat tsFormat = new SimpleDateFormat("yyyyMMddHHmmss.SSS");
1099         t.setParameter("messageControlID", String.valueOf(++msgCtrlid) );
1100         t.setParameter("messageDateTime", tsFormat.format(new Date()));
1101         int pos = receiver.indexOf('^');
1102         t.setParameter("receivingApplication", receiver.substring(0, pos++));
1103         t.setParameter("receivingFacility", receiver.substring(pos));
1104         if ( sender != null ) {
1105             pos = sender.indexOf('^');
1106             t.setParameter("sendingApplication", sender.substring(0, pos++));
1107             t.setParameter("sendingFacility", sender.substring(pos));
1108         } else {
1109             t.setParameter("sendingApplication", sendingApplication);
1110             t.setParameter("sendingFacility", sendingFacility);
1111         }
1112         return th;
