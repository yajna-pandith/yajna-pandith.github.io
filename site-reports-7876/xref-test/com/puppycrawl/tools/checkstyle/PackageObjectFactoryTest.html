336     @Test
337     @SuppressWarnings("unchecked")
338     public void testNameToFullModuleNameMap() throws Exception {
339         final Set<Class<?>> classes = CheckUtil.getCheckstyleModules();
340         final Class<PackageObjectFactory> packageObjectFactoryClass = PackageObjectFactory.class;
341         final Field field = packageObjectFactoryClass.getDeclaredField("NAME_TO_FULL_MODULE_NAME");
342         field.setAccessible(true);
343         final Collection<String> canonicalNames = ((Map<String, String>) field.get(null)).values();
344 
345         final Optional<Class<?>> optional1 = classes.stream()
346                 .filter(clazz -> !canonicalNames.contains(clazz.getCanonicalName())).findFirst();
347         if (optional1.isPresent()) {
348             fail("Invalid canonical name: " + optional1.get());
349         }
350         final Optional<String> optional2 = canonicalNames.stream().filter(canonicalName -> {
351             return classes.stream().map(Class::getCanonicalName)
352                     .noneMatch(clssCanonicalName -> clssCanonicalName.equals(canonicalName));
353         }).findFirst();
354         if (optional2.isPresent()) {
355             fail("Invalid class: " + optional2.get());
356         }
357     }
