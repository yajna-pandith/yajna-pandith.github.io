20  package com.puppycrawl.tools.checkstyle.api;
21  
22  import static org.junit.jupiter.api.Assertions.assertEquals;
23  import static org.junit.jupiter.api.Assertions.assertNull;
24  
25  import org.junit.jupiter.api.Test;
26  
27  public class AuditEventTest {
28  
29      @Test
30      public void test() {
31          final AuditEvent event = new AuditEvent(getClass());
32  
33          assertNull(event.getFileName(), "invalid file name");
34          assertNull(event.getLocalizedMessage(), "invalid localized message");
35          assertEquals(getClass(), event.getSource(), "invalid source");
36          assertEquals(SeverityLevel.INFO, event.getSeverityLevel(), "invalid severity");
37      }
38  
39      @Test
40      public void testFullConstructor() {
41          final LocalizedMessage message = new LocalizedMessage(1, 2, 3, "bundle", "key", null,
42                  SeverityLevel.ERROR, "moduleId", getClass(), "customMessage");
43          final AuditEvent event = new AuditEvent(getClass(), "fileName", message);
44  
45          assertEquals("fileName", event.getFileName(), "invalid file name");
46          assertEquals(message, event.getLocalizedMessage(), "invalid localized message");
47          assertEquals("customMessage", event.getMessage(), "invalid message");
48          assertEquals(getClass(), event.getSource(), "invalid source");
49          assertEquals(1, event.getLine(), "invalid line");
50          assertEquals(2, event.getColumn(), "invalid column");
51          assertEquals(SeverityLevel.ERROR, event.getSeverityLevel(), "invalid severity");
52          assertEquals("moduleId", event.getModuleId(), "invalid module id");
53          assertEquals("com.puppycrawl.tools.checkstyle.api.AuditEventTest",
54                  event.getSourceName(), "invalid source name");
55      }
56  }
