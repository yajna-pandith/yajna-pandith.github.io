65      @Test
66      public void testProcessSequential() throws Exception {
67          final DummyFileSetCheck check = new DummyFileSetCheck();
68          check.configure(new DefaultConfiguration("filesetcheck"));
69          check.setFileExtensions("tmp");
70          final File firstFile = new File("inputAbstractFileSetCheck.tmp");
71          final SortedSet<LocalizedMessage> firstFileMessages =
72              check.process(firstFile, new FileText(firstFile, Collections.emptyList()));
73  
74          assertEquals("File should not be empty.",
75              firstFileMessages.first().getMessage(), "Invalid message");
76  
77          final SortedSet<LocalizedMessage> internalMessages =
78                  check.getMessages();
79          assertTrue(internalMessages.isEmpty(), "Internal message should be empty, but was not");
80  
81          final File secondFile = new File("inputAbstractFileSetCheck.txt");
82          final List<String> lines = Arrays.asList("key=value", "ext=tmp");
83          final SortedSet<LocalizedMessage> secondFileMessages =
84              check.process(secondFile, new FileText(secondFile, lines));
85  
86          assertTrue(secondFileMessages.isEmpty(), "Message should be empty, but was not");
87      }

102     @Test
103     public void testProcessException() throws Exception {
104         final ExceptionFileSetCheck check = new ExceptionFileSetCheck();
105         check.configure(new DefaultConfiguration("filesetcheck"));
106         check.setFileExtensions("tmp");
107         final File firstFile = new File("inputAbstractFileSetCheck.tmp");
108 
109         try {
110             check.process(firstFile, new FileText(firstFile, Collections.emptyList()));
111             fail("Exception is expected");
112         }
113         catch (IllegalArgumentException ex) {
114             // exception is expected
115             assertEquals("Test", ex.getMessage(), "Invalid exception message");
116         }
117 
118         final SortedSet<LocalizedMessage> internalMessages =
119                 check.getMessages();
120         assertEquals(1, internalMessages.size(), "Internal message should only have 1");
121 
122         // again to prove only 1 violation exists
123         final File secondFile = new File("inputAbstractFileSetCheck.tmp");
124         try {
125             check.process(secondFile, new FileText(secondFile, Collections.emptyList()));
126             fail("Exception is expected");
127         }
128         catch (IllegalArgumentException ex) {
129             // exception is expected
130             assertEquals("Test", ex.getMessage(), "Invalid exception message");
131         }
132 
133         final SortedSet<LocalizedMessage> internalMessages2 =
134             check.getMessages();
135         assertEquals(1, internalMessages2.size(), "Internal message should only have 1 again");
136     }

198     @Test
199     public void testMultiFileFireErrors() throws Exception {
200         final MultiFileViolationFileSetCheck check = new MultiFileViolationFileSetCheck();
201         check.configure(new DefaultConfiguration("filesetcheck"));
202         final ViolationDispatcher dispatcher = new ViolationDispatcher();
203         check.setMessageDispatcher(dispatcher);
204 
205         check.finishProcessing();
206 
207         assertEquals("fileName", dispatcher.name, "Invalid fileName reported");
208 
209         assertEquals(1, dispatcher.errorList.size(), "errors should only have 1");
210 
211         final LocalizedMessage message = dispatcher.errorList.iterator().next();
212         assertEquals(1, message.getLineNo(), "expected line");
213         assertEquals(0, message.getColumnNo(), "expected column");
214 
215         // re-running erases previous errors
216 
217         check.finishProcessing();
218 
219         assertEquals(1, dispatcher.errorList.size(), "errors should still have 1 after re-run");
220         assertEquals(2, MultiFileViolationFileSetCheck.finishProcessingCount,
221                 "finishProcessing was called twice");
222     }

248     public static class MultiFileViolationFileSetCheck extends AbstractFileSetCheck {
249 
250         private static final String MSG_KEY = "Violation.";
251         private static int finishProcessingCount;
252 
253         @Override
254         protected void processFiltered(File file, FileText fileText) {
255             // no code needed
256         }
257 
258         @Override
259         public void finishProcessing() {
260             final String fileName = "fileName";
261 
262             log(1, MSG_KEY + finishProcessingCount);
263             fireErrors(fileName);
264 
265             finishProcessingCount++;
266         }
267 
268     }

284     public static class ViolationDispatcher implements MessageDispatcher {
285         private String name;
286         private SortedSet<LocalizedMessage> errorList;
287 
288         @Override
289         public void fireFileStarted(String fileName) {
290             // no code needed
291         }
292 
293         @Override
294         public void fireFileFinished(String fileName) {
295             // no code needed
296         }
297 
298         @Override
299         public void fireErrors(String fileName, SortedSet<LocalizedMessage> errors) {
300             name = fileName;
301             errorList = new TreeSet<>(errors);
302         }
