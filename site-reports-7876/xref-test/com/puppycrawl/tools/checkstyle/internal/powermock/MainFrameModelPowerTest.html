20  package com.puppycrawl.tools.checkstyle.internal.powermock;
21  
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.assertFalse;
24  import static org.junit.Assert.assertTrue;
25  import static org.junit.Assert.fail;
26  
27  import java.io.File;
28  import java.io.IOException;
29  
30  import org.junit.Before;
31  import org.junit.Test;
32  import org.junit.runner.RunWith;
33  import org.powermock.api.mockito.PowerMockito;
34  import org.powermock.core.classloader.annotations.PrepareForTest;
35  import org.powermock.modules.junit4.PowerMockRunner;
36  import org.powermock.reflect.Whitebox;
37  
38  import com.puppycrawl.tools.checkstyle.AbstractModuleTestSupport;
39  import com.puppycrawl.tools.checkstyle.api.CheckstyleException;
40  import com.puppycrawl.tools.checkstyle.gui.MainFrameModel;
41  import com.puppycrawl.tools.checkstyle.gui.MainFrameModel.ParseMode;
42  
43  @RunWith(PowerMockRunner.class)
44  @PrepareForTest(ParseMode.class)
45  public class MainFrameModelPowerTest extends AbstractModuleTestSupport {
46  
47      private static final String FILE_NAME_TEST_DATA = "InputMainFrameModel.java";
48      private static final String FILE_NAME_NON_JAVA = "NotJavaFile.notjava";
49      private static final String FILE_NAME_NON_EXISTENT = "non-existent.file";
50  
51      private MainFrameModel model;
52      private File testData;
53  
54      @Override
55      protected String getPackageLocation() {
56          return "com/puppycrawl/tools/checkstyle/gui/mainframemodel";
57      }
58  
59      @Before
60      public void prepareTestData() throws IOException {
61          model = new MainFrameModel();
62          testData = new File(getPath(FILE_NAME_TEST_DATA));
63      }
64  
65      @Test
66      public void testShouldAcceptFile() throws IOException {
67          final File directory = PowerMockito.mock(File.class);
68          PowerMockito.when(directory.isDirectory()).thenReturn(true);
69          assertTrue("MainFrame should accept directory",
70                  MainFrameModel.shouldAcceptFile(directory));
71  
72          final File javaFile = new File(getPath(FILE_NAME_TEST_DATA));
73          assertTrue("MainFrame should accept java file",
74                  MainFrameModel.shouldAcceptFile(javaFile));
75  
76          final File nonJavaFile = PowerMockito.mock(File.class);
77          PowerMockito.when(nonJavaFile.isDirectory()).thenReturn(false);
78          PowerMockito.when(nonJavaFile.getName()).thenReturn(FILE_NAME_NON_JAVA);
79          assertFalse("MainFrame should not accept nonJava file",
80                  MainFrameModel.shouldAcceptFile(nonJavaFile));
81  
82          final File nonExistentFile = new File(getPath(FILE_NAME_NON_EXISTENT));
83          assertFalse("MainFrame should not accept nonexistent file",
84                  MainFrameModel.shouldAcceptFile(nonExistentFile));
85      }
86  
87      @Test
88      public void testOpenFileWithUnknownParseMode() throws CheckstyleException {
89          final ParseMode unknownParseMode = PowerMockito.mock(ParseMode.class);
90          Whitebox.setInternalState(unknownParseMode, "ordinal", 3);
91  
92          PowerMockito.when(unknownParseMode.toString()).thenReturn("Unknown parse mode");
93          PowerMockito.mockStatic(ParseMode.class);
94          PowerMockito.when(ParseMode.values()).thenReturn(
95                  new ParseMode[] {
96                      ParseMode.PLAIN_JAVA, ParseMode.JAVA_WITH_COMMENTS,
97                      ParseMode.JAVA_WITH_JAVADOC_AND_COMMENTS, unknownParseMode, });
98  
99          try {
100             model.setParseMode(unknownParseMode);
101             model.openFile(testData);
102 
103             fail("Expected IllegalArgumentException is not thrown.");
104         }
105         catch (IllegalArgumentException ex) {
106             assertEquals("Invalid exception message",
107                     "Unknown mode: Unknown parse mode", ex.getMessage());
108         }
109     }
110 
111 }
